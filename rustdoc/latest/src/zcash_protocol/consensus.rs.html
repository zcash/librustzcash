13/// A wrapper type representing blockchain heights.
14///
15/// Safe conversion from various integer types, as well as addition and subtraction, are
16/// provided.
17#[repr(transparent)]
18#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
19pub struct BlockHeight(u32);
20
21#[cfg(feature = "std")]
22memuse::impl_no_dynamic_usage!(BlockHeight);
23
24/// The height of the genesis block on a network.
25pub const H0: BlockHeight = BlockHeight(0);
26
27impl BlockHeight {
28    pub const fn from_u32(v: u32) -> BlockHeight {
29        BlockHeight(v)
30    }
31
32    /// Subtracts the provided value from this height, returning `H0` if this would result in
33    /// underflow of the wrapped `u32`.
34    pub fn saturating_sub(self, v: u32) -> BlockHeight {
35        BlockHeight(self.0.saturating_sub(v))
36    }
37}

41        self.0.fmt(formatter)
42    }
43}

322/// Zcash consensus parameters.
323pub trait Parameters: Clone {
324    /// Returns the type of network configured by this set of consensus parameters.
325    fn network_type(&self) -> NetworkType;
326
327    /// Returns the activation height for a particular network upgrade,
328    /// if an activation height has been set.
329    fn activation_height(&self, nu: NetworkUpgrade) -> Option<BlockHeight>;
330
331    /// Determines whether the specified network upgrade is active as of the
332    /// provided block height on the network to which this Parameters value applies.
333    fn is_nu_active(&self, nu: NetworkUpgrade, height: BlockHeight) -> bool {
334        self.activation_height(nu).is_some_and(|h| h <= height)
335    }
336}

569/// A globally-unique identifier for a set of consensus rules within the Zcash chain.
570///
571/// Each branch ID in this enum corresponds to one of the epochs between a pair of Zcash
572/// network upgrades. For example, `BranchId::Overwinter` corresponds to the blocks
573/// starting at Overwinter activation, and ending the block before Sapling activation.
574///
575/// The main use of the branch ID is in signature generation: transactions commit to a
576/// specific branch ID by including it as part of [`signature_hash`]. This ensures
577/// two-way replay protection for transactions across network upgrades.
578///
579/// See [ZIP 200](https://zips.z.cash/zip-0200) for more details.
580///
581/// [`signature_hash`]: https://docs.rs/zcash_primitives/latest/zcash_primitives/transaction/sighash/fn.signature_hash.html
582#[derive(Clone, Copy, Debug, PartialEq, Eq)]
583pub enum BranchId {
584    /// The consensus rules at the launch of Zcash.
585    Sprout,
586    /// The consensus rules deployed by [`NetworkUpgrade::Overwinter`].
587    Overwinter,
588    /// The consensus rules deployed by [`NetworkUpgrade::Sapling`].
589    Sapling,
590    /// The consensus rules deployed by [`NetworkUpgrade::Blossom`].
591    Blossom,
592    /// The consensus rules deployed by [`NetworkUpgrade::Heartwood`].
593    Heartwood,
594    /// The consensus rules deployed by [`NetworkUpgrade::Canopy`].
595    Canopy,
596    /// The consensus rules deployed by [`NetworkUpgrade::Nu5`].
597    Nu5,
598    /// The consensus rules deployed by [`NetworkUpgrade::Nu6`].
599    Nu6,
600    /// Candidates for future consensus rules; this branch will never
601    /// activate on mainnet.
602    #[cfg(zcash_unstable = "zfuture")]
603    ZFuture,
604}
605
606#[cfg(feature = "std")]
607memuse::impl_no_dynamic_usage!(BranchId);
608
609impl TryFrom<u32> for BranchId {
610    type Error = &'static str;
611
612    fn try_from(value: u32) -> Result<Self, Self::Error> {
613        match value {
614            0 => Ok(BranchId::Sprout),
615            0x5ba8_1b19 => Ok(BranchId::Overwinter),
616            0x76b8_09bb => Ok(BranchId::Sapling),
617            0x2bb4_0e60 => Ok(BranchId::Blossom),
618            0xf5b9_230b => Ok(BranchId::Heartwood),
619            0xe9ff_75a6 => Ok(BranchId::Canopy),
620            0xc2d6_d0b4 => Ok(BranchId::Nu5),
621            0xc8e7_1055 => Ok(BranchId::Nu6),
622            #[cfg(zcash_unstable = "zfuture")]
623            0xffff_ffff => Ok(BranchId::ZFuture),
624            _ => Err("Unknown consensus branch ID"),
625        }
626    }
627}

643    }

645
646impl BranchId {
647    /// Returns the branch ID corresponding to the consensus rule set that is active at
648    /// the given height.
649    ///
650    /// This is the branch ID that should be used when creating transactions.
651    pub fn for_height<P: Parameters>(parameters: &P, height: BlockHeight) -> Self {
652        for nu in UPGRADES_IN_ORDER.iter().rev() {
653            if parameters.is_nu_active(*nu, height) {
654                return nu.branch_id();
655            }
656        }
657
658        // Sprout rules apply before any network upgrade
659        BranchId::Sprout
660    }
661
662    /// Returns the range of heights for the consensus epoch associated with this branch id.
663    ///
664    /// The resulting tuple implements the [`RangeBounds<BlockHeight>`] trait.
665    pub fn height_range<P: Parameters>(&self, params: &P) -> Option<impl RangeBounds<BlockHeight>> {
666        self.height_bounds(params).map(|(lower, upper)| {
667            (
668                Bound::Included(lower),
669                upper.map_or(Bound::Unbounded, Bound::Excluded),
670            )
671        })
672    }
673
674    /// Returns the range of heights for the consensus epoch associated with this branch id.
675    ///
676    /// The return type of this value is slightly more precise than [`Self::height_range`]:
677    /// - `Some((x, Some(y)))` means that the consensus rules corresponding to this branch id
678    ///   are in effect for the range `x..y`
679    /// - `Some((x, None))` means that the consensus rules corresponding to this branch id are
680    ///   in effect for the range `x..`
681    /// - `None` means that the consensus rules corresponding to this branch id are never in effect.
682    pub fn height_bounds<P: Parameters>(
683        &self,
684        params: &P,
685    ) -> Option<(BlockHeight, Option<BlockHeight>)> {
686        match self {
687            BranchId::Sprout => params
688                .activation_height(NetworkUpgrade::Overwinter)
689                .map(|upper| (BlockHeight(0), Some(upper))),
690            BranchId::Overwinter => params
691                .activation_height(NetworkUpgrade::Overwinter)
692                .map(|lower| (lower, params.activation_height(NetworkUpgrade::Sapling))),
693            BranchId::Sapling => params
694                .activation_height(NetworkUpgrade::Sapling)
695                .map(|lower| (lower, params.activation_height(NetworkUpgrade::Blossom))),
696            BranchId::Blossom => params
697                .activation_height(NetworkUpgrade::Blossom)
698                .map(|lower| (lower, params.activation_height(NetworkUpgrade::Heartwood))),
699            BranchId::Heartwood => params
700                .activation_height(NetworkUpgrade::Heartwood)
701                .map(|lower| (lower, params.activation_height(NetworkUpgrade::Canopy))),
702            BranchId::Canopy => params
703                .activation_height(NetworkUpgrade::Canopy)
704                .map(|lower| (lower, params.activation_height(NetworkUpgrade::Nu5))),
705            BranchId::Nu5 => params
706                .activation_height(NetworkUpgrade::Nu5)
707                .map(|lower| (lower, params.activation_height(NetworkUpgrade::Nu6))),
708            BranchId::Nu6 => params.activation_height(NetworkUpgrade::Nu6).map(|lower| {
709                #[cfg(zcash_unstable = "zfuture")]
710                let upper = params.activation_height(NetworkUpgrade::ZFuture);
711                #[cfg(not(zcash_unstable = "zfuture"))]
712                let upper = None;
713                (lower, upper)
714            }),
715            #[cfg(zcash_unstable = "zfuture")]
716            BranchId::ZFuture => params
717                .activation_height(NetworkUpgrade::ZFuture)
718                .map(|lower| (lower, None)),
719        }
720    }
721
722    pub fn sprout_uses_groth_proofs(&self) -> bool {
723        !matches!(self, BranchId::Sprout | BranchId::Overwinter)
724    }
725}
