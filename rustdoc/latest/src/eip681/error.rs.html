7/// Errors discovered during parsing.
8#[derive(Debug, Snafu)]
9#[snafu(visibility(pub(crate)))]
10#[non_exhaustive]
11pub enum ParseError<'a> {
12    /// A parsing error generated by `nom`.
13    #[snafu(display("{}", nom::error::Error::to_string(&nom::error::Error {
14        input: *input, code: *code
15    })))]
16    Nom {
17        input: &'a str,
18        code: nom::error::ErrorKind,
19        other: Option<String>,
20    },
21
22    /// The number of digits did not match or exceed the expected minimum.
23    #[snafu(display(
24        "Expected at least {min} digits, saw {} before {}",
25        digits.len(),
26        input.split_at(10).0
27    ))]
28    DigitsMinimum {
29        min: usize,
30        digits: Vec<u8>,
31        input: &'a str,
32    },
33
34    /// The ENS name was expected and missing.
35    #[snafu(display("Missing ENS name"))]
36    EnsMissing,
37
38    /// The ENS name was not a valid domain.
39    #[snafu(display("Not a domain"))]
40    EnsDomain,
41
42    /// The ENS name could not be normalized.
43    #[snafu(display("{source}"))]
44    EnsNormalization {
45        source: ens_normalize_rs::ProcessError,
46    },
47
48    /// The ENS should be normalized, but wasn't.
49    #[snafu(display("ENS name must be normalized. Expected '{expected}' but saw '{seen}'"))]
50    NotNormalized { expected: String, seen: String },
51
52    /// A parameter value had an unexpected value.
53    #[snafu(display("Invalid parameter value. Expected a {ty}."))]
54    InvalidParameterValue { ty: String },
55
56    /// A parameter key had an unexpected value.
57    #[snafu(display("Invalid parameter key '{key}'."))]
58    InvalidParameterKey { key: String },
59
60    /// Unexpected leftover parsing input.
61    #[snafu(display("Unexpected leftover parsing input '{input}'"))]
62    UnexpectedLeftoverInput { input: &'a str },
63}
64
65impl<'a> nom::error::ParseError<&'a str> for ParseError<'a> {
66    fn from_error_kind(input: &'a str, code: nom::error::ErrorKind) -> Self {
67        NomSnafu {
68            input,
69            code,
70            other: None,
71        }
72        .build()
73    }
74
75    fn append(input: &'a str, code: nom::error::ErrorKind, other: Self) -> Self {
76        NomSnafu {
77            input,
78            code,
79            other: Some(other.to_string()),
80        }
81        .build()
82    }
83}
84
85impl<'a> From<ParseError<'a>> for nom::Err<ParseError<'a>> {
86    fn from(value: ParseError<'a>) -> Self {
87        nom::Err::Error(value)
88    }
89}

91/// Errors discovered after parsing, usually when attempting to convert parsed
92/// types into numeric types.
93#[derive(Debug, Snafu, PartialEq)]
94#[non_exhaustive]
95#[snafu(visibility(pub(crate)))]
96pub enum ValidationError {
97    #[snafu(display("Exponent is too small, expected at least {expected}, saw {seen}"))]
98    SmallExponent { expected: usize, seen: u64 },
99
100    #[snafu(display("Exponent is too large, expected at most {expected}, saw {seen}"))]
101    LargeExponent { expected: usize, seen: u64 },
102
103    #[snafu(display(
104        "Arithmetic operation resulted in an overflow, exceeding the allowable range"
105    ))]
106    Overflow,
107
108    #[snafu(display("Could not decode url-encoded string: {source}"))]
109    UrlEncoding { source: std::str::Utf8Error },
110
111    /// An attempt to convert an integer failed.
112    #[snafu(display("Integer conversion failed: {source}"))]
113    Integer { source: std::num::TryFromIntError },
114
115    /// More than one value exists in a parameter list with a key
116    /// that denotes a single value.
117    #[snafu(display(
118        "More than one parameter by the key '{key}', saw: [{}]",
119        values.iter().map(|(k, v)| format!("{k}={v}")).collect::<Vec<_>>().join(", ")
120    ))]
121    MultipleParameterValues {
122        key: &'static str,
123        values: Vec<(&'static str, Value)>,
124    },
125}
