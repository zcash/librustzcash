20/// Succeeds if all the input has been consumed by its child parser.
21///
22/// Identical to [`nom::combinator::all_consuming`] except it only accepts `&str` inputs
23/// and returns a different error.
24fn all_consuming<'i, O, F>(
25    mut f: F,
26) -> impl FnMut(&'i str) -> nom::IResult<&'i str, O, ParseError<'i>>
27where
28    F: Parser<&'i str, O, ParseError<'i>>,
29{
30    move |input| {
31        let (input, res) = f.parse(input)?;
32        snafu::ensure!(input.is_empty(), UnexpectedLeftoverInputSnafu { input });
33        Ok((input, res))
34    }
35}

56impl Digits {
57    #[cfg(any(test, feature = "test-dependencies"))]
58    pub const MAX_PLACES: usize = u64::MAX.ilog10() as usize;

60    /// Construct a new `Digits` from a `u64`.
61    pub fn from_u64(mut v: u64) -> Self {
62        if v == 0 {
63            return Digits { places: vec![0] };
64        }
65
66        let mut places: Vec<u8> = vec![];
67        while v > 0 {
68            places.push((v % 10) as u8);
69            v /= 10;
70        }
71        places.reverse();
72        Digits { places }
73    }
74
75    /// Returns the `u64` representation.
76    ///
77    /// ## Errors
78    /// Errors if internal arithmetic operations overflow.
79    pub fn as_u64(&self) -> Result<u64, ValidationError> {
80        let mut total = 0u64;
81        for digit in &self.places {
82            total = total
83                .checked_mul(10)
84                .context(OverflowSnafu)?
85                .checked_add(*digit as u64)
86                .context(OverflowSnafu)?;
87        }
88        Ok(total)
89    }
90
91    /// Returns the [`U256`] representation.
92    ///
93    /// ## Errors
94    /// Errors if internal arithmetic operations overflow.
95    pub fn as_uint256(&self) -> Result<U256, ValidationError> {
96        let mut total = U256::zero();
97        for digit in &self.places {
98            total = total
99                .checked_mul(U256::from(10u64))
100                .context(OverflowSnafu)?
101                .checked_add(U256::from(*digit))
102                .context(OverflowSnafu)?;
103        }
104        Ok(total)
105    }
106
107    #[cfg(test)]
108    /// Returns the ratio corresponding to the decimal number `0.<digits>`,
109    /// i.e. the denominator will be `10^len(digits)`.
110    fn as_decimal_ratio(&self) -> Result<(u64, u64), ValidationError> {
111        let denominator = 10u64
112            .checked_pow(u32::try_from(self.places.len()).context(IntegerSnafu)?)
113            .context(OverflowSnafu)?;
114        Ok((self.as_u64()?, denominator))
115    }
116
117    /// Parse at least `min` digits.
118    pub fn parse_min(min: usize) -> impl Fn(&str) -> nom::IResult<&str, Self, ParseError<'_>> {
119        move |i| {
120            let (i, chars) = digit0(i)?;
121            let places = chars
122                .chars()
123                .map(|c| {
124                    c.to_digit(10)
125                        .expect("we already checked that this char was a digit")
126                        as u8
127                })
128                .collect::<Vec<_>>();
129            snafu::ensure!(
130                places.len() >= min,
131                DigitsMinimumSnafu {
132                    min,
133                    digits_len: places.len(),
134                    input: i
135                }
136            );
137            Ok((i, Digits { places }))
138        }
139    }

268    /// Parse at least `min` digits.
269    pub fn parse_min(min: usize) -> impl Fn(&str) -> nom::IResult<&str, Self, ParseError<'_>> {
270        move |i| {
271            let (i, data) = take_while(|c: char| c.is_ascii_hexdigit())(i)?;
272            snafu::ensure!(
273                data.len() >= min,
274                DigitsMinimumSnafu {
275                    min,
276                    digits_len: data.len(),
277                    input: i
278                }
279            );
280            let mut places = vec![];
281            for c in data.chars() {
282                let digit = CaseSensitiveHexDigit::from_char(c)?;
283                places.push(digit);
284            }
285            Ok((i, HexDigits { places }))
286        }
287    }

289    /// Returns whether all alphabetic digits are lowercase.
290    pub fn is_all_lowercase(&self) -> bool {
291        let uppers = CaseSensitiveHexDigit::UpperA..=CaseSensitiveHexDigit::UpperF;
292        for digit in self.places.iter() {
293            if uppers.contains(digit) {
294                return false;
295            }
296        }
297        true
298    }

301    pub fn is_all_uppercase(&self) -> bool {
302        let lowers = CaseSensitiveHexDigit::LowerA..=CaseSensitiveHexDigit::LowerF;
303        for digit in self.places.iter() {
304            if lowers.contains(digit) {
305                return false;
306            }
307        }
308        true
309    }

311    /// Validates the digits as an Ethereum address according to ERC-55, if possible.   
312    pub fn validate_erc55(&self) -> Result<(), ValidationError> {
313        snafu::ensure!(
314            self.places.len() == 40,
315            IncorrectEthAddressLenSnafu {
316                len: self.places.len()
317            }
318        );
319
320        // Construct an ERC-55 checksummed address and compare that to the input
321        let address_input = self.to_string();
322        let address_lowercase = address_input.to_ascii_lowercase();
323        let address_hashed = hex::encode(Keccak256::digest(&address_lowercase));
324        let address_checksummed = address_lowercase
325            .chars()
326            .enumerate()
327            .map(|(i, c)| {
328                if c.is_ascii_digit() {
329                    c
330                } else {
331                    let should_uppercase = u8::from_str_radix(&address_hashed[i..i + 1], 16)
332                        .expect("Always a hexadecimal ASCII char")
333                        >= 8;
334                    if should_uppercase {
335                        c.to_uppercase().next().expect("Always the char")
336                    } else {
337                        c
338                    }
339                }
340            })
341            .collect::<String>();
342        snafu::ensure!(
343            address_input == address_checksummed,
344            Erc55ValidationSnafu {
345                reason: {
346                    // It's possible that this was not a checksummed address. It's pretty rare that a checksummed
347                    // address is all lowercase or all uppercase (a probability of 0.0004942706034105575)
348                    if self.is_all_lowercase() {
349                        Erc55ValidationFailureReason::AllLowercase
350                    } else if self.is_all_uppercase() {
351                        Erc55ValidationFailureReason::AllUppercase
352                    } else {
353                        Erc55ValidationFailureReason::ChecksumDoesNotMatch {
354                            expected: address_checksummed,
355                            saw: address_input,
356                        }
357                    }
358                }
359            }
360        );
361
362        Ok(())
363    }

366/// A parsed number.
367///
368/// ```abnf
369/// number = [ "-" / "+" ] *DIGIT [ "." 1*DIGIT ] [ ( "e" / "E" ) [ 1*DIGIT ] ]
370/// ```
371///
372/// A number can be expressed in scientific notation, with a
373/// multiplier of a power of 10. Only integer numbers are allowed, so the
374/// exponent MUST be greater or equal to the number of decimals after the point.
375///
376/// ## Warning
377/// The ABNF notation is ambiguous, it allows for quite a few cases that don't
378/// make sense - for example:
379///
380/// 1. `*DIGIT` is missing and `[ "." 1*DIGIT ]` is present while `[ ( "e" / "E" ) [ 1*DIGIT] ]`
381///    is not
382/// 2. `[ ( "e" / "E" ) [ 1*DIGIT] ]` is missing the `[ 1*DIGIT ]` (eg, just "e")
383/// 3. only `"-" / "+"` is present
384///
385/// For this reason, in this library, parsing is separate from validation.
386///
387/// Other implementations use regular expressions instead of parsing, and only
388/// support very specific values.
389#[derive(Clone, Debug, PartialEq, Eq, Hash)]
390pub struct Number {
391    /// true for "+", false for "-"
392    signum: Option<bool>,
393    integer: Digits,
394    decimal: Option<Digits>,
395    exponent: Option<(
396        // true for "e", false for "E"
397        bool,
398        Option<Digits>,
399    )>,
400}
401
402impl core::fmt::Display for Number {
403    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
404        let this = &self;
405        let Number {
406            signum,
407            integer,
408            decimal,
409            exponent,
410        } = this;
411        if let Some(signum) = signum {
412            f.write_str(if *signum { "+" } else { "-" })?;
413        }
414        integer.fmt(f)?;
415        if let Some(dec) = decimal {
416            f.write_fmt(format_args!(".{dec}"))?;
417        };
418        if let Some((little_e, maybe_exp)) = exponent {
419            f.write_str(if *little_e { "e" } else { "E" })?;
420            if let Some(exp) = maybe_exp {
421                exp.fmt(f)?;
422            }
423        }
424        Ok(())
425    }
426}

428impl Number {
429    /// Parse a `Number`.
430    ///
431    /// ```abnf
432    /// number = [ "-" / "+" ] *DIGIT [ "." 1*DIGIT ] [ ( "e" / "E" ) [ 1*DIGIT ] ]
433    /// ```
434    pub fn parse(i: &str) -> nom::IResult<&str, Self, ParseError<'_>> {
435        // Parse [ "-" / "+" ], returning `true` if "+"
436        let parse_signum = opt(char('+').map(|_| true).or(char('-').map(|_| false)));
437
438        // Parse *DIGIT
439        let parse_integer = Digits::parse_min(0);
440
441        // Parse [ "." 1*DIGIT ]
442        let parse_decimal = opt(preceded(char('.'), Digits::parse_min(1)));
443
444        // Parse [ ( "e" / "E" ) [ 1*DIGIT ] ]
445        fn parse_exponent(
446            i: &str,
447        ) -> nom::IResult<&str, Option<(bool, Option<Digits>)>, ParseError<'_>> {
448            // Parse ( "e" / "E" ), returning `true` if "e"
449            let parse_e = char('e').map(|_| true).or(char('E').map(|_| false));
450
451            // Parse [ 1*DIGIT ]
452            let maybe_exp = opt(Digits::parse_min(1));
453
454            opt(tuple((parse_e, maybe_exp))).parse(i)
455        }
456
457        tuple((parse_signum, parse_integer, parse_decimal, parse_exponent))
458            .map(|(signum, integer, decimal, exponent)| Self {
459                signum,
460                integer,
461                decimal,
462                exponent,
463            })
464            .parse(i)
465    }
466
467    /// Returns the value of the integer portion of the number.
468    ///
469    /// ## Errors
470    /// Errors if internal arithmetic operations overflow.
471    fn integer(&self) -> Result<u64, ValidationError> {
472        self.integer.as_u64()
473    }
474
475    /// Convert this [`Number`] into an i128, if possible.
476    ///
477    /// ## Errors
478    /// Errors if internal arithmetic operations overflow.
479    pub fn as_i128(&self) -> Result<i128, ValidationError> {
480        let signum = self
481            .signum
482            .map_or(1i128, |is_positive| if is_positive { 1 } else { -1 });
483        let integer = self.integer()?;
484        let decimal_numerator = self
485            .decimal
486            .as_ref()
487            .map(|d| d.as_u64())
488            .transpose()?
489            .unwrap_or(0);
490        let decimal_places: u32 = self
491            .decimal
492            .as_ref()
493            .map(|d| d.places.len())
494            .unwrap_or(0)
495            .try_into()
496            .context(IntegerSnafu)?;
497        let exp: u32 = self
498            .exponent
499            .as_ref()
500            .and_then(|(_, maybe_exp)| maybe_exp.as_ref().map(|digits| digits.as_u64()))
501            .transpose()?
502            .unwrap_or(0)
503            .try_into()
504            .context(IntegerSnafu)?;
505        let multiplier = 10i128.checked_pow(exp).context(LargeExponentSnafu {
506            expected: u128::MAX.ilog10() as usize,
507            seen: exp as u64,
508        })?;
509        // The exponent must be >= the number of decimal places to yield an integer result.
510        let decimal_exp = exp
511            .checked_sub(decimal_places)
512            .with_context(|| SmallExponentSnafu {
513                expected: decimal_places as usize,
514                seen: exp,
515            });
516        // Since it's hard to see through all the function chaining, this is
517        // what's going on here:
518        // ```
519        // signum * (
520        //     (integer * 10^exp) +
521        //     (decimal_numerator * 10^(exp - decimal_places))
522        // )
523        // ```
524        let multiplied_integer = (integer as i128)
525            .checked_mul(multiplier)
526            .with_context(|| OverflowSnafu)?;
527        let decimal_multiplier = match decimal_exp {
528            Ok(d) => 10i128.checked_pow(d).context(OverflowSnafu)?,
529            Err(_) if decimal_numerator == 0 => 0,
530            Err(e) => return Err(e),
531        };
532        let multiplied_decimal = (decimal_numerator as i128)
533            .checked_mul(decimal_multiplier)
534            .with_context(|| OverflowSnafu)?;
535        let value = multiplied_integer
536            .checked_add(multiplied_decimal)
537            .context(OverflowSnafu)?;
538        signum.checked_mul(value).context(OverflowSnafu)
539    }
540
541    /// Convert this [`Number`] into a [`U256`], if possible.
542    ///
543    /// ## Errors
544    /// - Returns [`NegativeValueForUnsignedType`](ValidationError::NegativeValueForUnsignedType)
545    ///   if the number has a negative sign
546    /// - Returns [`Overflow`](ValidationError::Overflow) if internal arithmetic operations overflow
547    /// - Returns [`SmallExponent`](ValidationError::SmallExponent) if the exponent doesn't cover
548    ///   all non-zero decimal places (i.e., would require truncating non-zero decimal digits)
549    pub fn as_uint256(&self) -> Result<U256, ValidationError> {
550        // Reject negative numbers - U256 is unsigned
551        snafu::ensure!(
552            self.signum.is_none_or(|b| b),
553            NegativeValueForUnsignedTypeSnafu
554        );
555
556        let integer = self.integer.as_uint256()?;
557        let decimal_numerator = self
558            .decimal
559            .as_ref()
560            .map(|d| d.as_uint256())
561            .transpose()?
562            .unwrap_or(U256::zero());
563        let decimal_places: u32 = self
564            .decimal
565            .as_ref()
566            .map(|d| d.places.len())
567            .unwrap_or(0)
568            .try_into()
569            .context(IntegerSnafu)?;
570        let exp: u32 = self
571            .exponent
572            .as_ref()
573            .and_then(|(_, maybe_exp)| maybe_exp.as_ref().map(|digits| digits.as_u64()))
574            .transpose()?
575            .unwrap_or(0)
576            .try_into()
577            .context(IntegerSnafu)?;
578
579        let multiplier = U256::from(10u64)
580            .checked_pow(U256::from(exp))
581            .context(OverflowSnafu)?;
582
583        // The exponent must be >= the number of decimal places to yield an integer result.
584        let decimal_exp = exp
585            .checked_sub(decimal_places)
586            .with_context(|| SmallExponentSnafu {
587                expected: decimal_places as usize,
588                seen: exp,
589            });
590
591        // Formula: (integer * 10^exp) + (decimal_numerator * 10^(exp - decimal_places))
592        let multiplied_integer = integer.checked_mul(multiplier).context(OverflowSnafu)?;
593
594        let decimal_multiplier = match decimal_exp {
595            Ok(d) => U256::from(10u64)
596                .checked_pow(U256::from(d))
597                .context(OverflowSnafu)?,
598            Err(_) if decimal_numerator.is_zero() => U256::zero(),
599            Err(e) => return Err(e),
600        };
601
602        let multiplied_decimal = decimal_numerator
603            .checked_mul(decimal_multiplier)
604            .context(OverflowSnafu)?;
605
606        multiplied_integer
607            .checked_add(multiplied_decimal)
608            .context(OverflowSnafu)
609    }

612/// Ethereum Name Service name.
613///
614/// See [EIP-137](https://eips.ethereum.org/EIPS/eip-137).
615///
616/// Examples:
617/// * dao.eth
618/// * linea.eth
619///
620/// Uses:
621/// * <https://crates.io/crates/ens-normalize-rs>
622///
623/// ENS names must conform to the following syntax:
624///
625/// ## Name Syntax
626///
627/// ```not_abnf
628/// <domain> ::= <label> | <domain> "." <label>
629/// <label> ::= any valid string label per [UTS46](https://unicode.org/reports/tr46/)
630/// ```
631///
632/// In short, names consist of a series of dot-separated labels.
633///
634/// ## Note
635/// There's nothing in the spec that says an ENS name _must_ contain a '.', but for the
636/// purposes of this library it is required.
637#[derive(Clone, Debug, PartialEq)]
638pub struct EnsName(String);
639
640impl core::fmt::Display for EnsName {
641    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
642        self.0.fmt(f)
643    }
644}

646impl EnsName {
647    /// These delimiters are implicit in the ABNF syntax because they mark the
648    /// start of optional sections after the possible ENS name. There's nothing
649    /// in the spec that says these characters _cannot_ occur within an ENS name,
650    /// as the EIP-137 spec defers to the Unicode Technical Standard #46 to define
651    /// what valid labels are, and that pulls from
652    /// [this table](https://unicode.org/reports/tr46/#Table_Base_Valid_Set), which
653    /// specifically says "Add all ASCII except for '.'".
654    ///
655    /// But it's clear at a quick look that these delimiters _must_ end parsing of
656    /// an ENS name because if they did not, the parser would eat all optional sections.
657    ///
658    /// Also we must add '&' because an address could be used as the value of a parameter.
659    const DELIMITERS: &[char] = &['@', '/', '?', '&'];
660
661    /// Parse an `EnsName`.
662    pub fn parse(i: &str) -> nom::IResult<&str, Self, ParseError<'_>> {
663        fn continue_parsing(c: char) -> bool {
664            !c.is_whitespace() && !EnsName::DELIMITERS.contains(&c)
665        }
666
667        let (i, name) = take_till(|c| !continue_parsing(c))(i)?;
668        snafu::ensure!(!name.is_empty(), EnsMissingSnafu);
669        snafu::ensure!(name.contains('.'), EnsDomainSnafu);
670
671        // Now we have our name, normalize
672        let normalized_name = ens_normalize_rs::normalize(name).context(EnsNormalizationSnafu)?;
673        // According to <https://eips.ethereum.org/EIPS/eip-137#name-syntax> each label in a name
674        // must be a valid, normalized label, so the input `name` should match `normalized_name`.
675        snafu::ensure!(
676            normalized_name == name,
677            NotNormalizedSnafu {
678                expected: normalized_name,
679                seen: name.to_string(),
680            }
681        );
682
683        Ok((i, EnsName(normalized_name)))
684    }
685}

733                        e => Erc55ValidationSnafu { reason: e }.fail(),

736                    Ok(format!("0x{hex_digits}"))
737                }
738            }
739            AddressOrEnsName::Name(ens_name) => Ok(ens_name.to_string()),

777/// A parameter value.
778///
779/// ```abnf
780/// value = number / ethereum_address / STRING
781/// ```
782#[derive(Clone, Debug, PartialEq)]
783pub enum Value {
784    Number(Number),
785    Address(AddressOrEnsName),
786    String(UrlEncodedUnicodeString),
787}
788
789impl core::fmt::Display for Value {
790    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
791        match self {
792            Value::Number(n) => n.fmt(f),
793            Value::Address(a) => a.fmt(f),
794            Value::String(s) => s.fmt(f),
795        }
796    }
797}
798
799impl Value {
800    /// Parse a `Value`.
801    ///
802    /// ```abnf
803    /// value = number / ethereum_address / STRING
804    /// ```
805    pub fn parse(i: &str) -> nom::IResult<&str, Self, ParseError<'_>> {
806        let mut number = Number::parse.map(Value::Number);
807        let mut ethereum_address = AddressOrEnsName::parse.map(Value::Address);
808        let mut string = UrlEncodedUnicodeString::parse.map(Value::String);
809
810        // Here there are some interesting corner cases:
811        //
812        // 1. In `number`'s spec, absolutely everything is optional, so it _never_ fails.
813        // 2. `ethereum_address` may be prefixed with `0` (in the hex-address case), so both number
814        //   and address could be parsed. Because of 1, `number` always wins if it comes first, but
815        //   even if the minimum was 1 digit `number` would still win by parsing "0", so
816        //   we can't simply apply them with `number`.or(`address`).
817        // 3. `string` will always match a valid `number`.
818        //
819        // To solve all of these cases we run all three parsers and then return whichever one
820        // consumed the most input.
821        //
822        // In practice this should be fine because values either end with `&` (in the case of
823        // more parameters after it) or with the end of the input.
824
825        let mut results = [number.parse(i), ethereum_address.parse(i), string.parse(i)];
826        results.sort_by_key(|a| {
827            // Sort by the amount of input left over, smallest first
828            a.as_ref().ok().map(|(i, _)| i.len()).unwrap_or(usize::MAX)
829        });
830        results
831            .into_iter()
832            .next()
833            .expect("safe because this was an array of 3")
834    }
835
836    /// Return the value as a [`Number`], if possible.
837    pub fn as_number(&self) -> Option<&Number> {
838        match self {
839            Value::Number(n) => Some(n),
840            _ => None,
841        }
842    }
843
844    /// Return the value as an [`AddressOrEnsName`].
845    pub fn as_address_or_ens_name(&self) -> Option<&AddressOrEnsName> {
846        match self {
847            Value::Address(a) => Some(a),
848            _ => None,
849        }
850    }
851}

891/// A key-value pair, where the the type of the value depends upon the key.
892///
893/// ```abnf
894/// key = "value" / "gas" / "gasLimit" / "gasPrice" / TYPE
895/// ```
896///
897/// > If _key_ in the parameter list is "value", "gasLimit", "gasPrice" or "gas" then
898/// > _value_ MUST be a number. Otherwise, it must correspond to the TYPE string
899/// > used as key.
900///
901/// > ... gasLimit and gasPrice are suggested user-editable values for gas
902/// > limit and gas price, respectively, for the requested transaction. It is
903/// > acceptable to abbreviate gasLimit as gas, the two are treated synonymously.
904#[derive(Clone, Debug, PartialEq)]
905pub enum Parameter {
906    /// The amount to be paid, in the atomic unit of the native token of the blockchain.
907    ///
908    /// In most cases this will denote wei on the ether blockchain, but it depends on
909    /// context outside the scope of this library.
910    Value(Number),
911    /// Synonym for [`Self::GasLimit`].
912    Gas(Number),
913    /// Suggested user-editable value for the gas limit of the transaction.
914    GasLimit(Number),
915    /// Suggested user-editable value for the gas price of the transaction.
916    GasPrice(Number),
917    /// A "type" parameter denotes a positional parameter provided to the function
918    /// named by `function_name`. For information about `function_name` see the ABNF
919    /// spec in the module-level docs.
920    AbiType(EthereumAbiTypeName, Value),
921}
922
923impl core::fmt::Display for Parameter {
924    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
925        f.write_fmt(format_args!("{}={}", self.key(), self.value()))
926    }
927}
928
929impl Parameter {
930    /// Returns the parameter name or type.
931    pub fn key(&self) -> String {
932        match self {
933            Parameter::Value(_) => "value".to_string(),
934            Parameter::Gas(_) => "gas".to_string(),
935            Parameter::GasLimit(_) => "gasLimit".to_string(),
936            Parameter::GasPrice(_) => "gasPrice".to_string(),
937            Parameter::AbiType(ethereum_abi_type_name, _) => format!("{ethereum_abi_type_name}"),
938        }
939    }
940
941    /// Returns just the value of this parameter.
942    pub fn value(&self) -> Value {
943        match self {
944            Parameter::Value(number) => Value::Number(number.clone()),
945            Parameter::Gas(number) => Value::Number(number.clone()),
946            Parameter::GasLimit(number) => Value::Number(number.clone()),
947            Parameter::GasPrice(number) => Value::Number(number.clone()),
948            Parameter::AbiType(_, value) => value.clone(),
949        }
950    }
951
952    pub fn parse(i: &str) -> nom::IResult<&str, Self, ParseError<'_>> {
953        // Parse the key blob
954        let (i, key_blob) = terminated(take_till1(|c| c == '='), tag("="))(i)?;
955
956        fn parse_number(
957            f: fn(Number) -> Parameter,
958        ) -> impl Fn(&str) -> nom::IResult<&str, Parameter, ParseError<'_>> {
959            move |i| {
960                Number::parse.map(f).parse(i).map_err(|_| {
961                    ParseError::InvalidParameterValue {
962                        ty: "Number".to_string(),
963                    }
964                    .into()
965                })
966            }
967        }
968        // If key in the parameter list is value, gasLimit, gasPrice or gas then
969        // value MUST be a number. Otherwise, it must correspond to the TYPE
970        // string used as key.
971        Ok(match key_blob {
972            "value" => parse_number(Parameter::Value)(i)?,
973            "gas" => parse_number(Parameter::Gas)(i)?,
974            "gasLimit" => parse_number(Parameter::GasLimit)(i)?,
975            "gasPrice" => parse_number(Parameter::GasPrice)(i)?,
976            other_key_blob => {
977                let (remaining_name_input, type_name) = EthereumAbiTypeName::parse(other_key_blob)?;
978                snafu::ensure!(
979                    remaining_name_input.is_empty(),
980                    InvalidParameterKeySnafu {
981                        key: other_key_blob.to_string()
982                    }
983                );
984                let (i, value) =
985                    Value::parse(i).map_err(|_| ParseError::InvalidParameterValue {
986                        ty: type_name.to_string(),
987                    })?;
988                (i, Parameter::AbiType(type_name, value))
989            }
990        })
991    }
992}

1087            MultipleParameterValuesSnafu {

1187/// A raw, parsed EIP-681 transaction request.
1188///
1189/// ```abnf
1190/// request = schema_prefix target_address [ "@" chain_id ] [ "/" function_name ] [ "?" parameters ]
1191/// ```
1192///
1193/// ## Note
1194/// [`RawTransactionRequest::parse`] succeeds if any portion of the input can be parsed as a transaction.
1195/// You must take care to handle any leftover input properly, or otherwise fail downstream parsers.
1196#[derive(Clone, Debug, PartialEq)]
1197pub struct RawTransactionRequest {
1198    pub schema_prefix: SchemaPrefix,
1199    pub target_address: AddressOrEnsName,
1200    pub chain_id: Option<Digits>,
1201    pub function_name: Option<UrlEncodedUnicodeString>,
1202    pub parameters: Option<Parameters>,
1203}
1204
1205impl core::fmt::Display for RawTransactionRequest {
1206    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
1207        let Self {
1208            schema_prefix,
1209            target_address,
1210            chain_id,
1211            function_name,
1212            parameters,
1213        } = self;
1214        schema_prefix.fmt(f)?;
1215        target_address.fmt(f)?;
1216        if let Some(chain_id) = chain_id {
1217            f.write_str("@")?;
1218            chain_id.fmt(f)?;
1219        }
1220        if let Some(fn_name) = function_name {
1221            f.write_str("/")?;
1222            fn_name.fmt(f)?;
1223        }
1224        if let Some(parameters) = parameters {
1225            f.write_str("?")?;
1226            parameters.fmt(f)?;
1227        }
1228
1229        Ok(())
1230    }
1231}
1232
1233impl RawTransactionRequest {
1234    /// Parse a transaction request.
1235    pub fn parse(i: &str) -> nom::IResult<&str, Self, ParseError<'_>> {
1236        let chain_id = Digits::parse_min(1);
1237        let function_name = UrlEncodedUnicodeString::parse;
1238
1239        tuple((
1240            SchemaPrefix::parse,
1241            map_parser(is_not("@/?"), all_consuming(AddressOrEnsName::parse)),
1242            opt(preceded(tag("@"), chain_id)),
1243            opt(preceded(tag("/"), function_name)),
1244            opt(preceded(tag("?"), Parameters::parse)),
1245        ))
1246        .map(
1247            |(schema_prefix, target_address, chain_id, function_name, parameters)| Self {
1248                schema_prefix,
1249                target_address,
1250                chain_id,
1251                function_name,
1252                parameters,
1253            },
1254        )
1255        .parse(i)
1256    }

1454                    signum,
1455                    integer: integer.clone(),

1539    ///

1552    /// Produces "happy path" `EnsName`s that are guaranteed to be parseable,
1553    /// but are not exhaustive.
1554    ///
1555    /// Being exhaustive _and_ happy is hard here, because labels are almost
1556    /// arbitrary Unicode, but there are "recommendations" about what makes a
1557    /// good name. For added security we also validate the name after parsing
1558    /// to ensure it is properly normalized, but synthesizing Unicode that
1559    /// is guaranteed normalizable is outside the scope of this testsuite, for
1560    /// now.
1561    fn arb_happy_ens_name() -> impl Strategy<Value = EnsName> {
1562        /// Creates a "happy path" label according to
1563        /// [EIP-137](https://eips.ethereum.org/EIPS/eip-137):
1564        ///
1565        /// > Labels and domains may be of any length, but for compatibility with
1566        /// > legacy DNS, it is recommended that labels be restricted to no more than
1567        /// > 64 characters each, and complete ENS names to no more than 255
1568        /// > characters. For the same reason, it is recommended that labels do not
1569        /// > start or end with hyphens, or start with digits.
1570        ///
1571        fn arb_happy_label() -> impl Strategy<Value = String> {
1572            proptest::string::string_regex("[a-z][a-z0-9]{1,62}[a-z0-9]?").unwrap()
1573        }
1574
1575        fn arb_happy_label_maybe() -> impl Strategy<Value = Option<String>> {
1576            (any::<bool>(), arb_happy_label()).prop_map(|(is_some, label)| is_some.then_some(label))
1577        }
1578
1579        fn arb_happy_label_list() -> impl Strategy<Value = [Option<String>; 4]> {
1580            [
1581                // We require the ENS name to have at least two labels
1582                arb_happy_label().prop_map(Some).boxed(),
1583                arb_happy_label().prop_map(Some).boxed(),
1584                arb_happy_label_maybe().boxed(),
1585                arb_happy_label_maybe().boxed(),
1586            ]
1587        }
1588
1589        arb_happy_label_list()
1590            .prop_map(|list| EnsName(list.into_iter().flatten().collect::<Vec<_>>().join(".")))
1591    }

1714    fn arb_eth_type_name(recursions_max: usize) -> impl Strategy<Value = EthereumAbiTypeName> {
1715        fn arb_eth_base_type_name() -> impl Strategy<Value = EthereumAbiTypeName> {
1716            fn one_offs() -> impl Strategy<Value = EthereumAbiTypeName> {
1717                proptest::strategy::Union::new(
1718                    [
1719                        "uint", "int", "address", "bool", "fixed", "ufixed", "function", "bytes",
1720                        "string",
1721                    ]
1722                    .map(|s| proptest::strategy::Just(EthereumAbiTypeName { name: s.to_owned() })),
1723                )
1724            }
1725            fn arb_uint_or_int() -> impl Strategy<Value = EthereumAbiTypeName> {
1726                // 0 < M <= 256, M % 8 == 0
1727                ((0u32..=256 / 8), any::<bool>()).prop_map(|(n, is_uint)| EthereumAbiTypeName {
1728                    name: format!("{}{}", if is_uint { "uint" } else { "int" }, n * 8),
1729                })
1730            }
1731            fn arb_fixed_or_ufixed() -> impl Strategy<Value = EthereumAbiTypeName> {
1732                // 8 <= M <= 256, M % 8 == 0
1733                let m = 1u32..=256 / 8;
1734                // 0 < N <= 80
1735                let n = 1..=80;
1736                (m, n, any::<bool>()).prop_map(|(m, n, is_fixed)| EthereumAbiTypeName {
1737                    name: format!("{}{m}{n}", if is_fixed { "fixed" } else { "ufixed" }),
1738                })
1739            }
1740            fn arb_bytes() -> impl Strategy<Value = EthereumAbiTypeName> {
1741                // 0 < M <= 32
1742                (1..=32).prop_map(|m| EthereumAbiTypeName {
1743                    name: format!("bytes{m}"),
1744                })
1745            }
1746
1747            proptest::strategy::Union::new([
1748                one_offs().boxed(),
1749                arb_uint_or_int().boxed(),
1750                arb_fixed_or_ufixed().boxed(),
1751                arb_bytes().boxed(),
1752            ])
1753        }
1754
1755        // Don't allow infinite recursion, but ensure that we return _something_
1756        if recursions_max == 0 {
1757            return arb_eth_base_type_name().boxed();
1758        }
1759
1760        fn arb_eth_array_type_name(
1761            recursions_max: usize,
1762        ) -> impl Strategy<Value = EthereumAbiTypeName> {
1763            // M >= 0 but we also have to cover variable-length arrays
1764            (
1765                (0u32..),
1766                arb_eth_type_name(recursions_max.saturating_sub(1)),
1767            )
1768                .prop_map(|(m, ty)| EthereumAbiTypeName {
1769                    name: format!(
1770                        "<{ty}>[{}]",
1771                        if m > 0 { m.to_string() } else { String::new() }
1772                    ),
1773                })
1774        }

1776        fn arb_eth_n_tuple(
1777            recursions_max: usize,
1778            n_tuple: usize,
1779        ) -> impl Strategy<Value = EthereumAbiTypeName> {
1780            proptest::collection::vec(arb_eth_type_name(recursions_max.saturating_sub(1)), n_tuple)
1781                .prop_map(|types| EthereumAbiTypeName {
1782                    name: format!(
1783                        "({})",
1784                        types
1785                            .into_iter()
1786                            .map(|ty| ty.name)
1787                            .collect::<Vec<_>>()
1788                            .join(",")
1789                    ),
1790                })
1791        }

1796        }
1797
1798        proptest::strategy::Union::new([
1799            arb_eth_base_type_name().boxed(),
1800            arb_eth_array_type_name(recursions_max - 1).boxed(),
1801            arb_eth_tuple(recursions_max - 1).boxed(),

1803        .boxed()
1804    }
