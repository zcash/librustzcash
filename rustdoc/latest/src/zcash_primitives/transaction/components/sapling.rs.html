26/// Returns the enforcement policy for ZIP 212 at the given height.
27pub fn zip212_enforcement(params: &impl Parameters, height: BlockHeight) -> Zip212Enforcement {
28    if params.is_nu_active(NetworkUpgrade::Canopy, height) {
29        let grace_period_end_height =
30            params.activation_height(NetworkUpgrade::Canopy).unwrap() + ZIP212_GRACE_PERIOD;
31
32        if height < grace_period_end_height {
33            Zip212Enforcement::GracePeriod
34        } else {
35            Zip212Enforcement::On
36        }
37    } else {
38        Zip212Enforcement::Off
39    }
40}

42/// A map from one bundle authorization to another.
43///
44/// For use with [`TransactionData::map_authorization`].
45///
46/// [`TransactionData::map_authorization`]: crate::transaction::TransactionData::map_authorization
47pub trait MapAuth<A: Authorization, B: Authorization> {
48    fn map_spend_proof(&mut self, p: A::SpendProof) -> B::SpendProof;
49    fn map_output_proof(&mut self, p: A::OutputProof) -> B::OutputProof;
50    fn map_auth_sig(&mut self, s: A::AuthSig) -> B::AuthSig;
51    fn map_authorization(&mut self, a: A) -> B;
52}

87/// Consensus rules (§4.4) & (§4.5):
88/// - Canonical encoding is enforced here.
89/// - "Not small order" is enforced here.
90fn read_value_commitment<R: Read>(mut reader: R) -> io::Result<ValueCommitment> {
91    let mut bytes = [0u8; 32];
92    reader.read_exact(&mut bytes)?;
93    let cv = ValueCommitment::from_bytes_not_small_order(&bytes);
94
95    if cv.is_none().into() {
96        Err(io::Error::new(io::ErrorKind::InvalidInput, "invalid cv"))
97    } else {
98        Ok(cv.unwrap())
99    }
100}

102/// Consensus rules (§7.3) & (§7.4):
103/// - Canonical encoding is enforced here
104fn read_cmu<R: Read>(mut reader: R) -> io::Result<ExtractedNoteCommitment> {
105    let mut f = [0u8; 32];
106    reader.read_exact(&mut f)?;
107    Option::from(ExtractedNoteCommitment::from_bytes(&f))
108        .ok_or_else(|| io::Error::new(io::ErrorKind::InvalidInput, "cmu not in field"))
109}

111/// Consensus rules (§7.3) & (§7.4):
112/// - Canonical encoding is enforced here
113pub fn read_base<R: Read>(mut reader: R, _field: &str) -> io::Result<jubjub::Base> {
114    let mut f = [0u8; 32];
115    reader.read_exact(&mut f)?;
116    Option::from(jubjub::Base::from_repr(f)).ok_or_else(|| {
117        io::Error::new(
118            io::ErrorKind::InvalidInput,
119            "base value not a valid field element",
120        )
121    })
122}

124/// Consensus rules (§4.4) & (§4.5):
125/// - Canonical encoding is enforced by the API of SaplingVerificationContext::check_spend()
126///   and SaplingVerificationContext::check_output() due to the need to parse this into a
127///   bellman::groth16::Proof.
128/// - Proof validity is enforced in SaplingVerificationContext::check_spend()
129///   and SaplingVerificationContext::check_output()
130pub fn read_zkproof<R: Read>(mut reader: R) -> io::Result<GrothProofBytes> {
131    let mut zkproof = [0u8; GROTH_PROOF_SIZE];
132    reader.read_exact(&mut zkproof)?;
133    Ok(zkproof)
134}

136fn read_nullifier<R: Read>(mut reader: R) -> io::Result<Nullifier> {
137    let mut nullifier = Nullifier([0u8; 32]);
138    reader.read_exact(&mut nullifier.0)?;
139    Ok(nullifier)

142/// Consensus rules (§4.4):
143/// - Canonical encoding is enforced here.
144/// - "Not small order" is enforced in SaplingVerificationContext::check_spend()
145fn read_rk<R: Read>(mut reader: R) -> io::Result<redjubjub::VerificationKey<SpendAuth>> {
146    let mut bytes = [0; 32];
147    reader.read_exact(&mut bytes)?;
148    redjubjub::VerificationKey::try_from(bytes)
149        .map_err(|_| io::Error::new(io::ErrorKind::InvalidData, "verification key is invalid"))
150}

155fn read_spend_auth_sig<R: Read>(mut reader: R) -> io::Result<redjubjub::Signature<SpendAuth>> {
156    let mut sig = [0; 64];
157    reader.read_exact(&mut sig)?;

168fn read_spend_v4<R: Read>(mut reader: R) -> io::Result<SpendDescription<Authorized>> {
169    // Consensus rules (§4.4) & (§4.5):
170    // - Canonical encoding is enforced here.
171    // - "Not small order" is enforced in SaplingVerificationContext::(check_spend()/check_output())
172    //   (located in zcash_proofs::sapling::verifier).
173    let cv = read_value_commitment(&mut reader)?;
174    // Consensus rules (§7.3) & (§7.4):
175    // - Canonical encoding is enforced here
176    let anchor = read_base(&mut reader, "anchor")?;
177    let nullifier = read_nullifier(&mut reader)?;
178    let rk = read_rk(&mut reader)?;
179    let zkproof = read_zkproof(&mut reader)?;
180    let spend_auth_sig = read_spend_auth_sig(&mut reader)?;
181
182    Ok(SpendDescription::from_parts(
183        cv,
184        anchor,
185        nullifier,
186        rk,
187        zkproof,
188        spend_auth_sig,
189    ))
190}

192fn write_spend_v4<W: Write>(mut writer: W, spend: &SpendDescription<Authorized>) -> io::Result<()> {
193    writer.write_all(&spend.cv().to_bytes())?;
194    writer.write_all(spend.anchor().to_repr().as_ref())?;
195    writer.write_all(&spend.nullifier().0)?;
196    writer.write_all(&<[u8; 32]>::from(*spend.rk()))?;
197    writer.write_all(spend.zkproof())?;
198    writer.write_all(&<[u8; 64]>::from(*spend.spend_auth_sig()))
199}

210fn read_spend_v5<R: Read>(mut reader: &mut R) -> io::Result<SpendDescriptionV5> {

225fn read_output_v4<R: Read>(mut reader: &mut R) -> io::Result<OutputDescription<GrothProofBytes>> {
226    // Consensus rules (§4.5):
227    // - Canonical encoding is enforced here.
228    // - "Not small order" is enforced in SaplingVerificationContext::check_output()
229    //   (located in zcash_proofs::sapling::verifier).
230    let cv = read_value_commitment(&mut reader)?;
231
232    // Consensus rule (§7.4): Canonical encoding is enforced here
233    let cmu = read_cmu(&mut reader)?;
234
235    // Consensus rules (§4.5):
236    // - Canonical encoding is enforced in librustzcash_sapling_check_output by zcashd
237    // - "Not small order" is enforced in SaplingVerificationContext::check_output()
238    let mut ephemeral_key = EphemeralKeyBytes([0u8; 32]);
239    reader.read_exact(&mut ephemeral_key.0)?;
240
241    let mut enc_ciphertext = [0u8; ENC_CIPHERTEXT_SIZE];
242    let mut out_ciphertext = [0u8; OUT_CIPHERTEXT_SIZE];
243    reader.read_exact(&mut enc_ciphertext)?;
244    reader.read_exact(&mut out_ciphertext)?;
245
246    let zkproof = read_zkproof(&mut reader)?;
247
248    Ok(OutputDescription::from_parts(
249        cv,
250        cmu,
251        ephemeral_key,
252        enc_ciphertext,
253        out_ciphertext,
254        zkproof,
255    ))
256}

289fn read_output_v5<R: Read>(mut reader: &mut R) -> io::Result<OutputDescriptionV5> {
290    let cv = read_value_commitment(&mut reader)?;
291    let cmu = read_cmu(&mut reader)?;
292
293    // Consensus rules (§4.5):
294    // - Canonical encoding is enforced in librustzcash_sapling_check_output by zcashd
295    // - "Not small order" is enforced in SaplingVerificationContext::check_output()
296    let mut ephemeral_key = EphemeralKeyBytes([0u8; 32]);
297    reader.read_exact(&mut ephemeral_key.0)?;
298
299    let mut enc_ciphertext = [0u8; 580];
300    let mut out_ciphertext = [0u8; 80];
301    reader.read_exact(&mut enc_ciphertext)?;
302    reader.read_exact(&mut out_ciphertext)?;
303
304    Ok(OutputDescriptionV5::from_parts(
305        cv,
306        cmu,
307        ephemeral_key,
308        enc_ciphertext,
309        out_ciphertext,

327/// Reads the Sapling components of a v4 transaction.
328#[allow(clippy::type_complexity)]
329pub(crate) fn read_v4_components<R: Read>(
330    mut reader: R,
331    tx_has_sapling: bool,
332) -> io::Result<(
333    ZatBalance,
334    Vec<SpendDescription<Authorized>>,
335    Vec<OutputDescription<GrothProofBytes>>,
336)> {
337    if tx_has_sapling {
338        let vb = Transaction::read_amount(&mut reader)?;
339        #[allow(clippy::redundant_closure)]
340        let ss: Vec<SpendDescription<Authorized>> =
341            Vector::read(&mut reader, |r| read_spend_v4(r))?;
342        #[allow(clippy::redundant_closure)]
343        let so: Vec<OutputDescription<GrothProofBytes>> =
344            Vector::read(&mut reader, |r| read_output_v4(r))?;
345        Ok((vb, ss, so))
346    } else {
347        Ok((ZatBalance::zero(), vec![], vec![]))
348    }
349}

361/// Writes the Sapling components of a v4 transaction.
362pub(crate) fn write_v4_components<W: Write>(
363    mut writer: W,
364    bundle: Option<&Bundle<Authorized, ZatBalance>>,
365    tx_has_sapling: bool,
366) -> io::Result<()> {
367    if tx_has_sapling {
368        writer.write_all(
369            &bundle
370                .map_or(ZatBalance::zero(), |b| *b.value_balance())
371                .to_i64_le_bytes(),
372        )?;
373        Vector::write(
374            &mut writer,
375            bundle.map_or(&[], |b| b.shielded_spends()),
376            |w, e| write_spend_v4(w, e),
377        )?;
378        Vector::write(
379            &mut writer,
380            bundle.map_or(&[], |b| b.shielded_outputs()),
381            |w, e| write_output_v4(w, e),
382        )?;
383    } else if bundle.is_some() {
384        return Err(io::Error::new(
385            io::ErrorKind::InvalidInput,
386            "Sapling components may not be present if Sapling is not active.",
387        ));
388    }
389
390    Ok(())
391}

393/// Reads a [`Bundle`] from a v5 transaction format.
394#[allow(clippy::redundant_closure)]
395pub(crate) fn read_v5_bundle<R: Read>(
396    mut reader: R,
397) -> io::Result<Option<Bundle<Authorized, ZatBalance>>> {
398    let sd_v5s = Vector::read(&mut reader, read_spend_v5)?;
399    let od_v5s = Vector::read(&mut reader, read_output_v5)?;
400    let n_spends = sd_v5s.len();
401    let n_outputs = od_v5s.len();
402    let value_balance = if n_spends > 0 || n_outputs > 0 {
403        Transaction::read_amount(&mut reader)?
404    } else {
405        ZatBalance::zero()
406    };
407
408    let anchor = if n_spends > 0 {
409        Some(read_base(&mut reader, "anchor")?)
410    } else {
411        None
412    };
413
414    let v_spend_proofs = Array::read(&mut reader, n_spends, |r| read_zkproof(r))?;
415    let v_spend_auth_sigs = Array::read(&mut reader, n_spends, |r| read_spend_auth_sig(r))?;
416    let v_output_proofs = Array::read(&mut reader, n_outputs, |r| read_zkproof(r))?;
417
418    let binding_sig = if n_spends > 0 || n_outputs > 0 {
419        let mut sig = [0; 64];
420        reader.read_exact(&mut sig)?;
421        Some(redjubjub::Signature::from(sig))
422    } else {
423        None
424    };
425
426    let shielded_spends = sd_v5s
427        .into_iter()
428        .zip(v_spend_proofs.into_iter().zip(v_spend_auth_sigs))
429        .map(|(sd_5, (zkproof, spend_auth_sig))| {
430            // the following `unwrap` is safe because we know n_spends > 0.
431            sd_5.into_spend_description(anchor.unwrap(), zkproof, spend_auth_sig)
432        })
433        .collect();
434
435    let shielded_outputs = od_v5s
436        .into_iter()
437        .zip(v_output_proofs)
438        .map(|(od_5, zkproof)| od_5.into_output_description(zkproof))
439        .collect();
440
441    Ok(binding_sig.and_then(|binding_sig| {
442        Bundle::from_parts(
443            shielded_spends,
444            shielded_outputs,
445            value_balance,
446            Authorized { binding_sig },
447        )
448    }))
449}

451/// Writes a [`Bundle`] in the v5 transaction format.
452pub(crate) fn write_v5_bundle<W: Write>(
453    mut writer: W,
454    sapling_bundle: Option<&Bundle<Authorized, ZatBalance>>,
455) -> io::Result<()> {
456    if let Some(bundle) = sapling_bundle {
457        Vector::write(&mut writer, bundle.shielded_spends(), |w, e| {
458            write_spend_v5_without_witness_data(w, e)
459        })?;
460
461        Vector::write(&mut writer, bundle.shielded_outputs(), |w, e| {
462            write_output_v5_without_proof(w, e)
463        })?;
464
465        if !(bundle.shielded_spends().is_empty() && bundle.shielded_outputs().is_empty()) {
466            writer.write_all(&bundle.value_balance().to_i64_le_bytes())?;
467        }
468        if !bundle.shielded_spends().is_empty() {
469            writer.write_all(bundle.shielded_spends()[0].anchor().to_repr().as_ref())?;
470        }
471
472        Array::write(
473            &mut writer,
474            bundle.shielded_spends().iter().map(|s| &s.zkproof()[..]),
475            |w, e| w.write_all(e),
476        )?;
477        Array::write(
478            &mut writer,
479            bundle.shielded_spends().iter().map(|s| s.spend_auth_sig()),
480            |w, e| w.write_all(&<[u8; 64]>::from(**e)),
481        )?;
482
483        Array::write(
484            &mut writer,
485            bundle.shielded_outputs().iter().map(|s| &s.zkproof()[..]),
486            |w, e| w.write_all(e),
487        )?;
488
489        if !(bundle.shielded_spends().is_empty() && bundle.shielded_outputs().is_empty()) {
490            writer.write_all(&<[u8; 64]>::from(bundle.authorization().binding_sig))?;
491        }
492    } else {
493        CompactSize::write(&mut writer, 0)?;
494        CompactSize::write(&mut writer, 0)?;
495    }
496
497    Ok(())
498}
