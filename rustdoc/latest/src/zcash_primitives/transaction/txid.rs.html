65fn hasher(personal: &[u8; 16]) -> StateWrite {
66    StateWrite(Params::new().hash_length(32).personal(personal).to_state())
67}

69/// Sequentially append the serialized value of each transparent input
70/// to a hash personalized by ZCASH_PREVOUTS_HASH_PERSONALIZATION.
71/// In the case that no inputs are provided, this produces a default
72/// hash from just the personalization string.
73pub(crate) fn transparent_prevout_hash<TransparentAuth: transparent::Authorization>(
74    vin: &[TxIn<TransparentAuth>],
75) -> Blake2bHash {
76    let mut h = hasher(ZCASH_PREVOUTS_HASH_PERSONALIZATION);
77    for t_in in vin {
78        t_in.prevout.write(&mut h).unwrap();
79    }
80    h.finalize()
81}

133/// Implements [ZIP 244 section T.3a](https://zips.z.cash/zip-0244#t-3a-sapling-spends-digest)
134///
135/// Write disjoint parts of each Sapling shielded spend to a pair of hashes:
136/// * \[nullifier*\] - personalized with ZCASH_SAPLING_SPENDS_COMPACT_HASH_PERSONALIZATION
137/// * \[(cv, anchor, rk, zkproof)*\] - personalized with ZCASH_SAPLING_SPENDS_NONCOMPACT_HASH_PERSONALIZATION
138///
139/// Then, hash these together personalized by ZCASH_SAPLING_SPENDS_HASH_PERSONALIZATION
140pub(crate) fn hash_sapling_spends<A: sapling::bundle::Authorization>(
141    shielded_spends: &[SpendDescription<A>],
142) -> Blake2bHash {
143    let mut h = hasher(ZCASH_SAPLING_SPENDS_HASH_PERSONALIZATION);
144    if !shielded_spends.is_empty() {
145        let mut ch = hasher(ZCASH_SAPLING_SPENDS_COMPACT_HASH_PERSONALIZATION);
146        let mut nh = hasher(ZCASH_SAPLING_SPENDS_NONCOMPACT_HASH_PERSONALIZATION);
147        for s_spend in shielded_spends {
148            // we build the hash of nullifiers separately for compact blocks.
149            ch.write_all(s_spend.nullifier().as_ref()).unwrap();
150
151            nh.write_all(&s_spend.cv().to_bytes()).unwrap();
152            nh.write_all(&s_spend.anchor().to_repr()).unwrap();
153            nh.write_all(&<[u8; 32]>::from(*s_spend.rk())).unwrap();
154        }
155
156        let compact_digest = ch.finalize();
157        h.write_all(compact_digest.as_bytes()).unwrap();
158        let noncompact_digest = nh.finalize();
159        h.write_all(noncompact_digest.as_bytes()).unwrap();
160    }
161    h.finalize()
162}

164/// Implements [ZIP 244 section T.3b](https://zips.z.cash/zip-0244#t-3b-sapling-outputs-digest)
165///
166/// Write disjoint parts of each Sapling shielded output as 3 separate hashes:
167/// * \[(cmu, epk, enc_ciphertext\[..52\])*\] personalized with ZCASH_SAPLING_OUTPUTS_COMPACT_HASH_PERSONALIZATION
168/// * \[enc_ciphertext\[52..564\]*\] (memo ciphertexts) personalized with ZCASH_SAPLING_OUTPUTS_MEMOS_HASH_PERSONALIZATION
169/// * \[(cv, enc_ciphertext\[564..\], out_ciphertext, zkproof)*\] personalized with ZCASH_SAPLING_OUTPUTS_NONCOMPACT_HASH_PERSONALIZATION
170///
171/// Then, hash these together personalized with ZCASH_SAPLING_OUTPUTS_HASH_PERSONALIZATION
172pub(crate) fn hash_sapling_outputs<A>(shielded_outputs: &[OutputDescription<A>]) -> Blake2bHash {
173    let mut h = hasher(ZCASH_SAPLING_OUTPUTS_HASH_PERSONALIZATION);
174    if !shielded_outputs.is_empty() {
175        let mut ch = hasher(ZCASH_SAPLING_OUTPUTS_COMPACT_HASH_PERSONALIZATION);
176        let mut mh = hasher(ZCASH_SAPLING_OUTPUTS_MEMOS_HASH_PERSONALIZATION);
177        let mut nh = hasher(ZCASH_SAPLING_OUTPUTS_NONCOMPACT_HASH_PERSONALIZATION);
178        for s_out in shielded_outputs {
179            ch.write_all(s_out.cmu().to_bytes().as_ref()).unwrap();
180            ch.write_all(s_out.ephemeral_key().as_ref()).unwrap();
181            ch.write_all(&s_out.enc_ciphertext()[..52]).unwrap();
182
183            mh.write_all(&s_out.enc_ciphertext()[52..564]).unwrap();
184
185            nh.write_all(&s_out.cv().to_bytes()).unwrap();
186            nh.write_all(&s_out.enc_ciphertext()[564..]).unwrap();
187            nh.write_all(&s_out.out_ciphertext()[..]).unwrap();
188        }
189
190        h.write_all(ch.finalize().as_bytes()).unwrap();
191        h.write_all(mh.finalize().as_bytes()).unwrap();
192        h.write_all(nh.finalize().as_bytes()).unwrap();
193    }
194    h.finalize()
195}

219/// Implements [ZIP 244 section T.1](https://zips.z.cash/zip-0244#t-1-header-digest)
220fn hash_header_txid_data(
221    version: TxVersion,
222    // we commit to the consensus branch ID with the header
223    consensus_branch_id: BranchId,
224    lock_time: u32,
225    expiry_height: BlockHeight,
226) -> Blake2bHash {
227    let mut h = hasher(ZCASH_HEADERS_HASH_PERSONALIZATION);
228
229    h.write_u32_le(version.header()).unwrap();
230    h.write_u32_le(version.version_group_id()).unwrap();
231    h.write_u32_le(consensus_branch_id.into()).unwrap();
232    h.write_u32_le(lock_time).unwrap();
233    h.write_u32_le(expiry_height.into()).unwrap();
234
235    h.finalize()
236}

238/// Implements [ZIP 244 section T.2](https://zips.z.cash/zip-0244#t-2-transparent-digest)
239pub(crate) fn hash_transparent_txid_data(
240    t_digests: Option<&TransparentDigests<Blake2bHash>>,
241) -> Blake2bHash {
242    let mut h = hasher(ZCASH_TRANSPARENT_HASH_PERSONALIZATION);
243    if let Some(d) = t_digests {
244        h.write_all(d.prevouts_digest.as_bytes()).unwrap();
245        h.write_all(d.sequence_digest.as_bytes()).unwrap();
246        h.write_all(d.outputs_digest.as_bytes()).unwrap();
247    }
248    h.finalize()
249}

251/// Implements [ZIP 244 section T.3](https://zips.z.cash/zip-0244#t-3-sapling-digest)
252fn hash_sapling_txid_data<A: sapling::bundle::Authorization>(
253    bundle: &sapling::Bundle<A, ZatBalance>,
254) -> Blake2bHash {
255    let mut h = hasher(ZCASH_SAPLING_HASH_PERSONALIZATION);
256    if !(bundle.shielded_spends().is_empty() && bundle.shielded_outputs().is_empty()) {
257        h.write_all(hash_sapling_spends(bundle.shielded_spends()).as_bytes())
258            .unwrap();
259
260        h.write_all(hash_sapling_outputs(bundle.shielded_outputs()).as_bytes())
261            .unwrap();
262
263        h.write_all(&bundle.value_balance().to_i64_le_bytes())
264            .unwrap();
265    }
266    h.finalize()
267}
268
269fn hash_sapling_txid_empty() -> Blake2bHash {
270    hasher(ZCASH_SAPLING_HASH_PERSONALIZATION).finalize()
271}

273#[cfg(zcash_unstable = "zfuture")]
274fn hash_tze_txid_data(tze_digests: Option<&TzeDigests<Blake2bHash>>) -> Blake2bHash {
275    let mut h = hasher(ZCASH_TZE_HASH_PERSONALIZATION);
276    if let Some(d) = tze_digests {
277        h.write_all(d.inputs_digest.as_bytes()).unwrap();
278        h.write_all(d.outputs_digest.as_bytes()).unwrap();
279        if let Some(s) = d.per_input_digest {
280            h.write_all(s.as_bytes()).unwrap();
281        }
282    }
283    h.finalize()
284}

286/// A TransactionDigest implementation that commits to all of the effecting
287/// data of a transaction to produce a nonmalleable transaction identifier.
288///
289/// This expects and relies upon the existence of canonical encodings for
290/// each effecting component of a transaction.
291///
292/// This implements the [TxId Digest section of ZIP 244](https://zips.z.cash/zip-0244#txid-digest)
293pub struct TxIdDigester;
294
295impl<A: Authorization> TransactionDigest<A> for TxIdDigester {
296    type HeaderDigest = Blake2bHash;
297    type TransparentDigest = Option<TransparentDigests<Blake2bHash>>;
298    type SaplingDigest = Option<Blake2bHash>;
299    type OrchardDigest = Option<Blake2bHash>;
300
301    #[cfg(zcash_unstable = "zfuture")]
302    type TzeDigest = Option<TzeDigests<Blake2bHash>>;
303
304    type Digest = TxDigests<Blake2bHash>;
305
306    fn digest_header(
307        &self,
308        version: TxVersion,
309        consensus_branch_id: BranchId,
310        lock_time: u32,
311        expiry_height: BlockHeight,
312    ) -> Self::HeaderDigest {
313        hash_header_txid_data(version, consensus_branch_id, lock_time, expiry_height)
314    }
315
316    fn digest_transparent(
317        &self,
318        transparent_bundle: Option<&transparent::Bundle<A::TransparentAuth>>,
319    ) -> Self::TransparentDigest {
320        transparent_bundle.map(transparent_digests)
321    }
322
323    fn digest_sapling(
324        &self,
325        sapling_bundle: Option<&sapling::Bundle<A::SaplingAuth, ZatBalance>>,
326    ) -> Self::SaplingDigest {
327        sapling_bundle.map(hash_sapling_txid_data)
328    }
329
330    fn digest_orchard(
331        &self,
332        orchard_bundle: Option<&orchard::Bundle<A::OrchardAuth, ZatBalance>>,
333    ) -> Self::OrchardDigest {
334        orchard_bundle.map(|b| b.commitment().0)
335    }
336
337    #[cfg(zcash_unstable = "zfuture")]
338    fn digest_tze(&self, tze_bundle: Option<&tze::Bundle<A::TzeAuth>>) -> Self::TzeDigest {
339        tze_bundle.map(tze_digests)
340    }
341
342    fn combine(
343        &self,
344        header_digest: Self::HeaderDigest,
345        transparent_digests: Self::TransparentDigest,
346        sapling_digest: Self::SaplingDigest,
347        orchard_digest: Self::OrchardDigest,
348        #[cfg(zcash_unstable = "zfuture")] tze_digests: Self::TzeDigest,
349    ) -> Self::Digest {
350        TxDigests {
351            header_digest,
352            transparent_digests,
353            sapling_digest,
354            orchard_digest,
355            #[cfg(zcash_unstable = "zfuture")]
356            tze_digests,
357        }
358    }
359}

361pub(crate) fn to_hash(
362    _txversion: TxVersion,
363    consensus_branch_id: BranchId,
364    header_digest: Blake2bHash,
365    transparent_digest: Blake2bHash,
366    sapling_digest: Option<Blake2bHash>,
367    orchard_digest: Option<Blake2bHash>,
368    #[cfg(zcash_unstable = "zfuture")] tze_digests: Option<&TzeDigests<Blake2bHash>>,
369) -> Blake2bHash {
370    let mut personal = [0; 16];
371    personal[..12].copy_from_slice(ZCASH_TX_PERSONALIZATION_PREFIX);
372    (&mut personal[12..])
373        .write_u32_le(consensus_branch_id.into())
374        .unwrap();
375
376    let mut h = hasher(&personal);
377    h.write_all(header_digest.as_bytes()).unwrap();
378    h.write_all(transparent_digest.as_bytes()).unwrap();
379    h.write_all(
380        sapling_digest
381            .unwrap_or_else(hash_sapling_txid_empty)
382            .as_bytes(),
383    )
384    .unwrap();
385    h.write_all(
386        orchard_digest
387            .unwrap_or_else(orchard::commitments::hash_bundle_txid_empty)
388            .as_bytes(),
389    )
390    .unwrap();
391
392    #[cfg(zcash_unstable = "zfuture")]
393    if _txversion.has_tze() {
394        h.write_all(hash_tze_txid_data(tze_digests).as_bytes())
395            .unwrap();
396    }
397
398    h.finalize()
399}

401pub fn to_txid(
402    txversion: TxVersion,
403    consensus_branch_id: BranchId,
404    digests: &TxDigests<Blake2bHash>,
405) -> TxId {
406    let txid_digest = to_hash(
407        txversion,
408        consensus_branch_id,
409        digests.header_digest,
410        hash_transparent_txid_data(digests.transparent_digests.as_ref()),
411        digests.sapling_digest,
412        digests.orchard_digest,
413        #[cfg(zcash_unstable = "zfuture")]
414        digests.tze_digests.as_ref(),
415    );
416
417    TxId::from_bytes(<[u8; 32]>::try_from(txid_digest.as_bytes()).unwrap())
418}

420/// Digester which constructs a digest of only the witness data.
421/// This does not internally commit to the txid, so if that is
422/// desired it should be done using the result of this digest
423/// function.
424pub struct BlockTxCommitmentDigester;
425
426impl TransactionDigest<Authorized> for BlockTxCommitmentDigester {
427    /// We use the header digest to pass the transaction ID into
428    /// where it needs to be used for personalization string construction.
429    type HeaderDigest = BranchId;
430    type TransparentDigest = Blake2bHash;
431    type SaplingDigest = Blake2bHash;
432    type OrchardDigest = Blake2bHash;
433
434    #[cfg(zcash_unstable = "zfuture")]
435    type TzeDigest = Blake2bHash;
436
437    type Digest = Blake2bHash;
438
439    fn digest_header(
440        &self,
441        _version: TxVersion,
442        consensus_branch_id: BranchId,
443        _lock_time: u32,
444        _expiry_height: BlockHeight,
445    ) -> Self::HeaderDigest {
446        consensus_branch_id
447    }
448
449    fn digest_transparent(
450        &self,
451        transparent_bundle: Option<&transparent::Bundle<transparent::Authorized>>,
452    ) -> Blake2bHash {
453        let mut h = hasher(ZCASH_TRANSPARENT_SCRIPTS_HASH_PERSONALIZATION);
454        if let Some(bundle) = transparent_bundle {
455            for txin in &bundle.vin {
456                txin.script_sig.write(&mut h).unwrap();
457            }
458        }
459        h.finalize()
460    }
461
462    fn digest_sapling(
463        &self,
464        sapling_bundle: Option<&sapling::Bundle<sapling::bundle::Authorized, ZatBalance>>,
465    ) -> Blake2bHash {
466        let mut h = hasher(ZCASH_SAPLING_SIGS_HASH_PERSONALIZATION);
467        if let Some(bundle) = sapling_bundle {
468            for spend in bundle.shielded_spends() {
469                h.write_all(spend.zkproof()).unwrap();
470            }
471
472            for spend in bundle.shielded_spends() {
473                h.write_all(&<[u8; 64]>::from(*spend.spend_auth_sig()))
474                    .unwrap();
475            }
476
477            for output in bundle.shielded_outputs() {
478                h.write_all(output.zkproof()).unwrap();
479            }
480
481            h.write_all(&<[u8; 64]>::from(bundle.authorization().binding_sig))
482                .unwrap();
483        }
484        h.finalize()
485    }
486
487    fn digest_orchard(
488        &self,
489        orchard_bundle: Option<&orchard::Bundle<orchard::Authorized, ZatBalance>>,
490    ) -> Self::OrchardDigest {
491        orchard_bundle.map_or_else(orchard::commitments::hash_bundle_auth_empty, |b| {
492            b.authorizing_commitment().0
493        })
494    }
495
496    #[cfg(zcash_unstable = "zfuture")]
497    fn digest_tze(&self, tze_bundle: Option<&tze::Bundle<tze::Authorized>>) -> Blake2bHash {
498        let mut h = hasher(ZCASH_TZE_WITNESSES_HASH_PERSONALIZATION);
499        if let Some(bundle) = tze_bundle {
500            for tzein in &bundle.vin {
501                h.write_all(&tzein.witness.payload.0).unwrap();
502            }
503        }
504        h.finalize()
505    }
506
507    fn combine(
508        &self,
509        consensus_branch_id: Self::HeaderDigest,
510        transparent_digest: Self::TransparentDigest,
511        sapling_digest: Self::SaplingDigest,
512        orchard_digest: Self::OrchardDigest,
513        #[cfg(zcash_unstable = "zfuture")] tze_digest: Self::TzeDigest,
514    ) -> Self::Digest {
515        let digests = [transparent_digest, sapling_digest, orchard_digest];
516
517        let mut personal = [0; 16];
518        personal[..12].copy_from_slice(ZCASH_AUTH_PERSONALIZATION_PREFIX);
519        (&mut personal[12..])
520            .write_u32_le(consensus_branch_id.into())
521            .unwrap();
522
523        let mut h = hasher(&personal);
524        for digest in &digests {
525            h.write_all(digest.as_bytes()).unwrap();
526        }
