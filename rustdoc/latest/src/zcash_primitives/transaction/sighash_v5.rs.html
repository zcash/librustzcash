46/// Implements [ZIP 244 section S.2](https://zips.z.cash/zip-0244#s-2-transparent-sig-digest).
47fn transparent_sig_digest<A: TransparentAuthorizingContext>(
48    tx_data: Option<(&transparent::Bundle<A>, &TransparentDigests<Blake2bHash>)>,
49    input: &SignableInput<'_>,
50) -> Blake2bHash {
51    match tx_data {
52        // No transparent inputs or outputs.
53        None => hash_transparent_txid_data(None),
54        // No transparent inputs, or coinbase.
55        Some((bundle, txid_digests)) if bundle.is_coinbase() || bundle.vin.is_empty() => {
56            hash_transparent_txid_data(Some(txid_digests))
57        }
58        // Some transparent inputs, and not coinbase.
59        Some((bundle, txid_digests)) => {
60            let hash_type = input.hash_type();
61            let flag_anyonecanpay = hash_type & SIGHASH_ANYONECANPAY != 0;
62            let flag_single = hash_type & SIGHASH_MASK == SIGHASH_SINGLE;
63            let flag_none = hash_type & SIGHASH_MASK == SIGHASH_NONE;
64
65            let prevouts_digest = if flag_anyonecanpay {
66                transparent_prevout_hash::<A>(&[])
67            } else {
68                txid_digests.prevouts_digest
69            };
70
71            let amounts_digest = {
72                let mut h = hasher(ZCASH_TRANSPARENT_AMOUNTS_HASH_PERSONALIZATION);
73                if !flag_anyonecanpay {
74                    Array::write(&mut h, bundle.authorization.input_amounts(), |w, amount| {
75                        w.write_all(&amount.to_i64_le_bytes())
76                    })
77                    .unwrap();
78                }
79                h.finalize()
80            };
81
82            let scripts_digest = {
83                let mut h = hasher(ZCASH_TRANSPARENT_SCRIPTS_HASH_PERSONALIZATION);
84                if !flag_anyonecanpay {
85                    Array::write(
86                        &mut h,
87                        bundle.authorization.input_scriptpubkeys(),
88                        |w, script| script.write(w),
89                    )
90                    .unwrap();
91                }
92                h.finalize()
93            };
94
95            let sequence_digest = if flag_anyonecanpay {
96                transparent_sequence_hash::<A>(&[])
97            } else {
98                txid_digests.sequence_digest
99            };
100
101            let outputs_digest = if let SignableInput::Transparent(input) = input {
102                if flag_single {
103                    if *input.index() < bundle.vout.len() {
104                        transparent_outputs_hash(&[&bundle.vout[*input.index()]])
105                    } else {
106                        transparent_outputs_hash::<TxOut>(&[])
107                    }
108                } else if flag_none {
109                    transparent_outputs_hash::<TxOut>(&[])
110                } else {
111                    txid_digests.outputs_digest
112                }
113            } else {
114                txid_digests.outputs_digest
115            };
116
117            //S.2g.i:   prevout      (field encoding)
118            //S.2g.ii:  value        (8-byte signed little-endian)
119            //S.2g.iii: scriptPubKey (field encoding)
120            //S.2g.iv:  nSequence    (4-byte unsigned little-endian)
121            let mut ch = hasher(ZCASH_TRANSPARENT_INPUT_HASH_PERSONALIZATION);
122            if let SignableInput::Transparent(input) = input {
123                let txin = &bundle.vin[*input.index()];
124                txin.prevout.write(&mut ch).unwrap();
125                ch.write_all(&input.value().to_i64_le_bytes()).unwrap();
126                input.script_pubkey().write(&mut ch).unwrap();
127                ch.write_all(&txin.sequence.to_le_bytes()).unwrap();
128            }
129            let txin_sig_digest = ch.finalize();
130
131            let mut h = hasher(ZCASH_TRANSPARENT_HASH_PERSONALIZATION);
132            h.write_all(&[hash_type]).unwrap();
133            h.write_all(prevouts_digest.as_bytes()).unwrap();
134            h.write_all(amounts_digest.as_bytes()).unwrap();
135            h.write_all(scripts_digest.as_bytes()).unwrap();
136            h.write_all(sequence_digest.as_bytes()).unwrap();
137            h.write_all(outputs_digest.as_bytes()).unwrap();
138            h.write_all(txin_sig_digest.as_bytes()).unwrap();
139            h.finalize()
140        }
141    }
142}

144#[cfg(zcash_unstable = "zfuture")]
145fn tze_input_sigdigests<A: tze::Authorization>(
146    bundle: &tze::Bundle<A>,
147    input: &SignableInput<'_>,
148    txid_digests: &TzeDigests<Blake2bHash>,
149) -> TzeDigests<Blake2bHash> {
150    let mut ch = hasher(ZCASH_TZE_INPUT_HASH_PERSONALIZATION);
151    if let SignableInput::Tze {
152        index,
153        precondition,
154        value,
155    } = input
156    {
157        let tzein = &bundle.vin[*index];
158        tzein.prevout.write(&mut ch).unwrap();
159        CompactSize::write(&mut ch, precondition.extension_id.try_into().unwrap()).unwrap();
160        CompactSize::write(&mut ch, precondition.mode.try_into().unwrap()).unwrap();
161        Vector::write(&mut ch, &precondition.payload, |w, e| w.write_u8(*e)).unwrap();
162        ch.write_all(&value.to_i64_le_bytes()).unwrap();
163    }
164    let per_input_digest = ch.finalize();
165
166    TzeDigests {
167        inputs_digest: txid_digests.inputs_digest,
168        outputs_digest: txid_digests.outputs_digest,
169        per_input_digest: Some(per_input_digest),
170    }
171}

173/// Implements the [Signature Digest section of ZIP 244](https://zips.z.cash/zip-0244#signature-digest)
174pub fn v5_signature_hash<
175    TA: TransparentAuthorizingContext,
176    A: Authorization<TransparentAuth = TA>,
177>(
178    tx: &TransactionData<A>,
179    signable_input: &SignableInput<'_>,
180    txid_parts: &TxDigests<Blake2bHash>,
181) -> Blake2bHash {
182    // The caller must provide the transparent digests if and only if the transaction has a
183    // transparent component.
184    assert_eq!(
185        tx.transparent_bundle.is_some(),
186        txid_parts.transparent_digests.is_some()
187    );
188
189    to_hash(
190        tx.version,
191        tx.consensus_branch_id,
192        txid_parts.header_digest,
193        transparent_sig_digest(
194            tx.transparent_bundle
195                .as_ref()
196                .zip(txid_parts.transparent_digests.as_ref()),
197            signable_input,
198        ),
199        txid_parts.sapling_digest,
200        txid_parts.orchard_digest,
201        #[cfg(zcash_unstable = "zfuture")]
202        tx.tze_bundle
203            .as_ref()
204            .zip(txid_parts.tze_digests.as_ref())
205            .map(|(bundle, tze_digests)| tze_input_sigdigests(bundle, signable_input, tze_digests))
206            .as_ref(),
207    )
208}
