59/// The set of defined transaction format versions.
60///
61/// This is serialized in the first four or eight bytes of the transaction format, and
62/// represents valid combinations of the `(overwintered, version, version_group_id)`
63/// transaction fields. Note that this is not dependent on epoch, only on transaction encoding.
64/// For example, if a particular epoch defines a new transaction version but also allows the
65/// previous version, then only the new version would be added to this enum.
66#[derive(Clone, Copy, Debug, PartialEq, Eq)]
67pub enum TxVersion {
68    /// Transaction versions allowed prior to Overwinter activation. The argument MUST be
69    /// in the range `1..=0x7FFFFFFF`. Only versions 1 and 2 are defined; `3..=0x7FFFFFFF`
70    /// was allowed by consensus but considered equivalent to 2. This is specified in
71    /// [ยง 7.1 Transaction Encoding and Consensus](https://zips.z.cash/protocol/protocol.pdf#txnencoding).
72    Sprout(u32),
73    /// Transaction version 3, which was introduced by the Overwinter network upgrade
74    /// and allowed until Sapling activation. It is specified in
75    /// [ยง 7.1 Transaction Encoding and Consensus](https://zips.z.cash/protocol/protocol.pdf#txnencoding).
76    V3,
77    /// Transaction version 4, which was introduced by the Sapling network upgrade.
78    /// It is specified in [ยง 7.1 Transaction Encoding and Consensus](https://zips.z.cash/protocol/protocol.pdf#txnencoding).
79    V4,
80    /// Transaction version 5, which was introduced by the NU5 network upgrade.
81    /// It is specified in [ยง 7.1 Transaction Encoding and Consensus](https://zips.z.cash/protocol/protocol.pdf#txnencoding)
82    /// and [ZIP 225](https://zips.z.cash/zip-0225).
83    V5,
84    /// Transaction version 6, specified in [ZIP 230](https://zips.z.cash/zip-0230).
85    #[cfg(zcash_unstable = "nu7")]
86    V6,
87    /// This version is used exclusively for in-development transaction
88    /// serialization, and will never be active under the consensus rules.
89    /// When new consensus transaction versions are added, all call sites
90    /// using this constant should be inspected, and uses should be
91    /// removed as appropriate in favor of the new transaction version.
92    #[cfg(zcash_unstable = "zfuture")]
93    ZFuture,
94}
95
96impl TxVersion {
97    pub fn read<R: Read>(mut reader: R) -> io::Result<Self> {
98        let header = reader.read_u32_le()?;
99        let overwintered = (header >> 31) == 1;
100        let version = header & 0x7FFFFFFF;
101
102        if overwintered {
103            match (version, reader.read_u32_le()?) {
104                (V3_TX_VERSION, V3_VERSION_GROUP_ID) => Ok(TxVersion::V3),
105                (V4_TX_VERSION, V4_VERSION_GROUP_ID) => Ok(TxVersion::V4),
106                (V5_TX_VERSION, V5_VERSION_GROUP_ID) => Ok(TxVersion::V5),
107                #[cfg(zcash_unstable = "nu7")]
108                (V6_TX_VERSION, V6_VERSION_GROUP_ID) => Ok(TxVersion::V6),
109                #[cfg(zcash_unstable = "zfuture")]
110                (ZFUTURE_TX_VERSION, ZFUTURE_VERSION_GROUP_ID) => Ok(TxVersion::ZFuture),
111                _ => Err(io::Error::new(
112                    io::ErrorKind::InvalidData,
113                    "Unknown transaction format",
114                )),
115            }
116        } else if version >= 1 {
117            Ok(TxVersion::Sprout(version))
118        } else {
119            Err(io::Error::new(
120                io::ErrorKind::InvalidData,
121                "Unknown transaction format",
122            ))
123        }
124    }
125
126    pub fn header(&self) -> u32 {
127        // After Sprout, the overwintered bit is always set.
128        let overwintered = match self {
129            TxVersion::Sprout(_) => 0,
130            _ => 1 << 31,
131        };
132
133        overwintered
134            | match self {
135                TxVersion::Sprout(v) => *v,
136                TxVersion::V3 => V3_TX_VERSION,
137                TxVersion::V4 => V4_TX_VERSION,
138                TxVersion::V5 => V5_TX_VERSION,
139                #[cfg(zcash_unstable = "nu7")]
140                TxVersion::V6 => V6_TX_VERSION,
141                #[cfg(zcash_unstable = "zfuture")]
142                TxVersion::ZFuture => ZFUTURE_TX_VERSION,
143            }
144    }
145
146    pub fn version_group_id(&self) -> u32 {
147        match self {
148            TxVersion::Sprout(_) => 0,
149            TxVersion::V3 => V3_VERSION_GROUP_ID,
150            TxVersion::V4 => V4_VERSION_GROUP_ID,
151            TxVersion::V5 => V5_VERSION_GROUP_ID,
152            #[cfg(zcash_unstable = "nu7")]
153            TxVersion::V6 => V6_VERSION_GROUP_ID,
154            #[cfg(zcash_unstable = "zfuture")]
155            TxVersion::ZFuture => ZFUTURE_VERSION_GROUP_ID,
156        }
157    }
158
159    pub fn write<W: Write>(&self, mut writer: W) -> io::Result<()> {
160        writer.write_u32_le(self.header())?;
161        match self {
162            TxVersion::Sprout(_) => Ok(()),
163            _ => writer.write_u32_le(self.version_group_id()),
164        }
165    }
166
167    /// Returns `true` if this transaction version supports the Sprout protocol.
168    pub fn has_sprout(&self) -> bool {
169        match self {
170            TxVersion::Sprout(v) => *v >= 2u32,
171            TxVersion::V3 | TxVersion::V4 => true,
172            TxVersion::V5 => false,
173            #[cfg(zcash_unstable = "nu7")]
174            TxVersion::V6 => false,
175            #[cfg(zcash_unstable = "zfuture")]
176            TxVersion::ZFuture => false,
177        }
178    }
179
180    pub fn has_overwinter(&self) -> bool {
181        !matches!(self, TxVersion::Sprout(_))
182    }
183
184    /// Returns `true` if this transaction version supports the Sapling protocol.
185    pub fn has_sapling(&self) -> bool {
186        match self {
187            TxVersion::Sprout(_) | TxVersion::V3 => false,
188            TxVersion::V4 => true,
189            TxVersion::V5 => true,
190            #[cfg(zcash_unstable = "nu7")]
191            TxVersion::V6 => true,
192            #[cfg(zcash_unstable = "zfuture")]
193            TxVersion::ZFuture => true,
194        }
195    }
196
197    /// Returns `true` if this transaction version supports the Orchard protocol.
198    pub fn has_orchard(&self) -> bool {
199        match self {
200            TxVersion::Sprout(_) | TxVersion::V3 | TxVersion::V4 => false,
201            TxVersion::V5 => true,
202            #[cfg(zcash_unstable = "nu7")]
203            TxVersion::V6 => true,
204            #[cfg(zcash_unstable = "zfuture")]
205            TxVersion::ZFuture => true,
206        }
207    }

209    #[cfg(all(
210        any(zcash_unstable = "nu7", zcash_unstable = "zfuture"),
211        feature = "zip-233"
212    ))]
213    pub fn has_zip233(&self) -> bool {
214        match self {
215            TxVersion::Sprout(_) | TxVersion::V3 | TxVersion::V4 | TxVersion::V5 => false,
216            #[cfg(zcash_unstable = "nu7")]
217            TxVersion::V6 => true,
218            #[cfg(zcash_unstable = "zfuture")]
219            TxVersion::ZFuture => true,
220        }
221    }
222
223    #[cfg(zcash_unstable = "zfuture")]
224    pub fn has_tze(&self) -> bool {
225        matches!(self, TxVersion::ZFuture)
226    }

228    /// Suggests the transaction version that should be used in the given Zcash epoch.
229    pub fn suggested_for_branch(consensus_branch_id: BranchId) -> Self {
230        match consensus_branch_id {
231            BranchId::Sprout => TxVersion::Sprout(2),
232            BranchId::Overwinter => TxVersion::V3,
233            BranchId::Sapling | BranchId::Blossom | BranchId::Heartwood | BranchId::Canopy => {
234                TxVersion::V4
235            }
236            BranchId::Nu5 => TxVersion::V5,
237            BranchId::Nu6 => TxVersion::V5,
238            BranchId::Nu6_1 => TxVersion::V5,
239            #[cfg(zcash_unstable = "nu7")]
240            BranchId::Nu7 => TxVersion::V6,
241            #[cfg(zcash_unstable = "zfuture")]
242            BranchId::ZFuture => TxVersion::ZFuture,
243        }
244    }

444    /// Returns the total fees paid by the transaction, given a function that can be used to
445    /// retrieve the value of previous transactions' transparent outputs that are being spent in
446    /// this transaction.
447    pub fn fee_paid<E, F>(&self, get_prevout: F) -> Result<Option<Zatoshis>, E>
448    where
449        E: From<BalanceError>,
450        F: FnMut(&OutPoint) -> Result<Option<Zatoshis>, E>,
451    {
452        let transparent_balance = self.transparent_bundle.as_ref().map_or_else(
453            || Ok(Some(ZatBalance::zero())),
454            |b| b.value_balance(get_prevout),
455        )?;
456
457        transparent_balance
458            .map(|transparent_balance| {
459                let value_balances = [
460                    transparent_balance,
461                    self.sprout_bundle.as_ref().map_or_else(
462                        || Ok(ZatBalance::zero()),
463                        |b| b.value_balance().ok_or(BalanceError::Overflow),
464                    )?,
465                    self.sapling_bundle
466                        .as_ref()
467                        .map_or_else(ZatBalance::zero, |b| *b.value_balance()),
468                    self.orchard_bundle
469                        .as_ref()
470                        .map_or_else(ZatBalance::zero, |b| *b.value_balance()),
471                    #[cfg(all(
472                        any(zcash_unstable = "nu7", zcash_unstable = "zfuture"),
473                        feature = "zip-233"
474                    ))]
475                    -ZatBalance::from(self.zip233_amount),
476                ];
477
478                let overall_balance = value_balances
479                    .iter()
480                    .sum::<Option<_>>()
481                    .ok_or(BalanceError::Overflow)?;
482
483                Zatoshis::try_from(overall_balance).map_err(|_| BalanceError::Underflow)
484            })
485            .transpose()
486            .map_err(E::from)
487    }

612                .map(|b| b.map_authorization(f_transparent)),
613            sprout_bundle: self.sprout_bundle,
614            sapling_bundle: self.sapling_bundle.map(|b| {

616                    &mut f_sapling,
617                    |f, p| f.map_spend_proof(p),
618                    |f, p| f.map_output_proof(p),
619                    |f, s| f.map_auth_sig(s),
620                    |f, a| f.map_authorization(a),
621                )
622            }),
623            orchard_bundle: self.orchard_bundle.map(|b| {
624                b.map_authorization(
625                    &mut f_orchard,
626                    |f, _, s| f.map_spend_auth(s),
627                    |f, a| f.map_authorization(a),
628                )
629            }),
630            #[cfg(zcash_unstable = "zfuture")]
631            tze_bundle: self.tze_bundle.map(|b| b.map_authorization(f_tze)),
632        }
633    }

711    pub fn read<R: Read>(reader: R, consensus_branch_id: BranchId) -> io::Result<Self> {
712        let mut reader = HashReader::new(reader);
713
714        let version = TxVersion::read(&mut reader)?;
715        match version {
716            TxVersion::Sprout(_) | TxVersion::V3 | TxVersion::V4 => {
717                Self::read_v4(reader, version, consensus_branch_id)
718            }
719            TxVersion::V5 => Self::read_v5(reader.into_base_reader(), version),
720            #[cfg(zcash_unstable = "nu7")]
721            TxVersion::V6 => Self::read_v6(reader.into_base_reader(), version),
722            #[cfg(zcash_unstable = "zfuture")]
723            TxVersion::ZFuture => Self::read_v6(reader.into_base_reader(), version),
724        }
725    }

727    #[allow(clippy::redundant_closure)]
728    fn read_v4<R: Read>(
729        mut reader: HashReader<R>,
730        version: TxVersion,
731        consensus_branch_id: BranchId,
732    ) -> io::Result<Self> {
733        let transparent_bundle = Self::read_transparent(&mut reader)?;
734
735        let lock_time = reader.read_u32_le()?;
736        let expiry_height: BlockHeight = if version.has_overwinter() {
737            reader.read_u32_le()?.into()
738        } else {
739            0u32.into()
740        };
741
742        let (value_balance, shielded_spends, shielded_outputs) =
743            sapling_serialization::read_v4_components(&mut reader, version.has_sapling())?;
744
745        let sprout_bundle = if version.has_sprout() {
746            let joinsplits = Vector::read(&mut reader, |r| {
747                JsDescription::read(r, version.has_sapling())
748            })?;
749
750            if !joinsplits.is_empty() {
751                let mut bundle = sprout::Bundle {
752                    joinsplits,
753                    joinsplit_pubkey: [0; 32],
754                    joinsplit_sig: [0; 64],
755                };
756                reader.read_exact(&mut bundle.joinsplit_pubkey)?;
757                reader.read_exact(&mut bundle.joinsplit_sig)?;
758                Some(bundle)
759            } else {
760                None
761            }
762        } else {
763            None
764        };
765
766        let binding_sig = if version.has_sapling()
767            && !(shielded_spends.is_empty() && shielded_outputs.is_empty())
768        {
769            let mut sig = [0; 64];
770            reader.read_exact(&mut sig)?;
771            Some(redjubjub::Signature::from(sig))
772        } else {
773            None
774        };
775
776        let mut txid = [0; 32];
777        let hash_bytes = reader.into_hash();
778        txid.copy_from_slice(&hash_bytes);
779
780        Ok(Transaction {
781            txid: TxId::from_bytes(txid),
782            data: TransactionData {
783                version,
784                consensus_branch_id,
785                lock_time,
786                expiry_height,
787                #[cfg(all(
788                    any(zcash_unstable = "nu7", zcash_unstable = "zfuture"),
789                    feature = "zip-233"
790                ))]
791                zip233_amount: Zatoshis::ZERO,
792                transparent_bundle,
793                sprout_bundle,
794                sapling_bundle: binding_sig.and_then(|binding_sig| {
795                    sapling::Bundle::from_parts(
796                        shielded_spends,
797                        shielded_outputs,
798                        value_balance,
799                        sapling::bundle::Authorized { binding_sig },
800                    )
801                }),
802                orchard_bundle: None,
803                #[cfg(zcash_unstable = "zfuture")]
804                tze_bundle: None,
805            },
806        })
807    }

900    /// Utility function for reading header data common to v5 and v6 transactions.
901    fn read_header_fragment<R: Read>(mut reader: R) -> io::Result<(BranchId, u32, BlockHeight)> {
902        let consensus_branch_id = reader.read_u32_le().and_then(|value| {
903            BranchId::try_from(value).map_err(|_e| {
904                io::Error::new(
905                    io::ErrorKind::InvalidData,
906                    #[cfg(not(feature = "std"))]
907                    "invalid consensus branch id",
908                    #[cfg(feature = "std")]
909                    format!(
910                        "invalid consensus branch id 0x{}",
911                        hex::encode(value.to_be_bytes())
912                    ),
913                )
914            })
915        })?;
916        let lock_time = reader.read_u32_le()?;
917        let expiry_height: BlockHeight = reader.read_u32_le()?.into();
918        Ok((consensus_branch_id, lock_time, expiry_height))
919    }

967        match self.version {
968            TxVersion::Sprout(_) | TxVersion::V3 | TxVersion::V4 => self.write_v4(writer),
969            TxVersion::V5 => self.write_v5(writer),
970            #[cfg(zcash_unstable = "nu7")]
971            TxVersion::V6 => self.write_v6(writer),

973            TxVersion::ZFuture => self.write_v6(writer),

977    pub fn write_v4<W: Write>(&self, mut writer: W) -> io::Result<()> {
978        if self.orchard_bundle.is_some() {
979            return Err(io::Error::new(
980                io::ErrorKind::InvalidInput,
981                "Orchard components cannot be present when serializing to the V4 transaction format.",
982            ));
983        }
984
985        self.version.write(&mut writer)?;
986
987        self.write_transparent(&mut writer)?;
988        writer.write_u32_le(self.lock_time)?;
989        if self.version.has_overwinter() {
990            writer.write_u32_le(u32::from(self.expiry_height))?;
991        }
992
993        sapling_serialization::write_v4_components(
994            &mut writer,
995            self.sapling_bundle.as_ref(),
996            self.version.has_sapling(),
997        )?;
998
999        if self.version.has_sprout() {
1000            if let Some(bundle) = self.sprout_bundle.as_ref() {
1001                Vector::write(&mut writer, &bundle.joinsplits, |w, e| e.write(w))?;
1002                writer.write_all(&bundle.joinsplit_pubkey)?;
1003                writer.write_all(&bundle.joinsplit_sig)?;
1004            } else {
1005                CompactSize::write(&mut writer, 0)?;
1006            }
1007        }
1008
1009        if self.version.has_sapling() {
1010            if let Some(bundle) = self.sapling_bundle.as_ref() {
1011                writer.write_all(&<[u8; 64]>::from(bundle.authorization().binding_sig))?;
1012            }
1013        }
1014
1015        Ok(())
1016    }

1021            Vector::write(&mut writer, &bundle.vout, |w, e| e.write(w))?;
1022        } else {

1223    pub fn arb_tx_version(branch_id: BranchId) -> impl Strategy<Value = TxVersion> {
1224        match branch_id {
1225            BranchId::Sprout => (1..=2u32).prop_map(TxVersion::Sprout).boxed(),
1226            BranchId::Overwinter => Just(TxVersion::V3).boxed(),
1227            BranchId::Sapling | BranchId::Blossom | BranchId::Heartwood | BranchId::Canopy => {
1228                Just(TxVersion::V4).boxed()
1229            }
1230            BranchId::Nu5 => Just(TxVersion::V5).boxed(),
1231            BranchId::Nu6 => Just(TxVersion::V5).boxed(),
1232            BranchId::Nu6_1 => Just(TxVersion::V5).boxed(),
1233            #[cfg(zcash_unstable = "nu7")]
1234            BranchId::Nu7 => Just(TxVersion::V6).boxed(),
1235            #[cfg(zcash_unstable = "zfuture")]
1236            BranchId::ZFuture => Just(TxVersion::ZFuture).boxed(),
1237        }
1238    }
