17/// A trait that represents the ability to compute the fees that must be paid
18/// by a transaction having a specified set of inputs and outputs.
19pub trait FeeRule {
20    type Error;
21
22    /// Computes the total fee required for a transaction given the provided inputs and outputs.
23    ///
24    /// Implementations of this method should compute the fee amount given exactly the inputs and
25    /// outputs specified, and should NOT compute speculative fees given any additional change
26    /// outputs that may need to be created in order for inputs and outputs to balance.
27    #[allow(clippy::too_many_arguments)]
28    fn fee_required<P: consensus::Parameters>(
29        &self,
30        params: &P,
31        target_height: BlockHeight,
32        transparent_input_sizes: impl IntoIterator<Item = InputSize>,
33        transparent_output_sizes: impl IntoIterator<Item = usize>,
34        sapling_input_count: usize,
35        sapling_output_count: usize,
36        orchard_action_count: usize,
37    ) -> Result<Zatoshis, Self::Error>;
38}
39
40/// A trait that represents the ability to compute the fees that must be paid by a transaction
41/// having a specified set of inputs and outputs, for use when experimenting with the TZE feature.
42#[cfg(zcash_unstable = "zfuture")]
43pub trait FutureFeeRule: FeeRule {
44    /// Computes the total fee required for a transaction given the provided inputs and outputs.
45    ///
46    /// Implementations of this method should compute the fee amount given exactly the inputs and
47    /// outputs specified, and should NOT compute speculative fees given any additional change
48    /// outputs that may need to be created in order for inputs and outputs to balance.
49    #[allow(clippy::too_many_arguments)]
50    fn fee_required_zfuture<P: consensus::Parameters>(
51        &self,
52        params: &P,
53        target_height: BlockHeight,
54        transparent_input_sizes: impl IntoIterator<Item = InputSize>,
55        transparent_output_sizes: impl IntoIterator<Item = usize>,
56        sapling_input_count: usize,
57        sapling_output_count: usize,
58        orchard_action_count: usize,
59        tze_inputs: &[impl tze::InputView],
60        tze_outputs: &[impl tze::OutputView],
61    ) -> Result<Zatoshis, Self::Error>;
62}
