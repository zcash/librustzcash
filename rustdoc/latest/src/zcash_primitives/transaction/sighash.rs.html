15pub enum SignableInput<'a> {
16    Shielded,
17    Transparent(transparent::sighash::SignableInput<'a>),
18    #[cfg(zcash_unstable = "zfuture")]
19    Tze {
20        index: usize,
21        precondition: &'a Precondition,
22        value: Zatoshis,
23    },
24}
25
26impl SignableInput<'_> {
27    pub fn hash_type(&self) -> u8 {
28        match self {
29            SignableInput::Shielded => ::transparent::sighash::SIGHASH_ALL,
30            SignableInput::Transparent(input) => input.hash_type().encode(),
31            #[cfg(zcash_unstable = "zfuture")]
32            SignableInput::Tze { .. } => ::transparent::sighash::SIGHASH_ALL,
33        }
34    }
35}

45/// Computes the signature hash for an input to a transaction, given
46/// the full data of the transaction, the input being signed, and the
47/// set of precomputed hashes produced in the construction of the
48/// transaction ID.
49pub fn signature_hash<
50    TA: ::transparent::sighash::TransparentAuthorizingContext,
51    SA: sapling::bundle::Authorization<SpendProof = GrothProofBytes, OutputProof = GrothProofBytes>,
52    A: Authorization<SaplingAuth = SA, TransparentAuth = TA>,
53>(
54    tx: &TransactionData<A>,
55    signable_input: &SignableInput,
56    txid_parts: &TxDigests<Blake2bHash>,
57) -> SignatureHash {
58    SignatureHash(match tx.version {
59        TxVersion::Sprout(_) | TxVersion::V3 | TxVersion::V4 => {
60            v4_signature_hash(tx, signable_input)
61        }
62
63        TxVersion::V5 => v5_signature_hash(tx, signable_input, txid_parts),
64
65        #[cfg(zcash_unstable = "nu7")]
66        TxVersion::V6 => v6_signature_hash(tx, signable_input, txid_parts),
67        #[cfg(zcash_unstable = "zfuture")]
68        TxVersion::ZFuture => v6_signature_hash(tx, signable_input, txid_parts),
69    })
