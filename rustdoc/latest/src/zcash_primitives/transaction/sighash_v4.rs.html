18const ZCASH_SIGHASH_PERSONALIZATION_PREFIX: &[u8; 12] = b"ZcashSigHash";
19const ZCASH_PREVOUTS_HASH_PERSONALIZATION: &[u8; 16] = b"ZcashPrevoutHash";
20const ZCASH_SEQUENCE_HASH_PERSONALIZATION: &[u8; 16] = b"ZcashSequencHash";
21const ZCASH_OUTPUTS_HASH_PERSONALIZATION: &[u8; 16] = b"ZcashOutputsHash";
22const ZCASH_JOINSPLITS_HASH_PERSONALIZATION: &[u8; 16] = b"ZcashJSplitsHash";
23const ZCASH_SHIELDED_SPENDS_HASH_PERSONALIZATION: &[u8; 16] = b"ZcashSSpendsHash";
24const ZCASH_SHIELDED_OUTPUTS_HASH_PERSONALIZATION: &[u8; 16] = b"ZcashSOutputHash";

26macro_rules! update_hash {
27    ($h:expr, $cond:expr, $value:expr) => {
28        if $cond {
29            $h.update(&$value.as_ref());
30        } else {
31            $h.update(&[0; 32]);
32        }
33    };
34}

36fn prevout_hash<TA: transparent::Authorization>(vin: &[TxIn<TA>]) -> Blake2bHash {
37    let mut data = Vec::with_capacity(vin.len() * 36);
38    for t_in in vin {
39        t_in.prevout().write(&mut data).unwrap();
40    }
41    Blake2bParams::new()
42        .hash_length(32)
43        .personal(ZCASH_PREVOUTS_HASH_PERSONALIZATION)
44        .hash(&data)
45}
46
47fn sequence_hash<TA: transparent::Authorization>(vin: &[TxIn<TA>]) -> Blake2bHash {
48    let mut data = Vec::with_capacity(vin.len() * 4);
49    for t_in in vin {
50        data.extend_from_slice(&t_in.sequence().to_le_bytes());
51    }
52    Blake2bParams::new()
53        .hash_length(32)
54        .personal(ZCASH_SEQUENCE_HASH_PERSONALIZATION)
55        .hash(&data)
56}

58fn outputs_hash(vout: &[TxOut]) -> Blake2bHash {
59    let mut data = Vec::with_capacity(vout.len() * (4 + 1));
60    for t_out in vout {
61        t_out.write(&mut data).unwrap();
62    }
63    Blake2bParams::new()
64        .hash_length(32)
65        .personal(ZCASH_OUTPUTS_HASH_PERSONALIZATION)
66        .hash(&data)
67}

69fn single_output_hash(tx_out: &TxOut) -> Blake2bHash {
70    let mut data = vec![];
71    tx_out.write(&mut data).unwrap();
72    Blake2bParams::new()
73        .hash_length(32)
74        .personal(ZCASH_OUTPUTS_HASH_PERSONALIZATION)
75        .hash(&data)
76}

78fn joinsplits_hash(
79    consensus_branch_id: BranchId,
80    joinsplits: &[JsDescription],
81    joinsplit_pubkey: &[u8; 32],
82) -> Blake2bHash {
83    let mut data = Vec::with_capacity(
84        joinsplits.len()
85            * if consensus_branch_id.sprout_uses_groth_proofs() {
86                1698 // JSDescription with Groth16 proof
87            } else {
88                1802 // JsDescription with PHGR13 proof
89            },
90    );
91    for js in joinsplits {
92        js.write(&mut data).unwrap();
93    }
94    data.extend_from_slice(joinsplit_pubkey);
95    Blake2bParams::new()
96        .hash_length(32)
97        .personal(ZCASH_JOINSPLITS_HASH_PERSONALIZATION)
98        .hash(&data)
99}

101fn shielded_spends_hash<
102    A: sapling::bundle::Authorization<SpendProof = GrothProofBytes, OutputProof = GrothProofBytes>,
103>(
104    shielded_spends: &[SpendDescription<A>],
105) -> Blake2bHash {
106    let mut data = Vec::with_capacity(shielded_spends.len() * 384);
107    for s_spend in shielded_spends {
108        data.extend_from_slice(&s_spend.cv().to_bytes());
109        data.extend_from_slice(s_spend.anchor().to_repr().as_ref());
110        data.extend_from_slice(s_spend.nullifier().as_ref());
111        data.extend_from_slice(&<[u8; 32]>::from(*s_spend.rk()));
112        data.extend_from_slice(s_spend.zkproof());
113    }
114    Blake2bParams::new()
115        .hash_length(32)
116        .personal(ZCASH_SHIELDED_SPENDS_HASH_PERSONALIZATION)
117        .hash(&data)
118}

120fn shielded_outputs_hash(shielded_outputs: &[OutputDescription<GrothProofBytes>]) -> Blake2bHash {
121    let mut data = Vec::with_capacity(shielded_outputs.len() * 948);
122    for s_out in shielded_outputs {
123        sapling_serialization::write_output_v4(&mut data, s_out).unwrap();
124    }
125    Blake2bParams::new()
126        .hash_length(32)
127        .personal(ZCASH_SHIELDED_OUTPUTS_HASH_PERSONALIZATION)
128        .hash(&data)
129}

131pub fn v4_signature_hash<
132    SA: sapling::bundle::Authorization<SpendProof = GrothProofBytes, OutputProof = GrothProofBytes>,
133    A: Authorization<SaplingAuth = SA>,
134>(
135    tx: &TransactionData<A>,
136    signable_input: &SignableInput<'_>,
137) -> Blake2bHash {
138    let hash_type = signable_input.hash_type();
139    if tx.version.has_overwinter() {
140        let mut personal = [0; 16];
141        personal[..12].copy_from_slice(ZCASH_SIGHASH_PERSONALIZATION_PREFIX);
142        personal[12..].copy_from_slice(&u32::from(tx.consensus_branch_id).to_le_bytes());
143
144        let mut h = Blake2bParams::new()
145            .hash_length(32)
146            .personal(&personal)
147            .to_state();
148
149        h.update(&tx.version.header().to_le_bytes());
150        h.update(&tx.version.version_group_id().to_le_bytes());
151        update_hash!(
152            h,
153            hash_type & SIGHASH_ANYONECANPAY == 0,
154            prevout_hash(
155                tx.transparent_bundle
156                    .as_ref()
157                    .map_or(&[], |b| b.vin.as_slice())
158            )
159        );
160        update_hash!(
161            h,
162            (hash_type & SIGHASH_ANYONECANPAY) == 0
163                && (hash_type & SIGHASH_MASK) != SIGHASH_SINGLE
164                && (hash_type & SIGHASH_MASK) != SIGHASH_NONE,
165            sequence_hash(
166                tx.transparent_bundle
167                    .as_ref()
168                    .map_or(&[], |b| b.vin.as_slice())
169            )
170        );
171
172        if (hash_type & SIGHASH_MASK) != SIGHASH_SINGLE
173            && (hash_type & SIGHASH_MASK) != SIGHASH_NONE
174        {
175            h.update(
176                outputs_hash(
177                    tx.transparent_bundle
178                        .as_ref()
179                        .map_or(&[], |b| b.vout.as_slice()),
180                )
181                .as_bytes(),
182            );
183        } else if (hash_type & SIGHASH_MASK) == SIGHASH_SINGLE {
184            match (tx.transparent_bundle.as_ref(), signable_input) {
185                (Some(b), SignableInput::Transparent(input)) if input.index() < &b.vout.len() => {
186                    h.update(single_output_hash(&b.vout[*input.index()]).as_bytes())
187                }
188                _ => h.update(&[0; 32]),
189            };
190        } else {
191            h.update(&[0; 32]);
192        };
193
194        update_hash!(
195            h,
196            !tx.sprout_bundle
197                .as_ref()
198                .is_none_or(|b| b.joinsplits.is_empty()),
199            {
200                let bundle = tx.sprout_bundle.as_ref().unwrap();
201                joinsplits_hash(
202                    tx.consensus_branch_id,
203                    &bundle.joinsplits,
204                    &bundle.joinsplit_pubkey,
205                )
206            }
207        );
208
209        if tx.version.has_sapling() {
210            update_hash!(
211                h,
212                !tx.sapling_bundle
213                    .as_ref()
214                    .is_none_or(|b| b.shielded_spends().is_empty()),
215                shielded_spends_hash(tx.sapling_bundle.as_ref().unwrap().shielded_spends())
216            );
217            update_hash!(
218                h,
219                !tx.sapling_bundle
220                    .as_ref()
221                    .is_none_or(|b| b.shielded_outputs().is_empty()),
222                shielded_outputs_hash(tx.sapling_bundle.as_ref().unwrap().shielded_outputs())
223            );
224        }
225        h.update(&tx.lock_time.to_le_bytes());
226        h.update(&u32::from(tx.expiry_height).to_le_bytes());
227        if tx.version.has_sapling() {
228            h.update(&tx.sapling_value_balance().to_i64_le_bytes());
229        }
230        h.update(&u32::from(hash_type).to_le_bytes());
231
232        match signable_input {
233            SignableInput::Shielded => (),
234            SignableInput::Transparent(input) => {
235                if let Some(bundle) = tx.transparent_bundle.as_ref() {
236                    let mut data = vec![];
237                    bundle.vin[*input.index()]
238                        .prevout()
239                        .write(&mut data)
240                        .unwrap();
241                    input.script_code().write(&mut data).unwrap();
242                    data.extend_from_slice(&input.value().to_i64_le_bytes());
243                    data.extend_from_slice(&bundle.vin[*input.index()].sequence().to_le_bytes());
244                    h.update(&data);
245                } else {
246                    panic!(
247                        "A request has been made to sign a transparent input, but none are present."
248                    );
249                }
250            }
251
252            #[cfg(zcash_unstable = "zfuture")]
253            SignableInput::Tze { .. } => {
254                panic!(
255                    "A request has been made to sign a TZE input, but the transaction version is not ZFuture"
256                );
257            }
258        }
259
260        h.finalize()
261    } else {
262        panic!("Signature hashing for pre-overwinter transactions is not supported.")
263    }
