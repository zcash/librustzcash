16/// The size of a transparent input, or the outpoint corresponding to the input
17/// if the size of the script required to spend that input is unknown.
18#[derive(Clone, Debug, PartialEq, Eq)]
19pub enum InputSize {
20    /// The txin size is known.
21    Known(usize),
22    /// The size of the script required to spend this input (and therefore the txin size)
23    /// is unknown.
24    Unknown(OutPoint),
25}
26
27impl InputSize {
28    /// An `InputSize` corresponding to the upper bound on the size of a P2PKH input used by ZIP 317.
29    pub const STANDARD_P2PKH: InputSize = InputSize::Known(P2PKH_STANDARD_INPUT_SIZE);
30}
31
32/// This trait provides a minimized view of a transparent input suitable for use in
33/// fee and change computation.
34pub trait InputView: core::fmt::Debug {
35    /// The outpoint to which the input refers.
36    fn outpoint(&self) -> &OutPoint;
37
38    /// The previous output being spent.
39    fn coin(&self) -> &TxOut;
40
41    /// The size of the transparent script required to spend this input.
42    fn serialized_size(&self) -> InputSize {
43        match self.coin().script_pubkey.address() {
44            Some(TransparentAddress::PublicKeyHash(_)) => InputSize::STANDARD_P2PKH,
45            _ => InputSize::Unknown(self.outpoint().clone()),
46        }
47    }
48}
49
50#[cfg(feature = "transparent-inputs")]
51impl InputView for TransparentInputInfo {
52    fn outpoint(&self) -> &OutPoint {
53        self.outpoint()
54    }
55
56    fn coin(&self) -> &TxOut {
57        self.coin()
58    }
59}
60
61impl InputView for Infallible {
62    fn outpoint(&self) -> &OutPoint {
63        unreachable!()
64    }
65    fn coin(&self) -> &TxOut {
66        unreachable!()
67    }
68}
69
70/// This trait provides a minimized view of a transparent output suitable for use in
71/// fee and change computation.
72pub trait OutputView: core::fmt::Debug {
73    /// Returns the value of the output being created.
74    fn value(&self) -> Zatoshis;
75
76    /// Returns the script corresponding to the newly created output.
77    fn script_pubkey(&self) -> &Script;
78
79    /// Returns the serialized size of the txout.
80    fn serialized_size(&self) -> usize {
81        // The serialized size of a transparent `TxOut` is the serialized size of an amount
82        // plus the serialized size of the script pubkey.
83        8 + self.script_pubkey().serialized_size()
84    }
85}
86
87impl OutputView for TxOut {
88    fn value(&self) -> Zatoshis {
89        self.value
90    }
91
92    fn script_pubkey(&self) -> &Script {
93        &self.script_pubkey
94    }
95}
