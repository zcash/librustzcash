17/// The size of a transparent input, or the outpoint corresponding to the input
18/// if the size of the script required to spend that input is unknown.
19#[derive(Clone, Debug, PartialEq, Eq)]
20pub enum InputSize {
21    /// The txin size is known.
22    Known(usize),
23    /// The size of the script required to spend this input (and therefore the txin size)
24    /// is unknown.
25    Unknown(OutPoint),
26}
27
28impl InputSize {
29    /// An `InputSize` corresponding to the upper bound on the size of a P2PKH input used by ZIP 317.
30    pub const STANDARD_P2PKH: InputSize = InputSize::Known(P2PKH_STANDARD_INPUT_SIZE);
31}
32
33/// This trait provides a minimized view of a transparent input suitable for use in
34/// fee and change computation.
35pub trait InputView: core::fmt::Debug {
36    /// The outpoint to which the input refers.
37    fn outpoint(&self) -> &OutPoint;
38
39    /// The previous output being spent.
40    fn coin(&self) -> &TxOut;
41
42    /// The size of this transparent input in a transaction, as used in [ZIP 317].
43    ///
44    /// [ZIP 317]: https://zips.z.cash/zip-0317#rationale-for-the-chosen-parameters
45    fn serialized_size(&self) -> InputSize {
46        match script::PubKey::parse(&self.coin().script_pubkey().0)
47            .ok()
48            .as_ref()
49            .and_then(solver::standard)
50        {
51            Some(solver::ScriptKind::PubKeyHash { .. }) => InputSize::STANDARD_P2PKH,
52            _ => InputSize::Unknown(self.outpoint().clone()),
53        }
54    }
55}
56
57#[cfg(feature = "transparent-inputs")]
58impl InputView for TransparentInputInfo {
59    fn outpoint(&self) -> &OutPoint {
60        self.outpoint()
61    }
62
63    fn coin(&self) -> &TxOut {
64        self.coin()
65    }
66
67    fn serialized_size(&self) -> InputSize {
68        self.serialized_len().map_or(
69            InputSize::Unknown(self.outpoint().clone()),
70            InputSize::Known,
71        )
72    }
73}
74
75impl InputView for Infallible {
76    fn outpoint(&self) -> &OutPoint {
77        unreachable!()
78    }
79    fn coin(&self) -> &TxOut {
80        unreachable!()
81    }
82}

84/// This trait provides a minimized view of a transparent output suitable for use in
85/// fee and change computation.
86pub trait OutputView: core::fmt::Debug {
87    /// Returns the value of the output being created.
88    fn value(&self) -> Zatoshis;
89
90    /// Returns the script corresponding to the newly created output.
91    fn script_pubkey(&self) -> &Script;
92
93    /// Returns the serialized size of the txout.
94    fn serialized_size(&self) -> usize {
95        // The serialized size of a transparent `TxOut` is the serialized size of an amount
96        // plus the serialized size of the script pubkey.
97        8 + self.script_pubkey().serialized_size()
98    }
99}
100
101impl OutputView for TxOut {
102    fn value(&self) -> Zatoshis {
103        self.value()
104    }
105
106    fn script_pubkey(&self) -> &Script {
107        self.script_pubkey()
108    }
