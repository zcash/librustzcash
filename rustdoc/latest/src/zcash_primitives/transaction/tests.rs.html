30fn tx_read_write() {
31    let data = &self::data::tx_read_write::TX_READ_WRITE;
32    let tx = Transaction::read(&data[..], BranchId::Canopy).unwrap();
33    assert_eq!(
34        format!("{}", tx.txid()),
35        "64f0bd7fe30ce23753358fe3a2dc835b8fba9c0274c4e2c54a6f73114cb55639"
36    );
37
38    let mut encoded = Vec::with_capacity(data.len());
39    tx.write(&mut encoded).unwrap();
40    assert_eq!(&data[..], &encoded[..]);

43#[cfg(test)]
44fn check_roundtrip(tx: Transaction) -> Result<(), TestCaseError> {
45    let mut txn_bytes = vec![];
46    tx.write(&mut txn_bytes).unwrap();
47    let txo = Transaction::read(&txn_bytes[..], tx.consensus_branch_id).unwrap();
48
49    prop_assert_eq!(tx.version, txo.version);
50    #[cfg(zcash_unstable = "zfuture")]
51    prop_assert_eq!(tx.tze_bundle.as_ref(), txo.tze_bundle.as_ref());
52    prop_assert_eq!(tx.lock_time, txo.lock_time);
53    prop_assert_eq!(
54        tx.transparent_bundle.as_ref(),
55        txo.transparent_bundle.as_ref()
56    );
57    prop_assert_eq!(tx.sapling_value_balance(), txo.sapling_value_balance());
58    prop_assert_eq!(
59        tx.orchard_bundle.as_ref().map(|v| *v.value_balance()),
60        txo.orchard_bundle.as_ref().map(|v| *v.value_balance())
61    );
62    #[cfg(all(zcash_unstable = "nu7", feature = "zip-233"))]
63    if tx.version.has_zip233() {
64        prop_assert_eq!(tx.zip233_amount, txo.zip233_amount);
65    }
66    Ok(())
67}

227#[test]
228fn zip_0244() {
229    fn to_test_txdata(
230        tv: &self::data::zip_0244::TestVector,
231    ) -> (TransactionData<TestUnauthorized>, TxDigests<Blake2bHash>) {
232        let tx = Transaction::read(&tv.tx[..], BranchId::Nu5).unwrap();
233
234        assert_eq!(tx.txid.as_ref(), &tv.txid);
235        assert_eq!(tx.auth_commitment().as_ref(), &tv.auth_digest);
236
237        let txdata = tx.deref();
238
239        let input_amounts = tv
240            .amounts
241            .iter()
242            .map(|amount| Zatoshis::from_nonnegative_i64(*amount).unwrap())
243            .collect();
244        let input_scriptpubkeys = tv
245            .script_pubkeys
246            .iter()
247            .cloned()
248            .map(script::Code)
249            .map(Script)
250            .collect();
251
252        let test_bundle = txdata
253            .transparent_bundle
254            .as_ref()
255            .map(|b| transparent::Bundle {
256                // we have to do this map/clone to make the types line up, since the
257                // Authorization::ScriptSig type is bound to transparent::Authorized, and we need
258                // it to be bound to TestTransparentAuth.
259                vin: b
260                    .vin
261                    .iter()
262                    .map(|vin| {
263                        TxIn::from_parts(
264                            vin.prevout().clone(),
265                            vin.script_sig().clone(),
266                            vin.sequence(),
267                        )
268                    })
269                    .collect(),
270                vout: b.vout.clone(),
271                authorization: TestTransparentAuth {
272                    input_amounts,
273                    input_scriptpubkeys,
274                },
275            });
276
277        #[cfg(not(zcash_unstable = "zfuture"))]
278        let tdata = TransactionData::from_parts(
279            txdata.version(),
280            txdata.consensus_branch_id(),
281            txdata.lock_time(),
282            txdata.expiry_height(),
283            #[cfg(all(zcash_unstable = "nu7", feature = "zip-233"))]
284            txdata.zip233_amount,
285            test_bundle,
286            txdata.sprout_bundle().cloned(),
287            txdata.sapling_bundle().cloned(),
288            txdata.orchard_bundle().cloned(),
289        );
290        #[cfg(zcash_unstable = "zfuture")]
291        let tdata = TransactionData::from_parts_zfuture(
292            txdata.version(),
293            txdata.consensus_branch_id(),
294            txdata.lock_time(),
295            txdata.expiry_height(),
296            #[cfg(feature = "zip-233")]
297            txdata.zip233_amount,
298            test_bundle,
299            txdata.sprout_bundle().cloned(),
300            txdata.sapling_bundle().cloned(),
301            txdata.orchard_bundle().cloned(),
302            txdata.tze_bundle().cloned(),
303        );
304        (tdata, txdata.digest(TxIdDigester))
305    }
306
307    for tv in self::data::zip_0244::make_test_vectors() {
308        let (txdata, txid_parts) = to_test_txdata(&tv);
309
310        if let Some(index) = tv.transparent_input {
311            // nIn is a u32, but to actually use it we need a usize.
312            let index = index as usize;
313            let bundle = txdata.transparent_bundle().unwrap();
314            let value = bundle.authorization.input_amounts[index];
315            let script_pubkey = &bundle.authorization.input_scriptpubkeys[index];
316            let signable_input = |hash_type| {
317                SignableInput::Transparent(::transparent::sighash::SignableInput::from_parts(
318                    hash_type,
319                    index,
320                    script_pubkey,
321                    script_pubkey,
322                    value,
323                ))
324            };
325
326            assert_eq!(
327                v5_signature_hash(&txdata, &signable_input(SighashType::ALL), &txid_parts).as_ref(),
328                &tv.sighash_all.unwrap()
329            );
330
331            assert_eq!(
332                v5_signature_hash(&txdata, &signable_input(SighashType::NONE), &txid_parts)
333                    .as_ref(),
334                &tv.sighash_none.unwrap()
335            );
336
337            if index < bundle.vout.len() {
338                assert_eq!(
339                    v5_signature_hash(&txdata, &signable_input(SighashType::SINGLE), &txid_parts)
340                        .as_ref(),
341                    &tv.sighash_single.unwrap()
342                );
343            } else {
344                assert_eq!(tv.sighash_single, None);
345            }
346
347            assert_eq!(
348                v5_signature_hash(
349                    &txdata,
350                    &signable_input(SighashType::ALL_ANYONECANPAY),
351                    &txid_parts,
352                )
353                .as_ref(),
354                &tv.sighash_all_anyone.unwrap()
355            );
356
357            assert_eq!(
358                v5_signature_hash(
359                    &txdata,
360                    &signable_input(SighashType::NONE_ANYONECANPAY),
361                    &txid_parts,
362                )
363                .as_ref(),
364                &tv.sighash_none_anyone.unwrap()
365            );
366
367            if index < bundle.vout.len() {
368                assert_eq!(
369                    v5_signature_hash(
370                        &txdata,
371                        &signable_input(SighashType::SINGLE_ANYONECANPAY),
372                        &txid_parts,
373                    )
374                    .as_ref(),
375                    &tv.sighash_single_anyone.unwrap()
376                );
377            } else {
378                assert_eq!(tv.sighash_single_anyone, None);
379            }
380        };
381
382        assert_eq!(
383            v5_signature_hash(&txdata, &SignableInput::Shielded, &txid_parts).as_ref(),
384            tv.sighash_shielded
385        );
386    }
387}
