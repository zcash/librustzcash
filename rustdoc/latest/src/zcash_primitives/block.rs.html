17/// The identifier for a Zcash block.
18///
19/// This is the SHA-256d hash of the encoded [`BlockHeader`].
20#[derive(Clone, Copy, PartialEq, Eq, Hash)]
21pub struct BlockHash(pub [u8; 32]);
22
23memuse::impl_no_dynamic_usage!(BlockHash);
24
25impl fmt::Debug for BlockHash {
26    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
27        // The (byte-flipped) hex string is more useful than the raw bytes, because we can
28        // look that up in RPC methods and block explorers.
29        let block_hash_str = self.to_string();
30        f.debug_tuple("BlockHash").field(&block_hash_str).finish()
31    }
32}
33
34impl fmt::Display for BlockHash {
35    fn fmt(&self, formatter: &mut fmt::Formatter<'_>) -> fmt::Result {
36        let mut data = self.0;
37        data.reverse();
38        formatter.write_str(&hex::encode(data))
39    }
40}
41
42impl BlockHash {
43    /// Constructs a [`BlockHash`] from the given slice.
44    ///
45    /// # Panics
46    ///
47    /// This function will panic if the slice is not exactly 32 bytes.
48    pub fn from_slice(bytes: &[u8]) -> Self {
49        Self::try_from_slice(bytes).unwrap()
50    }
51
52    /// Constructs a [`BlockHash`] from the given slice.
53    ///
54    /// Returns `None` if `bytes` has any length other than 32
55    pub fn try_from_slice(bytes: &[u8]) -> Option<Self> {
56        if bytes.len() == 32 {
57            let mut hash = [0; 32];
58            hash.copy_from_slice(bytes);
59            Some(BlockHash(hash))
60        } else {
61            None
62        }
63    }
64}

80/// The information contained in a Zcash block header.
81pub struct BlockHeaderData {
82    pub version: i32,
83    pub prev_block: BlockHash,
84    pub merkle_root: [u8; 32],
85    pub final_sapling_root: [u8; 32],
86    pub time: u32,
87    pub bits: u32,
88    pub nonce: [u8; 32],
89    pub solution: Vec<u8>,
90}
91
92impl BlockHeaderData {
93    pub fn freeze(self) -> io::Result<BlockHeader> {
94        BlockHeader::from_data(self)
95    }
96}
97
98impl BlockHeader {
99    fn from_data(data: BlockHeaderData) -> io::Result<Self> {
100        let mut header = BlockHeader {
101            hash: BlockHash([0; 32]),
102            data,
103        };
104        let mut raw = vec![];
105        header.write(&mut raw)?;
106        header
107            .hash
108            .0
109            .copy_from_slice(&Sha256::digest(Sha256::digest(&raw)));
110        Ok(header)
111    }
112
113    /// Returns the hash of this header.
114    pub fn hash(&self) -> BlockHash {
115        self.hash
116    }
117
118    pub fn read<R: Read>(mut reader: R) -> io::Result<Self> {
119        let version = reader.read_i32_le()?;
120
121        let mut prev_block = BlockHash([0; 32]);
122        reader.read_exact(&mut prev_block.0)?;
123
124        let mut merkle_root = [0; 32];
125        reader.read_exact(&mut merkle_root)?;
126
127        let mut final_sapling_root = [0; 32];
128        reader.read_exact(&mut final_sapling_root)?;
129
130        let time = reader.read_u32_le()?;
131        let bits = reader.read_u32_le()?;
132
133        let mut nonce = [0; 32];
134        reader.read_exact(&mut nonce)?;
135
136        let solution = Vector::read(&mut reader, |r| r.read_u8())?;
137
138        BlockHeader::from_data(BlockHeaderData {
139            version,
140            prev_block,
141            merkle_root,
142            final_sapling_root,
143            time,
144            bits,
145            nonce,
146            solution,
147        })
148    }
149
150    pub fn write<W: Write>(&self, mut writer: W) -> io::Result<()> {
151        writer.write_i32_le(self.version)?;
152        writer.write_all(&self.prev_block.0)?;
153        writer.write_all(&self.merkle_root)?;
154        writer.write_all(&self.final_sapling_root)?;
155        writer.write_u32_le(self.time)?;
156        writer.write_u32_le(self.bits)?;
157        writer.write_all(&self.nonce)?;
158        Vector::write(&mut writer, &self.solution, |w, b| w.write_u8(*b))?;
159
160        Ok(())
161    }
162}
