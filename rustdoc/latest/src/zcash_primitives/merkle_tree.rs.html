15/// A hashable node within a Merkle tree.
16pub trait HashSer {
17    /// Parses a node from the given byte source.
18    fn read<R: Read>(reader: R) -> io::Result<Self>
19    where
20        Self: Sized;
21
22    /// Serializes this node.
23    fn write<W: Write>(&self, writer: W) -> io::Result<()>;
24}
25
26impl HashSer for sapling::Node {
27    fn read<R: Read>(mut reader: R) -> io::Result<Self> {
28        let mut repr = [0u8; 32];
29        reader.read_exact(&mut repr)?;
30        Option::from(Self::from_bytes(repr)).ok_or_else(|| {
31            io::Error::new(
32                io::ErrorKind::InvalidData,
33                "Non-canonical encoding of Jubjub base field value.",
34            )
35        })
36    }
37
38    fn write<W: Write>(&self, mut writer: W) -> io::Result<()> {
39        writer.write_all(&self.to_bytes())
40    }
41}

43impl HashSer for MerkleHashOrchard {
44    fn read<R: Read>(mut reader: R) -> io::Result<Self>
45    where
46        Self: Sized,
47    {

60    }

63/// Writes a usize value encoded as a u64 in little-endian order. Since usize
64/// is platform-dependent, we consistently represent it as u64 in serialized
65/// formats.
66pub fn write_usize_leu64<W: Write>(mut writer: W, value: usize) -> io::Result<()> {
67    writer.write_u64_le(u64::try_from(value).map_err(|_| {
68        io::Error::new(
69            io::ErrorKind::InvalidData,
70            "usize value was outside the representable range of a u64",
71        )
72    })?)

75/// Reads a usize value encoded as a u64 in little-endian order. Since usize
76/// is platform-dependent, we consistently represent it as u64 in serialized
77/// formats.
78pub fn read_leu64_usize<R: Read>(mut reader: R) -> io::Result<usize> {
79    let mut repr = [0u8; 8];
80    reader.read_exact(&mut repr)?;
81    usize::try_from(u64::from_le_bytes(repr)).map_err(|_| {
82        io::Error::new(
83            io::ErrorKind::InvalidData,
84            "usize could not be decoded from a 64-bit value",
85        )
86    })

116pub fn write_nonempty_frontier_v1<H: HashSer, W: Write>(
117    mut writer: W,
118    frontier: &NonEmptyFrontier<H>,
119) -> io::Result<()> {
120    write_position(&mut writer, frontier.position())?;
121    if frontier.position().is_right_child() {
122        // The v1 serialization wrote the sibling of a right-hand leaf as an optional value, rather
123        // than as part of the ommers vector.
124        frontier
125            .ommers()
126            .first()
127            .expect("ommers vector cannot be empty for right-hand nodes")
128            .write(&mut writer)?;
129        Optional::write(&mut writer, Some(frontier.leaf()), |w, n: &H| n.write(w))?;
130        Vector::write(&mut writer, &frontier.ommers()[1..], |w, e| e.write(w))?;
131    } else {
132        frontier.leaf().write(&mut writer)?;
133        Optional::write(&mut writer, None, |w, n: &H| n.write(w))?;
134        Vector::write(&mut writer, frontier.ommers(), |w, e| e.write(w))?;
135    }
136
137    Ok(())
138}

140#[allow(clippy::redundant_closure)]
141pub fn read_nonempty_frontier_v1<H: HashSer + Clone, R: Read>(
142    mut reader: R,
143) -> io::Result<NonEmptyFrontier<H>> {
144    let position = read_position(&mut reader)?;
145    let left = H::read(&mut reader)?;
146    let right = Optional::read(&mut reader, H::read)?;
147    let mut ommers = Vector::read(&mut reader, |r| H::read(r))?;
148
149    let leaf = if let Some(right) = right {
150        // if the frontier has a right leaf, then the left leaf is the first ommer
151        ommers.insert(0, left);
152        right
153    } else {
154        left
155    };
156
157    NonEmptyFrontier::from_parts(position, leaf, ommers).map_err(|_err| {
158        io::Error::new(
159            io::ErrorKind::InvalidData,
160            "Parsing resulted in an invalid Merkle frontier",
161        )
162    })
163}

165pub fn write_frontier_v1<H: HashSer, W: Write>(
166    writer: W,
167    frontier: &Frontier<H, 32>,
168) -> io::Result<()> {
169    Optional::write(writer, frontier.value(), write_nonempty_frontier_v1)
170}

172#[allow(clippy::redundant_closure)]
173pub fn read_frontier_v1<H: HashSer + Clone, R: Read>(reader: R) -> io::Result<Frontier<H, 32>> {
174    match Optional::read(reader, read_nonempty_frontier_v1)? {
175        None => Ok(Frontier::empty()),
176        Some(f) => Frontier::try_from(f).map_err(|_err| {
177            io::Error::new(
178                io::ErrorKind::InvalidData,
179                "Parsing resulted in an invalid Merkle frontier",
180            )
181        }),
182    }

185/// Reads a legacy `CommitmentTree` from its serialized form.
186pub fn read_commitment_tree<Node: HashSer, R: Read, const DEPTH: u8>(
187    mut reader: R,
188) -> io::Result<CommitmentTree<Node, DEPTH>> {
189    let left = Optional::read(&mut reader, Node::read)?;
190    let right = Optional::read(&mut reader, Node::read)?;
191    let parents = Vector::read(&mut reader, |r| Optional::read(r, Node::read))?;
192
193    CommitmentTree::from_parts(left, right, parents).map_err(|_| {
194        io::Error::new(
195            io::ErrorKind::InvalidData,
196            "parents vector exceeded tree depth",
197        )
198    })
199}

201/// Serializes a legacy `CommitmentTree` as an array of bytes.
202pub fn write_commitment_tree<Node: HashSer, W: Write, const DEPTH: u8>(
203    tree: &CommitmentTree<Node, DEPTH>,
204    mut writer: W,
205) -> io::Result<()> {
206    Optional::write(&mut writer, tree.left().as_ref(), |w, n| n.write(w))?;
207    Optional::write(&mut writer, tree.right().as_ref(), |w, n| n.write(w))?;
208    Vector::write(&mut writer, tree.parents(), |w, e| {
209        Optional::write(w, e.as_ref(), |w, n| n.write(w))
210    })
211}

213/// Reads an `IncrementalWitness` from its serialized form.
214#[allow(clippy::redundant_closure)]
215pub fn read_incremental_witness<Node: HashSer, R: Read, const DEPTH: u8>(
216    mut reader: R,
217) -> io::Result<IncrementalWitness<Node, DEPTH>> {
218    let tree = read_commitment_tree(&mut reader)?;
219    let filled = Vector::read(&mut reader, |r| Node::read(r))?;
220    let cursor = Optional::read(&mut reader, read_commitment_tree)?;
221
222    IncrementalWitness::from_parts(tree, filled, cursor).ok_or(io::Error::new(
223        io::ErrorKind::InvalidData,
224        "invalid witness: inconsistency detected between witness parts",
225    ))
226}

228/// Serializes an `IncrementalWitness` as an array of bytes.
229pub fn write_incremental_witness<Node: HashSer, W: Write, const DEPTH: u8>(
230    witness: &IncrementalWitness<Node, DEPTH>,
231    mut writer: W,
232) -> io::Result<()> {
233    write_commitment_tree(witness.tree(), &mut writer)?;
234    Vector::write(&mut writer, witness.filled(), |w, n| n.write(w))?;
235    Optional::write(&mut writer, witness.cursor().as_ref(), |w, t| {
236        write_commitment_tree(t, w)
237    })
238}

240/// Reads a Merkle path from its serialized form.
241pub fn merkle_path_from_slice<Node: HashSer, const DEPTH: u8>(
242    mut witness: &[u8],
243) -> io::Result<MerklePath<Node, DEPTH>> {
244    // Skip the first byte, which should be DEPTH to signify the length of
245    // the following vector of Pedersen hashes.
246    if witness[0] != DEPTH {
247        return Err(io::Error::new(
248            io::ErrorKind::InvalidData,
249            "depth is not as expected",
250        ));
251    }
252    witness = &witness[1..];
253
254    // Begin to construct the authentication path
255    let iter = witness.chunks_exact(33);
256    witness = iter.remainder();
257
258    // The vector works in reverse
259    let auth_path = iter
260        .rev()
261        .map(|bytes| {
262            // Length of inner vector should be the length of a Pedersen hash
263            if bytes[0] == 32 {
264                // Sibling node should be an element of Fr
265                Node::read(&bytes[1..])
266            } else {
267                Err(io::Error::new(
268                    io::ErrorKind::InvalidData,
269                    "length of auth path element is not the expected 32 bytes",
270                ))
271            }
272        })
273        .collect::<io::Result<Vec<_>>>()?;
274    if auth_path.len() != usize::from(DEPTH) {
275        return Err(io::Error::new(
276            io::ErrorKind::InvalidData,
277            "auth path has unexpected length",
278        ));
279    }
280
281    // Read the position from the witness
282    let position = witness.read_u64_le().map(Position::from)?;
283
284    // The witness should be empty now; if it wasn't, the caller would
285    // have provided more information than they should have, indicating
286    // a bug downstream
287    if witness.is_empty() {
288        MerklePath::from_parts(auth_path, position).map_err(|_| {
289            io::Error::new(
290                io::ErrorKind::InvalidData,
291                "auth path contained incorrect number of elements",
292            )
293        })
294    } else {
295        Err(io::Error::new(
296            io::ErrorKind::InvalidData,
297            "trailing data found in auth path decoding",
298        ))
299    }
300}

784            let mut tmp = [0u8; 32];

789        fn assert_tree_ser_eq(tree: &CommitmentTree<Node, TESTING_DEPTH>, expected: &str) {
790            // Check that the tree matches its encoding
791            let mut tmp = Vec::new();
792            write_commitment_tree(tree, &mut tmp).unwrap();
793            assert_eq!(hex::encode(&tmp[..]), expected);
794
795            // Check round-trip encoding
796            let decoded: CommitmentTree<Node, TESTING_DEPTH> =
797                read_commitment_tree(&hex::decode(expected).unwrap()[..]).unwrap();
798            tmp.clear();
799            write_commitment_tree(&decoded, &mut tmp).unwrap();
800            assert_eq!(hex::encode(tmp), expected);

803        fn assert_witness_ser_eq(
804            witness: &IncrementalWitness<Node, TESTING_DEPTH>,
805            expected: &str,
806        ) {
807            // Check that the witness matches its encoding
808            let mut tmp = Vec::new();
809            write_incremental_witness(witness, &mut tmp).unwrap();
810            assert_eq!(hex::encode(&tmp[..]), expected);
811
812            // Check round-trip encoding
813            let decoded: IncrementalWitness<Node, TESTING_DEPTH> =
814                read_incremental_witness(&hex::decode(expected).unwrap()[..]).unwrap();
815            tmp.clear();
816            write_incremental_witness(&decoded, &mut tmp).unwrap();
817            assert_eq!(hex::encode(tmp), expected);
818        }

820        let mut tree = CommitmentTree::<Node, TESTING_DEPTH>::empty();
821        assert_eq!(tree.size(), 0);
822
823        let mut witnesses = vec![];
824        let mut last_cmu = None;
825        let mut paths_i = 0;
826        let mut witness_ser_i = 0;
827        for i in 0..16 {
828            let cmu = hex::decode(commitments[i]).unwrap();
829
830            let cmu = Node::from_bytes(cmu[..].try_into().unwrap()).unwrap();
831
832            // Witness here
833            witnesses.push((
834                if tree.is_empty() {
835                    IncrementalWitness::invalid_empty_witness()
836                } else {
837                    IncrementalWitness::from_tree(tree.clone()).unwrap()
838                },
839                last_cmu,
840            ));

856                assert!(witness.append(cmu).is_ok());

869                }

879        }

883        assert!(tree.append(node).is_err());
884        for (witness, _) in witnesses.as_mut_slice() {

886        }
