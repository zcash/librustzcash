15/// A hashable node within a Merkle tree.
16pub trait HashSer {
17    /// Parses a node from the given byte source.
18    fn read<R: Read>(reader: R) -> io::Result<Self>
19    where
20        Self: Sized;
21
22    /// Serializes this node.
23    fn write<W: Write>(&self, writer: W) -> io::Result<()>;
24}
25
26impl HashSer for sapling::Node {
27    fn read<R: Read>(mut reader: R) -> io::Result<Self> {
28        let mut repr = [0u8; 32];
29        reader.read_exact(&mut repr)?;
30        Option::from(Self::from_bytes(repr)).ok_or_else(|| {
31            io::Error::new(
32                io::ErrorKind::InvalidData,
33                "Non-canonical encoding of Jubjub base field value.",
34            )
35        })
36    }
37
38    fn write<W: Write>(&self, mut writer: W) -> io::Result<()> {
39        writer.write_all(&self.to_bytes())
40    }
41}

43impl HashSer for MerkleHashOrchard {
44    fn read<R: Read>(mut reader: R) -> io::Result<Self>
45    where
46        Self: Sized,
47    {
48        let mut repr = [0u8; 32];
49        reader.read_exact(&mut repr)?;
50        <Option<_>>::from(Self::from_bytes(&repr)).ok_or_else(|| {

60    }

63/// Writes a usize value encoded as a u64 in little-endian order. Since usize
64/// is platform-dependent, we consistently represent it as u64 in serialized
65/// formats.
66pub fn write_usize_leu64<W: Write>(mut writer: W, value: usize) -> io::Result<()> {
67    writer.write_u64_le(u64::try_from(value).map_err(|_| {
68        io::Error::new(
69            io::ErrorKind::InvalidData,
70            "usize value was outside the representable range of a u64",
71        )
72    })?)

76/// is platform-dependent, we consistently represent it as u64 in serialized
77/// formats.
78pub fn read_leu64_usize<R: Read>(mut reader: R) -> io::Result<usize> {
79    let mut repr = [0u8; 8];
80    reader.read_exact(&mut repr)?;
81    usize::try_from(u64::from_le_bytes(repr)).map_err(|_| {
82        io::Error::new(
83            io::ErrorKind::InvalidData,
84            "usize could not be decoded from a 64-bit value",
85        )

116pub fn write_nonempty_frontier_v1<H: HashSer, W: Write>(
117    mut writer: W,
118    frontier: &NonEmptyFrontier<H>,
119) -> io::Result<()> {
120    write_position(&mut writer, frontier.position())?;
121    if frontier.position().is_right_child() {
122        // The v1 serialization wrote the sibling of a right-hand leaf as an optional value, rather
123        // than as part of the ommers vector.
124        frontier
125            .ommers()
126            .first()
127            .expect("ommers vector cannot be empty for right-hand nodes")
128            .write(&mut writer)?;
129        Optional::write(&mut writer, Some(frontier.leaf()), |w, n: &H| n.write(w))?;
130        Vector::write(&mut writer, &frontier.ommers()[1..], |w, e| e.write(w))?;
131    } else {
132        frontier.leaf().write(&mut writer)?;
133        Optional::write(&mut writer, None, |w, n: &H| n.write(w))?;
134        Vector::write(&mut writer, frontier.ommers(), |w, e| e.write(w))?;
135    }
136
137    Ok(())
138}

140pub fn read_nonempty_frontier_v1<H: HashSer + Clone, R: Read>(
141    mut reader: R,
142) -> io::Result<NonEmptyFrontier<H>> {
143    let position = read_position(&mut reader)?;
144    let left = H::read(&mut reader)?;
145    let right = Optional::read(&mut reader, H::read)?;
146    let mut ommers = Vector::read(&mut reader, |r| H::read(r))?;
147
148    let leaf = if let Some(right) = right {
149        // if the frontier has a right leaf, then the left leaf is the first ommer
150        ommers.insert(0, left);
151        right
152    } else {
153        left
154    };
155
156    NonEmptyFrontier::from_parts(position, leaf, ommers).map_err(|_err| {
157        io::Error::new(
158            io::ErrorKind::InvalidData,
159            "Parsing resulted in an invalid Merkle frontier",
160        )
161    })
162}

164pub fn write_frontier_v1<H: HashSer, W: Write>(
165    writer: W,
166    frontier: &Frontier<H, 32>,
167) -> io::Result<()> {
168    Optional::write(writer, frontier.value(), write_nonempty_frontier_v1)
169}

171pub fn read_frontier_v1<H: HashSer + Clone, R: Read>(reader: R) -> io::Result<Frontier<H, 32>> {
172    match Optional::read(reader, read_nonempty_frontier_v1)? {
173        None => Ok(Frontier::empty()),
174        Some(f) => Frontier::try_from(f).map_err(|_err| {
175            io::Error::new(
176                io::ErrorKind::InvalidData,
177                "Parsing resulted in an invalid Merkle frontier",
178            )
179        }),
180    }

183/// Reads a legacy `CommitmentTree` from its serialized form.
184pub fn read_commitment_tree<Node: HashSer, R: Read, const DEPTH: u8>(
185    mut reader: R,
186) -> io::Result<CommitmentTree<Node, DEPTH>> {
187    let left = Optional::read(&mut reader, Node::read)?;
188    let right = Optional::read(&mut reader, Node::read)?;
189    let parents = Vector::read(&mut reader, |r| Optional::read(r, Node::read))?;
190
191    CommitmentTree::from_parts(left, right, parents).map_err(|_| {
192        io::Error::new(
193            io::ErrorKind::InvalidData,
194            "parents vector exceeded tree depth",
195        )
196    })
197}

199/// Serializes a legacy `CommitmentTree` as an array of bytes.
200pub fn write_commitment_tree<Node: HashSer, W: Write, const DEPTH: u8>(
201    tree: &CommitmentTree<Node, DEPTH>,
202    mut writer: W,
203) -> io::Result<()> {
204    Optional::write(&mut writer, tree.left().as_ref(), |w, n| n.write(w))?;
205    Optional::write(&mut writer, tree.right().as_ref(), |w, n| n.write(w))?;
206    Vector::write(&mut writer, tree.parents(), |w, e| {
207        Optional::write(w, e.as_ref(), |w, n| n.write(w))
208    })
209}

211/// Reads an `IncrementalWitness` from its serialized form.
212pub fn read_incremental_witness<Node: HashSer, R: Read, const DEPTH: u8>(
213    mut reader: R,
214) -> io::Result<IncrementalWitness<Node, DEPTH>> {
215    let tree = read_commitment_tree(&mut reader)?;
216    let filled = Vector::read(&mut reader, |r| Node::read(r))?;
217    let cursor = Optional::read(&mut reader, read_commitment_tree)?;
218
219    IncrementalWitness::from_parts(tree, filled, cursor).ok_or(io::Error::new(
220        io::ErrorKind::InvalidData,
221        "invalid witness: inconsistency detected between witness parts",
222    ))
223}

225/// Serializes an `IncrementalWitness` as an array of bytes.
226pub fn write_incremental_witness<Node: HashSer, W: Write, const DEPTH: u8>(
227    witness: &IncrementalWitness<Node, DEPTH>,
228    mut writer: W,
229) -> io::Result<()> {
230    write_commitment_tree(witness.tree(), &mut writer)?;
231    Vector::write(&mut writer, witness.filled(), |w, n| n.write(w))?;
232    Optional::write(&mut writer, witness.cursor().as_ref(), |w, t| {
233        write_commitment_tree(t, w)
234    })
235}

237/// Reads a Merkle path from its serialized form.
238pub fn merkle_path_from_slice<Node: HashSer, const DEPTH: u8>(
239    mut witness: &[u8],
240) -> io::Result<MerklePath<Node, DEPTH>> {
241    // Skip the first byte, which should be DEPTH to signify the length of
242    // the following vector of Pedersen hashes.
243    if witness[0] != DEPTH {
244        return Err(io::Error::new(
245            io::ErrorKind::InvalidData,
246            "depth is not as expected",
247        ));
248    }
249    witness = &witness[1..];
250
251    // Begin to construct the authentication path
252    let iter = witness.chunks_exact(33);
253    witness = iter.remainder();
254
255    // The vector works in reverse
256    let auth_path = iter
257        .rev()
258        .map(|bytes| {
259            // Length of inner vector should be the length of a Pedersen hash
260            if bytes[0] == 32 {
261                // Sibling node should be an element of Fr
262                Node::read(&bytes[1..])
263            } else {
264                Err(io::Error::new(
265                    io::ErrorKind::InvalidData,
266                    "length of auth path element is not the expected 32 bytes",
267                ))
268            }
269        })
270        .collect::<io::Result<Vec<_>>>()?;
271    if auth_path.len() != usize::from(DEPTH) {
272        return Err(io::Error::new(
273            io::ErrorKind::InvalidData,
274            "auth path has unexpected length",
275        ));
276    }
277
278    // Read the position from the witness
279    let position = witness.read_u64_le().map(Position::from)?;
280
281    // The witness should be empty now; if it wasn't, the caller would
282    // have provided more information than they should have, indicating
283    // a bug downstream
284    if witness.is_empty() {
285        MerklePath::from_parts(auth_path, position).map_err(|_| {
286            io::Error::new(
287                io::ErrorKind::InvalidData,
288                "auth path contained incorrect number of elements",
289            )
290        })
291    } else {
292        Err(io::Error::new(
293            io::ErrorKind::InvalidData,
294            "trailing data found in auth path decoding",
295        ))
296    }
297}

786        fn assert_tree_ser_eq(tree: &CommitmentTree<Node, TESTING_DEPTH>, expected: &str) {
787            // Check that the tree matches its encoding
788            let mut tmp = Vec::new();
789            write_commitment_tree(tree, &mut tmp).unwrap();
790            assert_eq!(hex::encode(&tmp[..]), expected);
791
792            // Check round-trip encoding
793            let decoded: CommitmentTree<Node, TESTING_DEPTH> =
794                read_commitment_tree(&hex::decode(expected).unwrap()[..]).unwrap();
795            tmp.clear();
796            write_commitment_tree(&decoded, &mut tmp).unwrap();
797            assert_eq!(hex::encode(tmp), expected);
798        }

800        fn assert_witness_ser_eq(
801            witness: &IncrementalWitness<Node, TESTING_DEPTH>,
802            expected: &str,
803        ) {
804            // Check that the witness matches its encoding
805            let mut tmp = Vec::new();
806            write_incremental_witness(witness, &mut tmp).unwrap();
807            assert_eq!(hex::encode(&tmp[..]), expected);
808
809            // Check round-trip encoding
810            let decoded: IncrementalWitness<Node, TESTING_DEPTH> =
811                read_incremental_witness(&hex::decode(expected).unwrap()[..]).unwrap();
812            tmp.clear();
813            write_incremental_witness(&decoded, &mut tmp).unwrap();
814            assert_eq!(hex::encode(tmp), expected);
815        }
816
817        let mut tree = CommitmentTree::<Node, TESTING_DEPTH>::empty();
818        assert_eq!(tree.size(), 0);
819
820        let mut witnesses = vec![];
821        let mut last_cmu = None;
822        let mut paths_i = 0;
823        let mut witness_ser_i = 0;
824        for i in 0..16 {
825            let cmu = hex::decode(commitments[i]).unwrap();
826
827            let cmu = Node::from_bytes(cmu[..].try_into().unwrap()).unwrap();
828
829            // Witness here
830            witnesses.push((
831                if tree.is_empty() {
832                    IncrementalWitness::invalid_empty_witness()
833                } else {
834                    IncrementalWitness::from_tree(tree.clone()).unwrap()
835                },
836                last_cmu,
837            ));
838
839            // Now append a commitment to the tree
840            assert!(tree.append(cmu).is_ok());
841
842            // Size incremented by one.
843            assert_eq!(tree.size(), i + 1);
844
845            // Check tree root consistency

850
851            for (witness, leaf) in witnesses.as_mut_slice() {
852                // Append the same commitment to all the witnesses
853                assert!(witness.append(cmu).is_ok());
854
855                if let Some(leaf) = leaf {

858                        merkle_path_from_slice(&hex::decode(paths[paths_i]).unwrap()).unwrap();
859                    assert_eq!(path, expected);

863                } else {
864                    // The first witness can never form a path
865                    assert!(witness.path().is_none());
866                }
867
868                // Check witness serialization

871

873            }

875            last_cmu = Some(cmu);
876        }
877
878        // Tree should be full now
879        let node = Node::empty_leaf();
880        assert!(tree.append(node).is_err());
881        for (witness, _) in witnesses.as_mut_slice() {
882            assert!(witness.append(node).is_err());
883        }
