13impl OutputNote {

15        mut cs: CS,
16        a_pk: Option<PayingKey>,
17        value: &NoteValue,
18        r: Option<CommitmentRandomness>,
19        phi: &[Boolean],
20        h_sig: &[Boolean],
21        nonce: bool,
22    ) -> Result<Self, SynthesisError>
23    where
24        Scalar: PrimeField,

26    {

28
29        let a_pk = witness_u256(
30            cs.namespace(|| "a_pk"),
31            a_pk.as_ref().map(|a_pk| &a_pk.0[..]),
32        )?;
33
34        let r = witness_u256(cs.namespace(|| "r"), r.as_ref().map(|r| &r.0[..]))?;
35
36        let cm = note_comm(
37            cs.namespace(|| "cm computation"),
38            &a_pk,
39            &value.bits_le(),
40            &rho,
41            &r,
42        )?;
43
44        Ok(OutputNote { cm })
45    }
