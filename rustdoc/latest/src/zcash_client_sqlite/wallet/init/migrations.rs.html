360pub(super) fn verify_network_compatibility<P: consensus::Parameters>(
361    conn: &rusqlite::Connection,
362    params: &P,
363) -> Result<(), WalletMigrationError> {
364    // Ensure that the `ufvk_support` migration has been applied; if it hasn't, we won't be able to
365    // validate that the UFVKs in the wallet correspond to the network type that the wallet is
366    // being migrated for.
367    let has_ufvk = conn
368        .query_row(
369            &format!(
370                "SELECT 1 FROM {} WHERE id = :migration_id",
371                super::MIGRATIONS_TABLE
372            ),
373            named_params![":migration_id": &ufvk_support::MIGRATION_ID.as_bytes()[..]],
374            |row| row.get::<_, bool>(0),
375        )
376        .optional()?
377        == Some(true);
378
379    if has_ufvk {
380        let mut fvks_stmt = conn.prepare("SELECT ufvk FROM accounts")?;
381        let mut rows = fvks_stmt.query([])?;
382        while let Some(row) = rows.next()? {
383            let ufvk_str = row.get::<_, String>(0)?;
384            let (network, _) = Ufvk::decode(&ufvk_str).map_err(|e| {
385                WalletMigrationError::CorruptedData(format!("Unable to parse UFVK: {e}"))
386            })?;
387
388            if network != params.network_type() {
389                let network_name = |n| match n {
390                    consensus::NetworkType::Main => "mainnet",
391                    consensus::NetworkType::Test => "testnet",
392                    consensus::NetworkType::Regtest => "regtest",
393                };
394                return Err(WalletMigrationError::CorruptedData(format!(
395                    "Network type mismatch: account UFVK is for {} but attempting to initialize for {}.",
396                    network_name(network),
397                    network_name(params.network_type())
398                )));
399            }
400        }
401    }
402
403    Ok(())
404}

422    /// Tests that we can migrate from a completely empty wallet database to the target
423    /// migrations.

425        let data_file = NamedTempFile::new().unwrap();
426        let mut db_data = WalletDb::for_path(
427            data_file.path(),
428            Network::TestNetwork,
429            test_clock(),
430            test_rng(),
431        )
432        .unwrap();
433
434        let seed = [0xab; 32];
435        assert_matches!(
436            WalletMigrator::new()
437                .with_seed(Secret::new(seed.to_vec()))
438                .ignore_seed_relevance()
439                .init_or_migrate_to(&mut db_data, migrations),
440            Ok(_)
441        );
442    }

444    #[test]
445    fn migrate_between_releases_without_data() {
446        let data_file = NamedTempFile::new().unwrap();
447        let mut db_data = WalletDb::for_path(
448            data_file.path(),
449            Network::TestNetwork,
450            test_clock(),
451            test_rng(),
452        )
453        .unwrap();
454
455        let seed = [0xab; 32].to_vec();
456
457        let mut prev_state = HashSet::new();
458        let mut ensure_migration_state_changed = |conn: &Connection| {
459            let new_state = conn
460                .prepare_cached("SELECT * FROM schemer_migrations")
461                .unwrap()
462                .query_map([], |row| row.get::<_, [u8; 16]>(0).map(Uuid::from_bytes))
463                .unwrap()
464                .collect::<Result<HashSet<Uuid>, _>>()
465                .unwrap();
466            assert!(prev_state != new_state);
467            prev_state = new_state;
468        };
469
470        let mut prev_leaves: &[Uuid] = &[];
471        for migrations in super::PUBLIC_MIGRATION_STATES {
472            assert_matches!(
473                WalletMigrator::new()
474                    .with_seed(Secret::new(seed.clone()))
475                    .ignore_seed_relevance()
476                    .init_or_migrate_to(&mut db_data, migrations),
477                Ok(_)
478            );
479
480            // If we have any new leaves, ensure the migration state changed. This lets us
481            // represent releases that changed the graph edges without introducing any new
482            // migrations.
483            if migrations.iter().any(|m| !prev_leaves.contains(m)) {
484                ensure_migration_state_changed(&db_data.conn);
485            }
486
487            prev_leaves = *migrations;
488        }
489
490        // Now check that we can migrate from the last public release to the current
491        // migration state in this branch.
492        assert_matches!(
493            WalletMigrator::new()
494                .with_seed(Secret::new(seed))
495                .ignore_seed_relevance()
496                .init_or_migrate(&mut db_data),
497            Ok(_)
498        );
499        // We don't ensure that the migration state changed, because it may not have.
500    }
