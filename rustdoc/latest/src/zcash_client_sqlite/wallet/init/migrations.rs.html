354pub(super) fn verify_network_compatibility<P: consensus::Parameters>(
355    conn: &rusqlite::Connection,
356    params: &P,
357) -> Result<(), WalletMigrationError> {
358    // Ensure that the `ufvk_support` migration has been applied; if it hasn't, we won't be able to
359    // validate that the UFVKs in the wallet correspond to the network type that the wallet is
360    // being migrated for.
361    let has_ufvk = conn
362        .query_row(
363            &format!(
364                "SELECT 1 FROM {} WHERE id = :migration_id",
365                super::MIGRATIONS_TABLE
366            ),
367            named_params![":migration_id": &ufvk_support::MIGRATION_ID.as_bytes()[..]],
368            |row| row.get::<_, bool>(0),
369        )
370        .optional()?
371        == Some(true);
372
373    if has_ufvk {
374        let mut fvks_stmt = conn.prepare("SELECT ufvk FROM accounts")?;
375        let mut rows = fvks_stmt.query([])?;
376        while let Some(row) = rows.next()? {
377            let ufvk_str = row.get::<_, String>(0)?;
378            let (network, _) = Ufvk::decode(&ufvk_str).map_err(|e| {
379                WalletMigrationError::CorruptedData(format!("Unable to parse UFVK: {e}"))
380            })?;
381
382            if network != params.network_type() {
383                let network_name = |n| match n {
384                    consensus::NetworkType::Main => "mainnet",
385                    consensus::NetworkType::Test => "testnet",
386                    consensus::NetworkType::Regtest => "regtest",
387                };
388                return Err(WalletMigrationError::CorruptedData(format!(
389                    "Network type mismatch: account UFVK is for {} but attempting to initialize for {}.",
390                    network_name(network),
391                    network_name(params.network_type())
392                )));
393            }
394        }
395    }
396
397    Ok(())
398}

420        let mut db_data = WalletDb::for_path(
421            data_file.path(),
422            Network::TestNetwork,
423            test_clock(),

429        assert_matches!(

432                .ignore_seed_relevance()
433                .init_or_migrate_to(&mut db_data, migrations),
434            Ok(_)
435        );
436    }

438    #[test]
439    fn migrate_between_releases_without_data() {
440        let data_file = NamedTempFile::new().unwrap();
441        let mut db_data = WalletDb::for_path(
442            data_file.path(),
443            Network::TestNetwork,
444            test_clock(),
445            test_rng(),
446        )
447        .unwrap();
448
449        let seed = [0xab; 32].to_vec();
450
451        let mut prev_state = HashSet::new();
452        let mut ensure_migration_state_changed = |conn: &Connection| {
453            let new_state = conn
454                .prepare_cached("SELECT * FROM schemer_migrations")
455                .unwrap()
456                .query_map([], |row| row.get::<_, [u8; 16]>(0).map(Uuid::from_bytes))
457                .unwrap()
458                .collect::<Result<HashSet<Uuid>, _>>()
459                .unwrap();
460            assert!(prev_state != new_state);
461            prev_state = new_state;
462        };
463
464        let mut prev_leaves: &[Uuid] = &[];
465        for migrations in super::PUBLIC_MIGRATION_STATES {
466            assert_matches!(
467                WalletMigrator::new()
468                    .with_seed(Secret::new(seed.clone()))
469                    .ignore_seed_relevance()
470                    .init_or_migrate_to(&mut db_data, migrations),
471                Ok(_)
472            );
473
474            // If we have any new leaves, ensure the migration state changed. This lets us
475            // represent releases that changed the graph edges without introducing any new
476            // migrations.
477            if migrations.iter().any(|m| !prev_leaves.contains(m)) {
478                ensure_migration_state_changed(&db_data.conn);
479            }
480
481            prev_leaves = *migrations;
482        }
483
484        // Now check that we can migrate from the last public release to the current
485        // migration state in this branch.
486        assert_matches!(
487            WalletMigrator::new()
488                .with_seed(Secret::new(seed))
489                .ignore_seed_relevance()
490                .init_or_migrate(&mut db_data),
491            Ok(_)
492        );
493        // We don't ensure that the migration state changed, because it may not have.
494    }
