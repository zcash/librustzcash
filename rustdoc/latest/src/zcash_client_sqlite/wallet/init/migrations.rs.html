225pub(super) fn verify_network_compatibility<P: consensus::Parameters>(
226    conn: &rusqlite::Connection,
227    params: &P,
228) -> Result<(), WalletMigrationError> {
229    // Ensure that the `ufvk_support` migration has been applied; if it hasn't, we won't be able to
230    // validate that the UFVKs in the wallet correspond to the network type that the wallet is
231    // being migrated for.
232    let has_ufvk = conn
233        .query_row(
234            &format!(
235                "SELECT 1 FROM {} WHERE id = :migration_id",
236                super::MIGRATIONS_TABLE
237            ),
238            named_params![":migration_id": &ufvk_support::MIGRATION_ID.as_bytes()[..]],
239            |row| row.get::<_, bool>(0),
240        )
241        .optional()?
242        == Some(true);
243
244    if has_ufvk {
245        let mut fvks_stmt = conn.prepare("SELECT ufvk FROM accounts")?;
246        let mut rows = fvks_stmt.query([])?;
247        while let Some(row) = rows.next()? {
248            let ufvk_str = row.get::<_, String>(0)?;
249            let (network, _) = Ufvk::decode(&ufvk_str).map_err(|e| {
250                WalletMigrationError::CorruptedData(format!("Unable to parse UFVK: {e}"))
251            })?;
252
253            if network != params.network_type() {
254                let network_name = |n| match n {
255                    consensus::NetworkType::Main => "mainnet",
256                    consensus::NetworkType::Test => "testnet",
257                    consensus::NetworkType::Regtest => "regtest",
258                };
259                return Err(WalletMigrationError::CorruptedData(format!(
260                    "Network type mismatch: account UFVK is for {} but attempting to initialize for {}.",
261                    network_name(network),
262                    network_name(params.network_type())
263                )));
264            }
265        }
266    }
267
268    Ok(())
269}

301    #[test]
302    fn migrate_between_releases_without_data() {
303        let data_file = NamedTempFile::new().unwrap();
304        let mut db_data = WalletDb::for_path(data_file.path(), Network::TestNetwork).unwrap();
305
306        let seed = [0xab; 32].to_vec();
307
308        let mut prev_state = HashSet::new();
309        let mut ensure_migration_state_changed = |conn: &Connection| {
310            let new_state = conn
311                .prepare_cached("SELECT * FROM schemer_migrations")
312                .unwrap()
313                .query_map([], |row| row.get::<_, [u8; 16]>(0).map(Uuid::from_bytes))
314                .unwrap()
315                .collect::<Result<HashSet<Uuid>, _>>()
316                .unwrap();
317            assert!(prev_state != new_state);
318            prev_state = new_state;
319        };
320
321        let mut prev_leaves: &[Uuid] = &[];
322        for migrations in super::PUBLIC_MIGRATION_STATES {
323            assert_matches!(
324                init_wallet_db_internal(
325                    &mut db_data,
326                    Some(Secret::new(seed.clone())),
327                    migrations,
328                    false
329                ),
330                Ok(_)
331            );
332
333            // If we have any new leaves, ensure the migration state changed. This lets us
334            // represent releases that changed the graph edges without introducing any new
335            // migrations.
336            if migrations.iter().any(|m| !prev_leaves.contains(m)) {
337                ensure_migration_state_changed(&db_data.conn);
338            }
339
340            prev_leaves = *migrations;
341        }
342
343        // Now check that we can migrate from the last public release to the current
344        // migration state in this branch.
345        assert_matches!(
346            init_wallet_db_internal(&mut db_data, Some(Secret::new(seed)), &[], false),
347            Ok(_)
348        );
349        // We don't ensure that the migration state changed, because it may not have.
350    }
