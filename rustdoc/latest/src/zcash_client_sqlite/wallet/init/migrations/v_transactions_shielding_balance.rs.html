38            "DROP VIEW v_transactions;
39            CREATE VIEW v_transactions AS
40            WITH
41            notes AS (
42                SELECT sapling_received_notes.account        AS account_id,
43                       transactions.block                    AS block,
44                       transactions.txid                     AS txid,
45                       2                                     AS pool,
46                       sapling_received_notes.value          AS value,
47                       CASE
48                            WHEN sapling_received_notes.is_change THEN 1
49                            ELSE 0
50                       END AS is_change,
51                       CASE
52                            WHEN sapling_received_notes.is_change THEN 0
53                            ELSE 1
54                       END AS received_count,
55                       CASE
56                         WHEN (sapling_received_notes.memo IS NULL OR sapling_received_notes.memo = X'F6')
57                           THEN 0
58                         ELSE 1
59                       END AS memo_present
60                FROM sapling_received_notes
61                JOIN transactions
62                     ON transactions.id_tx = sapling_received_notes.tx
63                UNION
64                SELECT utxos.received_by_account     AS account_id,
65                       utxos.height                  AS block,
66                       utxos.prevout_txid            AS txid,
67                       0                             AS pool,
68                       utxos.value_zat               AS value,
69                       0                             AS is_change,
70                       1                             AS received_count,
71                       0                             AS memo_present
72                FROM utxos
73                UNION
74                SELECT sapling_received_notes.account        AS account_id,
75                       transactions.block                    AS block,
76                       transactions.txid                     AS txid,
77                       2                                     AS pool,
78                       -sapling_received_notes.value         AS value,
79                       0                             AS is_change,
80                       0                             AS received_count,
81                       0                             AS memo_present
82                FROM sapling_received_notes
83                JOIN transactions
84                     ON transactions.id_tx = sapling_received_notes.spent
85                UNION
86                SELECT utxos.received_by_account     AS account_id,
87                       transactions.block            AS block,
88                       transactions.txid             AS txid,
89                       0                             AS pool,
90                       -utxos.value_zat              AS value,
91                       0                             AS is_change,
92                       0                             AS received_count,
93                       0                             AS memo_present
94                FROM utxos
95                JOIN transactions
96                     ON transactions.id_tx = utxos.spent_in_tx
97            ),
98            sent_note_counts AS (
99                SELECT sent_notes.from_account AS account_id,
100                       transactions.txid       AS txid,
101                       COUNT(DISTINCT sent_notes.id_note) as sent_notes,
102                       SUM(
103                         CASE
104                           WHEN (sent_notes.memo IS NULL OR sent_notes.memo = X'F6' OR sapling_received_notes.tx IS NOT NULL)
105                             THEN 0
106                           ELSE 1
107                         END
108                       ) AS memo_count
109                FROM sent_notes
110                JOIN transactions
111                     ON transactions.id_tx = sent_notes.tx
112                LEFT JOIN sapling_received_notes
113                          ON (sent_notes.tx, sent_notes.output_pool, sent_notes.output_index) =
114                             (sapling_received_notes.tx, 2, sapling_received_notes.output_index)
115                WHERE COALESCE(sapling_received_notes.is_change, 0) = 0
116                GROUP BY account_id, txid
117            ),
118            blocks_max_height AS (
119                SELECT MAX(blocks.height) as max_height FROM blocks
120            )
121            SELECT notes.account_id                  AS account_id,
122                   notes.block                       AS mined_height,
123                   notes.txid                        AS txid,
124                   transactions.tx_index             AS tx_index,
125                   transactions.expiry_height        AS expiry_height,
126                   transactions.raw                  AS raw,
127                   SUM(notes.value)                  AS account_balance_delta,
128                   transactions.fee                  AS fee_paid,
129                   SUM(notes.is_change) > 0          AS has_change,
130                   MAX(COALESCE(sent_note_counts.sent_notes, 0))  AS sent_note_count,
131                   SUM(notes.received_count)         AS received_note_count,
132                   SUM(notes.memo_present) + MAX(COALESCE(sent_note_counts.memo_count, 0)) AS memo_count,
133                   blocks.time                       AS block_time,
134                   (
135                        blocks.height IS NULL
136                        AND transactions.expiry_height BETWEEN 1 AND blocks_max_height.max_height
137                   ) AS expired_unmined
138            FROM notes
139            LEFT JOIN transactions
140                 ON notes.txid = transactions.txid
141            JOIN blocks_max_height
142            LEFT JOIN blocks ON blocks.height = notes.block
143            LEFT JOIN sent_note_counts
144                      ON sent_note_counts.account_id = notes.account_id
145                      AND sent_note_counts.txid = notes.txid
146            GROUP BY notes.account_id, notes.txid;"
147        )?;
148
149        Ok(())
150    }
