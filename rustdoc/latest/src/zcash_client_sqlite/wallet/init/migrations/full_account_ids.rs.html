18/// The migration that switched from presumed seed-derived account IDs to supporting
19/// HD accounts and all sorts of imported keys.
20pub(super) const MIGRATION_ID: Uuid = Uuid::from_u128(0x6d02ec76_8720_4cc6_b646_c4e2ce69221c);
21
22pub(crate) struct Migration<P: consensus::Parameters> {
23    pub(super) seed: Option<Rc<SecretVec<u8>>>,
24    pub(super) params: P,
25}
26
27const DEPENDENCIES: &[Uuid] = &[
28    receiving_key_scopes::MIGRATION_ID,
29    add_account_birthdays::MIGRATION_ID,
30    v_transactions_note_uniqueness::MIGRATION_ID,
31    wallet_summaries::MIGRATION_ID,
32];
33
34impl<P: consensus::Parameters> schemerz::Migration<Uuid> for Migration<P> {
35    fn id(&self) -> Uuid {
36        MIGRATION_ID
37    }
38
39    fn dependencies(&self) -> HashSet<Uuid> {
40        DEPENDENCIES.iter().copied().collect()
41    }
42
43    fn description(&self) -> &'static str {
44        "Replaces the `account` column in the `accounts` table with columns to support all kinds of
45         account and key types."
46    }
47}
48
49impl<P: consensus::Parameters> RusqliteMigration for Migration<P> {
50    type Error = WalletMigrationError;
51
52    fn up(&self, transaction: &Transaction) -> Result<(), WalletMigrationError> {
53        let account_kind_derived = account_kind_code(&AccountSource::Derived {
54            derivation: Zip32Derivation::new(
55                SeedFingerprint::from_bytes([0; 32]),
56                zip32::AccountId::ZERO,
57            ),
58            key_source: None,
59        });
60        let account_kind_imported = account_kind_code(&AccountSource::Imported {
61            // the purpose here is irrelevant; we just use it to get the correct code
62            // for the account kind
63            purpose: AccountPurpose::ViewOnly,
64            key_source: None,
65        });
66        transaction.execute_batch(&format!(
67            r#"
68            CREATE TABLE accounts_new (
69                id INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
70                account_kind INTEGER NOT NULL DEFAULT {account_kind_derived},
71                hd_seed_fingerprint BLOB,
72                hd_account_index INTEGER,
73                ufvk TEXT,
74                uivk TEXT NOT NULL,
75                orchard_fvk_item_cache BLOB,
76                sapling_fvk_item_cache BLOB,
77                p2pkh_fvk_item_cache BLOB,
78                birthday_height INTEGER NOT NULL,
79                birthday_sapling_tree_size INTEGER,
80                birthday_orchard_tree_size INTEGER,
81                recover_until_height INTEGER,
82                CHECK (
83                  (
84                    account_kind = {account_kind_derived}
85                    AND hd_seed_fingerprint IS NOT NULL
86                    AND hd_account_index IS NOT NULL
87                    AND ufvk IS NOT NULL
88                  )
89                  OR
90                  (
91                    account_kind = {account_kind_imported}
92                    AND hd_seed_fingerprint IS NULL
93                    AND hd_account_index IS NULL
94                  )
95                )
96            );
97            CREATE UNIQUE INDEX hd_account ON accounts_new (hd_seed_fingerprint, hd_account_index);
98            CREATE UNIQUE INDEX accounts_uivk ON accounts_new (uivk);
99            CREATE UNIQUE INDEX accounts_ufvk ON accounts_new (ufvk);
100            "#
101        ))?;
102
103        // We require the seed *if* there are existing accounts in the table.
104        if transaction.query_row("SELECT COUNT(*) FROM accounts", [], |row| {
105            Ok(row.get::<_, u32>(0)? > 0)
106        })? {
107            if let Some(seed) = &self.seed {
108                let seed_id = SeedFingerprint::from_seed(seed.expose_secret())
109                    .expect("Seed is between 32 and 252 bytes in length.");
110
111                // We track whether we have determined seed relevance or not, in order to
112                // correctly report errors when checking the seed against an account:
113                //
114                // - If we encounter an error with the first account, we can assert that
115                //   the seed is not relevant to the wallet by assuming that:
116                //   - All accounts are from the same seed (which is historically the only
117                //     use case that this migration supported), and
118                //   - All accounts in the wallet must have been able to derive their USKs
119                //     (in order to derive UIVKs).
120                //
121                // - Once the seed has been determined to be relevant (because it matched
122                //   the first account), any subsequent account derivation failure is
123                //   proving wrong our second assumption above, and we report this as
124                //   corrupted data.
125                let mut seed_is_relevant = false;
126
127                let mut q = transaction.prepare("SELECT * FROM accounts")?;
128                let mut rows = q.query([])?;
129                while let Some(row) = rows.next()? {
130                    let account_index: u32 = row.get("account")?;
131                    let birthday_height: u32 = row.get("birthday_height")?;
132                    let recover_until_height: Option<u32> = row.get("recover_until_height")?;
133
134                    // Although 'id' is an AUTOINCREMENT column, we'll set it explicitly to match the old account value
135                    // strictly as a matter of convenience to make this migration script easier,
136                    // specifically around updating tables with foreign keys to this one.
137                    let account_id = account_index;
138
139                    // Verify that the UFVK is as expected by re-deriving it.
140                    let ufvk: String = row.get("ufvk")?;
141                    let ufvk_parsed = UnifiedFullViewingKey::decode(&self.params, &ufvk)
142                        .map_err(|_| WalletMigrationError::CorruptedData("Bad UFVK".to_string()))?;
143                    let usk = UnifiedSpendingKey::from_seed(
144                        &self.params,
145                        seed.expose_secret(),
146                        zip32::AccountId::try_from(account_index).map_err(|_| {
147                            WalletMigrationError::CorruptedData("Bad account index".to_string())
148                        })?,
149                    )
150                    .map_err(|_| {
151                        if seed_is_relevant {
152                            WalletMigrationError::CorruptedData(
153                                "Unable to derive spending key from seed.".to_string(),
154                            )
155                        } else {
156                            WalletMigrationError::SeedNotRelevant
157                        }
158                    })?;
159                    let expected_ufvk = usk.to_unified_full_viewing_key();
160                    if ufvk != expected_ufvk.encode(&self.params) {
161                        return Err(if seed_is_relevant {
162                            WalletMigrationError::CorruptedData(
163                                "UFVK does not match expected value.".to_string(),
164                            )
165                        } else {
166                            WalletMigrationError::SeedNotRelevant
167                        });
168                    }
169
170                    // We made it past one derived account, so the seed must be relevant.
171                    seed_is_relevant = true;
172
173                    let uivk = ufvk_parsed
174                        .to_unified_incoming_viewing_key()
175                        .encode(&self.params);
176
177                    #[cfg(feature = "orchard")]
178                    let orchard_item = ufvk_parsed.orchard().map(|k| k.to_bytes());
179                    #[cfg(not(feature = "orchard"))]
180                    let orchard_item: Option<Vec<u8>> = None;
181
182                    let sapling_item = ufvk_parsed.sapling().map(|k| k.to_bytes());
183
184                    #[cfg(feature = "transparent-inputs")]
185                    let transparent_item = ufvk_parsed.transparent().map(|k| k.serialize());
186                    #[cfg(not(feature = "transparent-inputs"))]
187                    let transparent_item: Option<Vec<u8>> = None;
188
189                    // Get the tree sizes for the birthday height from the blocks table, if
190                    // available.
191                    let (birthday_sapling_tree_size, birthday_orchard_tree_size) = transaction
192                        .query_row(
193                            "SELECT sapling_commitment_tree_size - sapling_output_count,
194                                    orchard_commitment_tree_size - orchard_action_count
195                             FROM blocks
196                             WHERE height = :birthday_height",
197                            named_params![":birthday_height": birthday_height],
198                            |row| {
199                                Ok(row
200                                    .get::<_, Option<u32>>(0)?
201                                    .zip(row.get::<_, Option<u32>>(1)?))
202                            },
203                        )
204                        .optional()?
205                        .flatten()
206                        .map_or((None, None), |(s, o)| (Some(s), Some(o)));
207
208                    transaction.execute(
209                        r#"
210                        INSERT INTO accounts_new (
211                            id, account_kind, hd_seed_fingerprint, hd_account_index,
212                            ufvk, uivk,
213                            orchard_fvk_item_cache, sapling_fvk_item_cache, p2pkh_fvk_item_cache,
214                            birthday_height, birthday_sapling_tree_size, birthday_orchard_tree_size,
215                            recover_until_height
216                        )
217                        VALUES (
218                            :account_id, :account_kind, :seed_id, :account_index,
219                            :ufvk, :uivk,
220                            :orchard_fvk_item_cache, :sapling_fvk_item_cache, :p2pkh_fvk_item_cache,
221                            :birthday_height, :birthday_sapling_tree_size, :birthday_orchard_tree_size,
222                            :recover_until_height
223                        );
224                        "#,
225                        named_params![
226                            ":account_id": account_id,
227                            ":account_kind": account_kind_derived,
228                            ":seed_id": seed_id.to_bytes(),
229                            ":account_index": account_index,
230                            ":ufvk": ufvk,
231                            ":uivk": uivk,
232                            ":orchard_fvk_item_cache": orchard_item,
233                            ":sapling_fvk_item_cache": sapling_item,
234                            ":p2pkh_fvk_item_cache": transparent_item,
235                            ":birthday_height": birthday_height,
236                            ":birthday_sapling_tree_size": birthday_sapling_tree_size,
237                            ":birthday_orchard_tree_size": birthday_orchard_tree_size,
238                            ":recover_until_height": recover_until_height,
239                        ],
240                    )?;
241                }
242            } else {
243                return Err(WalletMigrationError::SeedRequired);
244            }
245        }
246
247        transaction.execute_batch(r#"
248            PRAGMA legacy_alter_table = ON;
249
250            DROP TABLE accounts;
251            ALTER TABLE accounts_new RENAME TO accounts;
252
253            -- Migrate addresses table
254            CREATE TABLE addresses_new (
255                account_id INTEGER NOT NULL,
256                diversifier_index_be BLOB NOT NULL,
257                address TEXT NOT NULL,
258                cached_transparent_receiver_address TEXT,
259                FOREIGN KEY (account_id) REFERENCES accounts(id),
260                CONSTRAINT diversification UNIQUE (account_id, diversifier_index_be)
261            );
262            CREATE INDEX "addresses_accounts" ON "addresses_new" (
263                "account_id" ASC
264            );
265            INSERT INTO addresses_new (account_id, diversifier_index_be, address, cached_transparent_receiver_address)
266            SELECT account, diversifier_index_be, address, cached_transparent_receiver_address
267            FROM addresses;
268
269            DROP TABLE addresses;
270            ALTER TABLE addresses_new RENAME TO addresses;
271
272            -- Migrate sapling_received_notes table
273            CREATE TABLE sapling_received_notes_new (
274                id INTEGER PRIMARY KEY,
275                tx INTEGER NOT NULL,
276                output_index INTEGER NOT NULL,
277                account_id INTEGER NOT NULL,
278                diversifier BLOB NOT NULL,
279                value INTEGER NOT NULL,
280                rcm BLOB NOT NULL,
281                nf BLOB UNIQUE,
282                is_change INTEGER NOT NULL,
283                memo BLOB,
284                commitment_tree_position INTEGER,
285                recipient_key_scope INTEGER,
286                FOREIGN KEY (tx) REFERENCES transactions(id_tx),
287                FOREIGN KEY (account_id) REFERENCES accounts(id),
288                CONSTRAINT tx_output UNIQUE (tx, output_index)
289            );
290            CREATE INDEX "sapling_received_notes_account" ON "sapling_received_notes_new" (
291                "account_id" ASC
292            );
293            CREATE INDEX "sapling_received_notes_tx" ON "sapling_received_notes_new" (
294                "tx" ASC
295            );
296
297            -- Replace the `spent` column in `sapling_received_notes` with a junction table between
298            -- received notes and the transactions that spend them. This is necessary as otherwise
299            -- we cannot compute the correct value of transactions that expire unmined.
300            CREATE TABLE sapling_received_note_spends (
301                sapling_received_note_id INTEGER NOT NULL,
302                transaction_id INTEGER NOT NULL,
303                FOREIGN KEY (sapling_received_note_id)
304                    REFERENCES sapling_received_notes(id)
305                    ON DELETE CASCADE,
306                FOREIGN KEY (transaction_id)
307                    -- We do not delete transactions, so this does not cascade
308                    REFERENCES transactions(id_tx),
309                UNIQUE (sapling_received_note_id, transaction_id)
310            );
311
312            INSERT INTO sapling_received_note_spends (sapling_received_note_id, transaction_id)
313            SELECT id_note, spent
314            FROM sapling_received_notes
315            WHERE spent IS NOT NULL;
316
317            INSERT INTO sapling_received_notes_new (
318                id, tx, output_index, account_id,
319                diversifier, value, rcm, nf, is_change, memo, commitment_tree_position,
320                recipient_key_scope
321            )
322            SELECT
323                id_note, tx, output_index, account,
324                diversifier, value, rcm, nf, is_change, memo, commitment_tree_position,
325                recipient_key_scope
326            FROM sapling_received_notes;
327
328            DROP TABLE sapling_received_notes;
329            ALTER TABLE sapling_received_notes_new RENAME TO sapling_received_notes;
330
331            -- Migrate sent_notes table
332            CREATE TABLE sent_notes_new (
333                id INTEGER PRIMARY KEY,
334                tx INTEGER NOT NULL,
335                output_pool INTEGER NOT NULL,
336                output_index INTEGER NOT NULL,
337                from_account_id INTEGER NOT NULL,
338                to_address TEXT,
339                to_account_id INTEGER,
340                value INTEGER NOT NULL,
341                memo BLOB,
342                FOREIGN KEY (tx) REFERENCES transactions(id_tx),
343                FOREIGN KEY (from_account_id) REFERENCES accounts(id),
344                FOREIGN KEY (to_account_id) REFERENCES accounts(id),
345                CONSTRAINT tx_output UNIQUE (tx, output_pool, output_index),
346                CONSTRAINT note_recipient CHECK (
347                    (to_address IS NOT NULL) OR (to_account_id IS NOT NULL)
348                )
349            );
350            CREATE INDEX sent_notes_tx ON sent_notes_new (tx);
351            CREATE INDEX sent_notes_from_account ON sent_notes_new (from_account_id);
352            CREATE INDEX sent_notes_to_account ON sent_notes_new (to_account_id);
353            INSERT INTO sent_notes_new (id, tx, output_pool, output_index, from_account_id, to_address, to_account_id, value, memo)
354            SELECT id_note, tx, output_pool, output_index, from_account, to_address, to_account, value, memo
355            FROM sent_notes;
356
357            DROP TABLE sent_notes;
358            ALTER TABLE sent_notes_new RENAME TO sent_notes;
359
360            -- No one uses this table any more, and it contains a reference to columns we renamed.
361            DROP TABLE sapling_witnesses;
362
363            -- Migrate utxos table
364            CREATE TABLE utxos_new (
365                id INTEGER PRIMARY KEY,
366                received_by_account_id INTEGER NOT NULL,
367                address TEXT NOT NULL,
368                prevout_txid BLOB NOT NULL,
369                prevout_idx INTEGER NOT NULL,
370                script BLOB NOT NULL,
371                value_zat INTEGER NOT NULL,
372                height INTEGER NOT NULL,
373                FOREIGN KEY (received_by_account_id) REFERENCES accounts(id),
374                CONSTRAINT tx_outpoint UNIQUE (prevout_txid, prevout_idx)
375            );
376            CREATE INDEX utxos_received_by_account ON utxos_new (received_by_account_id);
377
378            INSERT INTO utxos_new (id, received_by_account_id, address, prevout_txid, prevout_idx, script, value_zat, height)
379            SELECT id_utxo, received_by_account, address, prevout_txid, prevout_idx, script, value_zat, height
380            FROM utxos;
381
382            -- Replace the `spent_in_tx` column in `utxos` with a junction table between received
383            -- outputs and the transactions that spend them. This is necessary as otherwise we
384            -- cannot compute the correct value of transactions that expire unmined.
385            CREATE TABLE transparent_received_output_spends (
386                transparent_received_output_id INTEGER NOT NULL,
387                transaction_id INTEGER NOT NULL,
388                FOREIGN KEY (transparent_received_output_id)
389                    REFERENCES utxos(id)
390                    ON DELETE CASCADE,
391                FOREIGN KEY (transaction_id)
392                    -- We do not delete transactions, so this does not cascade
393                    REFERENCES transactions(id_tx),
394                UNIQUE (transparent_received_output_id, transaction_id)
395            );
396
397            INSERT INTO transparent_received_output_spends (transparent_received_output_id, transaction_id)
398            SELECT id_utxo, spent_in_tx
399            FROM utxos
400            WHERE spent_in_tx IS NOT NULL;
401
402            DROP TABLE utxos;
403            ALTER TABLE utxos_new RENAME TO utxos;
404
405            PRAGMA legacy_alter_table = OFF;
406        "#)?;
407
408        // Rewrite v_transactions view
409        transaction.execute_batch("
410            DROP VIEW v_transactions;
411            CREATE VIEW v_transactions AS
412            WITH
413            notes AS (
414                SELECT sapling_received_notes.id             AS id,
415                       sapling_received_notes.account_id     AS account_id,
416                       transactions.block                    AS block,
417                       transactions.txid                     AS txid,
418                       2                                     AS pool,
419                       sapling_received_notes.value          AS value,
420                       CASE
421                            WHEN sapling_received_notes.is_change THEN 1
422                            ELSE 0
423                       END AS is_change,
424                       CASE
425                            WHEN sapling_received_notes.is_change THEN 0
426                            ELSE 1
427                       END AS received_count,
428                       CASE
429                         WHEN (sapling_received_notes.memo IS NULL OR sapling_received_notes.memo = X'F6')
430                           THEN 0
431                         ELSE 1
432                       END AS memo_present
433                FROM sapling_received_notes
434                JOIN transactions
435                     ON transactions.id_tx = sapling_received_notes.tx
436                UNION
437                SELECT utxos.id                      AS id,
438                       utxos.received_by_account_id  AS account_id,
439                       utxos.height                  AS block,
440                       utxos.prevout_txid            AS txid,
441                       0                             AS pool,
442                       utxos.value_zat               AS value,
443                       0                             AS is_change,
444                       1                             AS received_count,
445                       0                             AS memo_present
446                FROM utxos
447                UNION
448                SELECT sapling_received_notes.id             AS id,
449                       sapling_received_notes.account_id     AS account_id,
450                       transactions.block                    AS block,
451                       transactions.txid                     AS txid,
452                       2                                     AS pool,
453                       -sapling_received_notes.value         AS value,
454                       0                             AS is_change,
455                       0                             AS received_count,
456                       0                             AS memo_present
457                FROM sapling_received_notes
458                JOIN sapling_received_note_spends
459                     ON sapling_received_note_id = sapling_received_notes.id
460                JOIN transactions
461                     ON transactions.id_tx = sapling_received_note_spends.transaction_id
462                UNION
463                SELECT utxos.id                      AS id,
464                       utxos.received_by_account_id  AS account_id,
465                       transactions.block            AS block,
466                       transactions.txid             AS txid,
467                       0                             AS pool,
468                       -utxos.value_zat              AS value,
469                       0                             AS is_change,
470                       0                             AS received_count,
471                       0                             AS memo_present
472                FROM utxos
473            JOIN transparent_received_output_spends txo_spends
474                 ON txo_spends.transparent_received_output_id = txos.id
475            JOIN transactions
476                 ON transactions.id_tx = txo_spends.transaction_id
477            ),
478            sent_note_counts AS (
479                SELECT sent_notes.from_account_id AS account_id,
480                       transactions.txid       AS txid,
481                       COUNT(DISTINCT sent_notes.id) as sent_notes,
482                       SUM(
483                         CASE
484                           WHEN (sent_notes.memo IS NULL OR sent_notes.memo = X'F6' OR sapling_received_notes.tx IS NOT NULL)
485                             THEN 0
486                           ELSE 1
487                         END
488                       ) AS memo_count
489                FROM sent_notes
490                JOIN transactions
491                     ON transactions.id_tx = sent_notes.tx
492                LEFT JOIN sapling_received_notes
493                          ON (sent_notes.tx, sent_notes.output_pool, sent_notes.output_index) =
494                             (sapling_received_notes.tx, 2, sapling_received_notes.output_index)
495                WHERE COALESCE(sapling_received_notes.is_change, 0) = 0
496                GROUP BY account_id, txid
497            ),
498            blocks_max_height AS (
499                SELECT MAX(blocks.height) as max_height FROM blocks
500            )
501            SELECT notes.account_id                  AS account_id,
502                   notes.block                       AS mined_height,
503                   notes.txid                        AS txid,
504                   transactions.tx_index             AS tx_index,
505                   transactions.expiry_height        AS expiry_height,
506                   transactions.raw                  AS raw,
507                   SUM(notes.value)                  AS account_balance_delta,
508                   transactions.fee                  AS fee_paid,
509                   SUM(notes.is_change) > 0          AS has_change,
510                   MAX(COALESCE(sent_note_counts.sent_notes, 0))  AS sent_note_count,
511                   SUM(notes.received_count)         AS received_note_count,
512                   SUM(notes.memo_present) + MAX(COALESCE(sent_note_counts.memo_count, 0)) AS memo_count,
513                   blocks.time                       AS block_time,
514                   (
515                        blocks.height IS NULL
516                        AND transactions.expiry_height BETWEEN 1 AND blocks_max_height.max_height
517                   ) AS expired_unmined
518            FROM notes
519            LEFT JOIN transactions
520                 ON notes.txid = transactions.txid
521            JOIN blocks_max_height
522            LEFT JOIN blocks ON blocks.height = notes.block
523            LEFT JOIN sent_note_counts
524                      ON sent_note_counts.account_id = notes.account_id
525                      AND sent_note_counts.txid = notes.txid
526            GROUP BY notes.account_id, notes.txid;
527
528            DROP VIEW v_tx_outputs;
529            CREATE VIEW v_tx_outputs AS
530            SELECT transactions.txid                   AS txid,
531                   2                                   AS output_pool,
532                   sapling_received_notes.output_index AS output_index,
533                   sent_notes.from_account_id          AS from_account_id,
534                   sapling_received_notes.account_id   AS to_account_id,
535                   NULL                                AS to_address,
536                   sapling_received_notes.value        AS value,
537                   sapling_received_notes.is_change    AS is_change,
538                   sapling_received_notes.memo         AS memo
539            FROM sapling_received_notes
540            JOIN transactions
541                 ON transactions.id_tx = sapling_received_notes.tx
542            LEFT JOIN sent_notes
543                      ON (sent_notes.tx, sent_notes.output_pool, sent_notes.output_index) =
544                         (sapling_received_notes.tx, 2, sent_notes.output_index)
545            UNION
546            SELECT utxos.prevout_txid           AS txid,
547                   0                            AS output_pool,
548                   utxos.prevout_idx            AS output_index,
549                   NULL                         AS from_account_id,
550                   utxos.received_by_account_id AS to_account_id,
551                   utxos.address                AS to_address,
552                   utxos.value_zat              AS value,
553                   0                            AS is_change,
554                   NULL                         AS memo
555            FROM utxos
556            UNION
557            SELECT transactions.txid                 AS txid,
558                   sent_notes.output_pool            AS output_pool,
559                   sent_notes.output_index           AS output_index,
560                   sent_notes.from_account_id        AS from_account_id,
561                   sapling_received_notes.account_id AS to_account_id,
562                   sent_notes.to_address             AS to_address,
563                   sent_notes.value                  AS value,
564                   0                                 AS is_change,
565                   sent_notes.memo                   AS memo
566            FROM sent_notes
567            JOIN transactions
568                 ON transactions.id_tx = sent_notes.tx
569            LEFT JOIN sapling_received_notes
570                      ON (sent_notes.tx, sent_notes.output_pool, sent_notes.output_index) =
571                         (sapling_received_notes.tx, 2, sapling_received_notes.output_index)
572            WHERE COALESCE(sapling_received_notes.is_change, 0) = 0;
573        ")?;
574
575        Ok(())
576    }

578    fn down(&self, _transaction: &Transaction) -> Result<(), WalletMigrationError> {
579        Err(WalletMigrationError::CannotRevert(MIGRATION_ID))
580    }
