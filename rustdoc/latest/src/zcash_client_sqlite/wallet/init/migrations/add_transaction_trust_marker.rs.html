1//! Adds support for marking transactions as explicitly trusted for the purpose of satisfying the
2//! ZIP 315 confirmations policy
3use std::collections::HashSet;
4
5use schemerz_rusqlite::RusqliteMigration;
6use uuid::Uuid;
7
8use super::{fix_v_transactions_expired_unmined, tx_observation_height};
9use crate::wallet::init::WalletMigrationError;
10
11pub(super) const MIGRATION_ID: Uuid = Uuid::from_u128(0x4e68277f_6269_467e_9437_f3853cc4a41f);
12
13const DEPENDENCIES: &[Uuid] = &[
14    fix_v_transactions_expired_unmined::MIGRATION_ID,
15    tx_observation_height::MIGRATION_ID,
16];

18pub(super) struct Migration;
19
20impl schemerz::Migration<Uuid> for Migration {

23    }
24

31    }

34impl RusqliteMigration for Migration {
35    type Error = WalletMigrationError;
36
37    fn up(&self, transaction: &rusqlite::Transaction) -> Result<(), WalletMigrationError> {
38        transaction.execute_batch(
39            "ALTER TABLE transactions ADD COLUMN trust_status INTEGER;
40
41            DROP VIEW v_transactions;
42            CREATE VIEW v_transactions AS
43            WITH
44            notes AS (
45                -- Outputs received in this transaction
46                SELECT ro.account_id              AS account_id,
47                       transactions.mined_height  AS mined_height,
48                       transactions.txid          AS txid,
49                       ro.pool                    AS pool,
50                       id_within_pool_table,
51                       ro.value                   AS value,
52                       ro.value                   AS received_value,
53                       0                          AS spent_value,
54                       0                          AS spent_note_count,
55                       CASE
56                            WHEN ro.is_change THEN 1
57                            ELSE 0
58                       END AS change_note_count,
59                       CASE
60                            WHEN ro.is_change THEN 0
61                            ELSE 1
62                       END AS received_count,
63                       CASE
64                         WHEN (ro.memo IS NULL OR ro.memo = X'F6')
65                           THEN 0
66                         ELSE 1
67                       END AS memo_present,
68                       -- The wallet cannot receive transparent outputs in shielding transactions.
69                       CASE
70                         WHEN ro.pool = 0
71                           THEN 1
72                         ELSE 0
73                       END AS does_not_match_shielding
74                FROM v_received_outputs ro
75                JOIN transactions
76                     ON transactions.id_tx = ro.transaction_id
77                UNION
78                -- Outputs spent in this transaction
79                SELECT ro.account_id              AS account_id,
80                       transactions.mined_height  AS mined_height,
81                       transactions.txid          AS txid,
82                       ro.pool                    AS pool,
83                       id_within_pool_table,
84                       -ro.value                  AS value,
85                       0                          AS received_value,
86                       ro.value                   AS spent_value,
87                       1                          AS spent_note_count,
88                       0                          AS change_note_count,
89                       0                          AS received_count,
90                       0                          AS memo_present,
91                       -- The wallet cannot spend shielded outputs in shielding transactions.
92                       CASE
93                         WHEN ro.pool != 0
94                           THEN 1
95                         ELSE 0
96                       END AS does_not_match_shielding
97                FROM v_received_outputs ro
98                JOIN v_received_output_spends ros
99                     ON ros.pool = ro.pool
100                     AND ros.received_output_id = ro.id_within_pool_table
101                JOIN transactions
102                     ON transactions.id_tx = ros.transaction_id
103            ),
104            -- Obtain a count of the notes that the wallet created in each transaction,
105            -- not counting change notes.
106            sent_note_counts AS (
107                SELECT sent_notes.from_account_id     AS account_id,
108                       transactions.txid              AS txid,
109                       COUNT(DISTINCT sent_notes.id)  AS sent_notes,
110                       SUM(
111                         CASE
112                           WHEN (sent_notes.memo IS NULL OR sent_notes.memo = X'F6' OR ro.transaction_id IS NOT NULL)
113                             THEN 0
114                           ELSE 1
115                         END
116                       ) AS memo_count
117                FROM sent_notes
118                JOIN transactions
119                     ON transactions.id_tx = sent_notes.tx
120                LEFT JOIN v_received_outputs ro
121                     ON sent_notes.id = ro.sent_note_id
122                WHERE COALESCE(ro.is_change, 0) = 0
123                GROUP BY account_id, txid
124            ),
125            blocks_max_height AS (
126                SELECT MAX(blocks.height) AS max_height FROM blocks
127            )
128            SELECT accounts.uuid                AS account_uuid,
129                   notes.mined_height           AS mined_height,
130                   notes.txid                   AS txid,
131                   transactions.tx_index        AS tx_index,
132                   transactions.expiry_height   AS expiry_height,
133                   transactions.raw             AS raw,
134                   SUM(notes.value)             AS account_balance_delta,
135                   SUM(notes.spent_value)       AS total_spent,
136                   SUM(notes.received_value)    AS total_received,
137                   transactions.fee             AS fee_paid,
138                   SUM(notes.change_note_count) > 0  AS has_change,
139                   MAX(COALESCE(sent_note_counts.sent_notes, 0))  AS sent_note_count,
140                   SUM(notes.received_count)         AS received_note_count,
141                   SUM(notes.memo_present) + MAX(COALESCE(sent_note_counts.memo_count, 0)) AS memo_count,
142                   blocks.time                       AS block_time,
143                   (
144                        notes.mined_height IS NULL
145                        AND transactions.expiry_height BETWEEN 1 AND blocks_max_height.max_height
146                   ) AS expired_unmined,
147                   SUM(notes.spent_note_count) AS spent_note_count,
148                   (
149                        -- All of the wallet-spent and wallet-received notes are consistent with a
150                        -- shielding transaction.
151                        SUM(notes.does_not_match_shielding) = 0
152                        -- The transaction contains at least one wallet-spent output.
153                        AND SUM(notes.spent_note_count) > 0
154                        -- The transaction contains at least one wallet-received note.
155                        AND (SUM(notes.received_count) + SUM(notes.change_note_count)) > 0
156                        -- We do not know about any external outputs of the transaction.
157                        AND MAX(COALESCE(sent_note_counts.sent_notes, 0)) = 0
158                   ) AS is_shielding,
159                   transactions.trust_status
160            FROM notes
161            LEFT JOIN accounts ON accounts.id = notes.account_id
162            LEFT JOIN transactions
163                 ON notes.txid = transactions.txid
164            LEFT JOIN blocks_max_height
165            LEFT JOIN blocks ON blocks.height = notes.mined_height
166            LEFT JOIN sent_note_counts
167                 ON sent_note_counts.account_id = notes.account_id
168                 AND sent_note_counts.txid = notes.txid
169            GROUP BY notes.account_id, notes.txid"
170        )?;
171        Ok(())
172    }
173
174    fn down(&self, _transaction: &rusqlite::Transaction) -> Result<(), WalletMigrationError> {
175        Err(WalletMigrationError::CannotRevert(MIGRATION_ID))
176    }
177}
