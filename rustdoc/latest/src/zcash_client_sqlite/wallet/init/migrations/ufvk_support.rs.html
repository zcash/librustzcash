53impl<P: consensus::Parameters> RusqliteMigration for Migration<P> {
54    type Error = WalletMigrationError;
55
56    fn up(&self, transaction: &rusqlite::Transaction) -> Result<(), WalletMigrationError> {
57        //
58        // Update the accounts table to store ufvks rather than extfvks
59        //
60
61        transaction.execute_batch(
62            "CREATE TABLE accounts_new (
63                account INTEGER PRIMARY KEY,
64                ufvk TEXT NOT NULL,
65                address TEXT,
66                transparent_address TEXT
67            );",
68        )?;
69
70        let mut stmt_fetch_accounts =
71            transaction.prepare("SELECT account, address FROM accounts")?;
72
73        // We track whether we have determined seed relevance or not, in order to
74        // correctly report errors when checking the seed against an account:
75        //
76        // - If we encounter an error with the first account, we can assert that the seed
77        //   is not relevant to the wallet by assuming that:
78        //   - All accounts are from the same seed (which is historically the only use
79        //     case that this migration supported), and
80        //   - All accounts in the wallet must have been able to derive their USKs (in
81        //     order to derive UIVKs).
82        //
83        // - Once the seed has been determined to be relevant (because it matched the
84        //   first account), any subsequent account derivation failure is proving wrong
85        //   our second assumption above, and we report this as corrupted data.
86        let mut seed_is_relevant = false;
87
88        let ua_request = UnifiedAddressRequest::unsafe_custom(Omit, Require, UA_TRANSPARENT);
89        let mut rows = stmt_fetch_accounts.query([])?;
90        while let Some(row) = rows.next()? {
91            // We only need to check for the presence of the seed if we have keys that
92            // need to be migrated; otherwise, it's fine to not supply the seed if this
93            // migration is being used to initialize an empty database.
94            if let Some(seed) = &self.seed {
95                let account: u32 = row.get(0)?;
96                let account = AccountId::try_from(account).map_err(|_| {
97                    WalletMigrationError::CorruptedData("Account ID is invalid".to_owned())
98                })?;
99                let usk =
100                    UnifiedSpendingKey::from_seed(&self.params, seed.expose_secret(), account)
101                        .map_err(|_| {
102                            if seed_is_relevant {
103                                WalletMigrationError::CorruptedData(
104                                    "Unable to derive spending key from seed.".to_string(),
105                                )
106                            } else {
107                                WalletMigrationError::SeedNotRelevant
108                            }
109                        })?;
110                let ufvk = usk.to_unified_full_viewing_key();
111
112                let address: String = row.get(1)?;
113                let decoded = Address::decode(&self.params, &address).ok_or_else(|| {
114                    WalletMigrationError::CorruptedData(format!(
115                        "Could not decode {address} as a valid Zcash address."
116                    ))
117                })?;
118                match decoded {
119                    Address::Sapling(decoded_address) => {
120                        let dfvk = ufvk.sapling().ok_or_else(||
121                            WalletMigrationError::CorruptedData("Derivation should have produced a UFVK containing a Sapling component.".to_owned()))?;
122                        let (idx, expected_address) = dfvk.default_address();
123                        if decoded_address != expected_address {
124                            return Err(if seed_is_relevant {
125                                WalletMigrationError::CorruptedData(format!(
126                                    "Decoded Sapling address {} does not match the ufvk's Sapling address {} at {:?}.",
127                                    address,
128                                    Address::Sapling(expected_address).encode(&self.params),
129                                    idx
130                                ))
131                            } else {
132                                WalletMigrationError::SeedNotRelevant
133                            });
134                        }
135                    }
136                    Address::Transparent(_) | Address::Tex(_) => {
137                        return Err(WalletMigrationError::CorruptedData(
138                            "Address field value decoded to a transparent address; should have been Sapling or unified.".to_string()));
139                    }
140                    Address::Unified(decoded_address) => {
141                        let (expected_address, idx) = ufvk.default_address(ua_request)?;
142                        if decoded_address != expected_address {
143                            return Err(if seed_is_relevant {
144                                WalletMigrationError::CorruptedData(format!(
145                                    "Decoded unified address {} does not match the ufvk's default address {} at {:?}.",
146                                    address,
147                                    Address::Unified(expected_address).encode(&self.params),
148                                    idx
149                                ))
150                            } else {
151                                WalletMigrationError::SeedNotRelevant
152                            });
153                        }
154                    }
155                }
156
157                // We made it past one derived account, so the seed must be relevant.
158                seed_is_relevant = true;
159
160                let ufvk_str: String = ufvk.encode(&self.params);
161                let address_str: String = ufvk.default_address(ua_request)?.0.encode(&self.params);
162
163                // This migration, and the wallet behaviour before it, stored the default
164                // transparent address in the `accounts` table. This does not necessarily
165                // match the transparent receiver in the default Unified Address. Starting
166                // from `AddressesTableMigration` below, we no longer store transparent
167                // addresses directly, but instead extract them from the Unified Address
168                // (or from the UFVK if the UA was derived without a transparent receiver,
169                // which is not the case for UAs generated by this crate).
170                #[cfg(feature = "transparent-inputs")]
171                let taddress_str: Option<String> = ufvk.transparent().and_then(|k| {
172                    k.derive_external_ivk()
173                        .ok()
174                        .map(|k| k.default_address().0.encode(&self.params))
175                });
176                #[cfg(not(feature = "transparent-inputs"))]
177                let taddress_str: Option<String> = None;
178
179                transaction.execute(
180                    "INSERT INTO accounts_new (account, ufvk, address, transparent_address)
181                    VALUES (:account, :ufvk, :address, :transparent_address)",
182                    named_params![
183                        ":account": &<u32>::from(account),
184                        ":ufvk": &ufvk_str,
185                        ":address": &address_str,
186                        ":transparent_address": &taddress_str,
187                    ],
188                )?;
189            } else {
190                return Err(WalletMigrationError::SeedRequired);
191            }
192        }
193
194        transaction.execute_batch(
195            "DROP TABLE accounts;
196            ALTER TABLE accounts_new RENAME TO accounts;",
197        )?;
198
199        //
200        // Update the sent_notes table to include an output_pool column that
201        // is respected by the uniqueness constraint
202        //
203
204        transaction.execute_batch(
205            "CREATE TABLE sent_notes_new (
206                id_note INTEGER PRIMARY KEY,
207                tx INTEGER NOT NULL,
208                output_pool INTEGER NOT NULL ,
209                output_index INTEGER NOT NULL,
210                from_account INTEGER NOT NULL,
211                address TEXT NOT NULL,
212                value INTEGER NOT NULL,
213                memo BLOB,
214                FOREIGN KEY (tx) REFERENCES transactions(id_tx),
215                FOREIGN KEY (from_account) REFERENCES accounts(account),
216                CONSTRAINT tx_output UNIQUE (tx, output_pool, output_index)
217            );",
218        )?;
219
220        // we query in a nested scope so that the col_names iterator is correctly
221        // dropped and doesn't maintain a lock on the table.
222        let has_output_pool = {
223            let mut stmt_fetch_columns = transaction.prepare("PRAGMA TABLE_INFO('sent_notes')")?;
224            let mut col_names = stmt_fetch_columns.query_map([], |row| {
225                let col_name: String = row.get(1)?;
226                Ok(col_name)
227            })?;
228
229            col_names.any(|cname| cname == Ok("output_pool".to_string()))
230        };
231
232        if has_output_pool {
233            transaction.execute_batch(
234                "INSERT INTO sent_notes_new
235                    (id_note, tx, output_pool, output_index, from_account, address, value, memo)
236                    SELECT id_note, tx, output_pool, output_index, from_account, address, value, memo
237                    FROM sent_notes;"
238            )?;
239        } else {
240            let mut stmt_fetch_sent_notes = transaction.prepare(
241                "SELECT id_note, tx, output_index, from_account, address, value, memo
242                    FROM sent_notes",
243            )?;
244
245            let mut stmt_insert_sent_note = transaction.prepare(
246                "INSERT INTO sent_notes_new
247                    (id_note, tx, output_pool, output_index, from_account, address, value, memo)
248                    VALUES (?, ?, ?, ?, ?, ?, ?, ?)",
249            )?;
250
251            let mut rows = stmt_fetch_sent_notes.query([])?;
252            while let Some(row) = rows.next()? {
253                let id_note: i64 = row.get(0)?;
254                let tx_ref: i64 = row.get(1)?;
255                let output_index: i64 = row.get(2)?;
256                let account_id: u32 = row.get(3)?;
257                let address: String = row.get(4)?;
258                let value: i64 = row.get(5)?;
259                let memo: Option<Vec<u8>> = row.get(6)?;
260
261                let decoded_address = Address::decode(&self.params, &address).ok_or_else(|| {
262                    WalletMigrationError::CorruptedData(format!(
263                        "Could not decode {address} as a valid Zcash address."
264                    ))
265                })?;
266                let output_pool = match decoded_address {
267                    Address::Sapling(_) => Ok(pool_code(PoolType::SAPLING)),
268                    Address::Transparent(_) | Address::Tex(_) => {
269                        Ok(pool_code(PoolType::TRANSPARENT))
270                    }
271                    Address::Unified(_) => Err(WalletMigrationError::CorruptedData(
272                        "Unified addresses should not yet appear in the sent_notes table."
273                            .to_string(),
274                    )),
275                }?;
276
277                stmt_insert_sent_note.execute(params![
278                    id_note,
279                    tx_ref,
280                    output_pool,
281                    output_index,
282                    account_id,
283                    address,
284                    value,
285                    memo
286                ])?;
287            }
288        }
289
290        transaction.execute_batch(
291            "DROP TABLE sent_notes;
292            ALTER TABLE sent_notes_new RENAME TO sent_notes;",
293        )?;
294
295        Ok(())
296    }
297
298    fn down(&self, _transaction: &rusqlite::Transaction) -> Result<(), WalletMigrationError> {
299        Err(WalletMigrationError::CannotRevert(MIGRATION_ID))
300    }
301}
