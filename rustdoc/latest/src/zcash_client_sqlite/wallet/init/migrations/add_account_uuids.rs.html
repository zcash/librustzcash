1//! This migration adds a UUID to each account record, and adds `name` and `key_source` columns. In
2//! addition, imported account records are now permitted to include key derivation metadata.
3
4use std::collections::HashSet;
5
6use rusqlite::named_params;
7use schemerz_rusqlite::RusqliteMigration;
8use uuid::Uuid;
9use zcash_client_backend::data_api::{AccountPurpose, AccountSource, Zip32Derivation};
10use zip32::fingerprint::SeedFingerprint;
11
12use crate::wallet::{account_kind_code, init::WalletMigrationError};
13
14use super::support_legacy_sqlite;
15
16pub(super) const MIGRATION_ID: Uuid = Uuid::from_u128(0xcccc623f_3243_43c7_b884_ceef25149e04);
17
18const DEPENDENCIES: &[Uuid] = &[support_legacy_sqlite::MIGRATION_ID];
19
20pub(super) struct Migration;
21
22impl schemerz::Migration<Uuid> for Migration {
23    fn id(&self) -> Uuid {
24        MIGRATION_ID
25    }
26
27    fn dependencies(&self) -> HashSet<Uuid> {
28        DEPENDENCIES.iter().copied().collect()
29    }
30
31    fn description(&self) -> &'static str {
32        "Adds a UUID for each account."
33    }
34}
35
36impl RusqliteMigration for Migration {
37    type Error = WalletMigrationError;
38
39    fn up(&self, transaction: &rusqlite::Transaction) -> Result<(), Self::Error> {
40        let account_kind_derived = account_kind_code(&AccountSource::Derived {
41            derivation: Zip32Derivation::new(
42                SeedFingerprint::from_bytes([0; 32]),
43                zip32::AccountId::ZERO,
44                #[cfg(feature = "zcashd-compat")]
45                None,
46            ),
47            key_source: None,
48        });
49        let account_kind_imported = account_kind_code(&AccountSource::Imported {
50            // the purpose here is irrelevant; we just use it to get the correct code
51            // for the account kind
52            purpose: AccountPurpose::ViewOnly,
53            key_source: None,
54        });
55        transaction.execute_batch(&format!(
56            r#"
57            CREATE TABLE accounts_new (
58                id INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
59                name TEXT,
60                uuid BLOB NOT NULL,
61                account_kind INTEGER NOT NULL DEFAULT {account_kind_derived},
62                key_source TEXT,
63                hd_seed_fingerprint BLOB,
64                hd_account_index INTEGER,
65                ufvk TEXT,
66                uivk TEXT NOT NULL,
67                orchard_fvk_item_cache BLOB,
68                sapling_fvk_item_cache BLOB,
69                p2pkh_fvk_item_cache BLOB,
70                birthday_height INTEGER NOT NULL,
71                birthday_sapling_tree_size INTEGER,
72                birthday_orchard_tree_size INTEGER,
73                recover_until_height INTEGER,
74                has_spend_key INTEGER NOT NULL DEFAULT 1,
75                CHECK (
76                  (
77                    account_kind = {account_kind_derived}
78                    AND hd_seed_fingerprint IS NOT NULL
79                    AND hd_account_index IS NOT NULL
80                    AND ufvk IS NOT NULL
81                  )
82                  OR
83                  (
84                    account_kind = {account_kind_imported}
85                    AND (hd_seed_fingerprint IS NULL) = (hd_account_index IS NULL)
86                  )
87                )
88            );
89            "#
90        ))?;
91
92        let mut q = transaction.prepare("SELECT * FROM accounts")?;
93        let mut rows = q.query([])?;
94        while let Some(row) = rows.next()? {
95            let preserve = |idx: &str| row.get::<_, rusqlite::types::Value>(idx);
96            transaction.execute(
97                r#"
98                INSERT INTO accounts_new (
99                    id, uuid,
100                    account_kind, hd_seed_fingerprint, hd_account_index,
101                    ufvk, uivk,
102                    orchard_fvk_item_cache, sapling_fvk_item_cache, p2pkh_fvk_item_cache,
103                    birthday_height, birthday_sapling_tree_size, birthday_orchard_tree_size,
104                    recover_until_height,
105                    has_spend_key
106                )
107                VALUES (
108                    :account_id, :uuid,
109                    :account_kind, :hd_seed_fingerprint, :hd_account_index,
110                    :ufvk, :uivk,
111                    :orchard_fvk_item_cache, :sapling_fvk_item_cache, :p2pkh_fvk_item_cache,
112                    :birthday_height, :birthday_sapling_tree_size, :birthday_orchard_tree_size,
113                    :recover_until_height,
114                    :has_spend_key
115                );
116                "#,
117                named_params! {
118                    ":account_id": preserve("id")?,
119                    ":uuid": Uuid::new_v4(),
120                    ":account_kind": preserve("account_kind")?,
121                    ":hd_seed_fingerprint": preserve("hd_seed_fingerprint")?,
122                    ":hd_account_index": preserve("hd_account_index")?,
123                    ":ufvk": preserve("ufvk")?,
124                    ":uivk": preserve("uivk")?,
125                    ":orchard_fvk_item_cache": preserve("orchard_fvk_item_cache")?,
126                    ":sapling_fvk_item_cache": preserve("sapling_fvk_item_cache")?,
127                    ":p2pkh_fvk_item_cache": preserve("p2pkh_fvk_item_cache")?,
128                    ":birthday_height": preserve("birthday_height")?,
129                    ":birthday_sapling_tree_size": preserve("birthday_sapling_tree_size")?,
130                    ":birthday_orchard_tree_size": preserve("birthday_orchard_tree_size")?,
131                    ":recover_until_height": preserve("recover_until_height")?,
132                    ":has_spend_key": preserve("has_spend_key")?,
133                },
134            )?;
135        }
136
137        transaction.execute_batch(
138            "PRAGMA legacy_alter_table = ON;
139            DROP TABLE accounts;
140            ALTER TABLE accounts_new RENAME TO accounts;
141            PRAGMA legacy_alter_table = OFF;
142
143            -- Add the new index.
144            CREATE UNIQUE INDEX accounts_uuid ON accounts (uuid);
145
146            -- Recreate the existing indices now that the original ones have been deleted.
147            CREATE UNIQUE INDEX hd_account ON accounts (hd_seed_fingerprint, hd_account_index);
148            CREATE UNIQUE INDEX accounts_uivk ON accounts (uivk);
149            CREATE UNIQUE INDEX accounts_ufvk ON accounts (ufvk);
150
151            -- Replace accounts.id with accounts.uuid in v_transactions.
152            DROP VIEW v_transactions;
153            CREATE VIEW v_transactions AS
154            WITH
155            notes AS (
156                -- Outputs received in this transaction
157                SELECT ro.account_id              AS account_id,
158                       transactions.mined_height  AS mined_height,
159                       transactions.txid          AS txid,
160                       ro.pool                    AS pool,
161                       id_within_pool_table,
162                       ro.value                   AS value,
163                       0                          AS spent_note_count,
164                       CASE
165                            WHEN ro.is_change THEN 1
166                            ELSE 0
167                       END AS change_note_count,
168                       CASE
169                            WHEN ro.is_change THEN 0
170                            ELSE 1
171                       END AS received_count,
172                       CASE
173                         WHEN (ro.memo IS NULL OR ro.memo = X'F6')
174                           THEN 0
175                         ELSE 1
176                       END AS memo_present,
177                       -- The wallet cannot receive transparent outputs in shielding transactions.
178                       CASE
179                         WHEN ro.pool = 0
180                           THEN 1
181                         ELSE 0
182                       END AS does_not_match_shielding
183                FROM v_received_outputs ro
184                JOIN transactions
185                     ON transactions.id_tx = ro.transaction_id
186                UNION
187                -- Outputs spent in this transaction
188                SELECT ro.account_id              AS account_id,
189                       transactions.mined_height  AS mined_height,
190                       transactions.txid          AS txid,
191                       ro.pool                    AS pool,
192                       id_within_pool_table,
193                       -ro.value                  AS value,
194                       1                          AS spent_note_count,
195                       0                          AS change_note_count,
196                       0                          AS received_count,
197                       0                          AS memo_present,
198                       -- The wallet cannot spend shielded outputs in shielding transactions.
199                       CASE
200                         WHEN ro.pool != 0
201                           THEN 1
202                         ELSE 0
203                       END AS does_not_match_shielding
204                FROM v_received_outputs ro
205                JOIN v_received_output_spends ros
206                     ON ros.pool = ro.pool
207                     AND ros.received_output_id = ro.id_within_pool_table
208                JOIN transactions
209                     ON transactions.id_tx = ros.transaction_id
210            ),
211            -- Obtain a count of the notes that the wallet created in each transaction,
212            -- not counting change notes.
213            sent_note_counts AS (
214                SELECT sent_notes.from_account_id     AS account_id,
215                       transactions.txid              AS txid,
216                       COUNT(DISTINCT sent_notes.id)  AS sent_notes,
217                       SUM(
218                         CASE
219                           WHEN (sent_notes.memo IS NULL OR sent_notes.memo = X'F6' OR ro.transaction_id IS NOT NULL)
220                             THEN 0
221                           ELSE 1
222                         END
223                       ) AS memo_count
224                FROM sent_notes
225                JOIN transactions
226                     ON transactions.id_tx = sent_notes.tx
227                LEFT JOIN v_received_outputs ro
228                     ON sent_notes.id = ro.sent_note_id
229                WHERE COALESCE(ro.is_change, 0) = 0
230                GROUP BY account_id, txid
231            ),
232            blocks_max_height AS (
233                SELECT MAX(blocks.height) AS max_height FROM blocks
234            )
235            SELECT accounts.uuid                AS account_uuid,
236                   notes.mined_height           AS mined_height,
237                   notes.txid                   AS txid,
238                   transactions.tx_index        AS tx_index,
239                   transactions.expiry_height   AS expiry_height,
240                   transactions.raw             AS raw,
241                   SUM(notes.value)             AS account_balance_delta,
242                   transactions.fee             AS fee_paid,
243                   SUM(notes.change_note_count) > 0  AS has_change,
244                   MAX(COALESCE(sent_note_counts.sent_notes, 0))  AS sent_note_count,
245                   SUM(notes.received_count)         AS received_note_count,
246                   SUM(notes.memo_present) + MAX(COALESCE(sent_note_counts.memo_count, 0)) AS memo_count,
247                   blocks.time                       AS block_time,
248                   (
249                        blocks.height IS NULL
250                        AND transactions.expiry_height BETWEEN 1 AND blocks_max_height.max_height
251                   ) AS expired_unmined,
252                   SUM(notes.spent_note_count) AS spent_note_count,
253                   (
254                        -- All of the wallet-spent and wallet-received notes are consistent with a
255                        -- shielding transaction.
256                        SUM(notes.does_not_match_shielding) = 0
257                        -- The transaction contains at least one wallet-spent output.
258                        AND SUM(notes.spent_note_count) > 0
259                        -- The transaction contains at least one wallet-received note.
260                        AND (SUM(notes.received_count) + SUM(notes.change_note_count)) > 0
261                        -- We do not know about any external outputs of the transaction.
262                        AND MAX(COALESCE(sent_note_counts.sent_notes, 0)) = 0
263                   ) AS is_shielding
264            FROM notes
265            LEFT JOIN accounts ON accounts.id = notes.account_id
266            LEFT JOIN transactions
267                 ON notes.txid = transactions.txid
268            JOIN blocks_max_height
269            LEFT JOIN blocks ON blocks.height = notes.mined_height
270            LEFT JOIN sent_note_counts
271                 ON sent_note_counts.account_id = notes.account_id
272                 AND sent_note_counts.txid = notes.txid
273            GROUP BY notes.account_id, notes.txid;
274
275            -- Replace accounts.id with accounts.uuid in v_tx_outputs.
276            DROP VIEW v_tx_outputs;
277            CREATE VIEW v_tx_outputs AS
278            WITH unioned AS (
279                -- select all outputs received by the wallet
280                SELECT transactions.txid            AS txid,
281                       ro.pool                      AS output_pool,
282                       ro.output_index              AS output_index,
283                       from_account.uuid            AS from_account_uuid,
284                       to_account.uuid              AS to_account_uuid,
285                       NULL                         AS to_address,
286                       ro.value                     AS value,
287                       ro.is_change                 AS is_change,
288                       ro.memo                      AS memo
289                FROM v_received_outputs ro
290                JOIN transactions
291                    ON transactions.id_tx = ro.transaction_id
292                -- join to the sent_notes table to obtain `from_account_id`
293                LEFT JOIN sent_notes ON sent_notes.id = ro.sent_note_id
294                -- join on the accounts table to obtain account UUIDs
295                LEFT JOIN accounts from_account ON from_account.id = sent_notes.from_account_id
296                LEFT JOIN accounts to_account ON to_account.id = ro.account_id
297                UNION ALL
298                -- select all outputs sent from the wallet to external recipients
299                SELECT transactions.txid            AS txid,
300                       sent_notes.output_pool       AS output_pool,
301                       sent_notes.output_index      AS output_index,
302                       from_account.uuid            AS from_account_uuid,
303                       NULL                         AS to_account_uuid,
304                       sent_notes.to_address        AS to_address,
305                       sent_notes.value             AS value,
306                       0                            AS is_change,
307                       sent_notes.memo              AS memo
308                FROM sent_notes
309                JOIN transactions
310                    ON transactions.id_tx = sent_notes.tx
311                LEFT JOIN v_received_outputs ro ON ro.sent_note_id = sent_notes.id
312                -- join on the accounts table to obtain account UUIDs
313                LEFT JOIN accounts from_account ON from_account.id = sent_notes.from_account_id
314            )
315            -- merge duplicate rows while retaining maximum information
316            SELECT
317                txid,
318                output_pool,
319                output_index,
320                max(from_account_uuid) AS from_account_uuid,
321                max(to_account_uuid) AS to_account_uuid,
322                max(to_address) AS to_address,
323                max(value) AS value,
324                max(is_change) AS is_change,
325                max(memo) AS memo
326            FROM unioned
327            GROUP BY txid, output_pool, output_index",
328        )?;
329
330        Ok(())
331    }
332
333    fn down(&self, _transaction: &rusqlite::Transaction) -> Result<(), Self::Error> {
334        Err(WalletMigrationError::CannotRevert(MIGRATION_ID))
335    }
336}
