1//! This migration adds a UUID to each account record, and adds `name` and `key_source` columns. In
2//! addition, imported account records are now permitted to include key derivation metadata.
3
4use std::collections::HashSet;
5
6use rusqlite::named_params;
7use schemerz_rusqlite::RusqliteMigration;
8use uuid::Uuid;
9use zcash_client_backend::data_api::{AccountPurpose, AccountSource, Zip32Derivation};
10use zip32::fingerprint::SeedFingerprint;
11
12use crate::wallet::{account_kind_code, init::WalletMigrationError};
13
14use super::support_legacy_sqlite;
15
16pub(super) const MIGRATION_ID: Uuid = Uuid::from_u128(0xcccc623f_3243_43c7_b884_ceef25149e04);
17
18const DEPENDENCIES: &[Uuid] = &[support_legacy_sqlite::MIGRATION_ID];
19
20pub(super) struct Migration;
21
22impl schemerz::Migration<Uuid> for Migration {
23    fn id(&self) -> Uuid {
24        MIGRATION_ID
25    }
26
27    fn dependencies(&self) -> HashSet<Uuid> {
28        DEPENDENCIES.iter().copied().collect()
29    }
30
31    fn description(&self) -> &'static str {
32        "Adds a UUID for each account."
33    }
34}
35
36impl RusqliteMigration for Migration {
37    type Error = WalletMigrationError;
38
39    fn up(&self, transaction: &rusqlite::Transaction) -> Result<(), Self::Error> {
40        let account_kind_derived = account_kind_code(&AccountSource::Derived {
41            derivation: Zip32Derivation::new(
42                SeedFingerprint::from_bytes([0; 32]),
43                zip32::AccountId::ZERO,
44            ),
45            key_source: None,
46        });
47        let account_kind_imported = account_kind_code(&AccountSource::Imported {
48            // the purpose here is irrelevant; we just use it to get the correct code
49            // for the account kind
50            purpose: AccountPurpose::ViewOnly,
51            key_source: None,
52        });
53        transaction.execute_batch(&format!(
54            r#"
55            CREATE TABLE accounts_new (
56                id INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
57                name TEXT,
58                uuid BLOB NOT NULL,
59                account_kind INTEGER NOT NULL DEFAULT {account_kind_derived},
60                key_source TEXT,
61                hd_seed_fingerprint BLOB,
62                hd_account_index INTEGER,
63                ufvk TEXT,
64                uivk TEXT NOT NULL,
65                orchard_fvk_item_cache BLOB,
66                sapling_fvk_item_cache BLOB,
67                p2pkh_fvk_item_cache BLOB,
68                birthday_height INTEGER NOT NULL,
69                birthday_sapling_tree_size INTEGER,
70                birthday_orchard_tree_size INTEGER,
71                recover_until_height INTEGER,
72                has_spend_key INTEGER NOT NULL DEFAULT 1,
73                CHECK (
74                  (
75                    account_kind = {account_kind_derived}
76                    AND hd_seed_fingerprint IS NOT NULL
77                    AND hd_account_index IS NOT NULL
78                    AND ufvk IS NOT NULL
79                  )
80                  OR
81                  (
82                    account_kind = {account_kind_imported}
83                    AND (hd_seed_fingerprint IS NULL) = (hd_account_index IS NULL)
84                  )
85                )
86            );
87            "#
88        ))?;
89
90        let mut q = transaction.prepare("SELECT * FROM accounts")?;
91        let mut rows = q.query([])?;
92        while let Some(row) = rows.next()? {
93            let preserve = |idx: &str| row.get::<_, rusqlite::types::Value>(idx);
94            transaction.execute(
95                r#"
96                INSERT INTO accounts_new (
97                    id, uuid,
98                    account_kind, hd_seed_fingerprint, hd_account_index,
99                    ufvk, uivk,
100                    orchard_fvk_item_cache, sapling_fvk_item_cache, p2pkh_fvk_item_cache,
101                    birthday_height, birthday_sapling_tree_size, birthday_orchard_tree_size,
102                    recover_until_height,
103                    has_spend_key
104                )
105                VALUES (
106                    :account_id, :uuid,
107                    :account_kind, :hd_seed_fingerprint, :hd_account_index,
108                    :ufvk, :uivk,
109                    :orchard_fvk_item_cache, :sapling_fvk_item_cache, :p2pkh_fvk_item_cache,
110                    :birthday_height, :birthday_sapling_tree_size, :birthday_orchard_tree_size,
111                    :recover_until_height,
112                    :has_spend_key
113                );
114                "#,
115                named_params! {
116                    ":account_id": preserve("id")?,
117                    ":uuid": Uuid::new_v4(),
118                    ":account_kind": preserve("account_kind")?,
119                    ":hd_seed_fingerprint": preserve("hd_seed_fingerprint")?,
120                    ":hd_account_index": preserve("hd_account_index")?,
121                    ":ufvk": preserve("ufvk")?,
122                    ":uivk": preserve("uivk")?,
123                    ":orchard_fvk_item_cache": preserve("orchard_fvk_item_cache")?,
124                    ":sapling_fvk_item_cache": preserve("sapling_fvk_item_cache")?,
125                    ":p2pkh_fvk_item_cache": preserve("p2pkh_fvk_item_cache")?,
126                    ":birthday_height": preserve("birthday_height")?,
127                    ":birthday_sapling_tree_size": preserve("birthday_sapling_tree_size")?,
128                    ":birthday_orchard_tree_size": preserve("birthday_orchard_tree_size")?,
129                    ":recover_until_height": preserve("recover_until_height")?,
130                    ":has_spend_key": preserve("has_spend_key")?,
131                },
132            )?;
133        }
134
135        transaction.execute_batch(
136            "PRAGMA legacy_alter_table = ON;
137            DROP TABLE accounts;
138            ALTER TABLE accounts_new RENAME TO accounts;
139            PRAGMA legacy_alter_table = OFF;
140
141            -- Add the new index.
142            CREATE UNIQUE INDEX accounts_uuid ON accounts (uuid);
143
144            -- Recreate the existing indices now that the original ones have been deleted.
145            CREATE UNIQUE INDEX hd_account ON accounts (hd_seed_fingerprint, hd_account_index);
146            CREATE UNIQUE INDEX accounts_uivk ON accounts (uivk);
147            CREATE UNIQUE INDEX accounts_ufvk ON accounts (ufvk);
148
149            -- Replace accounts.id with accounts.uuid in v_transactions.
150            DROP VIEW v_transactions;
151            CREATE VIEW v_transactions AS
152            WITH
153            notes AS (
154                -- Outputs received in this transaction
155                SELECT ro.account_id              AS account_id,
156                       transactions.mined_height  AS mined_height,
157                       transactions.txid          AS txid,
158                       ro.pool                    AS pool,
159                       id_within_pool_table,
160                       ro.value                   AS value,
161                       0                          AS spent_note_count,
162                       CASE
163                            WHEN ro.is_change THEN 1
164                            ELSE 0
165                       END AS change_note_count,
166                       CASE
167                            WHEN ro.is_change THEN 0
168                            ELSE 1
169                       END AS received_count,
170                       CASE
171                         WHEN (ro.memo IS NULL OR ro.memo = X'F6')
172                           THEN 0
173                         ELSE 1
174                       END AS memo_present,
175                       -- The wallet cannot receive transparent outputs in shielding transactions.
176                       CASE
177                         WHEN ro.pool = 0
178                           THEN 1
179                         ELSE 0
180                       END AS does_not_match_shielding
181                FROM v_received_outputs ro
182                JOIN transactions
183                     ON transactions.id_tx = ro.transaction_id
184                UNION
185                -- Outputs spent in this transaction
186                SELECT ro.account_id              AS account_id,
187                       transactions.mined_height  AS mined_height,
188                       transactions.txid          AS txid,
189                       ro.pool                    AS pool,
190                       id_within_pool_table,
191                       -ro.value                  AS value,
192                       1                          AS spent_note_count,
193                       0                          AS change_note_count,
194                       0                          AS received_count,
195                       0                          AS memo_present,
196                       -- The wallet cannot spend shielded outputs in shielding transactions.
197                       CASE
198                         WHEN ro.pool != 0
199                           THEN 1
200                         ELSE 0
201                       END AS does_not_match_shielding
202                FROM v_received_outputs ro
203                JOIN v_received_output_spends ros
204                     ON ros.pool = ro.pool
205                     AND ros.received_output_id = ro.id_within_pool_table
206                JOIN transactions
207                     ON transactions.id_tx = ros.transaction_id
208            ),
209            -- Obtain a count of the notes that the wallet created in each transaction,
210            -- not counting change notes.
211            sent_note_counts AS (
212                SELECT sent_notes.from_account_id     AS account_id,
213                       transactions.txid              AS txid,
214                       COUNT(DISTINCT sent_notes.id)  AS sent_notes,
215                       SUM(
216                         CASE
217                           WHEN (sent_notes.memo IS NULL OR sent_notes.memo = X'F6' OR ro.transaction_id IS NOT NULL)
218                             THEN 0
219                           ELSE 1
220                         END
221                       ) AS memo_count
222                FROM sent_notes
223                JOIN transactions
224                     ON transactions.id_tx = sent_notes.tx
225                LEFT JOIN v_received_outputs ro
226                     ON sent_notes.id = ro.sent_note_id
227                WHERE COALESCE(ro.is_change, 0) = 0
228                GROUP BY account_id, txid
229            ),
230            blocks_max_height AS (
231                SELECT MAX(blocks.height) AS max_height FROM blocks
232            )
233            SELECT accounts.uuid                AS account_uuid,
234                   notes.mined_height           AS mined_height,
235                   notes.txid                   AS txid,
236                   transactions.tx_index        AS tx_index,
237                   transactions.expiry_height   AS expiry_height,
238                   transactions.raw             AS raw,
239                   SUM(notes.value)             AS account_balance_delta,
240                   transactions.fee             AS fee_paid,
241                   SUM(notes.change_note_count) > 0  AS has_change,
242                   MAX(COALESCE(sent_note_counts.sent_notes, 0))  AS sent_note_count,
243                   SUM(notes.received_count)         AS received_note_count,
244                   SUM(notes.memo_present) + MAX(COALESCE(sent_note_counts.memo_count, 0)) AS memo_count,
245                   blocks.time                       AS block_time,
246                   (
247                        blocks.height IS NULL
248                        AND transactions.expiry_height BETWEEN 1 AND blocks_max_height.max_height
249                   ) AS expired_unmined,
250                   SUM(notes.spent_note_count) AS spent_note_count,
251                   (
252                        -- All of the wallet-spent and wallet-received notes are consistent with a
253                        -- shielding transaction.
254                        SUM(notes.does_not_match_shielding) = 0
255                        -- The transaction contains at least one wallet-spent output.
256                        AND SUM(notes.spent_note_count) > 0
257                        -- The transaction contains at least one wallet-received note.
258                        AND (SUM(notes.received_count) + SUM(notes.change_note_count)) > 0
259                        -- We do not know about any external outputs of the transaction.
260                        AND MAX(COALESCE(sent_note_counts.sent_notes, 0)) = 0
261                   ) AS is_shielding
262            FROM notes
263            LEFT JOIN accounts ON accounts.id = notes.account_id
264            LEFT JOIN transactions
265                 ON notes.txid = transactions.txid
266            JOIN blocks_max_height
267            LEFT JOIN blocks ON blocks.height = notes.mined_height
268            LEFT JOIN sent_note_counts
269                 ON sent_note_counts.account_id = notes.account_id
270                 AND sent_note_counts.txid = notes.txid
271            GROUP BY notes.account_id, notes.txid;
272
273            -- Replace accounts.id with accounts.uuid in v_tx_outputs.
274            DROP VIEW v_tx_outputs;
275            CREATE VIEW v_tx_outputs AS
276            WITH unioned AS (
277                -- select all outputs received by the wallet
278                SELECT transactions.txid            AS txid,
279                       ro.pool                      AS output_pool,
280                       ro.output_index              AS output_index,
281                       from_account.uuid            AS from_account_uuid,
282                       to_account.uuid              AS to_account_uuid,
283                       NULL                         AS to_address,
284                       ro.value                     AS value,
285                       ro.is_change                 AS is_change,
286                       ro.memo                      AS memo
287                FROM v_received_outputs ro
288                JOIN transactions
289                    ON transactions.id_tx = ro.transaction_id
290                -- join to the sent_notes table to obtain `from_account_id`
291                LEFT JOIN sent_notes ON sent_notes.id = ro.sent_note_id
292                -- join on the accounts table to obtain account UUIDs
293                LEFT JOIN accounts from_account ON from_account.id = sent_notes.from_account_id
294                LEFT JOIN accounts to_account ON to_account.id = ro.account_id
295                UNION ALL
296                -- select all outputs sent from the wallet to external recipients
297                SELECT transactions.txid            AS txid,
298                       sent_notes.output_pool       AS output_pool,
299                       sent_notes.output_index      AS output_index,
300                       from_account.uuid            AS from_account_uuid,
301                       NULL                         AS to_account_uuid,
302                       sent_notes.to_address        AS to_address,
303                       sent_notes.value             AS value,
304                       0                            AS is_change,
305                       sent_notes.memo              AS memo
306                FROM sent_notes
307                JOIN transactions
308                    ON transactions.id_tx = sent_notes.tx
309                LEFT JOIN v_received_outputs ro ON ro.sent_note_id = sent_notes.id
310                -- join on the accounts table to obtain account UUIDs
311                LEFT JOIN accounts from_account ON from_account.id = sent_notes.from_account_id
312            )
313            -- merge duplicate rows while retaining maximum information
314            SELECT
315                txid,
316                output_pool,
317                output_index,
318                max(from_account_uuid) AS from_account_uuid,
319                max(to_account_uuid) AS to_account_uuid,
320                max(to_address) AS to_address,
321                max(value) AS value,
322                max(is_change) AS is_change,
323                max(memo) AS memo
324            FROM unioned
325            GROUP BY txid, output_pool, output_index",
326        )?;
327
328        Ok(())
329    }
330

332        Err(WalletMigrationError::CannotRevert(MIGRATION_ID))
333    }
