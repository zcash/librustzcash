1//! The migration that records ephemeral addresses for each account.
2use std::collections::HashSet;
3
4use schemerz_rusqlite::RusqliteMigration;
5use uuid::Uuid;
6
7use crate::wallet::init::WalletMigrationError;
8
9use super::full_account_ids;
10
11pub(super) const MIGRATION_ID: Uuid = Uuid::from_u128(0x07610aac_b0e3_4ba8_aaa6_cda606f0fd7b);
12
13const DEPENDENCIES: &[Uuid] = &[full_account_ids::MIGRATION_ID];
14
15#[allow(dead_code)]
16pub(super) struct Migration;
17
18impl schemerz::Migration<Uuid> for Migration {
19    fn id(&self) -> Uuid {
20        MIGRATION_ID
21    }
22
23    fn dependencies(&self) -> HashSet<Uuid> {
24        DEPENDENCIES.iter().copied().collect()
25    }
26
27    fn description(&self) -> &'static str {
28        "Track which accounts have associated spending keys."
29    }
30}
31
32impl RusqliteMigration for Migration {
33    type Error = WalletMigrationError;
34
35    fn up(&self, transaction: &rusqlite::Transaction) -> Result<(), WalletMigrationError> {
36        transaction.execute_batch(
37            "ALTER TABLE accounts ADD COLUMN has_spend_key INTEGER NOT NULL DEFAULT 1",
38        )?;
39        Ok(())
40    }
41
42    fn down(&self, transaction: &rusqlite::Transaction) -> Result<(), WalletMigrationError> {
43        transaction.execute_batch("ALTER TABLE accounts DROP COLUMN has_spend_key")?;
44        Ok(())
45    }
46}
