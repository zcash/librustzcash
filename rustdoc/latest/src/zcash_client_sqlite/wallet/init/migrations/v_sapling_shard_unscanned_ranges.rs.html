38impl<P: consensus::Parameters> RusqliteMigration for Migration<P> {
39    type Error = WalletMigrationError;
40
41    fn up(&self, transaction: &rusqlite::Transaction) -> Result<(), Self::Error> {
42        transaction.execute_batch(&format!(
43            "CREATE VIEW v_sapling_shard_scan_ranges AS
44                SELECT
45                    shard.shard_index,
46                    shard.shard_index << {} AS start_position,
47                    (shard.shard_index + 1) << {} AS end_position_exclusive,
48                    IFNULL(prev_shard.subtree_end_height, {}) AS subtree_start_height,
49                    shard.subtree_end_height,
50                    shard.contains_marked,
51                    scan_queue.block_range_start,
52                    scan_queue.block_range_end,
53                    scan_queue.priority
54                FROM sapling_tree_shards shard
55                LEFT OUTER JOIN sapling_tree_shards prev_shard
56                    ON shard.shard_index = prev_shard.shard_index + 1
57                -- Join with scan ranges that overlap with the subtree's involved blocks.
58                INNER JOIN scan_queue ON (
59                    subtree_start_height < scan_queue.block_range_end AND
60                    (
61                        scan_queue.block_range_start <= shard.subtree_end_height OR
62                        shard.subtree_end_height IS NULL
63                    )
64                )",
65            SAPLING_SHARD_HEIGHT,
66            SAPLING_SHARD_HEIGHT,
67            u32::from(
68                self.params
69                    .activation_height(NetworkUpgrade::Sapling)
70                    .unwrap()
71            ),
72        ))?;
73
74        transaction.execute_batch(&format!(
75            "CREATE VIEW v_sapling_shard_unscanned_ranges AS
76                WITH wallet_birthday AS (SELECT MIN(birthday_height) AS height FROM accounts)
77                SELECT
78                    shard_index,
79                    start_position,
80                    end_position_exclusive,
81                    subtree_start_height,
82                    subtree_end_height,
83                    contains_marked,
84                    block_range_start,
85                    block_range_end,
86                    priority
87                FROM v_sapling_shard_scan_ranges
88                INNER JOIN wallet_birthday
89                WHERE priority > {}
90                AND block_range_end > wallet_birthday.height;",
91            priority_code(&ScanPriority::Scanned),
92        ))?;
93
94        Ok(())
95    }
96
97    fn down(&self, transaction: &rusqlite::Transaction) -> Result<(), Self::Error> {
98        transaction.execute_batch("DROP VIEW v_sapling_shard_unscanned_ranges;")?;
99        Ok(())
100    }
101}
