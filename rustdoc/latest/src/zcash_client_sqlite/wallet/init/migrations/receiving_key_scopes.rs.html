58#[allow(clippy::type_complexity)]
59fn select_note_scope<S: ShardStore<H = sapling::Node, CheckpointId = BlockHeight>>(
60    commitment_tree: &mut ShardTree<
61        S,
62        { sapling::NOTE_COMMITMENT_TREE_DEPTH },
63        SAPLING_SHARD_HEIGHT,
64    >,
65    dfvk: &DiversifiableFullViewingKey,
66    diversifier: &sapling::Diversifier,
67    value: &sapling::value::NoteValue,
68    rseed: &sapling::Rseed,
69    note_commitment_tree_position: Position,
70) -> Result<Option<Scope>, WalletMigrationError> {
71    // Attempt to reconstruct the note being spent using both the internal and external dfvks
72    // corresponding to the unified spending key, checking against the witness we are using
73    // to spend the note that we've used the correct key.
74    let external_note = dfvk
75        .diversified_address(*diversifier)
76        .map(|addr| addr.create_note(*value, *rseed));
77    let internal_note = dfvk
78        .diversified_change_address(*diversifier)
79        .map(|addr| addr.create_note(*value, *rseed));
80
81    if let Some(recorded_node) = commitment_tree
82        .get_marked_leaf(note_commitment_tree_position)
83        .map_err(|e| {
84            WalletMigrationError::CorruptedData(format!(
85                "Error querying note commitment tree: {e:?}"
86            ))
87        })?
88    {
89        if external_note.map(|n| Node::from_cmu(&n.cmu())) == Some(recorded_node) {
90            Ok(Some(Scope::External))
91        } else if internal_note.map(|n| Node::from_cmu(&n.cmu())) == Some(recorded_node) {
92            Ok(Some(Scope::Internal))
93        } else {
94            Err(WalletMigrationError::CorruptedData(
95                "Unable to reconstruct note.".to_owned(),
96            ))
97        }
98    } else {
99        Ok(None)
100    }
101}

106    fn up(&self, transaction: &rusqlite::Transaction) -> Result<(), WalletMigrationError> {
107        transaction.execute_batch(
108            &format!(
109                "ALTER TABLE sapling_received_notes ADD COLUMN recipient_key_scope INTEGER NOT NULL DEFAULT {};",
110                KeyScope::EXTERNAL.encode()
111            )
112        )?;
113
114        // For all notes we have to determine whether they were actually sent to the internal key
115        // or the external key for the account, so we trial-decrypt the original output with the
116        // internal IVK and update the persisted scope value if necessary. We check all notes,
117        // rather than just change notes, because shielding notes may not have been considered
118        // change.
119        let mut stmt_select_notes = transaction.prepare(
120            "SELECT
121                id_note,
122                output_index,
123                transactions.raw,
124                transactions.block,
125                transactions.expiry_height,
126                accounts.ufvk,
127                diversifier,
128                value,
129                rcm,
130                commitment_tree_position
131             FROM sapling_received_notes
132             INNER JOIN accounts on accounts.account = sapling_received_notes.account
133             INNER JOIN transactions ON transactions.id_tx = sapling_received_notes.tx",
134        )?;
135
136        // In the case that we don't have the raw transaction
137        let mut commitment_tree = ShardTree::new(
138            SqliteShardStore::<_, _, SAPLING_SHARD_HEIGHT>::from_connection(
139                transaction,
140                SAPLING_TABLES_PREFIX,
141            )?,
142            PRUNING_DEPTH as usize,
143        );
144
145        let mut rows = stmt_select_notes.query([])?;
146        while let Some(row) = rows.next()? {
147            let note_id: i64 = row.get(0)?;
148            let output_index: usize = row.get(1)?;
149            let tx_data_opt: Option<Vec<u8>> = row.get(2)?;
150
151            let tx_height = row.get::<_, Option<u32>>(3)?.map(BlockHeight::from);
152            let tx_expiry = row.get::<_, Option<u32>>(4)?;
153            let zip212_height = tx_height.map_or_else(
154                || {
155                    tx_expiry.filter(|h| *h != 0).map_or_else(
156                        || chain_tip_height(transaction),
157                        |h| Ok(Some(BlockHeight::from(h))),
158                    )
159                },
160                |h| Ok(Some(h)),
161            )?;
162
163            let zip212_enforcement = zip212_height.map_or_else(
164                || {
165                    // If the transaction has not been mined and the expiry height is set to 0 (no
166                    // expiry) an no chain tip information is available, then we assume it can only
167                    // be mined under ZIP 212 enforcement rules, so we default to `On`
168                    Zip212Enforcement::On
169                },
170                |h| zip212_enforcement(&self.params, h),
171            );
172
173            let ufvk_str: String = row.get(5)?;
174            let ufvk = UnifiedFullViewingKey::decode(&self.params, &ufvk_str).map_err(|e| {
175                WalletMigrationError::CorruptedData(format!("Stored UFVK was invalid: {e:?}"))
176            })?;
177
178            let dfvk = ufvk.sapling().ok_or_else(|| {
179                WalletMigrationError::CorruptedData(
180                    "UFVK must have a Sapling component to have received Sapling notes.".to_owned(),
181                )
182            })?;
183
184            // We previously set the default to external scope, so we now verify whether the output
185            // is decryptable using the intenally-scoped IVK and, if so, mark it as such.
186            if let Some(tx_data) = tx_data_opt {
187                let tx = Transaction::read(&tx_data[..], BranchId::Canopy).map_err(|e| {
188                    WalletMigrationError::CorruptedData(format!(
189                        "Unable to parse raw transaction: {e:?}"
190                    ))
191                })?;
192                let output = tx
193                    .sapling_bundle()
194                    .and_then(|b| b.shielded_outputs().get(output_index))
195                    .unwrap_or_else(|| {
196                        panic!("A Sapling output must exist at index {output_index}")
197                    });
198
199                let pivk = PreparedIncomingViewingKey::new(&dfvk.to_ivk(Scope::Internal));
200                if try_sapling_note_decryption(&pivk, output, zip212_enforcement).is_some() {
201                    transaction.execute(
202                        "UPDATE sapling_received_notes SET recipient_key_scope = :scope
203                         WHERE id_note = :note_id",
204                        named_params! {":scope": KeyScope::INTERNAL.encode(), ":note_id": note_id},
205                    )?;
206                }
207            } else {
208                let diversifier = {
209                    let d: Vec<_> = row.get(6)?;
210                    Diversifier(d[..].try_into().map_err(|_| {
211                        WalletMigrationError::CorruptedData(
212                            "Invalid diversifier length".to_string(),
213                        )
214                    })?)
215                };
216
217                let note_value = Zatoshis::from_nonnegative_i64(row.get(7)?).map_err(|_e| {
218                    WalletMigrationError::CorruptedData(
219                        "Note values must be nonnegative".to_string(),
220                    )
221                })?;
222
223                let rseed = {
224                    let rcm_bytes: [u8; 32] =
225                        row.get::<_, Vec<u8>>(8)?[..].try_into().map_err(|_| {
226                            WalletMigrationError::CorruptedData(format!(
227                                "Note {note_id} is invalid"
228                            ))
229                        })?;
230
231                    let rcm = Option::from(jubjub::Fr::from_repr(rcm_bytes)).ok_or_else(|| {
232                        WalletMigrationError::CorruptedData(format!("Note {note_id} is invalid"))
233                    })?;
234
235                    // The wallet database always stores the `rcm` value, and not `rseed`,
236                    // so for note reconstruction we always use `BeforeZip212`.
237                    Rseed::BeforeZip212(rcm)
238                };
239
240                let note_commitment_tree_position =
241                    Position::from(u64::try_from(row.get::<_, i64>(9)?).map_err(|_| {
242                        WalletMigrationError::CorruptedData(
243                            "Note commitment tree position invalid.".to_string(),
244                        )
245                    })?);
246
247                let scope = select_note_scope(
248                    &mut commitment_tree,
249                    dfvk,
250                    &diversifier,
251                    &sapling::value::NoteValue::from_raw(note_value.into_u64()),
252                    &rseed,
253                    note_commitment_tree_position,
254                )?;
255
256                if scope == Some(Scope::Internal) {
257                    transaction.execute(
258                        "UPDATE sapling_received_notes SET recipient_key_scope = :scope
259                         WHERE id_note = :note_id",
260                        named_params! {":scope": KeyScope::INTERNAL.encode(), ":note_id": note_id},
261                    )?;
262                }
263            }
264        }
265
266        Ok(())
267    }
268
269    fn down(&self, _transaction: &rusqlite::Transaction) -> Result<(), WalletMigrationError> {
270        Err(WalletMigrationError::CannotRevert(MIGRATION_ID))
271    }
272}

425    fn put_received_note_before_migration<T: ReceivedSaplingOutput<AccountId = AccountRef>>(
426        conn: &Connection,
427        output: &T,
428        tx_ref: i64,
429        spent_in: Option<i64>,
430    ) -> Result<(), SqliteClientError> {
431        let mut stmt_upsert_received_note = conn.prepare_cached(
432            "INSERT INTO sapling_received_notes
433            (tx, output_index, account, diversifier, value, rcm, memo, nf,
434             is_change, spent, commitment_tree_position)
435            VALUES (
436                :tx,
437                :output_index,
438                :account,
439                :diversifier,
440                :value,
441                :rcm,
442                :memo,
443                :nf,
444                :is_change,
445                :spent,
446                :commitment_tree_position
447            )
448            ON CONFLICT (tx, output_index) DO UPDATE
449            SET account = :account,
450                diversifier = :diversifier,
451                value = :value,
452                rcm = :rcm,
453                nf = IFNULL(:nf, nf),
454                memo = IFNULL(:memo, memo),
455                is_change = IFNULL(:is_change, is_change),
456                spent = IFNULL(:spent, spent),
457                commitment_tree_position = IFNULL(:commitment_tree_position, commitment_tree_position)",
458        )?;
459
460        let rcm = output.note().rcm().to_bytes();
461        let to = output.note().recipient();
462        let diversifier = to.diversifier();
463
464        let account = output.account_id();
465        let sql_args = named_params![
466            ":tx": &tx_ref,
467            ":output_index": i64::try_from(output.index()).expect("output indices are representable as i64"),
468            ":account": account.0,
469            ":diversifier": &diversifier.0,
470            ":value": output.note().value().inner(),
471            ":rcm": &rcm,
472            ":nf": output.nullifier().map(|nf| nf.0),
473            ":memo": memo_repr(output.memo()),
474            ":is_change": output.is_change(),
475            ":spent": spent_in,
476            ":commitment_tree_position": output.note_commitment_tree_position().map(u64::from),
477        ];
478
479        stmt_upsert_received_note
480            .execute(sql_args)
481            .map_err(SqliteClientError::from)?;
482
483        Ok(())
484    }

486    /// This reproduces [`crate::wallet::put_tx_data`] as it was at the time
487    /// of the creation of this migration.
488    fn put_tx_data(
489        conn: &rusqlite::Connection,
490        tx: &Transaction,
491        fee: Option<Zatoshis>,
492        created_at: Option<time::OffsetDateTime>,
493    ) -> Result<TxRef, SqliteClientError> {
494        let mut stmt_upsert_tx_data = conn.prepare_cached(
495            "INSERT INTO transactions (txid, created, expiry_height, raw, fee)
496            VALUES (:txid, :created_at, :expiry_height, :raw, :fee)
497            ON CONFLICT (txid) DO UPDATE
498            SET expiry_height = :expiry_height,
499                raw = :raw,
500                fee = IFNULL(:fee, fee)
501            RETURNING id_tx",
502        )?;
503
504        let txid = tx.txid();
505        let mut raw_tx = vec![];
506        tx.write(&mut raw_tx)?;
507
508        let tx_params = named_params![
509            ":txid": &txid.as_ref()[..],
510            ":created_at": created_at,
511            ":expiry_height": u32::from(tx.expiry_height()),
512            ":raw": raw_tx,
513            ":fee": fee.map(u64::from),
514        ];
515
516        stmt_upsert_tx_data
517            .query_row(tx_params, |row| row.get::<_, i64>(0).map(TxRef))
518            .map_err(SqliteClientError::from)
519    }

522    fn receiving_key_scopes_migration_enhanced() {
523        let params = Network::TestNetwork;
524
525        // Create wallet upgraded to just before the current migration.
526        let data_file = NamedTempFile::new().unwrap();
527        let mut db_data =
528            WalletDb::for_path(data_file.path(), params, test_clock(), test_rng()).unwrap();
529        WalletMigrator::new()
530            .ignore_seed_relevance()
531            .init_or_migrate_to(
532                &mut db_data,
533                &[
534                    add_account_birthdays::MIGRATION_ID,
535                    shardtree_support::MIGRATION_ID,
536                ],
537            )
538            .unwrap();
539
540        let (ufvk0, height, res) = prepare_wallet_state(&mut db_data);
541        let tx = res.transaction();
542        let account_id = AccountRef(0);
543
544        // We can't use `decrypt_and_store_transaction` because we haven't migrated yet.
545        // Replicate its relevant innards here.
546        let d_tx = decrypt_transaction(
547            &params,
548            Some(height),
549            None,
550            tx,
551            &[(account_id, ufvk0)].into_iter().collect(),
552        );
553
554        db_data
555            .transactionally::<_, _, rusqlite::Error>(|wdb| {
556                let tx_ref = put_tx_data(wdb.conn.0, d_tx.tx(), None, None).unwrap();
557
558                let mut spending_account_id: Option<AccountRef> = None;
559
560                // Orchard outputs were not supported as of the wallet states that could require this
561                // migration.
562                for output in d_tx.sapling_outputs() {
563                    match output.transfer_type() {
564                        TransferType::Outgoing | TransferType::WalletInternal => {
565                            // Don't need to bother with sent outputs for this test.
566                            if output.transfer_type() != TransferType::Outgoing {
567                                put_received_note_before_migration(
568                                    wdb.conn.0, output, tx_ref.0, None,
569                                )
570                                .unwrap();
571                            }
572                        }
573                        TransferType::Incoming => {
574                            match spending_account_id {
575                                Some(id) => assert_eq!(id, *output.account()),
576                                None => {
577                                    spending_account_id = Some(*output.account());
578                                }
579                            }
580
581                            put_received_note_before_migration(wdb.conn.0, output, tx_ref.0, None)
582                                .unwrap();
583                        }
584                    }
585                }
586
587                Ok(())
588            })
589            .unwrap();
590
591        // Apply the current migration
592        WalletMigrator::new()
593            .ignore_seed_relevance()
594            .init_or_migrate_to(&mut db_data, &[super::MIGRATION_ID])
595            .unwrap();
596
597        // There should be two rows in the `sapling_received_notes` table with correct scopes.
598        let mut q = db_data
599            .conn
600            .prepare(
601                "SELECT value, recipient_key_scope
602                FROM sapling_received_notes",
603            )
604            .unwrap();
605        let mut rows = q.query([]).unwrap();
606        let mut row_count = 0;
607        while let Some(row) = rows.next().unwrap() {
608            row_count += 1;
609            let value: u64 = row.get(0).unwrap();
610            let scope = KeyScope::decode(row.get(1).unwrap()).unwrap();
611            match value {
612                EXTERNAL_VALUE => assert_eq!(scope, KeyScope::EXTERNAL),
613                INTERNAL_VALUE => assert_eq!(scope, KeyScope::INTERNAL),
614                _ => {
615                    panic!(
616                        "(Value, Scope) pair {:?} is not expected to exist in the wallet.",
617                        (value, scope),
618                    );
619                }
620            }
621        }
622        assert_eq!(row_count, 2);
623    }

650        };
651        for output in tx.sapling_bundle().unwrap().shielded_outputs() {
652            compact_tx.outputs.push(output.into());
653        }
654        let prev_hash = BlockHash([4; 32]);
655        let mut block = CompactBlock {
656            height: height.into(),
657            hash: vec![7; 32],
658            prev_hash: prev_hash.0[..].into(),
659            ..Default::default()
660        };
661        block.vtx.push(compact_tx);
662        let scanning_keys = ScanningKeys::from_account_ufvks([(AccountRef(0), ufvk0)]);
663
664        let scanned_block = scan_block(
665            &params,
666            block,
667            &scanning_keys,
668            &Nullifiers::empty(),
669            Some(&BlockMetadata::from_parts(
670                height - 1,
671                prev_hash,
672                Some(0),
673                #[cfg(feature = "orchard")]
674                Some(0),
675            )),
676        )
677        .unwrap();
678
679        // We can't use `put_blocks` because we haven't migrated yet.
680        // Replicate its relevant innards here.
681        let blocks = [scanned_block];
682        db_data
683            .transactionally(|wdb| {
684                let start_positions = blocks.first().map(|block| {
685                    (
686                        block.height(),
687                        Position::from(
688                            u64::from(block.sapling().final_tree_size())
689                                - u64::try_from(block.sapling().commitments().len()).unwrap(),
690                        ),
691                    )
692                });
693                let mut sapling_commitments = vec![];
694                let mut last_scanned_height = None;
695                let mut note_positions = vec![];
696                for block in blocks.into_iter() {
697                    if last_scanned_height
698                        .iter()
699                        .any(|prev| block.height() != *prev + 1)
700                    {
701                        return Err(SqliteClientError::NonSequentialBlocks);
702                    }
703
704                    // Insert the block into the database.
705                    put_block(
706                        wdb.conn.0,
707                        block.height(),
708                        block.block_hash(),
709                        block.block_time(),
710                        block.sapling().final_tree_size(),
711                        block.sapling().commitments().len().try_into().unwrap(),
712                        #[cfg(feature = "orchard")]
713                        block.orchard().final_tree_size(),
714                        #[cfg(feature = "orchard")]
715                        block.orchard().commitments().len().try_into().unwrap(),
716                    )?;
717
718                    for tx in block.transactions() {
719                        let tx_row = put_tx_meta(wdb.conn.0, tx, block.height())?;
720
721                        for output in tx.sapling_outputs() {
722                            put_received_note_before_migration(wdb.conn.0, output, tx_row, None)?;
723                        }
724                    }
725
726                    note_positions.extend(block.transactions().iter().flat_map(|wtx| {
727                        wtx.sapling_outputs()
728                            .iter()
729                            .map(|out| out.note_commitment_tree_position())
730                    }));
731
732                    last_scanned_height = Some(block.height());
733                    let block_commitments = block.into_commitments();
734                    sapling_commitments.extend(block_commitments.sapling.into_iter().map(Some));
735                }
736
737                // We will have a start position and a last scanned height in all cases where
738                // `blocks` is non-empty.
739                if let Some(((_, start_position), _)) = start_positions.zip(last_scanned_height) {
740                    // Create subtrees from the note commitments in parallel.
741                    const CHUNK_SIZE: usize = 1024;
742                    let subtrees = sapling_commitments
743                        .par_chunks_mut(CHUNK_SIZE)
744                        .enumerate()
745                        .filter_map(|(i, chunk)| {
746                            let start = start_position + (i * CHUNK_SIZE) as u64;
747                            let end = start + chunk.len() as u64;
748
749                            shardtree::LocatedTree::from_iter(
750                                start..end,
751                                SAPLING_SHARD_HEIGHT.into(),
752                                chunk.iter_mut().map(|n| n.take().expect("always Some")),
753                            )
754                        })
755                        .map(|res| (res.subtree, res.checkpoints))
756                        .collect::<Vec<_>>();
757
758                    // Update the Sapling note commitment tree with all newly read note commitments
759                    let mut subtrees = subtrees.into_iter();
760                    wdb.with_sapling_tree_mut::<_, _, SqliteClientError>(move |sapling_tree| {
761                        for (tree, checkpoints) in &mut subtrees {
762                            sapling_tree.insert_tree(tree, checkpoints)?;
763                        }
764
765                        Ok(())
766                    })?;
767                }
768
769                Ok(())
770            })
771            .unwrap();
772
773        // Apply the current migration
774        WalletMigrator::new()
775            .ignore_seed_relevance()
776            .init_or_migrate_to(&mut db_data, &[super::MIGRATION_ID])
777            .unwrap();
778
779        // There should be two rows in the `sapling_received_notes` table with correct scopes.
780        let mut q = db_data
781            .conn
782            .prepare(
783                "SELECT value, recipient_key_scope
784                FROM sapling_received_notes",
785            )
786            .unwrap();
787        let mut rows = q.query([]).unwrap();
788        let mut row_count = 0;
789        while let Some(row) = rows.next().unwrap() {
790            row_count += 1;
791            let value: u64 = row.get(0).unwrap();
792            let scope = KeyScope::decode(row.get(1).unwrap()).unwrap();
793            match value {
794                EXTERNAL_VALUE => assert_eq!(scope, KeyScope::EXTERNAL),
795                INTERNAL_VALUE => assert_eq!(scope, KeyScope::INTERNAL),
796                _ => {
797                    panic!(
798                        "(Value, Scope) pair {:?} is not expected to exist in the wallet.",
799                        (value, scope),
800                    );
801                }
802            }
803        }
804        assert_eq!(row_count, 2);
805    }

807    /// This is a copy of [`crate::wallet::put_block`] as of the expected database
808    /// state corresponding to this migration. It is duplicated here as later
809    /// updates to the database schema require incompatible changes to `put_block`.
810    #[allow(clippy::too_many_arguments)]
811    fn put_block(
812        conn: &rusqlite::Transaction<'_>,
813        block_height: BlockHeight,
814        block_hash: BlockHash,
815        block_time: u32,
816        sapling_commitment_tree_size: u32,
817        sapling_output_count: u32,
818        #[cfg(feature = "orchard")] orchard_commitment_tree_size: u32,
819        #[cfg(feature = "orchard")] orchard_action_count: u32,
820    ) -> Result<(), SqliteClientError> {
821        let block_hash_data = conn
822            .query_row(
823                "SELECT hash FROM blocks WHERE height = ?",
824                [u32::from(block_height)],
825                |row| row.get::<_, Vec<u8>>(0),
826            )
827            .optional()?;
828
829        // Ensure that in the case of an upsert, we don't overwrite block data
830        // with information for a block with a different hash.
831        if let Some(bytes) = block_hash_data {
832            let expected_hash = BlockHash::try_from_slice(&bytes).ok_or_else(|| {
833                SqliteClientError::CorruptedData(format!(
834                    "Invalid block hash at height {}",
835                    u32::from(block_height)
836                ))
837            })?;
838            if expected_hash != block_hash {
839                return Err(SqliteClientError::BlockConflict(block_height));
840            }
841        }
842
843        let mut stmt_upsert_block = conn.prepare_cached(
844            "INSERT INTO blocks (
845                height,
846                hash,
847                time,
848                sapling_commitment_tree_size,
849                sapling_output_count,
850                sapling_tree,
851                orchard_commitment_tree_size,
852                orchard_action_count
853            )
854            VALUES (
855                :height,
856                :hash,
857                :block_time,
858                :sapling_commitment_tree_size,
859                :sapling_output_count,
860                x'00',
861                :orchard_commitment_tree_size,
862                :orchard_action_count
863            )
864            ON CONFLICT (height) DO UPDATE
865            SET hash = :hash,
866                time = :block_time,
867                sapling_commitment_tree_size = :sapling_commitment_tree_size,
868                sapling_output_count = :sapling_output_count,
869                orchard_commitment_tree_size = :orchard_commitment_tree_size,
870                orchard_action_count = :orchard_action_count",
871        )?;
872
873        #[cfg(not(feature = "orchard"))]
874        let orchard_commitment_tree_size: Option<u32> = None;
875        #[cfg(not(feature = "orchard"))]
876        let orchard_action_count: Option<u32> = None;
877
878        stmt_upsert_block.execute(named_params![
879            ":height": u32::from(block_height),
880            ":hash": &block_hash.0[..],
881            ":block_time": block_time,
882            ":sapling_commitment_tree_size": sapling_commitment_tree_size,
883            ":sapling_output_count": sapling_output_count,
884            ":orchard_commitment_tree_size": orchard_commitment_tree_size,
885            ":orchard_action_count": orchard_action_count,
886        ])?;
887
888        Ok(())
889    }

891    /// This is a copy of [`crate::wallet::put_tx_meta`] as of the expected database
892    /// state corresponding to this migration. It is duplicated here as later
893    /// updates to the database schema require incompatible changes to `put_tx_meta`.
894    pub(crate) fn put_tx_meta(
895        conn: &rusqlite::Connection,
896        tx: &WalletTx<AccountRef>,
897        height: BlockHeight,
898    ) -> Result<i64, SqliteClientError> {
899        // It isn't there, so insert our transaction into the database.
900        let mut stmt_upsert_tx_meta = conn.prepare_cached(
901            "INSERT INTO transactions (txid, block, tx_index)
902            VALUES (:txid, :block, :tx_index)
903            ON CONFLICT (txid) DO UPDATE
904            SET block = :block,
905                tx_index = :tx_index
906            RETURNING id_tx",
907        )?;
908
909        let txid_bytes = tx.txid();
910        let tx_params = named_params![
911            ":txid": &txid_bytes.as_ref()[..],
912            ":block": u32::from(height),
913            ":tx_index": i64::try_from(tx.block_index()).expect("transaction indices are representable as i64"),
914        ];
915
916        stmt_upsert_tx_meta
917            .query_row(tx_params, |row| row.get::<_, i64>(0))
918            .map_err(SqliteClientError::from)
919    }
