59#[allow(clippy::type_complexity)]
60fn select_note_scope<S: ShardStore<H = sapling::Node, CheckpointId = BlockHeight>>(
61    commitment_tree: &mut ShardTree<
62        S,
63        { sapling::NOTE_COMMITMENT_TREE_DEPTH },
64        SAPLING_SHARD_HEIGHT,
65    >,
66    dfvk: &DiversifiableFullViewingKey,
67    diversifier: &sapling::Diversifier,
68    value: &sapling::value::NoteValue,
69    rseed: &sapling::Rseed,
70    note_commitment_tree_position: Position,
71) -> Result<Option<Scope>, WalletMigrationError> {
72    // Attempt to reconstruct the note being spent using both the internal and external dfvks
73    // corresponding to the unified spending key, checking against the witness we are using
74    // to spend the note that we've used the correct key.
75    let external_note = dfvk
76        .diversified_address(*diversifier)
77        .map(|addr| addr.create_note(*value, *rseed));
78    let internal_note = dfvk
79        .diversified_change_address(*diversifier)
80        .map(|addr| addr.create_note(*value, *rseed));
81
82    if let Some(recorded_node) = commitment_tree
83        .get_marked_leaf(note_commitment_tree_position)
84        .map_err(|e| {
85            WalletMigrationError::CorruptedData(format!(
86                "Error querying note commitment tree: {:?}",
87                e
88            ))
89        })?
90    {
91        if external_note.map(|n| Node::from_cmu(&n.cmu())) == Some(recorded_node) {
92            Ok(Some(Scope::External))
93        } else if internal_note.map(|n| Node::from_cmu(&n.cmu())) == Some(recorded_node) {
94            Ok(Some(Scope::Internal))
95        } else {
96            Err(WalletMigrationError::CorruptedData(
97                "Unable to reconstruct note.".to_owned(),
98            ))
99        }
100    } else {
101        Ok(None)
102    }
103}

108    fn up(&self, transaction: &rusqlite::Transaction) -> Result<(), WalletMigrationError> {
109        transaction.execute_batch(
110            &format!(
111                "ALTER TABLE sapling_received_notes ADD COLUMN recipient_key_scope INTEGER NOT NULL DEFAULT {};",
112                scope_code(Scope::External)
113            )
114        )?;
115
116        // For all notes we have to determine whether they were actually sent to the internal key
117        // or the external key for the account, so we trial-decrypt the original output with the
118        // internal IVK and update the persisted scope value if necessary. We check all notes,
119        // rather than just change notes, because shielding notes may not have been considered
120        // change.
121        let mut stmt_select_notes = transaction.prepare(
122            "SELECT
123                id_note,
124                output_index,
125                transactions.raw,
126                transactions.block,
127                transactions.expiry_height,
128                accounts.ufvk,
129                diversifier,
130                value,
131                rcm,
132                commitment_tree_position
133             FROM sapling_received_notes
134             INNER JOIN accounts on accounts.account = sapling_received_notes.account
135             INNER JOIN transactions ON transactions.id_tx = sapling_received_notes.tx",
136        )?;
137
138        // In the case that we don't have the raw transaction
139        let mut commitment_tree = ShardTree::new(
140            SqliteShardStore::<_, _, SAPLING_SHARD_HEIGHT>::from_connection(
141                transaction,
142                SAPLING_TABLES_PREFIX,
143            )?,
144            PRUNING_DEPTH as usize,
145        );
146
147        let mut rows = stmt_select_notes.query([])?;
148        while let Some(row) = rows.next()? {
149            let note_id: i64 = row.get(0)?;
150            let output_index: usize = row.get(1)?;
151            let tx_data_opt: Option<Vec<u8>> = row.get(2)?;
152
153            let tx_height = row.get::<_, Option<u32>>(3)?.map(BlockHeight::from);
154            let tx_expiry = row.get::<_, Option<u32>>(4)?;
155            let zip212_height = tx_height.map_or_else(
156                || {
157                    tx_expiry.filter(|h| *h != 0).map_or_else(
158                        || chain_tip_height(transaction),
159                        |h| Ok(Some(BlockHeight::from(h))),
160                    )
161                },
162                |h| Ok(Some(h)),
163            )?;
164
165            let zip212_enforcement = zip212_height.map_or_else(
166                || {
167                    // If the transaction has not been mined and the expiry height is set to 0 (no
168                    // expiry) an no chain tip information is available, then we assume it can only
169                    // be mined under ZIP 212 enforcement rules, so we default to `On`
170                    Zip212Enforcement::On
171                },
172                |h| zip212_enforcement(&self.params, h),
173            );
174
175            let ufvk_str: String = row.get(5)?;
176            let ufvk = UnifiedFullViewingKey::decode(&self.params, &ufvk_str).map_err(|e| {
177                WalletMigrationError::CorruptedData(format!("Stored UFVK was invalid: {:?}", e))
178            })?;
179
180            let dfvk = ufvk.sapling().ok_or_else(|| {
181                WalletMigrationError::CorruptedData(
182                    "UFVK must have a Sapling component to have received Sapling notes.".to_owned(),
183                )
184            })?;
185
186            // We previously set the default to external scope, so we now verify whether the output
187            // is decryptable using the intenally-scoped IVK and, if so, mark it as such.
188            if let Some(tx_data) = tx_data_opt {
189                let tx = Transaction::read(&tx_data[..], BranchId::Canopy).map_err(|e| {
190                    WalletMigrationError::CorruptedData(format!(
191                        "Unable to parse raw transaction: {:?}",
192                        e
193                    ))
194                })?;
195                let output = tx
196                    .sapling_bundle()
197                    .and_then(|b| b.shielded_outputs().get(output_index))
198                    .unwrap_or_else(|| {
199                        panic!("A Sapling output must exist at index {}", output_index)
200                    });
201
202                let pivk = PreparedIncomingViewingKey::new(&dfvk.to_ivk(Scope::Internal));
203                if try_sapling_note_decryption(&pivk, output, zip212_enforcement).is_some() {
204                    transaction.execute(
205                        "UPDATE sapling_received_notes SET recipient_key_scope = :scope
206                         WHERE id_note = :note_id",
207                        named_params! {":scope": scope_code(Scope::Internal), ":note_id": note_id},
208                    )?;
209                }
210            } else {
211                let diversifier = {
212                    let d: Vec<_> = row.get(6)?;
213                    Diversifier(d[..].try_into().map_err(|_| {
214                        WalletMigrationError::CorruptedData(
215                            "Invalid diversifier length".to_string(),
216                        )
217                    })?)
218                };
219
220                let note_value = Zatoshis::from_nonnegative_i64(row.get(7)?).map_err(|_e| {
221                    WalletMigrationError::CorruptedData(
222                        "Note values must be nonnegative".to_string(),
223                    )
224                })?;
225
226                let rseed = {
227                    let rcm_bytes: [u8; 32] =
228                        row.get::<_, Vec<u8>>(8)?[..].try_into().map_err(|_| {
229                            WalletMigrationError::CorruptedData(format!(
230                                "Note {} is invalid",
231                                note_id
232                            ))
233                        })?;
234
235                    let rcm = Option::from(jubjub::Fr::from_repr(rcm_bytes)).ok_or_else(|| {
236                        WalletMigrationError::CorruptedData(format!("Note {} is invalid", note_id))
237                    })?;
238
239                    // The wallet database always stores the `rcm` value, and not `rseed`,
240                    // so for note reconstruction we always use `BeforeZip212`.
241                    Rseed::BeforeZip212(rcm)
242                };
243
244                let note_commitment_tree_position =
245                    Position::from(u64::try_from(row.get::<_, i64>(9)?).map_err(|_| {
246                        WalletMigrationError::CorruptedData(
247                            "Note commitment tree position invalid.".to_string(),
248                        )
249                    })?);
250
251                let scope = select_note_scope(
252                    &mut commitment_tree,
253                    dfvk,
254                    &diversifier,
255                    &sapling::value::NoteValue::from_raw(note_value.into_u64()),
256                    &rseed,
257                    note_commitment_tree_position,
258                )?;
259
260                if scope == Some(Scope::Internal) {
261                    transaction.execute(
262                        "UPDATE sapling_received_notes SET recipient_key_scope = :scope
263                         WHERE id_note = :note_id",
264                        named_params! {":scope": scope_code(Scope::Internal), ":note_id": note_id},
265                    )?;
266                }
267            }
268        }
269
270        Ok(())
271    }
272
273    fn down(&self, _transaction: &rusqlite::Transaction) -> Result<(), WalletMigrationError> {
274        Err(WalletMigrationError::CannotRevert(MIGRATION_ID))
275    }

427    fn put_received_note_before_migration<T: ReceivedSaplingOutput<AccountId = AccountRef>>(
428        conn: &Connection,
429        output: &T,
430        tx_ref: i64,
431        spent_in: Option<i64>,
432    ) -> Result<(), SqliteClientError> {
433        let mut stmt_upsert_received_note = conn.prepare_cached(
434            "INSERT INTO sapling_received_notes
435            (tx, output_index, account, diversifier, value, rcm, memo, nf,
436             is_change, spent, commitment_tree_position)
437            VALUES (
438                :tx,
439                :output_index,
440                :account,
441                :diversifier,
442                :value,
443                :rcm,
444                :memo,
445                :nf,
446                :is_change,
447                :spent,
448                :commitment_tree_position
449            )
450            ON CONFLICT (tx, output_index) DO UPDATE
451            SET account = :account,
452                diversifier = :diversifier,
453                value = :value,
454                rcm = :rcm,
455                nf = IFNULL(:nf, nf),
456                memo = IFNULL(:memo, memo),
457                is_change = IFNULL(:is_change, is_change),
458                spent = IFNULL(:spent, spent),
459                commitment_tree_position = IFNULL(:commitment_tree_position, commitment_tree_position)",
460        )?;
461
462        let rcm = output.note().rcm().to_bytes();
463        let to = output.note().recipient();
464        let diversifier = to.diversifier();
465
466        let account = output.account_id();
467        let sql_args = named_params![
468            ":tx": &tx_ref,
469            ":output_index": i64::try_from(output.index()).expect("output indices are representable as i64"),
470            ":account": account.0,
471            ":diversifier": &diversifier.0,
472            ":value": output.note().value().inner(),
473            ":rcm": &rcm,
474            ":nf": output.nullifier().map(|nf| nf.0),
475            ":memo": memo_repr(output.memo()),
476            ":is_change": output.is_change(),
477            ":spent": spent_in,
478            ":commitment_tree_position": output.note_commitment_tree_position().map(u64::from),
479        ];
480
481        stmt_upsert_received_note
482            .execute(sql_args)
483            .map_err(SqliteClientError::from)?;
484
485        Ok(())
486    }

488    /// This reproduces [`crate::wallet::put_tx_data`] as it was at the time
489    /// of the creation of this migration.
490    fn put_tx_data(
491        conn: &rusqlite::Connection,
492        tx: &Transaction,
493        fee: Option<Zatoshis>,
494        created_at: Option<time::OffsetDateTime>,
495    ) -> Result<TxRef, SqliteClientError> {
496        let mut stmt_upsert_tx_data = conn.prepare_cached(
497            "INSERT INTO transactions (txid, created, expiry_height, raw, fee)
498            VALUES (:txid, :created_at, :expiry_height, :raw, :fee)
499            ON CONFLICT (txid) DO UPDATE
500            SET expiry_height = :expiry_height,
501                raw = :raw,
502                fee = IFNULL(:fee, fee)
503            RETURNING id_tx",
504        )?;
505
506        let txid = tx.txid();
507        let mut raw_tx = vec![];
508        tx.write(&mut raw_tx)?;
509
510        let tx_params = named_params![
511            ":txid": &txid.as_ref()[..],
512            ":created_at": created_at,
513            ":expiry_height": u32::from(tx.expiry_height()),
514            ":raw": raw_tx,
515            ":fee": fee.map(u64::from),
516        ];
517
518        stmt_upsert_tx_data
519            .query_row(tx_params, |row| row.get::<_, i64>(0).map(TxRef))
520            .map_err(SqliteClientError::from)
521    }

525        let params = Network::TestNetwork;
526
527        // Create wallet upgraded to just before the current migration.
528        let data_file = NamedTempFile::new().unwrap();
529        let mut db_data = WalletDb::for_path(data_file.path(), params).unwrap();
530        init_wallet_db_internal(
531            &mut db_data,

533            &[
534                add_account_birthdays::MIGRATION_ID,

537            false,
538        )
539        .unwrap();
540
541        let (ufvk0, height, res) = prepare_wallet_state(&mut db_data);
542        let tx = res.transaction();
543        let account_id = AccountRef(0);
544
545        // We can't use `decrypt_and_store_transaction` because we haven't migrated yet.
546        // Replicate its relevant innards here.
547        let d_tx = decrypt_transaction(
548            &params,
549            height,
550            tx,
551            &[(account_id, ufvk0)].into_iter().collect(),
552        );
553
554        db_data
555            .transactionally::<_, _, rusqlite::Error>(|wdb| {
556                let tx_ref = put_tx_data(wdb.conn.0, d_tx.tx(), None, None).unwrap();
557
558                let mut spending_account_id: Option<AccountRef> = None;
559
560                // Orchard outputs were not supported as of the wallet states that could require this
561                // migration.
562                for output in d_tx.sapling_outputs() {
563                    match output.transfer_type() {
564                        TransferType::Outgoing | TransferType::WalletInternal => {
565                            // Don't need to bother with sent outputs for this test.
566                            if output.transfer_type() != TransferType::Outgoing {
567                                put_received_note_before_migration(
568                                    wdb.conn.0, output, tx_ref.0, None,
569                                )
570                                .unwrap();
571                            }
572                        }
573                        TransferType::Incoming => {
574                            match spending_account_id {
575                                Some(id) => assert_eq!(id, *output.account()),
576                                None => {
577                                    spending_account_id = Some(*output.account());
578                                }
579                            }
580
581                            put_received_note_before_migration(wdb.conn.0, output, tx_ref.0, None)
582                                .unwrap();
583                        }
584                    }
585                }
586
587                Ok(())
588            })
589            .unwrap();
590
591        // Apply the current migration
592        init_wallet_db_internal(&mut db_data, None, &[super::MIGRATION_ID], false).unwrap();
593
594        // There should be two rows in the `sapling_received_notes` table with correct scopes.
595        let mut q = db_data
596            .conn
597            .prepare(
598                "SELECT value, recipient_key_scope
599                FROM sapling_received_notes",
600            )
601            .unwrap();
602        let mut rows = q.query([]).unwrap();
603        let mut row_count = 0;
604        while let Some(row) = rows.next().unwrap() {
605            row_count += 1;
606            let value: u64 = row.get(0).unwrap();
607            let scope = parse_scope(row.get(1).unwrap());
608            match value {
609                EXTERNAL_VALUE => assert_eq!(scope, Some(Scope::External)),
610                INTERNAL_VALUE => assert_eq!(scope, Some(Scope::Internal)),
611                _ => {
612                    panic!(
613                        "(Value, Scope) pair {:?} is not expected to exist in the wallet.",
614                        (value, scope),
615                    );
616                }
617            }
618        }
619        assert_eq!(row_count, 2);
620    }

623    fn receiving_key_scopes_migration_non_enhanced() {
624        let params = Network::TestNetwork;
625
626        // Create wallet upgraded to just before the current migration.
627        let data_file = NamedTempFile::new().unwrap();
628        let mut db_data = WalletDb::for_path(data_file.path(), params).unwrap();

630            &mut db_data,

633                wallet_summaries::MIGRATION_ID,

638        .unwrap();
639
640        let (ufvk0, height, res) = prepare_wallet_state(&mut db_data);
641        let tx = res.transaction();
642
643        let mut compact_tx = CompactTx {
644            hash: tx.txid().as_ref()[..].into(),
645            ..Default::default()
646        };
647        for output in tx.sapling_bundle().unwrap().shielded_outputs() {
648            compact_tx.outputs.push(output.into());
649        }
650        let prev_hash = BlockHash([4; 32]);
651        let mut block = CompactBlock {
652            height: height.into(),
653            hash: vec![7; 32],
654            prev_hash: prev_hash.0[..].into(),
655            ..Default::default()
656        };
657        block.vtx.push(compact_tx);
658        let scanning_keys = ScanningKeys::from_account_ufvks([(AccountRef(0), ufvk0)]);
659
660        let scanned_block = scan_block(
661            &params,
662            block,
663            &scanning_keys,
664            &Nullifiers::empty(),
665            Some(&BlockMetadata::from_parts(
666                height - 1,
667                prev_hash,
668                Some(0),
669                #[cfg(feature = "orchard")]
670                Some(0),
671            )),
672        )
673        .unwrap();
674
675        // We can't use `put_blocks` because we haven't migrated yet.
676        // Replicate its relevant innards here.
677        let blocks = [scanned_block];
678        db_data
679            .transactionally(|wdb| {
680                let start_positions = blocks.first().map(|block| {
681                    (
682                        block.height(),
683                        Position::from(
684                            u64::from(block.sapling().final_tree_size())
685                                - u64::try_from(block.sapling().commitments().len()).unwrap(),
686                        ),
687                    )
688                });
689                let mut sapling_commitments = vec![];
690                let mut last_scanned_height = None;
691                let mut note_positions = vec![];
692                for block in blocks.into_iter() {
693                    if last_scanned_height
694                        .iter()
695                        .any(|prev| block.height() != *prev + 1)
696                    {
697                        return Err(SqliteClientError::NonSequentialBlocks);
698                    }
699
700                    // Insert the block into the database.
701                    put_block(
702                        wdb.conn.0,
703                        block.height(),
704                        block.block_hash(),
705                        block.block_time(),
706                        block.sapling().final_tree_size(),
707                        block.sapling().commitments().len().try_into().unwrap(),
708                        #[cfg(feature = "orchard")]
709                        block.orchard().final_tree_size(),
710                        #[cfg(feature = "orchard")]
711                        block.orchard().commitments().len().try_into().unwrap(),
712                    )?;
713
714                    for tx in block.transactions() {
715                        let tx_row = put_tx_meta(wdb.conn.0, tx, block.height())?;
716
717                        for output in tx.sapling_outputs() {
718                            put_received_note_before_migration(wdb.conn.0, output, tx_row, None)?;
719                        }
720                    }
721
722                    note_positions.extend(block.transactions().iter().flat_map(|wtx| {
723                        wtx.sapling_outputs()
724                            .iter()
725                            .map(|out| out.note_commitment_tree_position())
726                    }));
727
728                    last_scanned_height = Some(block.height());
729                    let block_commitments = block.into_commitments();
730                    sapling_commitments.extend(block_commitments.sapling.into_iter().map(Some));
731                }
732
733                // We will have a start position and a last scanned height in all cases where
734                // `blocks` is non-empty.
735                if let Some(((_, start_position), _)) = start_positions.zip(last_scanned_height) {
736                    // Create subtrees from the note commitments in parallel.
737                    const CHUNK_SIZE: usize = 1024;
738                    let subtrees = sapling_commitments
739                        .par_chunks_mut(CHUNK_SIZE)
740                        .enumerate()
741                        .filter_map(|(i, chunk)| {
742                            let start = start_position + (i * CHUNK_SIZE) as u64;
743                            let end = start + chunk.len() as u64;
744
745                            shardtree::LocatedTree::from_iter(
746                                start..end,
747                                SAPLING_SHARD_HEIGHT.into(),
748                                chunk.iter_mut().map(|n| n.take().expect("always Some")),
749                            )
750                        })
751                        .map(|res| (res.subtree, res.checkpoints))
752                        .collect::<Vec<_>>();
753
754                    // Update the Sapling note commitment tree with all newly read note commitments
755                    let mut subtrees = subtrees.into_iter();
756                    wdb.with_sapling_tree_mut::<_, _, SqliteClientError>(move |sapling_tree| {
757                        for (tree, checkpoints) in &mut subtrees {
758                            sapling_tree.insert_tree(tree, checkpoints)?;
759                        }
760
761                        Ok(())
762                    })?;
763                }
764
765                Ok(())
766            })
767            .unwrap();
768
769        // Apply the current migration
770        init_wallet_db_internal(&mut db_data, None, &[super::MIGRATION_ID], false).unwrap();
771
772        // There should be two rows in the `sapling_received_notes` table with correct scopes.
773        let mut q = db_data
774            .conn
775            .prepare(
776                "SELECT value, recipient_key_scope
777                FROM sapling_received_notes",
778            )
779            .unwrap();
780        let mut rows = q.query([]).unwrap();
781        let mut row_count = 0;
782        while let Some(row) = rows.next().unwrap() {
783            row_count += 1;
784            let value: u64 = row.get(0).unwrap();
785            let scope = parse_scope(row.get(1).unwrap());
786            match value {
787                EXTERNAL_VALUE => assert_eq!(scope, Some(Scope::External)),
788                INTERNAL_VALUE => assert_eq!(scope, Some(Scope::Internal)),
789                _ => {
790                    panic!(
791                        "(Value, Scope) pair {:?} is not expected to exist in the wallet.",
792                        (value, scope),
793                    );
794                }
795            }
796        }
797        assert_eq!(row_count, 2);
798    }

800    /// This is a copy of [`crate::wallet::put_block`] as of the expected database
801    /// state corresponding to this migration. It is duplicated here as later
802    /// updates to the database schema require incompatible changes to `put_block`.
803    #[allow(clippy::too_many_arguments)]
804    fn put_block(
805        conn: &rusqlite::Transaction<'_>,
806        block_height: BlockHeight,
807        block_hash: BlockHash,
808        block_time: u32,
809        sapling_commitment_tree_size: u32,
810        sapling_output_count: u32,
811        #[cfg(feature = "orchard")] orchard_commitment_tree_size: u32,
812        #[cfg(feature = "orchard")] orchard_action_count: u32,
813    ) -> Result<(), SqliteClientError> {
814        let block_hash_data = conn
815            .query_row(
816                "SELECT hash FROM blocks WHERE height = ?",
817                [u32::from(block_height)],
818                |row| row.get::<_, Vec<u8>>(0),
819            )
820            .optional()?;
821
822        // Ensure that in the case of an upsert, we don't overwrite block data
823        // with information for a block with a different hash.
824        if let Some(bytes) = block_hash_data {
825            let expected_hash = BlockHash::try_from_slice(&bytes).ok_or_else(|| {
826                SqliteClientError::CorruptedData(format!(
827                    "Invalid block hash at height {}",
828                    u32::from(block_height)
829                ))
830            })?;
831            if expected_hash != block_hash {
832                return Err(SqliteClientError::BlockConflict(block_height));
833            }
834        }
835
836        let mut stmt_upsert_block = conn.prepare_cached(
837            "INSERT INTO blocks (
838                height,
839                hash,
840                time,
841                sapling_commitment_tree_size,
842                sapling_output_count,
843                sapling_tree,
844                orchard_commitment_tree_size,
845                orchard_action_count
846            )
847            VALUES (
848                :height,
849                :hash,
850                :block_time,
851                :sapling_commitment_tree_size,
852                :sapling_output_count,
853                x'00',
854                :orchard_commitment_tree_size,
855                :orchard_action_count
856            )
857            ON CONFLICT (height) DO UPDATE
858            SET hash = :hash,
859                time = :block_time,
860                sapling_commitment_tree_size = :sapling_commitment_tree_size,
861                sapling_output_count = :sapling_output_count,
862                orchard_commitment_tree_size = :orchard_commitment_tree_size,
863                orchard_action_count = :orchard_action_count",
864        )?;
865
866        #[cfg(not(feature = "orchard"))]
867        let orchard_commitment_tree_size: Option<u32> = None;
868        #[cfg(not(feature = "orchard"))]
869        let orchard_action_count: Option<u32> = None;
870
871        stmt_upsert_block.execute(named_params![
872            ":height": u32::from(block_height),
873            ":hash": &block_hash.0[..],
874            ":block_time": block_time,
875            ":sapling_commitment_tree_size": sapling_commitment_tree_size,
876            ":sapling_output_count": sapling_output_count,
877            ":orchard_commitment_tree_size": orchard_commitment_tree_size,
878            ":orchard_action_count": orchard_action_count,
879        ])?;
880
881        Ok(())
882    }

884    /// This is a copy of [`crate::wallet::put_tx_meta`] as of the expected database
885    /// state corresponding to this migration. It is duplicated here as later
886    /// updates to the database schema require incompatible changes to `put_tx_meta`.
887    pub(crate) fn put_tx_meta(
888        conn: &rusqlite::Connection,
889        tx: &WalletTx<AccountRef>,
890        height: BlockHeight,
891    ) -> Result<i64, SqliteClientError> {
892        // It isn't there, so insert our transaction into the database.
893        let mut stmt_upsert_tx_meta = conn.prepare_cached(
894            "INSERT INTO transactions (txid, block, tx_index)
895            VALUES (:txid, :block, :tx_index)
896            ON CONFLICT (txid) DO UPDATE
897            SET block = :block,
898                tx_index = :tx_index
899            RETURNING id_tx",
900        )?;
901
902        let txid_bytes = tx.txid();
903        let tx_params = named_params![
904            ":txid": &txid_bytes.as_ref()[..],
905            ":block": u32::from(height),
906            ":tx_index": i64::try_from(tx.block_index()).expect("transaction indices are representable as i64"),
907        ];
908
909        stmt_upsert_tx_meta
910            .query_row(tx_params, |row| row.get::<_, i64>(0))
911            .map_err(SqliteClientError::from)
912    }
