1//! Adds minimum observation height to transaction records.
2
3use rusqlite::Transaction;
4use schemerz_rusqlite::RusqliteMigration;
5use std::collections::HashSet;
6use uuid::Uuid;
7use zcash_protocol::consensus::{self, BlockHeight};
8
9use crate::wallet::{
10    init::{WalletMigrationError, migrations::fix_transparent_received_outputs},
11    mempool_height,
12};
13
14pub(super) const MIGRATION_ID: Uuid = Uuid::from_u128(0xab1be47e_dbfd_439a_876a_55a7e4a0ea0b);
15
16const DEPENDENCIES: &[Uuid] = &[fix_transparent_received_outputs::MIGRATION_ID];
17
18pub(super) struct Migration<P> {
19    pub(super) params: P,
20}
21
22impl<P> schemerz::Migration<Uuid> for Migration<P> {
23    fn id(&self) -> Uuid {
24        MIGRATION_ID
25    }
26
27    fn dependencies(&self) -> HashSet<Uuid> {
28        DEPENDENCIES.iter().copied().collect()
29    }
30
31    fn description(&self) -> &'static str {
32        "Adds minimum observation height to transaction records."
33    }
34}
35
36impl<P: consensus::Parameters> RusqliteMigration for Migration<P> {
37    type Error = WalletMigrationError;
38
39    fn up(&self, conn: &Transaction) -> Result<(), WalletMigrationError> {
40        // We use the target height as the minimum observed height for transactions
41        // for which we don't already have any other height information.
42        //
43        // chain_tip_height should only be `None` for a brand new wallet, in which case there
44        // should be no entries in the `transactions` table and the Sapling activation height will
45        // end up unused; we fall back to it just for defense in depth.
46        let fallback_height = u32::from(mempool_height(conn)?.map_or_else(
47            || {
48                self.params
49                    .activation_height(consensus::NetworkUpgrade::Sapling)
50                    .expect("sapling network upgrade has activated")
51            },
52            BlockHeight::from,
53        ));
54
55        conn.execute_batch(&format!(
56            r#"
57            PRAGMA legacy_alter_table = ON;
58
59            CREATE TABLE transactions_new (
60                id_tx INTEGER PRIMARY KEY,
61                txid BLOB NOT NULL UNIQUE,
62                created TEXT,
63                block INTEGER,
64                mined_height INTEGER,
65                tx_index INTEGER,
66                expiry_height INTEGER,
67                raw BLOB,
68                fee INTEGER,
69                target_height INTEGER,
70                min_observed_height INTEGER NOT NULL,
71                confirmed_unmined_at_height INTEGER,
72                FOREIGN KEY (block) REFERENCES blocks(height),
73                CONSTRAINT height_consistency CHECK (
74                    block IS NULL OR mined_height = block
75                ),
76                CONSTRAINT min_observed_consistency CHECK (
77                    mined_height IS NULL OR min_observed_height <= mined_height
78                ),
79                CONSTRAINT confirmed_unmined_consistency CHECK (
80                    confirmed_unmined_at_height IS NULL OR mined_height IS NULL
81                )
82            );
83
84            INSERT INTO transactions_new
85            SELECT
86                id_tx,
87                txid,
88                created,
89                block,
90                mined_height,
91                tx_index,
92                expiry_height,
93                raw,
94                fee,
95                target_height,
96                MIN(
97                    IFNULL(target_height, {fallback_height}),
98                    IFNULL(mined_height, {fallback_height}),
99                    {fallback_height}
100                ),
101                NULL  -- no transactions are initially confirmed as having expired without being mined
102            FROM transactions;
103
104            DROP TABLE transactions;
105            ALTER TABLE transactions_new RENAME TO transactions;
106
107            PRAGMA legacy_alter_table = OFF;
108            "#
109        ))?;
110
111        Ok(())
112    }
113
114    fn down(&self, _conn: &Transaction) -> Result<(), WalletMigrationError> {
115        Err(WalletMigrationError::CannotRevert(MIGRATION_ID))
116    }
117}
