1//! This migration adds tables to the wallet database that are needed to persist Sapling note
2//! commitment tree data using the `shardtree` crate, and migrates existing witness data into these
3//! data structures.
4

31pub(super) const MIGRATION_ID: Uuid = Uuid::from_u128(0x7da6489d_e835_4657_8be5_f512bcce6cbf);
32
33const DEPENDENCIES: &[Uuid] = &[received_notes_nullable_nf::MIGRATION_ID];
34
35pub(super) struct Migration<P> {
36    pub(super) params: P,
37}
38
39impl<P> schemerz::Migration<Uuid> for Migration<P> {
40    fn id(&self) -> Uuid {
41        MIGRATION_ID
42    }
43
44    fn dependencies(&self) -> HashSet<Uuid> {
45        DEPENDENCIES.iter().copied().collect()
46    }
47
48    fn description(&self) -> &'static str {
49        "Add support for receiving storage of note commitment tree data using the `shardtree` crate."
50    }
51}
52
53impl<P: consensus::Parameters> RusqliteMigration for Migration<P> {
54    type Error = WalletMigrationError;
55
56    fn up(&self, transaction: &rusqlite::Transaction) -> Result<(), WalletMigrationError> {
57        // Add commitment tree sizes to block metadata.
58        debug!("Adding new columns");
59        transaction.execute_batch(
60            "ALTER TABLE blocks ADD COLUMN sapling_commitment_tree_size INTEGER;
61             ALTER TABLE blocks ADD COLUMN orchard_commitment_tree_size INTEGER;
62             ALTER TABLE sapling_received_notes ADD COLUMN commitment_tree_position INTEGER;",
63        )?;
64
65        // Add shard persistence
66        debug!("Creating tables for shard persistence");
67        transaction.execute_batch(
68            "CREATE TABLE sapling_tree_shards (
69                shard_index INTEGER PRIMARY KEY,
70                subtree_end_height INTEGER,
71                root_hash BLOB,
72                shard_data BLOB,
73                contains_marked INTEGER,
74                CONSTRAINT root_unique UNIQUE (root_hash)
75            );
76            CREATE TABLE sapling_tree_cap (
77                -- cap_id exists only to be able to take advantage of `ON CONFLICT`
78                -- upsert functionality; the table will only ever contain one row
79                cap_id INTEGER PRIMARY KEY,
80                cap_data BLOB NOT NULL
81            );",
82        )?;
83
84        // Add checkpoint persistence
85        debug!("Creating tables for checkpoint persistence");
86        transaction.execute_batch(
87            "CREATE TABLE sapling_tree_checkpoints (
88                checkpoint_id INTEGER PRIMARY KEY,
89                position INTEGER
90            );
91            CREATE TABLE sapling_tree_checkpoint_marks_removed (
92                checkpoint_id INTEGER NOT NULL,
93                mark_removed_position INTEGER NOT NULL,
94                FOREIGN KEY (checkpoint_id) REFERENCES sapling_tree_checkpoints(checkpoint_id)
95                ON DELETE CASCADE,
96                CONSTRAINT spend_position_unique UNIQUE (checkpoint_id, mark_removed_position)
97            );",
98        )?;
99
100        let block_height_extrema = block_height_extrema(transaction)?;
101
102        let shard_store =
103            SqliteShardStore::<_, sapling::Node, SAPLING_SHARD_HEIGHT>::from_connection(
104                transaction,
105                SAPLING_TABLES_PREFIX,
106            )?;
107        let shard_store = CachingShardStore::load(shard_store).map_err(ShardTreeError::Storage)?;
108        let mut shard_tree: ShardTree<
109            _,
110            { sapling::NOTE_COMMITMENT_TREE_DEPTH },
111            SAPLING_SHARD_HEIGHT,
112        > = ShardTree::new(shard_store, PRUNING_DEPTH.try_into().unwrap());
113        // Insert all the tree information that we can get from block-end commitment trees
114        {
115            let mut stmt_blocks = transaction.prepare("SELECT height, sapling_tree FROM blocks")?;
116            let mut stmt_update_block_sapling_tree_size = transaction
117                .prepare("UPDATE blocks SET sapling_commitment_tree_size = ? WHERE height = ?")?;
118
119            let mut block_rows = stmt_blocks.query([])?;
120            while let Some(row) = block_rows.next()? {
121                let block_height: u32 = row.get(0)?;
122                let sapling_tree_data: Vec<u8> = row.get(1)?;
123
124                let block_end_tree = read_commitment_tree::<
125                    sapling::Node,
126                    _,
127                    { sapling::NOTE_COMMITMENT_TREE_DEPTH },
128                >(&sapling_tree_data[..])
129                .map_err(|e| {
130                    rusqlite::Error::FromSqlConversionFailure(
131                        sapling_tree_data.len(),
132                        rusqlite::types::Type::Blob,
133                        Box::new(e),
134                    )
135                })?;
136
137                if block_height % 1000 == 0 {
138                    debug!(height = block_height, "Migrating tree data to shardtree");
139                }
140                trace!(
141                    height = block_height,
142                    size = block_end_tree.size(),
143                    "Storing Sapling commitment tree size"
144                );
145                stmt_update_block_sapling_tree_size
146                    .execute(params![block_end_tree.size(), block_height])?;
147
148                // We only need to load frontiers into the ShardTree that are close enough
149                // to the wallet's known chain tip to fill `PRUNING_DEPTH` checkpoints, so
150                // that ShardTree's witness generation will be able to correctly handle
151                // anchor depths. Loading frontiers further back than this doesn't add any
152                // useful nodes to the ShardTree (as we don't support rollbacks beyond
153                // `PRUNING_DEPTH`, and we won't be finding notes in earlier blocks), and
154                // hurts performance (as frontier importing has a significant Merkle tree
155                // hashing cost).
156                if let Some((nonempty_frontier, scanned_range)) = block_end_tree
157                    .to_frontier()
158                    .value()
159                    .zip(block_height_extrema.as_ref())
160                {
161                    let block_height = BlockHeight::from(block_height);
162                    if block_height + PRUNING_DEPTH >= *scanned_range.end() {
163                        trace!(
164                            height = u32::from(block_height),
165                            frontier = ?nonempty_frontier,
166                            "Inserting frontier nodes",
167                        );
168                        shard_tree
169                            .insert_frontier_nodes(
170                                nonempty_frontier.clone(),
171                                Retention::Checkpoint {
172                                    id: block_height,
173                                    marking: Marking::Reference,
174                                },
175                            )
176                            .map_err(|e| match e {
177                                ShardTreeError::Query(e) => ShardTreeError::Query(e),
178                                ShardTreeError::Insert(e) => ShardTreeError::Insert(e),
179                                ShardTreeError::Storage(_) => unreachable!(),
180                            })?
181                    }
182                }
183            }
184        }
185
186        // Insert all the tree information that we can get from existing incremental witnesses
187        debug!("Migrating witness data to shardtree");
188        {
189            let mut stmt_blocks =
190                transaction.prepare("SELECT note, block, witness FROM sapling_witnesses")?;
191            let mut stmt_set_note_position = transaction.prepare(
192                "UPDATE sapling_received_notes
193                SET commitment_tree_position = :position
194                WHERE id_note = :note_id",
195            )?;
196            let mut updated_note_positions = BTreeSet::new();
197            let mut rows = stmt_blocks.query([])?;
198            while let Some(row) = rows.next()? {
199                let note_id: i64 = row.get(0)?;
200                let block_height: u32 = row.get(1)?;
201                let row_data: Vec<u8> = row.get(2)?;
202                let witness = read_incremental_witness::<
203                    sapling::Node,
204                    _,
205                    { sapling::NOTE_COMMITMENT_TREE_DEPTH },
206                >(&row_data[..])
207                .map_err(|e| {
208                    rusqlite::Error::FromSqlConversionFailure(
209                        row_data.len(),
210                        rusqlite::types::Type::Blob,
211                        Box::new(e),
212                    )
213                })?;
214
215                let witnessed_position = witness.witnessed_position();
216                if !updated_note_positions.contains(&witnessed_position) {
217                    stmt_set_note_position.execute(named_params![
218                        ":note_id": note_id,
219                        ":position": u64::from(witnessed_position)
220                    ])?;
221                    updated_note_positions.insert(witnessed_position);
222                }
223
224                shard_tree
225                    .insert_witness_nodes(witness, BlockHeight::from(block_height))
226                    .map_err(|e| match e {
227                        ShardTreeError::Query(e) => ShardTreeError::Query(e),
228                        ShardTreeError::Insert(e) => ShardTreeError::Insert(e),
229                        ShardTreeError::Storage(_) => unreachable!(),
230                    })?;
231            }
232        }
233
234        shard_tree
235            .into_store()
236            .flush()
237            .map_err(ShardTreeError::Storage)?;
238
239        // Establish the scan queue & wallet history table.
240        // block_range_end is exclusive.
241        debug!("Creating table for scan queue");
242        transaction.execute_batch(
243            "CREATE TABLE scan_queue (
244                block_range_start INTEGER NOT NULL,
245                block_range_end INTEGER NOT NULL,
246                priority INTEGER NOT NULL,
247                CONSTRAINT range_start_uniq UNIQUE (block_range_start),
248                CONSTRAINT range_end_uniq UNIQUE (block_range_end),
249                CONSTRAINT range_bounds_order CHECK (
250                    block_range_start < block_range_end
251                )
252            );",
253        )?;
254
255        if let Some(scanned_range) = block_height_extrema {
256            // `ScanRange` uses an exclusive upper bound.
257            let start = *scanned_range.start();
258            let chain_end = *scanned_range.end() + 1;
259            let ignored_range =
260                self.params
261                    .activation_height(NetworkUpgrade::Sapling)
262                    .map(|sapling_activation| {
263                        let ignored_range_start = std::cmp::min(sapling_activation, start);
264                        ScanRange::from_parts(ignored_range_start..start, ScanPriority::Ignored)
265                    });
266            let scanned_range = ScanRange::from_parts(start..chain_end, ScanPriority::Scanned);
267            insert_queue_entries(
268                transaction,
269                ignored_range.iter().chain(Some(scanned_range).iter()),
270            )?;
271        }
272
273        Ok(())
274    }
275
276    fn down(&self, _transaction: &rusqlite::Transaction) -> Result<(), WalletMigrationError> {
277        Err(WalletMigrationError::CannotRevert(MIGRATION_ID))
278    }
279}
