62impl<P: consensus::Parameters> RusqliteMigration for Migration<P> {
63    type Error = WalletMigrationError;
64
65    fn up(&self, conn: &Transaction) -> Result<(), WalletMigrationError> {
66        conn.execute_batch(
67            "CREATE TABLE tx_retrieval_queue (
68                txid BLOB NOT NULL UNIQUE,
69                query_type INTEGER NOT NULL,
70                dependent_transaction_id INTEGER,
71                FOREIGN KEY (dependent_transaction_id) REFERENCES transactions(id_tx)
72            );
73
74            ALTER TABLE transactions ADD COLUMN target_height INTEGER;
75
76            CREATE TABLE transparent_spend_search_queue (
77                address TEXT NOT NULL,
78                transaction_id INTEGER NOT NULL,
79                output_index INTEGER NOT NULL,
80                FOREIGN KEY (transaction_id) REFERENCES transactions(id_tx),
81                CONSTRAINT value_received_height UNIQUE (transaction_id, output_index)
82            );
83
84            CREATE TABLE transparent_spend_map (
85                spending_transaction_id INTEGER NOT NULL,
86                prevout_txid BLOB NOT NULL,
87                prevout_output_index INTEGER NOT NULL,
88                FOREIGN KEY (spending_transaction_id) REFERENCES transactions(id_tx)
89                -- NOTE: We can't create a unique constraint on just (prevout_txid, prevout_output_index) 
90                -- because the same output may be attempted to be spent in multiple transactions, even 
91                -- though only one will ever be mined.
92                CONSTRAINT transparent_spend_map_unique UNIQUE (
93                    spending_transaction_id, prevout_txid, prevout_output_index
94                )
95            );",
96        )?;
97
98        // Add estimated target height information for each transaction we know to
99        // have been created by the wallet; transactions that were discovered via
100        // chain scanning will have their `created` field set to `NULL`.
101        conn.execute(
102            "UPDATE transactions
103             SET target_height = expiry_height - :default_expiry_delta
104             WHERE expiry_height > :default_expiry_delta
105             AND created IS NOT NULL",
106            named_params![":default_expiry_delta": DEFAULT_TX_EXPIRY_DELTA],
107        )?;
108
109        // Populate the enhancement queues with any transparent history information that we don't
110        // already have.
111        #[cfg(feature = "transparent-inputs")]
112        {
113            let mut stmt_transactions =
114                conn.prepare("SELECT id_tx, raw, mined_height FROM transactions")?;
115            let mut rows = stmt_transactions.query([])?;
116            while let Some(row) = rows.next()? {
117                let tx_ref = row.get(0).map(TxRef)?;
118                let tx_data = row.get::<_, Option<Vec<u8>>>(1)?;
119                let mined_height = row.get::<_, Option<u32>>(2)?.map(BlockHeight::from);
120
121                if let Some(tx_data) = tx_data {
122                    let tx = zcash_primitives::transaction::Transaction::read(
123                        &tx_data[..],
124                        // We assume unmined transactions are created with the current consensus branch ID.
125                        mined_height.map_or(BranchId::Sapling, |h| {
126                            BranchId::for_height(&self._params, h)
127                        }),
128                    )
129                    .map_err(|_| {
130                        WalletMigrationError::CorruptedData(
131                            "Could not read serialized transaction data.".to_owned(),
132                        )
133                    })?;
134
135                    for (txout, output_index) in tx
136                        .transparent_bundle()
137                        .iter()
138                        .flat_map(|b| b.vout.iter())
139                        .zip(0u32..)
140                    {
141                        if let Some(address) = txout.recipient_address() {
142                            let find_address_account = || {
143                                conn.query_row(
144                                    "SELECT account_id FROM addresses
145                                     WHERE cached_transparent_receiver_address = :address
146                                     UNION
147                                     SELECT account_id from ephemeral_addresses
148                                     WHERE address = :address",
149                                    named_params![":address": address.encode(&self._params)],
150                                    |row| row.get(0).map(AccountRef),
151                                )
152                                .optional()
153                            };
154                            let find_legacy_address_account =
155                                || -> Result<Option<AccountRef>, SqliteClientError> {
156                                    let mut stmt = conn.prepare("SELECT id, uivk FROM accounts")?;
157                                    let mut rows = stmt.query([])?;
158                                    while let Some(row) = rows.next()? {
159                                        let account_id = row.get(0).map(AccountRef)?;
160                                        let uivk_str = row.get::<_, String>(1)?;
161
162                                        if let Some((legacy_taddr, _)) =
163                                            uivk_legacy_transparent_address(
164                                                &self._params,
165                                                &uivk_str,
166                                            )?
167                                        {
168                                            if legacy_taddr == address {
169                                                return Ok(Some(account_id));
170                                            }
171                                        }
172                                    }
173
174                                    Ok(None)
175                                };
176
177                            if find_address_account()?.is_some()
178                                || find_legacy_address_account()?.is_some()
179                            {
180                                queue_transparent_spend_detection(
181                                    conn,
182                                    &self._params,
183                                    address,
184                                    tx_ref,
185                                    output_index,
186                                )?
187                            }
188                        }
189                    }
190
191                    let d_tx = DecryptedTransaction::<'_, Infallible>::new(
192                        mined_height,
193                        &tx,
194                        vec![],
195                        #[cfg(feature = "orchard")]
196                        vec![],
197                    );
198
199                    queue_transparent_input_retrieval(conn, tx_ref, &d_tx)?;
200                    queue_unmined_tx_retrieval(conn, &d_tx)?;
201                }
202            }
203        }
204
205        Ok(())
206    }
207
208    fn down(&self, conn: &Transaction) -> Result<(), WalletMigrationError> {
209        conn.execute_batch(
210            "DROP TABLE transparent_spend_map;
211             DROP TABLE transparent_spend_search_queue;
212             ALTER TABLE transactions DROP COLUMN target_height;
213             DROP TABLE tx_retrieval_queue;",
214        )?;
215
216        Ok(())
217    }
218}

250    fn migrate_with_data() {
251        let data_file = NamedTempFile::new().unwrap();
252        let mut db_data = WalletDb::for_path(
253            data_file.path(),
254            Network::TestNetwork,
255            test_clock(),
256            test_rng(),
257        )
258        .unwrap();
259
260        let seed_bytes = vec![0xab; 32];
261
262        // Migrate to database state just prior to this migration.
263        WalletMigrator::new()
264            .with_seed(Secret::new(seed_bytes.clone()))
265            .ignore_seed_relevance()
266            .init_or_migrate_to(&mut db_data, DEPENDENCIES)
267            .unwrap();
268
269        // Add transactions to the wallet that exercise the data migration.
270        let add_tx_to_wallet = |tx: TransactionData<Authorized>| {
271            let tx = tx.freeze().unwrap();
272            let txid = tx.txid();
273            let mut raw_tx = vec![];
274            tx.write(&mut raw_tx).unwrap();
275            db_data
276                .conn
277                .execute(
278                    r#"INSERT INTO transactions (txid, raw) VALUES (:txid, :raw);"#,
279                    named_params! {":txid": txid.as_ref(), ":raw": raw_tx},
280                )
281                .unwrap();
282        };
283        add_tx_to_wallet(TransactionData::from_parts(
284            TxVersion::V5,
285            BranchId::Nu5,
286            0,
287            12345678.into(),
288            Some(transparent::bundle::Bundle {
289                vin: vec![TxIn {
290                    prevout: OutPoint::fake(),
291                    script_sig: Script(vec![]),
292                    sequence: 0,
293                }],
294                vout: vec![TxOut {
295                    value: Zatoshis::const_from_u64(10_000),
296                    script_pubkey: TransparentAddress::PublicKeyHash([7; 20]).script(),
297                }],
298                authorization: transparent::bundle::Authorized,
299            }),
300            None,
301            None,
302            None,
303        ));
304
305        // Check that we can apply this migration.
306        WalletMigrator::new()
307            .with_seed(Secret::new(seed_bytes))
308            .ignore_seed_relevance()
309            .init_or_migrate_to(&mut db_data, &[MIGRATION_ID])
310            .unwrap();
311    }
