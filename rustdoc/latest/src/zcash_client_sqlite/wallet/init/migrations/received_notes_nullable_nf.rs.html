1//! A migration that renames the `received_notes` table to `sapling_received_notes`
2//! and makes the `nf` column nullable. This allows change notes to be added to the
3//! table prior to being mined.
4use std::collections::HashSet;
5
6use schemerz_rusqlite::RusqliteMigration;
7use uuid::Uuid;
8
9use super::v_transactions_net;
10use crate::wallet::init::WalletMigrationError;
11
12pub(super) const MIGRATION_ID: Uuid = Uuid::from_u128(0xbdcdcedc_7b29_4f1c_8307_35f937f0d32a);
13
14const DEPENDENCIES: &[Uuid] = &[v_transactions_net::MIGRATION_ID];
15
16pub(crate) struct Migration;
17
18impl schemerz::Migration<Uuid> for Migration {
19    fn id(&self) -> Uuid {
20        MIGRATION_ID
21    }
22
23    fn dependencies(&self) -> HashSet<Uuid> {
24        DEPENDENCIES.iter().copied().collect()
25    }
26
27    fn description(&self) -> &'static str {
28        "Rename `received_notes` to `sapling_received_notes` and make the `nf` column nullable."
29    }
30}
31
32impl RusqliteMigration for Migration {
33    type Error = WalletMigrationError;
34
35    fn up(&self, transaction: &rusqlite::Transaction) -> Result<(), WalletMigrationError> {
36        // As of this migration, the `received_notes` table only stores Sapling notes, so
37        // we can hard-code the `output_pool` value.
38        transaction.execute_batch(
39            "CREATE TABLE sapling_received_notes (
40                id_note INTEGER PRIMARY KEY,
41                tx INTEGER NOT NULL,
42                output_index INTEGER NOT NULL,
43                account INTEGER NOT NULL,
44                diversifier BLOB NOT NULL,
45                value INTEGER NOT NULL,
46                rcm BLOB NOT NULL,
47                nf BLOB UNIQUE,
48                is_change INTEGER NOT NULL,
49                memo BLOB,
50                spent INTEGER,
51                FOREIGN KEY (tx) REFERENCES transactions(id_tx),
52                FOREIGN KEY (account) REFERENCES accounts(account),
53                FOREIGN KEY (spent) REFERENCES transactions(id_tx),
54                CONSTRAINT tx_output UNIQUE (tx, output_index)
55            );
56            INSERT INTO sapling_received_notes SELECT * FROM received_notes;",
57        )?;
58
59        transaction.execute_batch(
60            "ALTER TABLE sapling_witnesses RENAME TO sapling_witnesses_old;
61            CREATE TABLE sapling_witnesses (
62                id_witness INTEGER PRIMARY KEY,
63                note INTEGER NOT NULL,
64                block INTEGER NOT NULL,
65                witness BLOB NOT NULL,
66                FOREIGN KEY (note) REFERENCES sapling_received_notes(id_note),
67                FOREIGN KEY (block) REFERENCES blocks(height),
68                CONSTRAINT witness_height UNIQUE (note, block)
69            );
70            INSERT INTO sapling_witnesses SELECT * FROM sapling_witnesses_old;
71            DROP TABLE sapling_witnesses_old;",
72        )?;
73
74        transaction.execute_batch(
75            "DROP VIEW v_tx_outputs;
76            CREATE VIEW v_tx_outputs AS
77            SELECT sapling_received_notes.tx           AS id_tx,
78                   2                                   AS output_pool,
79                   sapling_received_notes.output_index AS output_index,
80                   sent_notes.from_account             AS from_account,
81                   sapling_received_notes.account      AS to_account,
82                   NULL                                AS to_address,
83                   sapling_received_notes.value        AS value,
84                   sapling_received_notes.is_change    AS is_change,
85                   sapling_received_notes.memo         AS memo
86            FROM sapling_received_notes
87            LEFT JOIN sent_notes
88                      ON (sent_notes.tx, sent_notes.output_pool, sent_notes.output_index) =
89                         (sapling_received_notes.tx, 2, sent_notes.output_index)
90            UNION
91            SELECT transactions.id_tx          AS id_tx,
92                   0                           AS output_pool,
93                   utxos.prevout_idx           AS output_index,
94                   NULL                        AS from_account,
95                   utxos.received_by_account   AS to_account,
96                   utxos.address               AS to_address,
97                   utxos.value_zat             AS value,
98                   false                       AS is_change,
99                   NULL                        AS memo
100            FROM utxos
101            JOIN transactions
102                 ON transactions.txid = utxos.prevout_txid
103            UNION
104            SELECT sent_notes.tx                  AS id_tx,
105                   sent_notes.output_pool         AS output_pool,
106                   sent_notes.output_index        AS output_index,
107                   sent_notes.from_account        AS from_account,
108                   sapling_received_notes.account AS to_account,
109                   sent_notes.to_address          AS to_address,
110                   sent_notes.value               AS value,
111                   false                          AS is_change,
112                   sent_notes.memo                AS memo
113            FROM sent_notes
114            LEFT JOIN sapling_received_notes
115                      ON (sent_notes.tx, sent_notes.output_pool, sent_notes.output_index) =
116                         (sapling_received_notes.tx, 2, sapling_received_notes.output_index)
117            WHERE  sapling_received_notes.is_change IS NULL
118               OR  sapling_received_notes.is_change = 0",
119        )?;
120
121        transaction.execute_batch(
122            "DROP VIEW v_transactions;
123            CREATE VIEW v_transactions AS
124            WITH
125            notes AS (
126                SELECT sapling_received_notes.account        AS account_id,
127                       sapling_received_notes.tx             AS id_tx,
128                       2                             AS pool,
129                       sapling_received_notes.value          AS value,
130                       CASE
131                            WHEN sapling_received_notes.is_change THEN 1
132                            ELSE 0
133                       END AS is_change,
134                       CASE
135                            WHEN sapling_received_notes.is_change THEN 0
136                            ELSE 1
137                       END AS received_count,
138                       CASE
139                           WHEN sapling_received_notes.memo IS NULL THEN 0
140                           ELSE 1
141                       END AS memo_present
142                FROM   sapling_received_notes
143                UNION
144                SELECT utxos.received_by_account     AS account_id,
145                       transactions.id_tx            AS id_tx,
146                       0                             AS pool,
147                       utxos.value_zat               AS value,
148                       0                             AS is_change,
149                       1                             AS received_count,
150                       0                             AS memo_present
151                FROM utxos
152                JOIN transactions
153                     ON transactions.txid = utxos.prevout_txid
154                UNION
155                SELECT sapling_received_notes.account        AS account_id,
156                       sapling_received_notes.spent          AS id_tx,
157                       2                             AS pool,
158                       -sapling_received_notes.value         AS value,
159                       0                             AS is_change,
160                       0                             AS received_count,
161                       0                             AS memo_present
162                FROM   sapling_received_notes
163                WHERE  sapling_received_notes.spent IS NOT NULL
164            ),
165            sent_note_counts AS (
166                SELECT sent_notes.from_account AS account_id,
167                       sent_notes.tx AS id_tx,
168                       COUNT(DISTINCT sent_notes.id_note) as sent_notes,
169                       SUM(
170                         CASE
171                             WHEN sent_notes.memo IS NULL THEN 0
172                             ELSE 1
173                         END
174                       ) AS memo_count
175                FROM sent_notes
176                LEFT JOIN sapling_received_notes
177                          ON (sent_notes.tx, sent_notes.output_pool, sent_notes.output_index) =
178                             (sapling_received_notes.tx, 2, sapling_received_notes.output_index)
179                WHERE  sapling_received_notes.is_change IS NULL
180                   OR  sapling_received_notes.is_change = 0
181                GROUP BY account_id, id_tx
182            ),
183            blocks_max_height AS (
184                SELECT MAX(blocks.height) as max_height FROM blocks
185            )
186            SELECT notes.account_id                  AS account_id,
187                   transactions.id_tx                AS id_tx,
188                   transactions.block                AS mined_height,
189                   transactions.tx_index             AS tx_index,
190                   transactions.txid                 AS txid,
191                   transactions.expiry_height        AS expiry_height,
192                   transactions.raw                  AS raw,
193                   SUM(notes.value)                  AS account_balance_delta,
194                   transactions.fee                  AS fee_paid,
195                   SUM(notes.is_change) > 0          AS has_change,
196                   MAX(COALESCE(sent_note_counts.sent_notes, 0))  AS sent_note_count,
197                   SUM(notes.received_count)         AS received_note_count,
198                   SUM(notes.memo_present) + MAX(COALESCE(sent_note_counts.memo_count, 0)) AS memo_count,
199                   blocks.time                       AS block_time,
200                   (
201                        blocks.height IS NULL
202                        AND transactions.expiry_height <= blocks_max_height.max_height
203                   ) AS expired_unmined
204            FROM transactions
205            JOIN notes ON notes.id_tx = transactions.id_tx
206            JOIN blocks_max_height
207            LEFT JOIN blocks ON blocks.height = transactions.block
208            LEFT JOIN sent_note_counts
209                      ON sent_note_counts.account_id = notes.account_id
210                      AND sent_note_counts.id_tx = notes.id_tx
211            GROUP BY notes.account_id, transactions.id_tx",
212        )?;
213
214        transaction.execute_batch("DROP TABLE received_notes;")?;
215
216        Ok(())
217    }
218
219    fn down(&self, _transaction: &rusqlite::Transaction) -> Result<(), WalletMigrationError> {
220        Err(WalletMigrationError::CannotRevert(MIGRATION_ID))
221    }
222}

224#[cfg(test)]
225mod tests {
226    use rusqlite::{self, params};
227    use tempfile::NamedTempFile;
228
229    use zcash_keys::keys::UnifiedSpendingKey;
230    use zcash_protocol::consensus::Network;
231    use zip32::AccountId;
232
233    use crate::{
234        WalletDb,
235        testing::db::{test_clock, test_rng},
236        wallet::init::{WalletMigrator, migrations::v_transactions_net},
237    };
238
239    #[test]
240    fn received_notes_nullable_migration() {
241        let data_file = NamedTempFile::new().unwrap();
242        let mut db_data = WalletDb::for_path(
243            data_file.path(),
244            Network::TestNetwork,
245            test_clock(),
246            test_rng(),
247        )
248        .unwrap();
249        WalletMigrator::new()
250            .ignore_seed_relevance()
251            .init_or_migrate_to(&mut db_data, &[v_transactions_net::MIGRATION_ID])
252            .unwrap();
253
254        // Create an account in the wallet
255        let usk0 = UnifiedSpendingKey::from_seed(&db_data.params, &[0u8; 32][..], AccountId::ZERO)
256            .unwrap();
257        let ufvk0 = usk0.to_unified_full_viewing_key();
258        db_data
259            .conn
260            .execute(
261                "INSERT INTO accounts (account, ufvk) VALUES (0, ?)",
262                params![ufvk0.encode(&db_data.params)],
263            )
264            .unwrap();
265
266        // Tx 0 contains two received notes of 2 and 5 zatoshis that are controlled by account 0.
267        db_data.conn.execute_batch(
268            "INSERT INTO blocks (height, hash, time, sapling_tree) VALUES (0, 0, 0, x'00');
269            INSERT INTO transactions (block, id_tx, txid) VALUES (0, 0, 'tx0');
270
271            INSERT INTO received_notes (tx, output_index, account, diversifier, value, rcm, nf, is_change)
272            VALUES (0, 0, 0, '', 2, '', 'nf_a', false);
273            INSERT INTO received_notes (tx, output_index, account, diversifier, value, rcm, nf, is_change)
274            VALUES (0, 3, 0, '', 5, '', 'nf_b', false);").unwrap();
275
276        // Apply the current migration
277        WalletMigrator::new()
278            .ignore_seed_relevance()
279            .init_or_migrate_to(&mut db_data, &[super::MIGRATION_ID])
280            .unwrap();
281
282        {
283            let mut q = db_data
284                .conn
285                .prepare(
286                    "SELECT account_id, id_tx, account_balance_delta, has_change, memo_count, sent_note_count, received_note_count
287                    FROM v_transactions",
288                )
289                .unwrap();
290            let mut rows = q.query([]).unwrap();
291            let mut row_count = 0;
292            while let Some(row) = rows.next().unwrap() {
293                row_count += 1;
294                let account: i64 = row.get(0).unwrap();
295                let tx: i64 = row.get(1).unwrap();
296                let account_balance_delta: i64 = row.get(2).unwrap();
297                let has_change: bool = row.get(3).unwrap();
298                let memo_count: i64 = row.get(4).unwrap();
299                let sent_note_count: i64 = row.get(5).unwrap();
300                let received_note_count: i64 = row.get(6).unwrap();
301                match (account, tx) {
302                    (0, 0) => {
303                        assert_eq!(account_balance_delta, 7);
304                        assert!(!has_change);
305                        assert_eq!(memo_count, 0);
306                        assert_eq!(sent_note_count, 0);
307                        assert_eq!(received_note_count, 2);
308                    }
309                    other => {
310                        panic!(
311                            "(Account, Transaction) pair {other:?} is not expected to exist in the wallet."
312                        );
313                    }
314                }
315            }
316            assert_eq!(row_count, 1);
317        }
318
319        // Now create an unmined transaction that spends both of our notes
320        db_data.conn.execute_batch(
321            "INSERT INTO transactions (id_tx, txid) VALUES (1, 'tx1');
322
323            -- Mark our existing notes as spent
324            UPDATE sapling_received_notes SET spent = 1 WHERE tx = 0;
325
326            -- The note sent to the external recipient.
327            INSERT INTO sent_notes (tx, output_pool, output_index, from_account, to_account, to_address, value)
328            VALUES (1, 2, 0, 0, NULL, 'zfake', 4);
329
330            -- The change notes. We send two notes to ensure that having multiple rows with NULL nullifiers
331            -- does not violate the uniqueness constraint
332            INSERT INTO sent_notes (tx, output_pool, output_index, from_account, to_account, to_address, value)
333            VALUES (1, 2, 1, 0, 0, NULL, 1);
334            INSERT INTO sent_notes (tx, output_pool, output_index, from_account, to_account, to_address, value)
335            VALUES (1, 2, 2, 0, 0, NULL, 1);
336            INSERT INTO sapling_received_notes (tx, output_index, account, diversifier, value, rcm, nf, is_change)
337            VALUES (1,    1, 0, '', 1, '', NULL, true);
338            INSERT INTO sapling_received_notes (tx, output_index, account, diversifier, value, rcm, nf, is_change)
339            VALUES (1,    2, 0, '', 1, '', NULL, true);
340            ").unwrap();
341        {
342            let mut q = db_data
343                .conn
344                .prepare(
345                    "SELECT account_id, id_tx, account_balance_delta, has_change, memo_count, sent_note_count, received_note_count
346                    FROM v_transactions",
347                )
348                .unwrap();
349            let mut rows = q.query([]).unwrap();
350            let mut row_count = 0;
351            while let Some(row) = rows.next().unwrap() {
352                row_count += 1;
353                let account: i64 = row.get(0).unwrap();
354                let tx: i64 = row.get(1).unwrap();
355                let account_balance_delta: i64 = row.get(2).unwrap();
356                let has_change: bool = row.get(3).unwrap();
357                let memo_count: i64 = row.get(4).unwrap();
358                let sent_note_count: i64 = row.get(5).unwrap();
359                let received_note_count: i64 = row.get(6).unwrap();
360                match (account, tx) {
361                    (0, 0) => {
362                        assert_eq!(account_balance_delta, 7);
363                        assert!(!has_change);
364                        assert_eq!(memo_count, 0);
365                        assert_eq!(sent_note_count, 0);
366                        assert_eq!(received_note_count, 2);
367                    }
368                    (0, 1) => {
369                        assert_eq!(account_balance_delta, -6);
370                        assert!(has_change);
371                        assert_eq!(memo_count, 0);
372                        assert_eq!(sent_note_count, 1);
373                        assert_eq!(received_note_count, 0);
374                    }
375                    other => {
376                        panic!(
377                            "(Account, Transaction) pair {other:?} is not expected to exist in the wallet."
378                        );
379                    }
380                }
381            }
382            assert_eq!(row_count, 2);
383        }
384    }
