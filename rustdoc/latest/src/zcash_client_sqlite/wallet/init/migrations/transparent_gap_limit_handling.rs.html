63// For each account, ensure that all diversifier indexes prior to that for the default
64// address have corresponding cached transparent addresses.
65#[cfg(feature = "transparent-inputs")]
66pub(super) fn insert_initial_transparent_addrs<P: consensus::Parameters>(
67    conn: &rusqlite::Transaction,
68    params: &P,
69) -> Result<(), WalletMigrationError> {
70    let mut min_addr_diversifiers = conn.prepare(
71        r#"
72        SELECT accounts.id AS account_id,
73               MIN(addresses.transparent_child_index) AS transparent_child_index,
74               MIN(addresses.diversifier_index_be) AS diversifier_index_be
75        FROM accounts
76        LEFT OUTER JOIN addresses
77            ON accounts.id = addresses.account_id
78            AND addresses.key_scope = :key_scope_external
79        GROUP BY accounts.id
80        "#,
81    )?;
82
83    let mut min_addr_rows = min_addr_diversifiers.query(named_params![
84        ":key_scope_external": KeyScope::EXTERNAL.encode()
85    ])?;
86    while let Some(row) = min_addr_rows.next()? {
87        let account_id = AccountRef(row.get("account_id")?);
88
89        let min_transparent_idx = row
90            .get::<_, Option<u32>>("transparent_child_index")?
91            .map(|i| {
92                NonHardenedChildIndex::from_index(i).ok_or(WalletMigrationError::CorruptedData(
93                    format!("{i} is not a valid transparent child index."),
94                ))
95            })
96            .transpose()?;
97
98        let min_diversifier_idx = row
99            .get::<_, Option<Vec<u8>>>("diversifier_index_be")?
100            .map(|b| decode_diversifier_index_be(&b[..]))
101            .transpose()?
102            .and_then(|di| NonHardenedChildIndex::try_from(di).ok());
103
104        // Ensure that there is an address for each possible external address index prior to the
105        // default UA for the account. If the default address has a diversifier index greater than
106        // the gap limit, we generate transparent addresses up to that index but not beyond.
107        let start = NonHardenedChildIndex::const_from_index(0);
108        let end = std::cmp::min(
109            min_transparent_idx
110                .or(min_diversifier_idx)
111                // guarantee that we have an entry at index 0; this address will have previously
112                // been provided explicitly as one of the wallet's addresses in response to a call
113                // to `get_transparent_receivers, even if we have no other addresses generated
114                // (which shouldn't ordinarily be the case anyway)
115                .unwrap_or(NonHardenedChildIndex::const_from_index(1)),
116            NonHardenedChildIndex::from_index(GapLimits::default().external())
117                .expect("default external gap limit fits in non-hardened child index space."),
118        );
119
120        generate_address_range(
121            conn,
122            params,
123            account_id,
124            KeyScope::EXTERNAL,
125            UnifiedAddressRequest::ALLOW_ALL,
126            start..end,
127            false,
128        )?;
129    }
130
131    Ok(())
132}

134impl<P: consensus::Parameters, C: Clock, R: RngCore> RusqliteMigration for Migration<P, C, R> {
135    type Error = WalletMigrationError;
136
137    fn up(&self, conn: &Transaction) -> Result<(), WalletMigrationError> {
138        let decode_uivk = |uivk_str: String| {
139            UnifiedIncomingViewingKey::decode(&self.params, &uivk_str).map_err(|e| {
140                WalletMigrationError::CorruptedData(format!(
141                    "Invalid UIVK encoding {uivk_str}: {e}"
142                ))
143            })
144        };
145
146        let external_scope_code = KeyScope::EXTERNAL.encode();
147
148        conn.execute_batch(&format!(
149            r#"
150            ALTER TABLE addresses ADD COLUMN key_scope INTEGER NOT NULL DEFAULT {external_scope_code};
151            ALTER TABLE addresses ADD COLUMN transparent_child_index INTEGER;
152            ALTER TABLE addresses ADD COLUMN exposed_at_height INTEGER;
153            ALTER TABLE addresses ADD COLUMN receiver_flags INTEGER;
154            "#
155        ))?;
156
157        let mut account_ids = HashSet::new();
158
159        {
160            // If the diversifier index is in the valid range of non-hardened child indices, set
161            // `transparent_child_index` so that we can use it for gap limit handling.
162            // No `DISTINCT` is necessary here due to the preexisting UNIQUE(account_id,
163            // diversifier_index_be) constraint.
164            let mut di_query = conn.prepare(
165                r#"
166                SELECT
167                    account_id,
168                    address,
169                    accounts.uivk AS uivk,
170                    diversifier_index_be,
171                    accounts.birthday_height
172                FROM addresses
173                JOIN accounts ON accounts.id = account_id
174                "#,
175            )?;
176            let mut rows = di_query.query([])?;
177            while let Some(row) = rows.next()? {
178                let account_id: i64 = row.get("account_id")?;
179                account_ids.insert(account_id);
180
181                let addr_str: String = row.get("address")?;
182                let address = ZcashAddress::try_from_encoded(&addr_str).map_err(|e| {
183                    WalletMigrationError::CorruptedData(format!(
184                        "Encoded address {addr_str} is not a valid zcash address: {e}"
185                    ))
186                })?;
187                let receiver_flags = address.convert::<ReceiverFlags>().map_err(|_| {
188                    WalletMigrationError::CorruptedData("Unexpected address type".to_string())
189                })?;
190                let di_be: Vec<u8> = row.get("diversifier_index_be")?;
191                let account_birthday: i64 = row.get("birthday_height")?;
192
193                let update_without_taddr = || {
194                    conn.execute(
195                        r#"
196                        UPDATE addresses
197                        SET exposed_at_height = :account_birthday,
198                            receiver_flags = :receiver_flags
199                        WHERE account_id = :account_id
200                        AND diversifier_index_be = :diversifier_index_be
201                        "#,
202                        named_params! {
203                            ":account_id": account_id,
204                            ":diversifier_index_be": &di_be[..],
205                            ":account_birthday": account_birthday,
206                            ":receiver_flags": receiver_flags.bits(),
207                        },
208                    )
209                };
210
211                #[cfg(feature = "transparent-inputs")]
212                {
213                    let uivk = decode_uivk(row.get("uivk")?)?;
214                    let diversifier_index = decode_diversifier_index_be(&di_be)?;
215                    let transparent_external = NonHardenedChildIndex::try_from(diversifier_index)
216                        .ok()
217                        .and_then(|idx| {
218                            uivk.transparent()
219                                .as_ref()
220                                .and_then(|external_ivk| external_ivk.derive_address(idx).ok())
221                                .map(|t_addr| (idx, t_addr.encode(&self.params)))
222                        });
223
224                    // Add transparent address index metadata and the transparent address corresponding
225                    // to the index to the addresses table. We unconditionally set the cached
226                    // transparent receiver address in order to simplify gap limit handling; even if a
227                    // unified address is generated without a transparent receiver, we still assume
228                    // that a transparent-only wallet for which we have imported the seed may have
229                    // generated an address at that index.
230                    if let Some((idx, t_addr)) = transparent_external {
231                        conn.execute(
232                            r#"
233                            UPDATE addresses
234                            SET transparent_child_index = :transparent_child_index,
235                                cached_transparent_receiver_address = :t_addr,
236                                exposed_at_height = :account_birthday,
237                                receiver_flags = :receiver_flags
238                            WHERE account_id = :account_id
239                            AND diversifier_index_be = :diversifier_index_be
240                            "#,
241                            named_params! {
242                                ":account_id": account_id,
243                                ":diversifier_index_be": &di_be[..],
244                                ":transparent_child_index": idx.index(),
245                                ":t_addr": t_addr,
246                                ":account_birthday": account_birthday,
247                                ":receiver_flags": receiver_flags.bits(),
248                            },
249                        )?;
250                    } else {
251                        update_without_taddr()?;
252                    }
253                }
254
255                #[cfg(not(feature = "transparent-inputs"))]
256                {
257                    update_without_taddr()?;
258                }
259            }
260        };
261
262        // We now have to re-create the `addresses` table in order to fix the constraints. Note
263        // that we do not include the `seen_in_tx` column as this is duplicative of information
264        // that can be discovered via joins with the various `*_received_{notes|outputs}` tables,
265        // which we will create a view to perform below. The `used_in_tx` column data is used only
266        // to determine the height at which the address was exposed (for which we use the target
267        // height for the transaction.)
268        conn.execute_batch(r#"
269            CREATE TABLE addresses_new (
270                id INTEGER NOT NULL PRIMARY KEY,
271                account_id INTEGER NOT NULL,
272                key_scope INTEGER NOT NULL,
273                diversifier_index_be BLOB NOT NULL,
274                address TEXT NOT NULL,
275                transparent_child_index INTEGER,
276                cached_transparent_receiver_address TEXT,
277                exposed_at_height INTEGER,
278                receiver_flags INTEGER NOT NULL,
279                transparent_receiver_next_check_time INTEGER,
280                FOREIGN KEY (account_id) REFERENCES accounts(id),
281                CONSTRAINT diversification UNIQUE (account_id, key_scope, diversifier_index_be),
282                CONSTRAINT transparent_index_consistency CHECK (
283                    (transparent_child_index IS NOT NULL) == (cached_transparent_receiver_address IS NOT NULL)
284                )
285            );
286
287            -- we will only set `transparent_receiver_next_check_time` for ephemeral addresses
288            INSERT INTO addresses_new (
289                account_id, key_scope, diversifier_index_be, address,
290                transparent_child_index, cached_transparent_receiver_address,
291                exposed_at_height, receiver_flags
292            )
293            SELECT
294                account_id, key_scope, diversifier_index_be, address,
295                transparent_child_index, cached_transparent_receiver_address,
296                exposed_at_height, receiver_flags
297            FROM addresses;
298            "#)?;
299
300        // Now, we add the ephemeral addresses to the newly unified `addresses` table.
301        #[cfg(feature = "transparent-inputs")]
302        {
303            let mut ea_insert = conn.prepare(
304                r#"
305                INSERT INTO addresses_new (
306                    account_id, key_scope, diversifier_index_be, address,
307                    transparent_child_index, cached_transparent_receiver_address,
308                    exposed_at_height, receiver_flags,
309                    transparent_receiver_next_check_time
310                ) VALUES (
311                    :account_id, :key_scope, :diversifier_index_be, :address,
312                    :transparent_child_index, :cached_transparent_receiver_address,
313                    :exposed_at_height, :receiver_flags,
314                    :transparent_receiver_next_check_time
315                )
316                "#,
317            )?;
318
319            let mut ea_query = conn.prepare(
320                r#"
321                SELECT
322                    account_id, address_index, address,
323                    t.expiry_height - :expiry_delta AS exposed_at_height
324                FROM ephemeral_addresses ea
325                LEFT OUTER JOIN transactions t ON t.id_tx = ea.used_in_tx
326                "#,
327            )?;
328            let rows = ea_query
329                .query_and_then(
330                    named_params! {":expiry_delta": DEFAULT_TX_EXPIRY_DELTA },
331                    |row| {
332                        let account_id: i64 = row.get("account_id")?;
333                        let transparent_child_index = row.get::<_, i64>("address_index")?;
334                        let diversifier_index = DiversifierIndex::from(
335                            u32::try_from(transparent_child_index)
336                                .ok()
337                                .and_then(NonHardenedChildIndex::from_index)
338                                .ok_or(WalletMigrationError::CorruptedData(
339                                    "ephemeral address indices must be in the range of `u31`"
340                                        .to_owned(),
341                                ))?
342                                .index(),
343                        );
344                        let address: String = row.get("address")?;
345                        let exposed_at_height: Option<i64> = row.get("exposed_at_height")?;
346                        Ok((
347                            account_id,
348                            diversifier_index,
349                            transparent_child_index,
350                            address,
351                            exposed_at_height,
352                        ))
353                    },
354                )?
355                .collect::<Result<Vec<_>, WalletMigrationError>>()?;
356
357            let ephemeral_address_count =
358                u32::try_from(rows.len()).expect("number of ephemeral addrs fits into u32");
359            let mut check_time = self._clock.now();
360            for (
361                account_id,
362                diversifier_index,
363                transparent_child_index,
364                address,
365                exposed_at_height,
366            ) in rows
367            {
368                // Compute a next check time for the address such that, when considered in the
369                // context of all other allocated ephemeral addresses, it will be checked once per
370                // day.
371                let next_check_time = {
372                    let rng = self
373                        ._rng
374                        .lock()
375                        .expect("can obtain write lock to shared rng");
376
377                    next_check_time(rng, check_time, (24 * 60 * 60) / ephemeral_address_count)
378                        .expect("computed next check time is valid")
379                };
380                let next_check_epoch_seconds = epoch_seconds(next_check_time).unwrap();
381
382                // We set both the `address` column and the `cached_transparent_receiver_address`
383                // column to the same value here; there is no Unified address that corresponds to
384                // this transparent address.
385                ea_insert.execute(named_params! {
386                    ":account_id": account_id,
387                    ":key_scope": KeyScope::Ephemeral.encode(),
388                    ":diversifier_index_be": encode_diversifier_index_be(diversifier_index),
389                    ":address": address,
390                    ":transparent_child_index": transparent_child_index,
391                    ":cached_transparent_receiver_address": address,
392                    ":exposed_at_height": exposed_at_height,
393                    ":receiver_flags": ReceiverFlags::P2PKH.bits(),
394                    ":transparent_receiver_next_check_time": next_check_epoch_seconds
395                })?;
396
397                account_ids.insert(account_id);
398                check_time = next_check_time;
399            }
400        }
401
402        conn.execute_batch(
403            r#"
404            PRAGMA legacy_alter_table = ON;
405
406            DROP TABLE addresses;
407            ALTER TABLE addresses_new RENAME TO addresses;
408            CREATE INDEX idx_addresses_accounts ON addresses (
409                account_id ASC
410            );
411            CREATE INDEX idx_addresses_indices ON addresses (
412                diversifier_index_be ASC
413            );
414            CREATE INDEX idx_addresses_t_indices ON addresses (
415                transparent_child_index ASC
416            );
417
418            DROP TABLE ephemeral_addresses;
419
420            PRAGMA legacy_alter_table = OFF;
421            "#,
422        )?;
423
424        #[cfg(feature = "transparent-inputs")]
425        insert_initial_transparent_addrs(conn, &self.params)?;
426
427        // Add foreign key references from the *_received_{notes|outputs} tables to the addresses
428        // table to make it possible to identify which address was involved. These foreign key
429        // columns must be nullable as for shielded account-internal. Ideally the foreign key
430        // relationship between `transparent_received_outputs` and `addresses` would not be
431        // nullable, but we allow it to be so here in order to avoid having to re-create that
432        // table.
433        //
434        // While it would be possible to only add the address reference to
435        // `transparent_received_outputs`, that would mean that a note received at a shielded
436        // component of a diversified Unified Address would not update the position of the
437        // transparent "address gap". Since we will include shielded address indices in the gap
438        // computation, transparent-only wallets may not be able to discover all transparent funds,
439        // but users of shielded wallets will be guaranteed to be able to recover all of their
440        // funds.
441        conn.execute_batch(
442            r#"
443            ALTER TABLE orchard_received_notes
444                ADD COLUMN address_id INTEGER REFERENCES addresses(id);
445            ALTER TABLE sapling_received_notes
446                ADD COLUMN address_id INTEGER REFERENCES addresses(id);
447            ALTER TABLE transparent_received_outputs
448                ADD COLUMN address_id INTEGER REFERENCES addresses(id);
449            "#,
450        )?;
451
452        // Ensure that an address exists for each received Orchard note, and populate the
453        // `address_id` column.
454        #[cfg(feature = "orchard")]
455        {
456            let mut stmt_rn_diversifiers = conn.prepare(
457                r#"
458                SELECT orn.id, orn.account_id, accounts.uivk,
459                       orn.recipient_key_scope, orn.diversifier, t.mined_height
460                FROM orchard_received_notes orn
461                JOIN accounts ON accounts.id = account_id
462                JOIN transactions t on t.id_tx = orn.tx
463                "#,
464            )?;
465
466            let mut rows = stmt_rn_diversifiers.query([])?;
467            while let Some(row) = rows.next()? {
468                let scope = KeyScope::decode(row.get("recipient_key_scope")?)?;
469                // for Orchard and Sapling, we only store addresses for externally-scoped keys.
470                if scope == KeyScope::EXTERNAL {
471                    let row_id: i64 = row.get("id")?;
472                    let account_id = AccountRef(row.get("account_id")?);
473                    let mined_height = row
474                        .get::<_, Option<u32>>("mined_height")?
475                        .map(BlockHeight::from);
476
477                    let uivk = decode_uivk(row.get("uivk")?)?;
478                    let diversifier =
479                        orchard::keys::Diversifier::from_bytes(row.get("diversifier")?);
480
481                    // TODO: It's annoying that `IncomingViewingKey` doesn't expose the ability to
482                    // decrypt the diversifier to find the index directly, and doesn't provide an
483                    // accessor for `dk`. We already know we have the right IVK.
484                    let ivk = uivk
485                        .orchard()
486                        .as_ref()
487                        .expect("previously received an Orchard output");
488                    let di = ivk
489                        .diversifier_index(&ivk.address(diversifier))
490                        .expect("roundtrip");
491                    let ua = uivk.address(di, UnifiedAddressRequest::ALLOW_ALL)?;
492                    let address_id = wallet::upsert_address(
493                        conn,
494                        &self.params,
495                        account_id,
496                        di,
497                        &ua,
498                        mined_height,
499                        false,
500                    )?;
501
502                    conn.execute(
503                        "UPDATE orchard_received_notes
504                         SET address_id = :address_id
505                         WHERE id = :row_id",
506                        named_params! {
507                            ":address_id": address_id.0,
508                            ":row_id": row_id
509                        },
510                    )?;
511                }
512            }
513        }
514
515        // Ensure that an address exists for each received Sapling note, and populate the
516        // `address_id` column.
517        {
518            let mut stmt_rn_diversifiers = conn.prepare(
519                r#"
520                SELECT srn.id, srn.account_id, accounts.uivk,
521                       srn.recipient_key_scope, srn.diversifier, t.mined_height
522                FROM sapling_received_notes srn
523                JOIN accounts ON accounts.id = account_id
524                JOIN transactions t ON t.id_tx = srn.tx
525                "#,
526            )?;
527
528            let mut rows = stmt_rn_diversifiers.query([])?;
529            while let Some(row) = rows.next()? {
530                let scope = KeyScope::decode(row.get("recipient_key_scope")?)?;
531                // for Orchard and Sapling, we only store addresses for externally-scoped keys.
532                if scope == KeyScope::EXTERNAL {
533                    let row_id: i64 = row.get("id")?;
534                    let account_id = AccountRef(row.get("account_id")?);
535                    let mined_height = row
536                        .get::<_, Option<u32>>("mined_height")?
537                        .map(BlockHeight::from);
538
539                    let uivk = decode_uivk(row.get("uivk")?)?;
540                    let diversifier = sapling::Diversifier(row.get("diversifier")?);
541
542                    // TODO: It's annoying that `IncomingViewingKey` doesn't expose the ability to
543                    // decrypt the diversifier to find the index directly, and doesn't provide an
544                    // accessor for `dk`. We already know we have the right IVK.
545                    let ivk = uivk
546                        .sapling()
547                        .as_ref()
548                        .expect("previously received a Sapling output");
549                    let di = ivk
550                        .decrypt_diversifier(
551                            &ivk.address(diversifier)
552                                .expect("previously generated an address"),
553                        )
554                        .expect("roundtrip");
555                    let ua = uivk.address(di, UnifiedAddressRequest::ALLOW_ALL)?;
556                    let address_id = wallet::upsert_address(
557                        conn,
558                        &self.params,
559                        account_id,
560                        di,
561                        &ua,
562                        mined_height,
563                        false,
564                    )?;
565
566                    conn.execute(
567                        "UPDATE sapling_received_notes
568                         SET address_id = :address_id
569                         WHERE id = :row_id",
570                        named_params! {
571                            ":address_id": address_id.0,
572                            ":row_id": row_id
573                        },
574                    )?;
575                }
576            }
577        }
578
579        // At this point, every address on which we've received a transparent output should have a
580        // corresponding row in the `addresses` table with a valid
581        // `cached_transparent_receiver_address` entry, because we will only have queried the light
582        // wallet server for outputs from exactly these addresses. So for transparent outputs, we
583        // join to the addresses table using the address itself in order to obtain the address index.
584        #[cfg(feature = "transparent-inputs")]
585        {
586            conn.execute_batch(
587                r#"
588                PRAGMA legacy_alter_table = ON;
589
590                UPDATE transparent_received_outputs
591                SET address_id = addresses.id
592                FROM addresses
593                WHERE addresses.cached_transparent_receiver_address = transparent_received_outputs.address;
594
595                CREATE TABLE transparent_received_outputs_new (
596                    id INTEGER PRIMARY KEY,
597                    transaction_id INTEGER NOT NULL,
598                    output_index INTEGER NOT NULL,
599                    account_id INTEGER NOT NULL,
600                    address TEXT NOT NULL,
601                    script BLOB NOT NULL,
602                    value_zat INTEGER NOT NULL,
603                    max_observed_unspent_height INTEGER,
604                    address_id INTEGER NOT NULL REFERENCES addresses(id),
605                    FOREIGN KEY (transaction_id) REFERENCES transactions(id_tx),
606                    FOREIGN KEY (account_id) REFERENCES accounts(id),
607                    CONSTRAINT transparent_output_unique UNIQUE (transaction_id, output_index)
608                );
609                INSERT INTO transparent_received_outputs_new SELECT * FROM transparent_received_outputs;
610
611                DROP TABLE transparent_received_outputs;
612                ALTER TABLE transparent_received_outputs_new RENAME TO transparent_received_outputs;
613
614                PRAGMA legacy_alter_table = OFF;
615                "#,
616            )?;
617        }

619        // Construct a view that identifies the minimum block height at which each address was
