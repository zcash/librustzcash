63// For each account, ensure that all diversifier indexes prior to that for the default
64// address have corresponding cached transparent addresses.
65#[cfg(feature = "transparent-inputs")]
66pub(super) fn insert_initial_transparent_addrs<P: consensus::Parameters>(
67    conn: &rusqlite::Transaction,
68    params: &P,
69) -> Result<(), WalletMigrationError> {
70    let mut min_addr_diversifiers = conn.prepare(
71        r#"
72        SELECT accounts.id AS account_id,
73               MIN(addresses.transparent_child_index) AS transparent_child_index,
74               MIN(addresses.diversifier_index_be) AS diversifier_index_be
75        FROM accounts
76        LEFT OUTER JOIN addresses
77            ON accounts.id = addresses.account_id
78            AND addresses.key_scope = :key_scope_external
79        GROUP BY accounts.id
80        "#,
81    )?;
82
83    let mut min_addr_rows = min_addr_diversifiers.query(named_params![
84        ":key_scope_external": KeyScope::EXTERNAL.encode()
85    ])?;
86    while let Some(row) = min_addr_rows.next()? {
87        let account_id = AccountRef(row.get::<_, u32>("account_id")?);
88
89        let min_transparent_idx = row
90            .get::<_, Option<u32>>("transparent_child_index")?
91            .map(|i| {
92                NonHardenedChildIndex::from_index(i).ok_or(WalletMigrationError::CorruptedData(
93                    format!("{} is not a valid transparent child index.", i),
94                ))
95            })
96            .transpose()?;
97
98        let min_diversifier_idx = row
99            .get::<_, Option<Vec<u8>>>("diversifier_index_be")?
100            .map(|b| decode_diversifier_index_be(&b[..]))
101            .transpose()?
102            .and_then(|di| NonHardenedChildIndex::try_from(di).ok());
103
104        // Ensure that there is an address for each possible external address index prior to the
105        // default UA for the account. If the default address has a diversifier index greater than
106        // the gap limit, we generate transparent addresses up to that index but not beyond.
107        let start = NonHardenedChildIndex::const_from_index(0);
108        let end = std::cmp::min(
109            min_transparent_idx
110                .or(min_diversifier_idx)
111                // guarantee that we have an entry at index 0; this address will have previously
112                // been provided explicitly as one of the wallet's addresses in response to a call
113                // to `get_transparent_receivers, even if we have no other addresses generated
114                // (which shouldn't ordinarily be the case anyway)
115                .unwrap_or(NonHardenedChildIndex::const_from_index(1)),
116            NonHardenedChildIndex::from_index(GapLimits::default().external())
117                .expect("default external gap limit fits in non-hardened child index space."),
118        );
119
120        generate_address_range(
121            conn,
122            params,
123            account_id,
124            KeyScope::EXTERNAL,
125            UnifiedAddressRequest::ALLOW_ALL,
126            start..end,
127            false,
128        )?;
129    }
130
131    Ok(())
132}

134impl<P: consensus::Parameters, C: Clock, R: RngCore> RusqliteMigration for Migration<P, C, R> {
135    type Error = WalletMigrationError;
136
137    fn up(&self, conn: &Transaction) -> Result<(), WalletMigrationError> {
138        let decode_uivk = |uivk_str: String| {
139            UnifiedIncomingViewingKey::decode(&self.params, &uivk_str).map_err(|e| {
140                WalletMigrationError::CorruptedData(format!(
141                    "Invalid UIVK encoding {}: {}",
142                    uivk_str, e
143                ))
144            })
145        };
146
147        let external_scope_code = KeyScope::EXTERNAL.encode();
148
149        conn.execute_batch(&format!(
150            r#"
151            ALTER TABLE addresses ADD COLUMN key_scope INTEGER NOT NULL DEFAULT {external_scope_code};
152            ALTER TABLE addresses ADD COLUMN transparent_child_index INTEGER;
153            ALTER TABLE addresses ADD COLUMN exposed_at_height INTEGER;
154            ALTER TABLE addresses ADD COLUMN receiver_flags INTEGER;
155            "#
156        ))?;
157
158        let mut account_ids = HashSet::new();
159
160        {
161            // If the diversifier index is in the valid range of non-hardened child indices, set
162            // `transparent_child_index` so that we can use it for gap limit handling.
163            // No `DISTINCT` is necessary here due to the preexisting UNIQUE(account_id,
164            // diversifier_index_be) constraint.
165            let mut di_query = conn.prepare(
166                r#"
167                SELECT
168                    account_id,
169                    address,
170                    accounts.uivk AS uivk,
171                    diversifier_index_be,
172                    accounts.birthday_height
173                FROM addresses
174                JOIN accounts ON accounts.id = account_id
175                "#,
176            )?;
177            let mut rows = di_query.query([])?;
178            while let Some(row) = rows.next()? {
179                let account_id: i64 = row.get("account_id")?;
180                account_ids.insert(account_id);
181
182                let addr_str: String = row.get("address")?;
183                let address = ZcashAddress::try_from_encoded(&addr_str).map_err(|e| {
184                    WalletMigrationError::CorruptedData(format!(
185                        "Encoded address {} is not a valid zcash address: {}",
186                        addr_str, e
187                    ))
188                })?;
189                let receiver_flags = address.convert::<ReceiverFlags>().map_err(|_| {
190                    WalletMigrationError::CorruptedData("Unexpected address type".to_string())
191                })?;
192                let di_be: Vec<u8> = row.get("diversifier_index_be")?;
193                let account_birthday: i64 = row.get("birthday_height")?;
194
195                let update_without_taddr = || {
196                    conn.execute(
197                        r#"
198                        UPDATE addresses
199                        SET exposed_at_height = :account_birthday,
200                            receiver_flags = :receiver_flags
201                        WHERE account_id = :account_id
202                        AND diversifier_index_be = :diversifier_index_be
203                        "#,
204                        named_params! {
205                            ":account_id": account_id,
206                            ":diversifier_index_be": &di_be[..],
207                            ":account_birthday": account_birthday,
208                            ":receiver_flags": receiver_flags.bits(),
209                        },
210                    )
211                };
212
213                #[cfg(feature = "transparent-inputs")]
214                {
215                    let uivk = decode_uivk(row.get("uivk")?)?;
216                    let diversifier_index = decode_diversifier_index_be(&di_be)?;
217                    let transparent_external = NonHardenedChildIndex::try_from(diversifier_index)
218                        .ok()
219                        .and_then(|idx| {
220                            uivk.transparent()
221                                .as_ref()
222                                .and_then(|external_ivk| external_ivk.derive_address(idx).ok())
223                                .map(|t_addr| (idx, t_addr.encode(&self.params)))
224                        });
225
226                    // Add transparent address index metadata and the transparent address corresponding
227                    // to the index to the addresses table. We unconditionally set the cached
228                    // transparent receiver address in order to simplify gap limit handling; even if a
229                    // unified address is generated without a transparent receiver, we still assume
230                    // that a transparent-only wallet for which we have imported the seed may have
231                    // generated an address at that index.
232                    if let Some((idx, t_addr)) = transparent_external {
233                        conn.execute(
234                            r#"
235                            UPDATE addresses
236                            SET transparent_child_index = :transparent_child_index,
237                                cached_transparent_receiver_address = :t_addr,
238                                exposed_at_height = :account_birthday,
239                                receiver_flags = :receiver_flags
240                            WHERE account_id = :account_id
241                            AND diversifier_index_be = :diversifier_index_be
242                            "#,
243                            named_params! {
244                                ":account_id": account_id,
245                                ":diversifier_index_be": &di_be[..],
246                                ":transparent_child_index": idx.index(),
247                                ":t_addr": t_addr,
248                                ":account_birthday": account_birthday,
249                                ":receiver_flags": receiver_flags.bits(),
250                            },
251                        )?;
252                    } else {
253                        update_without_taddr()?;
254                    }
255                }
256
257                #[cfg(not(feature = "transparent-inputs"))]
258                {
259                    update_without_taddr()?;
260                }
261            }
262        };
263
264        // We now have to re-create the `addresses` table in order to fix the constraints. Note
265        // that we do not include the `seen_in_tx` column as this is duplicative of information
266        // that can be discovered via joins with the various `*_received_{notes|outputs}` tables,
267        // which we will create a view to perform below. The `used_in_tx` column data is used only
268        // to determine the height at which the address was exposed (for which we use the target
269        // height for the transaction.)
270        conn.execute_batch(r#"
271            CREATE TABLE addresses_new (
272                id INTEGER NOT NULL PRIMARY KEY,
273                account_id INTEGER NOT NULL,
274                key_scope INTEGER NOT NULL,
275                diversifier_index_be BLOB NOT NULL,
276                address TEXT NOT NULL,
277                transparent_child_index INTEGER,
278                cached_transparent_receiver_address TEXT,
279                exposed_at_height INTEGER,
280                receiver_flags INTEGER NOT NULL,
281                transparent_receiver_next_check_time INTEGER,
282                FOREIGN KEY (account_id) REFERENCES accounts(id),
283                CONSTRAINT diversification UNIQUE (account_id, key_scope, diversifier_index_be),
284                CONSTRAINT transparent_index_consistency CHECK (
285                    (transparent_child_index IS NOT NULL) == (cached_transparent_receiver_address IS NOT NULL)
286                )
287            );
288
289            -- we will only set `transparent_receiver_next_check_time` for ephemeral addresses
290            INSERT INTO addresses_new (
291                account_id, key_scope, diversifier_index_be, address,
292                transparent_child_index, cached_transparent_receiver_address,
293                exposed_at_height, receiver_flags
294            )
295            SELECT
296                account_id, key_scope, diversifier_index_be, address,
297                transparent_child_index, cached_transparent_receiver_address,
298                exposed_at_height, receiver_flags
299            FROM addresses;
300            "#)?;
301
302        // Now, we add the ephemeral addresses to the newly unified `addresses` table.
303        #[cfg(feature = "transparent-inputs")]
304        {
305            let mut ea_insert = conn.prepare(
306                r#"
307                INSERT INTO addresses_new (
308                    account_id, key_scope, diversifier_index_be, address,
309                    transparent_child_index, cached_transparent_receiver_address,
310                    exposed_at_height, receiver_flags,
311                    transparent_receiver_next_check_time
312                ) VALUES (
313                    :account_id, :key_scope, :diversifier_index_be, :address,
314                    :transparent_child_index, :cached_transparent_receiver_address,
315                    :exposed_at_height, :receiver_flags,
316                    :transparent_receiver_next_check_time
317                )
318                "#,
319            )?;
320
321            let mut ea_query = conn.prepare(
322                r#"
323                SELECT
324                    account_id, address_index, address,
325                    t.expiry_height - :expiry_delta AS exposed_at_height
326                FROM ephemeral_addresses ea
327                LEFT OUTER JOIN transactions t ON t.id_tx = ea.used_in_tx
328                "#,
329            )?;
330            let rows = ea_query
331                .query_and_then(
332                    named_params! {":expiry_delta": DEFAULT_TX_EXPIRY_DELTA },
333                    |row| {
334                        let account_id: i64 = row.get("account_id")?;
335                        let transparent_child_index = row.get::<_, i64>("address_index")?;
336                        let diversifier_index = DiversifierIndex::from(
337                            u32::try_from(transparent_child_index)
338                                .ok()
339                                .and_then(NonHardenedChildIndex::from_index)
340                                .ok_or(WalletMigrationError::CorruptedData(
341                                    "ephemeral address indices must be in the range of `u31`"
342                                        .to_owned(),
343                                ))?
344                                .index(),
345                        );
346                        let address: String = row.get("address")?;
347                        let exposed_at_height: Option<i64> = row.get("exposed_at_height")?;
348                        Ok((
349                            account_id,
350                            diversifier_index,
351                            transparent_child_index,
352                            address,
353                            exposed_at_height,
354                        ))
355                    },
356                )?
357                .collect::<Result<Vec<_>, WalletMigrationError>>()?;
358
359            let ephemeral_address_count =
360                u32::try_from(rows.len()).expect("number of ephemeral addrs fits into u32");
361            let mut check_time = self._clock.now();
362            for (
363                account_id,
364                diversifier_index,
365                transparent_child_index,
366                address,
367                exposed_at_height,
368            ) in rows
369            {
370                // Compute a next check time for the address such that, when considered in the
371                // context of all other allocated ephemeral addresses, it will be checked once per
372                // day.
373                let next_check_time = {
374                    let rng = self
375                        ._rng
376                        .lock()
377                        .expect("can obtain write lock to shared rng");
378
379                    next_check_time(rng, check_time, (24 * 60 * 60) / ephemeral_address_count)
380                        .expect("computed next check time is valid")
381                };
382                let next_check_epoch_seconds = epoch_seconds(next_check_time).unwrap();
383
384                // We set both the `address` column and the `cached_transparent_receiver_address`
385                // column to the same value here; there is no Unified address that corresponds to
386                // this transparent address.
387                ea_insert.execute(named_params! {
388                    ":account_id": account_id,
389                    ":key_scope": KeyScope::Ephemeral.encode(),
390                    ":diversifier_index_be": encode_diversifier_index_be(diversifier_index),
391                    ":address": address,
392                    ":transparent_child_index": transparent_child_index,
393                    ":cached_transparent_receiver_address": address,
394                    ":exposed_at_height": exposed_at_height,
395                    ":receiver_flags": ReceiverFlags::P2PKH.bits(),
396                    ":transparent_receiver_next_check_time": next_check_epoch_seconds
397                })?;
398
399                account_ids.insert(account_id);
400                check_time = next_check_time;
401            }
402        }
403
404        conn.execute_batch(
405            r#"
406            PRAGMA legacy_alter_table = ON;
407
408            DROP TABLE addresses;
409            ALTER TABLE addresses_new RENAME TO addresses;
410            CREATE INDEX idx_addresses_accounts ON addresses (
411                account_id ASC
412            );
413            CREATE INDEX idx_addresses_indices ON addresses (
414                diversifier_index_be ASC
415            );
416            CREATE INDEX idx_addresses_t_indices ON addresses (
417                transparent_child_index ASC
418            );
419
420            DROP TABLE ephemeral_addresses;
421
422            PRAGMA legacy_alter_table = OFF;
423            "#,
424        )?;
425
426        #[cfg(feature = "transparent-inputs")]
427        insert_initial_transparent_addrs(conn, &self.params)?;
428
429        // Add foreign key references from the *_received_{notes|outputs} tables to the addresses
430        // table to make it possible to identify which address was involved. These foreign key
431        // columns must be nullable as for shielded account-internal. Ideally the foreign key
432        // relationship between `transparent_received_outputs` and `addresses` would not be
433        // nullable, but we allow it to be so here in order to avoid having to re-create that
434        // table.
435        //
436        // While it would be possible to only add the address reference to
437        // `transparent_received_outputs`, that would mean that a note received at a shielded
438        // component of a diversified Unified Address would not update the position of the
439        // transparent "address gap". Since we will include shielded address indices in the gap
440        // computation, transparent-only wallets may not be able to discover all transparent funds,
441        // but users of shielded wallets will be guaranteed to be able to recover all of their
442        // funds.
443        conn.execute_batch(
444            r#"
445            ALTER TABLE orchard_received_notes
446                ADD COLUMN address_id INTEGER REFERENCES addresses(id);
447            ALTER TABLE sapling_received_notes
448                ADD COLUMN address_id INTEGER REFERENCES addresses(id);
449            ALTER TABLE transparent_received_outputs
450                ADD COLUMN address_id INTEGER REFERENCES addresses(id);
451            "#,
452        )?;
453
454        // Ensure that an address exists for each received Orchard note, and populate the
455        // `address_id` column.
456        #[cfg(feature = "orchard")]
457        {
458            let mut stmt_rn_diversifiers = conn.prepare(
459                r#"
460                SELECT orn.id, orn.account_id, accounts.uivk,
461                       orn.recipient_key_scope, orn.diversifier, t.mined_height
462                FROM orchard_received_notes orn
463                JOIN accounts ON accounts.id = account_id
464                JOIN transactions t on t.id_tx = orn.tx
465                "#,
466            )?;
467
468            let mut rows = stmt_rn_diversifiers.query([])?;
469            while let Some(row) = rows.next()? {
470                let scope = KeyScope::decode(row.get("recipient_key_scope")?)?;
471                // for Orchard and Sapling, we only store addresses for externally-scoped keys.
472                if scope == KeyScope::EXTERNAL {
473                    let row_id: i64 = row.get("id")?;
474                    let account_id = AccountRef(row.get("account_id")?);
475                    let mined_height = row
476                        .get::<_, Option<u32>>("mined_height")?
477                        .map(BlockHeight::from);
478
479                    let uivk = decode_uivk(row.get("uivk")?)?;
480                    let diversifier =
481                        orchard::keys::Diversifier::from_bytes(row.get("diversifier")?);
482
483                    // TODO: It's annoying that `IncomingViewingKey` doesn't expose the ability to
484                    // decrypt the diversifier to find the index directly, and doesn't provide an
485                    // accessor for `dk`. We already know we have the right IVK.
486                    let ivk = uivk
487                        .orchard()
488                        .as_ref()
489                        .expect("previously received an Orchard output");
490                    let di = ivk
491                        .diversifier_index(&ivk.address(diversifier))
492                        .expect("roundtrip");
493                    let ua = uivk.address(di, UnifiedAddressRequest::ALLOW_ALL)?;
494                    let address_id = wallet::upsert_address(
495                        conn,
496                        &self.params,
497                        account_id,
498                        di,
499                        &ua,
500                        mined_height,
501                        false,
502                    )?;
503
504                    conn.execute(
505                        "UPDATE orchard_received_notes
506                         SET address_id = :address_id
507                         WHERE id = :row_id",
508                        named_params! {
509                            ":address_id": address_id.0,
510                            ":row_id": row_id
511                        },
512                    )?;
513                }
514            }
515        }
516
517        // Ensure that an address exists for each received Sapling note, and populate the
518        // `address_id` column.
519        {
520            let mut stmt_rn_diversifiers = conn.prepare(
521                r#"
522                SELECT srn.id, srn.account_id, accounts.uivk,
523                       srn.recipient_key_scope, srn.diversifier, t.mined_height
524                FROM sapling_received_notes srn
525                JOIN accounts ON accounts.id = account_id
526                JOIN transactions t ON t.id_tx = srn.tx
527                "#,
528            )?;
529
530            let mut rows = stmt_rn_diversifiers.query([])?;
531            while let Some(row) = rows.next()? {
532                let scope = KeyScope::decode(row.get("recipient_key_scope")?)?;
533                // for Orchard and Sapling, we only store addresses for externally-scoped keys.
534                if scope == KeyScope::EXTERNAL {
535                    let row_id: i64 = row.get("id")?;
536                    let account_id = AccountRef(row.get("account_id")?);
537                    let mined_height = row
538                        .get::<_, Option<u32>>("mined_height")?
539                        .map(BlockHeight::from);
540
541                    let uivk = decode_uivk(row.get("uivk")?)?;
542                    let diversifier = sapling::Diversifier(row.get("diversifier")?);
543
544                    // TODO: It's annoying that `IncomingViewingKey` doesn't expose the ability to
545                    // decrypt the diversifier to find the index directly, and doesn't provide an
546                    // accessor for `dk`. We already know we have the right IVK.
547                    let ivk = uivk
548                        .sapling()
549                        .as_ref()
550                        .expect("previously received a Sapling output");
551                    let di = ivk
552                        .decrypt_diversifier(
553                            &ivk.address(diversifier)
554                                .expect("previously generated an address"),
555                        )
556                        .expect("roundtrip");
557                    let ua = uivk.address(di, UnifiedAddressRequest::ALLOW_ALL)?;
558                    let address_id = wallet::upsert_address(
559                        conn,
560                        &self.params,
561                        account_id,
562                        di,
563                        &ua,
564                        mined_height,
565                        false,
566                    )?;
567
568                    conn.execute(
569                        "UPDATE sapling_received_notes
570                         SET address_id = :address_id
571                         WHERE id = :row_id",
572                        named_params! {
573                            ":address_id": address_id.0,
574                            ":row_id": row_id
575                        },
576                    )?;
577                }
578            }
579        }
580
581        // At this point, every address on which we've received a transparent output should have a
582        // corresponding row in the `addresses` table with a valid
583        // `cached_transparent_receiver_address` entry, because we will only have queried the light
584        // wallet server for outputs from exactly these addresses. So for transparent outputs, we
585        // join to the addresses table using the address itself in order to obtain the address index.
586        #[cfg(feature = "transparent-inputs")]
587        {
588            conn.execute_batch(
589                r#"
590                PRAGMA legacy_alter_table = ON;
591
592                UPDATE transparent_received_outputs
593                SET address_id = addresses.id
594                FROM addresses
595                WHERE addresses.cached_transparent_receiver_address = transparent_received_outputs.address;
596
597                CREATE TABLE transparent_received_outputs_new (
598                    id INTEGER PRIMARY KEY,
599                    transaction_id INTEGER NOT NULL,
600                    output_index INTEGER NOT NULL,
601                    account_id INTEGER NOT NULL,
602                    address TEXT NOT NULL,
603                    script BLOB NOT NULL,
604                    value_zat INTEGER NOT NULL,
605                    max_observed_unspent_height INTEGER,
606                    address_id INTEGER NOT NULL REFERENCES addresses(id),
607                    FOREIGN KEY (transaction_id) REFERENCES transactions(id_tx),
608                    FOREIGN KEY (account_id) REFERENCES accounts(id),
609                    CONSTRAINT transparent_output_unique UNIQUE (transaction_id, output_index)
610                );
611                INSERT INTO transparent_received_outputs_new SELECT * FROM transparent_received_outputs;
612
613                DROP TABLE transparent_received_outputs;
614                ALTER TABLE transparent_received_outputs_new RENAME TO transparent_received_outputs;
615
616                PRAGMA legacy_alter_table = OFF;
617                "#,
618            )?;
619        }
620
621        // Construct a view that identifies the minimum block height at which each address was
622        // first used
623        conn.execute_batch(
624            r#"
625            CREATE VIEW v_address_uses AS
626                SELECT orn.address_id, orn.account_id, orn.tx AS transaction_id, t.mined_height,
627                       a.key_scope, a.diversifier_index_be, a.transparent_child_index
628                FROM orchard_received_notes orn
629                JOIN addresses a ON a.id = orn.address_id
630                JOIN transactions t ON t.id_tx = orn.tx
631            UNION
632                SELECT srn.address_id, srn.account_id, srn.tx AS transaction_id, t.mined_height,
633                       a.key_scope, a.diversifier_index_be, a.transparent_child_index
634                FROM sapling_received_notes srn
635                JOIN addresses a ON a.id = srn.address_id
636                JOIN transactions t ON t.id_tx = srn.tx
637            UNION
638                SELECT tro.address_id, tro.account_id, tro.transaction_id, t.mined_height,
639                       a.key_scope, a.diversifier_index_be, a.transparent_child_index
640                FROM transparent_received_outputs tro
641                JOIN addresses a ON a.id = tro.address_id
642                JOIN transactions t ON t.id_tx = tro.transaction_id;
643
644            CREATE VIEW v_address_first_use AS
645            SELECT
646                address_id,
647                account_id,
648                key_scope,
649                diversifier_index_be,
650                transparent_child_index,
651                MIN(mined_height) AS first_use_height
652            FROM v_address_uses
653            GROUP BY
654                address_id, account_id, key_scope,
655                diversifier_index_be, transparent_child_index;
656
657            DROP VIEW v_received_outputs;
658            CREATE VIEW v_received_outputs AS
659                SELECT
660                    sapling_received_notes.id AS id_within_pool_table,
661                    sapling_received_notes.tx AS transaction_id,
662                    2 AS pool,
663                    sapling_received_notes.output_index,
664                    account_id,
665                    sapling_received_notes.value,
666                    is_change,
667                    sapling_received_notes.memo,
668                    sent_notes.id AS sent_note_id,
669                    sapling_received_notes.address_id
670                FROM sapling_received_notes
671                LEFT JOIN sent_notes
672                ON (sent_notes.tx, sent_notes.output_pool, sent_notes.output_index) =
673                   (sapling_received_notes.tx, 2, sapling_received_notes.output_index)
674            UNION
675                SELECT
676                    orchard_received_notes.id AS id_within_pool_table,
677                    orchard_received_notes.tx AS transaction_id,
678                    3 AS pool,
679                    orchard_received_notes.action_index AS output_index,
680                    account_id,
681                    orchard_received_notes.value,
682                    is_change,
683                    orchard_received_notes.memo,
684                    sent_notes.id AS sent_note_id,
685                    orchard_received_notes.address_id
686                FROM orchard_received_notes
687                LEFT JOIN sent_notes
688                ON (sent_notes.tx, sent_notes.output_pool, sent_notes.output_index) =
689                   (orchard_received_notes.tx, 3, orchard_received_notes.action_index)
690            UNION
691                SELECT
692                    u.id AS id_within_pool_table,
693                    u.transaction_id,
694                    0 AS pool,
695                    u.output_index,
696                    u.account_id,
697                    u.value_zat AS value,
698                    0 AS is_change,
699                    NULL AS memo,
700                    sent_notes.id AS sent_note_id,
701                    u.address_id
702                FROM transparent_received_outputs u
703                LEFT JOIN sent_notes
704                ON (sent_notes.tx, sent_notes.output_pool, sent_notes.output_index) =
705                   (u.transaction_id, 0, u.output_index);
706            "#,
707        )?;
708
709        // At this point, we have completed updating the infrastructure for gap limit handling,
710        // so we can regenerate the gap limit worth of addresses for each account that we
711        // recorded.
712        #[cfg(feature = "transparent-inputs")]
713        for account_id in account_ids {
714            for key_scope in [KeyScope::EXTERNAL, KeyScope::INTERNAL] {
715                use ReceiverRequirement::*;
716                generate_gap_addresses(
717                    conn,
718                    &self.params,
719                    AccountRef(account_id.try_into().unwrap()),
720                    key_scope,
721                    &GapLimits::default(),
722                    UnifiedAddressRequest::unsafe_custom(Allow, Allow, Require),
723                    false,
724                )?;
725            }
726        }
727
728        Ok(())
