77/// Generates an SQL condition that a transaction is unexpired.
78///
79/// # Usage requirements
80/// - `tx` must be set to the SQL variable name for the transaction in the parent.
81/// - The parent must provide `:target_height` as a named argument.
82/// - The parent is responsible for enclosing this condition in parentheses as appropriate.
83///
84/// If the wallet doesn't know an actual mined height or expiry height for a transaction, it will
85/// be treated as unexpired _only_ if we just observed it in the last DEFAULT_TX_EXPIRY_DELTA
86/// blocks, guessing that the wallet creating the transaction used the same expiry delta as our
87/// default. If our guess is wrong (and the wallet used a larger expiry delta or disabled expiry),
88/// then the transaction will be treated as unexpired when it shouldn't be for as long as it takes
89/// this wallet to either observe the transaction being mined, or enhance it to learn its expiry
90/// height.
91pub(crate) fn tx_unexpired_condition(tx: &str) -> String {
92    format!(
93        r#"
94        {tx}.mined_height < :target_height  -- the transaction is mined
95        OR {tx}.expiry_height = 0  -- the tx will not expire
96        OR {tx}.expiry_height >= :target_height  -- the tx is unexpired
97        OR (
98            {tx}.expiry_height IS NULL -- the expiry height is unknown
99            AND {tx}.min_observed_height + {DEFAULT_TX_EXPIRY_DELTA} >= :target_height
100        )
101        "#
102    )
103}

105// Generates a SQL expression that returns the identifiers of all spent notes in the wallet.
106///
107/// # Usage requirements
108/// - `table_prefix` must be set to the table prefix for the shielded protocol under which the
109///   query is being performed.
110/// - The parent must provide `:target_height` as a named argument.
111/// - The parent is responsible for enclosing this condition in parentheses as appropriate.
112pub(crate) fn spent_notes_clause(table_prefix: &str) -> String {
113    format!(
114        r#"
115        SELECT rns.{table_prefix}_received_note_id
116        FROM {table_prefix}_received_note_spends rns
117        JOIN transactions stx ON stx.id_tx = rns.transaction_id
118        WHERE {}
119        "#,
120        tx_unexpired_condition("stx")
121    )
122}

124fn unscanned_tip_exists(
125    conn: &Connection,
126    anchor_height: BlockHeight,
127    table_prefix: &'static str,
128) -> Result<bool, rusqlite::Error> {
129    // v_sapling_shard_unscanned_ranges only returns ranges ending on or after wallet birthday, so
130    // we don't need to refer to the birthday in this query.
131    conn.query_row(
132        &format!(
133            "SELECT EXISTS (
134                 SELECT 1 FROM v_{table_prefix}_shard_unscanned_ranges range
135                 WHERE range.block_range_start <= :anchor_height
136                 AND :anchor_height BETWEEN
137                    range.subtree_start_height
138                    AND IFNULL(range.subtree_end_height, :anchor_height)
139             )"
140        ),
141        named_params![":anchor_height": u32::from(anchor_height),],
142        |row| row.get::<_, bool>(0),
143    )
144}

146/// Retrieves the set of nullifiers for "potentially spendable" notes that the wallet is tracking.
147///
148/// "Potentially spendable" means:
149/// - The transaction in which the note was created has been observed as mined.
150/// - No transaction in which the note's nullifier appears has been observed as mined.
151///
152/// This may over-select nullifiers and return those that have been spent in un-mined transactions
153/// that have not yet expired, or for which the expiry height is unknown. This is fine because
154/// these nullifiers are primarily used to detect the spends of our own notes in scanning; if we
155/// select a few too many nullifiers, it's not a big deal.
156pub(crate) fn get_nullifiers<N, F: Fn(&[u8]) -> Result<N, SqliteClientError>>(
157    conn: &Connection,
158    protocol: ShieldedProtocol,
159    query: NullifierQuery,
160    parse_nf: F,
161) -> Result<Vec<(AccountUuid, N)>, SqliteClientError> {
162    let TableConstants { table_prefix, .. } = table_constants::<SqliteClientError>(protocol)?;
163
164    // Get the nullifiers for the notes we are tracking
165    let mut stmt_fetch_nullifiers = match query {
166        NullifierQuery::Unspent => conn.prepare(&format!(
167            // See the method documentation for why this does not use `spent_notes_clause`.
168            // We prefer to be more restrictive in determining whether a note is spent here.
169            "SELECT a.uuid, rn.nf
170                 FROM {table_prefix}_received_notes rn
171                 JOIN accounts a ON a.id = rn.account_id
172                 JOIN transactions tx ON tx.id_tx = rn.tx
173                 WHERE rn.nf IS NOT NULL
174                 AND tx.mined_height IS NOT NULL
175                 AND rn.id NOT IN (
176                   SELECT rns.{table_prefix}_received_note_id
177                   FROM {table_prefix}_received_note_spends rns
178                   JOIN transactions stx ON stx.id_tx = rns.transaction_id
179                   WHERE stx.mined_height IS NOT NULL  -- the spending tx is mined
180                   OR stx.expiry_height = 0 -- the spending tx will not expire
181                 )"
182        )),
183        NullifierQuery::All => conn.prepare(&format!(
184            "SELECT a.uuid, rn.nf
185             FROM {table_prefix}_received_notes rn
186             JOIN accounts a ON a.id = rn.account_id
187             WHERE nf IS NOT NULL",
188        )),
189    }?;
190
191    let nullifiers = stmt_fetch_nullifiers.query_and_then([], |row| {
192        let account = AccountUuid(row.get(0)?);
193        let nf_bytes: Vec<u8> = row.get(1)?;
194        Ok::<_, SqliteClientError>((account, parse_nf(&nf_bytes)?))
195    })?;
196
197    let res: Vec<_> = nullifiers.collect::<Result<_, _>>()?;
198    Ok(res)
199}
200// The `clippy::let_and_return` lint is explicitly allowed here because a bug in Clippy
201// (https://github.com/rust-lang/rust-clippy/issues/11308) means it fails to identify that the `result` temporary
202// is required in order to resolve the borrows involved in the `query_and_then` call.
203#[allow(clippy::let_and_return)]
204pub(crate) fn get_spendable_note<P: consensus::Parameters, F, Note>(
205    conn: &Connection,
206    params: &P,
207    txid: &TxId,
208    index: u32,
209    protocol: ShieldedProtocol,
210    target_height: TargetHeight,
211    to_spendable_note: F,
212) -> Result<Option<ReceivedNote<ReceivedNoteId, Note>>, SqliteClientError>
213where
214    F: Fn(&P, &Row) -> Result<Option<ReceivedNote<ReceivedNoteId, Note>>, SqliteClientError>,
215{
216    let TableConstants {
217        table_prefix,
218        output_index_col,
219        note_reconstruction_cols,
220        ..
221    } = table_constants::<SqliteClientError>(protocol)?;
222
223    let result = conn.query_row_and_then(
224        &format!(
225            "SELECT rn.id, t.txid, rn.{output_index_col},
226                rn.diversifier, rn.value, {note_reconstruction_cols}, rn.commitment_tree_position,
227                accounts.ufvk, rn.recipient_key_scope, t.mined_height,
228                MAX(tt.mined_height) AS max_shielding_input_height
229             FROM {table_prefix}_received_notes rn
230             INNER JOIN accounts ON accounts.id = rn.account_id
231             INNER JOIN transactions t ON t.id_tx = rn.tx
232             LEFT OUTER JOIN transparent_received_output_spends ros
233                ON ros.transaction_id = t.id_tx
234             LEFT OUTER JOIN transparent_received_outputs tro
235                ON tro.id = ros.transparent_received_output_id
236                AND tro.account_id = accounts.id
237             LEFT OUTER JOIN transactions tt
238                ON tt.id_tx = tro.transaction_id
239             WHERE t.txid = :txid
240             AND t.block IS NOT NULL
241             AND rn.{output_index_col} = :output_index
242             AND accounts.ufvk IS NOT NULL
243             AND rn.recipient_key_scope IS NOT NULL
244             AND rn.nf IS NOT NULL
245             AND rn.commitment_tree_position IS NOT NULL
246             AND rn.id NOT IN ({})
247             GROUP BY rn.id",
248            spent_notes_clause(table_prefix)
249        ),
250        named_params![
251           ":txid": txid.as_ref(),
252           ":output_index": index,
253           ":target_height": u32::from(target_height),
254        ],
255        |row| to_spendable_note(params, row),
256    );
257
258    // `OptionalExtension` doesn't work here because the error type of `Result` is already
259    // `SqliteClientError`
260    match result {
261        Ok(r) => Ok(r),
262        Err(SqliteClientError::DbError(rusqlite::Error::QueryReturnedNoRows)) => Ok(None),
263        Err(e) => Err(e),
264    }
265}

267/// A directive that specifies how `select_unspent_notes` should filter and/or error on unspendable
268/// notes.
269#[derive(Debug, Clone, Copy)]
270pub(crate) enum NoteRequest {
271    /// Retrieve all currently spendable notes, ignoring those for which the wallet does not yet
272    /// have enough information to construct spends, given the provided anchor height.
273    Spendable { anchor_height: BlockHeight },
274    /// Retrieve all currently unspent notes, including those for which the wallet does not yet
275    /// have enough information to construct spends.
276    Unspent,
277    /// Retrieve all currently unspent notes, or an error if any notes exist for which the wallet
278    /// does not yet have enough information to construct spends.
279    UnspentOrError { anchor_height: BlockHeight },
280}
281
282impl NoteRequest {
283    pub(crate) fn from_max_spend_mode(value: MaxSpendMode, anchor_height: BlockHeight) -> Self {
284        match value {
285            MaxSpendMode::MaxSpendable => NoteRequest::Spendable { anchor_height },
286            MaxSpendMode::Everything => NoteRequest::UnspentOrError { anchor_height },
287        }
288    }
289
290    pub(crate) fn anchor_height(&self) -> Option<BlockHeight> {
291        match self {
292            NoteRequest::Spendable { anchor_height } => Some(*anchor_height),
293            NoteRequest::Unspent => None,
294            NoteRequest::UnspentOrError { anchor_height } => Some(*anchor_height),
295        }
296    }
297}

299#[allow(clippy::too_many_arguments)]
300pub(crate) fn select_spendable_notes<P: consensus::Parameters, F, Note>(
301    conn: &Connection,
302    params: &P,
303    account: AccountUuid,
304    target_value: TargetValue,
305    target_height: TargetHeight,
306    confirmations_policy: ConfirmationsPolicy,
307    exclude: &[ReceivedNoteId],
308    protocol: ShieldedProtocol,
309    to_spendable_note: F,
310) -> Result<Vec<ReceivedNote<ReceivedNoteId, Note>>, SqliteClientError>
311where
312    F: Fn(&P, &Row) -> Result<Option<ReceivedNote<ReceivedNoteId, Note>>, SqliteClientError>,
313{
314    let Some(anchor_height) =
315        get_anchor_height(conn, target_height, confirmations_policy.trusted())?
316    else {
317        return Ok(vec![]);
318    };
319
320    match target_value {
321        TargetValue::AllFunds(mode) => select_unspent_notes(
322            conn,
323            params,
324            account,
325            target_height,
326            confirmations_policy,
327            exclude,
328            protocol,
329            &to_spendable_note,
330            NoteRequest::from_max_spend_mode(mode, anchor_height),
331        ),
332        TargetValue::AtLeast(zats) => select_spendable_notes_matching_value(
333            conn,
334            params,
335            account,
336            zats,
337            target_height,
338            anchor_height,
339            confirmations_policy,
340            exclude,
341            protocol,
342            &to_spendable_note,
343        ),
344    }
345}
346/// Selects all the unspent notes with value greater than [`zip317::MARGINAL_FEE`] and for the
347/// specified shielded protocols from a given account, excepting any explicitly excluded note
348/// identifiers.
349///
350/// Implementation details:
351///
352/// - Notes with individual value *below* the ``MARGINAL_FEE`` will be ignored
353/// - Note spendability is determined using the `target_height`. If the note is mined at a height
354///   greater than or equal to the target height, it will still be returned by this query.
355/// - The `to_spendable_note` function is expected to return `Ok(None)` in the case that spending
356///   key details cannot be determined.
357#[allow(clippy::too_many_arguments)]
358pub(crate) fn select_unspent_notes<P: consensus::Parameters, F, Note>(
359    conn: &Connection,
360    params: &P,
361    account: AccountUuid,
362    target_height: TargetHeight,
363    confirmations_policy: ConfirmationsPolicy,
364    exclude: &[ReceivedNoteId],
365    protocol: ShieldedProtocol,
366    to_received_note: F,
367    note_request: NoteRequest,
368) -> Result<Vec<ReceivedNote<ReceivedNoteId, Note>>, SqliteClientError>
369where
370    F: Fn(&P, &Row) -> Result<Option<ReceivedNote<ReceivedNoteId, Note>>, SqliteClientError>,
371{
372    let TableConstants {
373        table_prefix,
374        output_index_col,
375        note_reconstruction_cols,
376        ..
377    } = table_constants::<SqliteClientError>(protocol)?;
378
379    // Select all unspent notes belonging to the given account, ignoring dust notes.
380    let mut stmt_select_notes = conn.prepare_cached(&format!(
381        "SELECT
382             rn.id AS id, t.txid, rn.{output_index_col},
383             rn.diversifier, rn.value, {note_reconstruction_cols}, rn.commitment_tree_position,
384             accounts.ufvk as ufvk, rn.recipient_key_scope,
385             t.block AS mined_height,
386             scan_state.max_priority,
387             IFNULL(t.trust_status, 0) AS trust_status,
388             MAX(tt.mined_height) AS max_shielding_input_height,
389             MIN(IFNULL(tt.trust_status, 0)) AS min_shielding_input_trust
390         FROM {table_prefix}_received_notes rn
391         INNER JOIN accounts ON accounts.id = rn.account_id
392         INNER JOIN transactions t ON t.id_tx = rn.tx
393         LEFT OUTER JOIN v_{table_prefix}_shards_scan_state scan_state
394            ON rn.commitment_tree_position >= scan_state.start_position
395            AND rn.commitment_tree_position < scan_state.end_position_exclusive
396         LEFT OUTER JOIN transparent_received_output_spends ros
397            ON ros.transaction_id = t.id_tx
398         LEFT OUTER JOIN transparent_received_outputs tro
399            ON tro.id = ros.transparent_received_output_id
400            AND tro.account_id = accounts.id
401         LEFT OUTER JOIN transactions tt
402            ON tt.id_tx = tro.transaction_id
403         WHERE accounts.uuid = :account_uuid
404         AND rn.value > :min_value
405         AND accounts.ufvk IS NOT NULL
406         AND recipient_key_scope IS NOT NULL
407         AND nf IS NOT NULL
408         AND ({})  -- the transaction is unexpired
409         AND rn.id NOT IN rarray(:exclude)  -- the note is not excluded
410         AND rn.id NOT IN ({})  -- the note is unspent
411         GROUP BY rn.id",
412        tx_unexpired_condition("t"),
413        spent_notes_clause(table_prefix)
414    ))?;
415
416    let excluded: Vec<Value> = exclude
417        .iter()
418        .filter_map(|ReceivedNoteId(p, n)| {
419            if *p == protocol {
420                Some(Value::from(*n))
421            } else {
422                None
423            }
424        })
425        .collect();
426    let excluded_ptr = Rc::new(excluded);
427
428    let row_results = stmt_select_notes.query_and_then(
429        named_params![
430            ":account_uuid": account.0,
431            ":target_height": &u32::from(target_height),
432            ":exclude": &excluded_ptr,
433            ":min_value": u64::from(zip317::MARGINAL_FEE)
434        ],
435        |row| -> Result<_, SqliteClientError> {
436            let result_note = to_received_note(params, row)?;
437            let max_priority_raw = row.get::<_, Option<i64>>("max_priority")?;
438            let tx_trust_status = row.get::<_, bool>("trust_status")?;
439            let tx_shielding_inputs_trusted = row.get::<_, bool>("min_shielding_input_trust")?;
440            let shard_scan_priority = max_priority_raw
441                .map(|code| {
442                    parse_priority_code(code).ok_or_else(|| {
443                        SqliteClientError::CorruptedData(format!(
444                            "Priority code {code} not recognized."
445                        ))
446                    })
447                })
448                .transpose()?;
449
450            Ok((
451                result_note,
452                shard_scan_priority,
453                tx_trust_status,
454                tx_shielding_inputs_trusted,
455            ))
456        },
457    )?;
458
459    let trusted_height = target_height.saturating_sub(u32::from(confirmations_policy.trusted()));
460    let untrusted_height =
461        target_height.saturating_sub(u32::from(confirmations_policy.untrusted()));
462
463    row_results
464        .map(|t| match t? {
465            (Some(note), max_shard_priority, trusted, tx_shielding_inputs_trusted) => {
466                let shard_scanned = max_shard_priority
467                    .iter()
468                    .any(|p| *p <= ScanPriority::Scanned);
469
470                let mined_at_anchor = note
471                    .mined_height()
472                    .zip(note_request.anchor_height())
473                    .is_some_and(|(h, ah)| h <= ah);
474
475                let has_confirmations = match (note.mined_height(), note.spending_key_scope()) {
476                    (None, _) => false,
477                    (Some(received_height), Scope::Internal) => {
478                        // The note has the required number of confirmations for a trusted note.
479                        received_height <= trusted_height &&
480                        // if the note was the output of a shielding transaction
481                        note.max_shielding_input_height().iter().all(|h| {
482                            // its inputs have at least `untrusted` confirmations
483                            h <= &untrusted_height ||
484                            // or its inputs are trusted and have at least `trusted` confirmations
485                            (h <= &trusted_height && tx_shielding_inputs_trusted)
486                        })
487                    }
488                    (Some(received_height), Scope::External) => {
489                        // The note has the required number of confirmations for an untrusted note.
490                        received_height <= untrusted_height ||
491                        // or it is the output of an explicitly trusted tx and has at least
492                        // `trusted` confirmations
493                        (received_height <= trusted_height && trusted)
494                    }
495                };
496
497                match (
498                    note_request,
499                    shard_scanned && mined_at_anchor && has_confirmations,
500                ) {
501                    (NoteRequest::UnspentOrError { .. }, false) => {
502                        Err(SqliteClientError::IneligibleNotes)
503                    }
504                    (NoteRequest::Spendable { .. }, false) => Ok(None),
505                    (NoteRequest::Unspent, false) | (_, true) => Ok(Some(note)),
506                }
507            }
508            _ => Err(SqliteClientError::IneligibleNotes),
509        })
510        .filter_map(|r| r.transpose())
511        .collect()
512}

514/// Selects the set of spendable notes whose sum will be equal or greater that the
515/// specified ``target_value`` in Zatoshis from the specified shielded protocols excluding
516/// the ones present in the ``exclude`` slice.
517///
518/// - Implementation details
519///   - Notes with individual value *below* the ``MARGINAL_FEE`` will be ignored
520///   - Note spendability is determined using the `anchor_height`
521#[allow(clippy::too_many_arguments)]
522fn select_spendable_notes_matching_value<P: consensus::Parameters, F, Note>(
523    conn: &Connection,
524    params: &P,
525    account: AccountUuid,
526    target_value: Zatoshis,
527    target_height: TargetHeight,
528    anchor_height: BlockHeight,
529    confirmations_policy: ConfirmationsPolicy,
530    exclude: &[ReceivedNoteId],
531    protocol: ShieldedProtocol,
532    to_spendable_note: F,
533) -> Result<Vec<ReceivedNote<ReceivedNoteId, Note>>, SqliteClientError>
534where
535    F: Fn(&P, &Row) -> Result<Option<ReceivedNote<ReceivedNoteId, Note>>, SqliteClientError>,
536{
537    let TableConstants {
538        table_prefix,
539        output_index_col,
540        note_reconstruction_cols,
541        ..
542    } = table_constants::<SqliteClientError>(protocol)?;
543    if unscanned_tip_exists(conn, anchor_height, table_prefix)? {
544        return Ok(vec![]);
545    }
546
547    // The goal of this SQL statement is to select the oldest notes until the required
548    // value has been reached.
549    // 1) Use a window function to create a view of all notes, ordered from oldest to
550    //    newest, with an additional column containing a running sum:
551    //    - Unspent notes accumulate the values of all unspent notes in that note's
552    //      account, up to itself.
553    //    - Spent notes accumulate the values of all notes in the transaction they were
554    //      spent in, up to itself.
555    //
556    // 2) Select all unspent notes in the desired account, along with their running sum.
557    //
558    // 3) Select all notes for which the running sum was less than the required value, as
559    //    well as a single note for which the sum was greater than or equal to the
560    //    required value, bringing the sum of all selected notes across the threshold.
561    let mut stmt_select_notes = conn.prepare_cached(&format!(
562        "WITH eligible AS (
563             SELECT
564                 rn.id AS id, t.txid, rn.{output_index_col},
565                 rn.diversifier, rn.value,
566                 {note_reconstruction_cols}, rn.commitment_tree_position,
567                 SUM(value) OVER (ROWS UNBOUNDED PRECEDING) AS so_far,
568                 accounts.ufvk as ufvk, rn.recipient_key_scope,
569                 t.block AS mined_height,
570                 IFNULL(t.trust_status, 0) AS trust_status,
571                 MAX(tt.mined_height) AS max_shielding_input_height,
572                 MIN(IFNULL(tt.trust_status, 0)) AS min_shielding_input_trust
573             FROM {table_prefix}_received_notes rn
574             INNER JOIN accounts ON accounts.id = rn.account_id
575             INNER JOIN transactions t ON t.id_tx = rn.tx
576             LEFT OUTER JOIN v_{table_prefix}_shards_scan_state scan_state
577                ON rn.commitment_tree_position >= scan_state.start_position
578                AND rn.commitment_tree_position < scan_state.end_position_exclusive
579             LEFT OUTER JOIN transparent_received_output_spends ros
580                ON ros.transaction_id = t.id_tx
581             LEFT OUTER JOIN transparent_received_outputs tro
582                ON tro.id = ros.transparent_received_output_id
583                AND tro.account_id = accounts.id
584             LEFT OUTER JOIN transactions tt
585                ON tt.id_tx = tro.transaction_id
586             WHERE accounts.uuid = :account_uuid
587             AND rn.value > :min_value
588             AND accounts.ufvk IS NOT NULL
589             AND recipient_key_scope IS NOT NULL
590             AND nf IS NOT NULL
591             -- the shard containing the note is fully scanned; this condition will exclude
592             -- notes for which `scan_state.max_priority IS NULL` (which will also arise if
593             -- `rn.commitment_tree_position IS NULL`; hence we don't need that explicit filter)
594             AND scan_state.max_priority <= :scanned_priority
595             AND t.block <= :anchor_height
596             AND rn.id NOT IN rarray(:exclude)
597             AND rn.id NOT IN ({})
598             GROUP BY rn.id
599         )
600         SELECT id, txid, {output_index_col},
601                diversifier, value, {note_reconstruction_cols}, commitment_tree_position,
602                ufvk, recipient_key_scope,
603                mined_height, trust_status,
604                max_shielding_input_height, min_shielding_input_trust
605         FROM eligible WHERE so_far < :target_value
606         UNION
607         SELECT id, txid, {output_index_col},
608                diversifier, value, {note_reconstruction_cols}, commitment_tree_position,
609                ufvk, recipient_key_scope,
610                mined_height, trust_status,
611                max_shielding_input_height, min_shielding_input_trust
612         FROM (SELECT * from eligible WHERE so_far >= :target_value LIMIT 1)",
613        spent_notes_clause(table_prefix)
614    ))?;
615
616    let excluded: Vec<Value> = exclude
617        .iter()
618        .filter_map(|ReceivedNoteId(p, n)| {
619            if *p == protocol {
620                Some(Value::from(*n))
621            } else {
622                None
623            }
624        })
625        .collect();
626    let excluded_ptr = Rc::new(excluded);
627
628    let notes = stmt_select_notes.query_and_then(
629        named_params![
630            ":account_uuid": account.0,
631            ":anchor_height": &u32::from(anchor_height),
632            ":target_height": &u32::from(target_height),
633            ":target_value": &u64::from(target_value),
634            ":exclude": &excluded_ptr,
635            ":scanned_priority": priority_code(&ScanPriority::Scanned),
636            ":min_value": u64::from(zip317::MARGINAL_FEE)
637        ],
638        |row| {
639            let tx_trust_status = row.get::<_, bool>("trust_status")?;
640            let tx_shielding_inputs_trusted = row.get::<_, bool>("min_shielding_input_trust")?;
641            let note = to_spendable_note(params, row)?;
642
643            Ok(note.map(|n| (n, tx_trust_status, tx_shielding_inputs_trusted)))
644        },
645    )?;
646
647    let trusted_height = target_height.saturating_sub(u32::from(confirmations_policy.trusted()));
648    let untrusted_height =
649        target_height.saturating_sub(u32::from(confirmations_policy.untrusted()));
650
651    notes
652        .filter_map(|result_maybe_note| {
653            let result_note = result_maybe_note.transpose()?;
654            result_note
655                .map(|(note, trusted, tx_shielding_inputs_trusted)| {
656                    let received_height = note
657                        .mined_height()
658                        .expect("mined height checked to be non-null");
659
660                    let has_confirmations = match note.spending_key_scope() {
661                        Scope::Internal => {
662                            // The note was has at least `trusted` confirmations.
663                            received_height <= trusted_height &&
664                            // And, if the note was the output of a shielding transaction, its
665                            // transparent inputs have at least `untrusted` confirmations.
666                            note.max_shielding_input_height().iter().all(|h| {
667                                // its inputs have at least `untrusted` confirmations
668                                h <= &untrusted_height ||
669                                // or its inputs are trusted and have at least `trusted` confirmations
670                                (h <= &trusted_height && tx_shielding_inputs_trusted)
671                            })
672                        }
673                        Scope::External => {
674                            // The note has the required number of confirmations for an untrusted note.
675                            received_height <= untrusted_height ||
676                            // or it is the output of an explicitly trusted tx and has at least
677                            // `trusted` confirmations
678                            (received_height <= trusted_height && trusted)
679                        }
680                    };
681
682                    has_confirmations.then_some(note)
683                })
684                .transpose()
685        })
686        .collect::<Result<Vec<_>, _>>()
687}

721pub(crate) fn select_unspent_note_meta(
722    conn: &rusqlite::Connection,
723    protocol: ShieldedProtocol,
724    wallet_birthday: BlockHeight,
725    anchor_height: BlockHeight,
726) -> Result<Vec<UnspentNoteMeta>, SqliteClientError> {
727    let TableConstants {
728        table_prefix,
729        output_index_col,
730        ..
731    } = table_constants::<SqliteClientError>(protocol)?;
732
733    // This query is effectively the same as the internal `eligible` subquery
734    // used in `select_spendable_notes`.
735    //
736    // TODO: Deduplicate this in the future by introducing a view?
737    let mut stmt = conn.prepare_cached(&format!(
738        "SELECT {table_prefix}_received_notes.id AS id, txid, {output_index_col},
739                commitment_tree_position, value
740         FROM {table_prefix}_received_notes rn
741         INNER JOIN transactions ON transactions.id_tx = rn.tx
742         WHERE value > 5000 -- FIXME #1316, allow selection of dust inputs
743         AND recipient_key_scope IS NOT NULL
744         AND nf IS NOT NULL
745         AND commitment_tree_position IS NOT NULL
746         AND rn.id NOT IN ({})
747         AND NOT EXISTS (
748            SELECT 1 FROM v_{table_prefix}_shard_unscanned_ranges unscanned
749            -- select all the unscanned ranges involving the shard containing this note
750            WHERE rn.commitment_tree_position >= unscanned.start_position
751            AND rn.commitment_tree_position < unscanned.end_position_exclusive
752            -- exclude unscanned ranges that start above the anchor height (they don't affect spendability)
753            AND unscanned.block_range_start <= :anchor_height
754            -- exclude unscanned ranges that end below the wallet birthday
755            AND unscanned.block_range_end > :wallet_birthday
756         )",
757         spent_notes_clause(table_prefix)
758    ))?;
759
760    let res = stmt
761        .query_and_then::<_, SqliteClientError, _, _>(
762            named_params![
763                ":wallet_birthday": u32::from(wallet_birthday),
764                ":anchor_height": u32::from(anchor_height),
765            ],
766            |row| {
767                Ok(UnspentNoteMeta {
768                    note_id: row.get("id").map(|id| ReceivedNoteId(protocol, id))?,
769                    txid: row.get("txid").map(TxId::from_bytes)?,
770                    output_index: row.get(output_index_col)?,
771                    commitment_tree_position: row
772                        .get::<_, u64>("commitment_tree_position")
773                        .map(Position::from)?,
774                    value: Zatoshis::from_nonnegative_i64(row.get("value")?)?,
775                })
776            },
777        )?
778        .collect::<Result<Vec<_>, _>>()?;
779
780    Ok(res)
781}

783pub(crate) fn unspent_notes_meta(
784    conn: &rusqlite::Connection,
785    protocol: ShieldedProtocol,
786    target_height: TargetHeight,
787    account: AccountUuid,
788    filter: &NoteFilter,
789    exclude: &[ReceivedNoteId],
790) -> Result<Option<PoolMeta>, SqliteClientError> {
791    let TableConstants { table_prefix, .. } = table_constants::<SqliteClientError>(protocol)?;
792
793    let excluded: Vec<Value> = exclude
794        .iter()
795        .filter_map(|ReceivedNoteId(p, n)| {
796            if *p == protocol {
797                Some(Value::from(*n))
798            } else {
799                None
800            }
801        })
802        .collect();
803    let excluded_ptr = Rc::new(excluded);
804
805    fn zatoshis(value: i64) -> Result<Zatoshis, SqliteClientError> {
806        Zatoshis::from_nonnegative_i64(value).map_err(|_| {
807            SqliteClientError::CorruptedData(format!("Negative received note value: {value}"))
808        })
809    }
810
811    let run_selection = |min_value| {
812        conn.query_row_and_then::<_, SqliteClientError, _, _>(
813            &format!(
814                "SELECT COUNT(*), SUM(rn.value)
815                 FROM {table_prefix}_received_notes rn
816                 INNER JOIN accounts a ON a.id = rn.account_id
817                 INNER JOIN transactions ON transactions.id_tx = rn.tx
818                 WHERE a.uuid = :account_uuid
819                 AND a.ufvk IS NOT NULL
820                 AND rn.value >= :min_value
821                 AND transactions.mined_height IS NOT NULL
822                 AND rn.id NOT IN rarray(:exclude)
823                 AND rn.id NOT IN ({})",
824                spent_notes_clause(table_prefix)
825            ),
826            named_params![
827                ":account_uuid": account.0,
828                ":min_value": u64::from(min_value),
829                ":exclude": &excluded_ptr,
830                ":target_height": u32::from(target_height)
831            ],
832            |row| {
833                Ok((
834                    row.get::<_, usize>(0)?,
835                    row.get::<_, Option<i64>>(1)?.map(zatoshis).transpose()?,
836                ))
837            },
838        )
839    };
840
841    // Evaluates the provided note filter conditions against the wallet database in order to
842    // determine the minimum value of notes to be produced by note splitting.
843    fn min_note_value(
844        conn: &rusqlite::Connection,
845        account: AccountUuid,
846        filter: &NoteFilter,
847        target_height: TargetHeight,
848    ) -> Result<Option<Zatoshis>, SqliteClientError> {
849        match filter {
850            NoteFilter::ExceedsMinValue(v) => Ok(Some(*v)),
851            NoteFilter::ExceedsPriorSendPercentile(n) => {
852                let mut bucket_query = conn.prepare(
853                    "WITH bucketed AS (
854                        SELECT s.value, NTILE(10) OVER (ORDER BY s.value) AS bucket_index
855                        FROM sent_notes s
856                        JOIN transactions t ON s.tx = t.id_tx
857                        JOIN accounts a on a.id = s.from_account_id
858                        WHERE a.uuid = :account_uuid
859                        -- only count mined transactions
860                        AND t.mined_height IS NOT NULL
861                        -- exclude change and account-internal sends
862                        AND (s.to_account_id IS NULL OR s.from_account_id != s.to_account_id)
863                    )
864                    SELECT MAX(value) as value
865                    FROM bucketed
866                    GROUP BY bucket_index
867                    ORDER BY bucket_index",
868                )?;
869
870                let bucket_maxima = bucket_query
871                    .query_and_then::<_, SqliteClientError, _, _>(
872                        named_params![":account_uuid": account.0],
873                        |row| {
874                            Zatoshis::from_nonnegative_i64(row.get::<_, i64>(0)?).map_err(|_| {
875                                SqliteClientError::CorruptedData(format!(
876                                    "Negative received note value: {}",
877                                    n.value()
878                                ))
879                            })
880                        },
881                    )?
882                    .collect::<Result<Vec<_>, _>>()?;
883
884                // Pick a bucket index by scaling the requested percentile to the number of buckets
885                let i = (bucket_maxima.len() * usize::from(*n) / 100).saturating_sub(1);
886                Ok(bucket_maxima.get(i).copied())
887            }
888            NoteFilter::ExceedsBalancePercentage(p) => {
889                let balance = conn.query_row_and_then::<_, SqliteClientError, _, _>(
890                    &format!(
891                        "SELECT SUM(rn.value)
892                         FROM v_received_outputs rn
893                         INNER JOIN accounts a ON a.id = rn.account_id
894                         INNER JOIN transactions ON transactions.id_tx = rn.transaction_id
895                         WHERE a.uuid = :account_uuid
896                         AND a.ufvk IS NOT NULL
897                         AND transactions.mined_height IS NOT NULL
898                         AND rn.pool != :transparent_pool
899                         AND (rn.pool, rn.id_within_pool_table) NOT IN (
900                            SELECT rns.pool, rns.received_output_id
901                            FROM v_received_output_spends rns
902                            JOIN transactions stx ON stx.id_tx = rns.transaction_id
903                            WHERE ({})  -- the spending transaction is unexpired
904                         )",
905                        tx_unexpired_condition("stx")
906                    ),
907                    named_params![
908                        ":account_uuid": account.0,
909                        ":transparent_pool": pool_code(PoolType::Transparent),
910                        ":target_height": u32::from(target_height),
911                    ],
912                    |row| row.get::<_, Option<i64>>(0)?.map(zatoshis).transpose(),
913                )?;
914
915                Ok(match balance {
916                    None => None,
917                    Some(b) => {
918                        let numerator = (b * u64::from(p.value())).ok_or(BalanceError::Overflow)?;
919                        Some(numerator / NonZeroU64::new(100).expect("Constant is nonzero."))
920                    }
921                })
922            }
923            NoteFilter::Combine(a, b) => {
924                // All the existing note selectors set lower bounds on note value, so the "and"
925                // operation is just taking the maximum of the two lower bounds.
926                let a_min_value = min_note_value(conn, account, a.as_ref(), target_height)?;
927                let b_min_value = min_note_value(conn, account, b.as_ref(), target_height)?;
928                Ok(a_min_value
929                    .zip(b_min_value)
930                    .map(|(av, bv)| std::cmp::max(av, bv))
931                    .or(a_min_value)
932                    .or(b_min_value))
933            }
934            NoteFilter::Attempt {
935                condition,
936                fallback,
937            } => {
938                let cond = min_note_value(conn, account, condition.as_ref(), target_height)?;
939                if cond.is_none() {
940                    min_note_value(conn, account, fallback, target_height)
941                } else {
942                    Ok(cond)
943                }
944            }
945        }
946    }
947
948    // TODO: Simplify the query before executing it. Not worrying about this now because queries
949    // will be developer-configured, not end-user defined.
950    if let Some(min_value) = min_note_value(conn, account, filter, target_height)? {
951        let (note_count, total_value) = run_selection(min_value)?;
952
953        Ok(Some(PoolMeta::new(
954            note_count,
955            total_value.unwrap_or(Zatoshis::ZERO),
956        )))
957    } else {
958        Ok(None)
959    }
960}
