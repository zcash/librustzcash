25pub(crate) fn per_protocol_names(
26    protocol: ShieldedProtocol,
27) -> (&'static str, &'static str, &'static str) {
28    match protocol {
29        ShieldedProtocol::Sapling => (SAPLING_TABLES_PREFIX, "output_index", "rcm"),
30        #[cfg(feature = "orchard")]
31        ShieldedProtocol::Orchard => (ORCHARD_TABLES_PREFIX, "action_index", "rho, rseed"),
32        #[cfg(not(feature = "orchard"))]
33        ShieldedProtocol::Orchard => {
34            unreachable!("Should never be called unless the `orchard` feature is enabled")
35        }
36    }
37}

39fn unscanned_tip_exists(
40    conn: &Connection,
41    anchor_height: BlockHeight,
42    table_prefix: &'static str,
43) -> Result<bool, rusqlite::Error> {
44    // v_sapling_shard_unscanned_ranges only returns ranges ending on or after wallet birthday, so
45    // we don't need to refer to the birthday in this query.
46    conn.query_row(
47        &format!(
48            "SELECT EXISTS (
49                 SELECT 1 FROM v_{table_prefix}_shard_unscanned_ranges range
50                 WHERE range.block_range_start <= :anchor_height
51                 AND :anchor_height BETWEEN
52                    range.subtree_start_height
53                    AND IFNULL(range.subtree_end_height, :anchor_height)
54             )"
55        ),
56        named_params![":anchor_height": u32::from(anchor_height),],
57        |row| row.get::<_, bool>(0),
58    )
59}

61// The `clippy::let_and_return` lint is explicitly allowed here because a bug in Clippy
62// (https://github.com/rust-lang/rust-clippy/issues/11308) means it fails to identify that the `result` temporary
63// is required in order to resolve the borrows involved in the `query_and_then` call.
64#[allow(clippy::let_and_return)]
65pub(crate) fn get_spendable_note<P: consensus::Parameters, F, Note>(
66    conn: &Connection,
67    params: &P,
68    txid: &TxId,
69    index: u32,
70    protocol: ShieldedProtocol,
71    to_spendable_note: F,
72) -> Result<Option<ReceivedNote<ReceivedNoteId, Note>>, SqliteClientError>
73where
74    F: Fn(&P, &Row) -> Result<Option<ReceivedNote<ReceivedNoteId, Note>>, SqliteClientError>,
75{
76    let (table_prefix, index_col, note_reconstruction_cols) = per_protocol_names(protocol);
77    let result = conn.query_row_and_then(
78        &format!(
79            "SELECT rn.id, txid, {index_col},
80                diversifier, value, {note_reconstruction_cols}, commitment_tree_position,
81                accounts.ufvk, recipient_key_scope
82             FROM {table_prefix}_received_notes rn
83             INNER JOIN accounts ON accounts.id = rn.account_id
84             INNER JOIN transactions ON transactions.id_tx = rn.tx
85             WHERE txid = :txid
86             AND transactions.block IS NOT NULL
87             AND {index_col} = :output_index
88             AND accounts.ufvk IS NOT NULL
89             AND recipient_key_scope IS NOT NULL
90             AND nf IS NOT NULL
91             AND commitment_tree_position IS NOT NULL
92             AND rn.id NOT IN (
93               SELECT {table_prefix}_received_note_id
94               FROM {table_prefix}_received_note_spends
95               JOIN transactions stx ON stx.id_tx = transaction_id
96               WHERE stx.block IS NOT NULL -- the spending tx is mined
97               OR stx.expiry_height IS NULL -- the spending tx will not expire
98             )"
99        ),
100        named_params![
101           ":txid": txid.as_ref(),
102           ":output_index": index,
103        ],
104        |row| to_spendable_note(params, row),
105    );
106
107    // `OptionalExtension` doesn't work here because the error type of `Result` is already
108    // `SqliteClientError`
109    match result {
110        Ok(r) => Ok(r),
111        Err(SqliteClientError::DbError(rusqlite::Error::QueryReturnedNoRows)) => Ok(None),
112        Err(e) => Err(e),
113    }
114}

116#[allow(clippy::too_many_arguments)]
117pub(crate) fn select_spendable_notes<P: consensus::Parameters, F, Note>(
118    conn: &Connection,
119    params: &P,
120    account: AccountUuid,
121    target_value: Zatoshis,
122    anchor_height: BlockHeight,
123    exclude: &[ReceivedNoteId],
124    protocol: ShieldedProtocol,
125    to_spendable_note: F,
126) -> Result<Vec<ReceivedNote<ReceivedNoteId, Note>>, SqliteClientError>
127where
128    F: Fn(&P, &Row) -> Result<Option<ReceivedNote<ReceivedNoteId, Note>>, SqliteClientError>,
129{
130    let birthday_height = match wallet_birthday(conn)? {
131        Some(birthday) => birthday,
132        None => {
133            // the wallet birthday can only be unknown if there are no accounts in the wallet; in
134            // such a case, the wallet has no notes to spend.
135            return Ok(vec![]);
136        }
137    };
138
139    let (table_prefix, index_col, note_reconstruction_cols) = per_protocol_names(protocol);
140    if unscanned_tip_exists(conn, anchor_height, table_prefix)? {
141        return Ok(vec![]);
142    }
143
144    // The goal of this SQL statement is to select the oldest notes until the required
145    // value has been reached.
146    // 1) Use a window function to create a view of all notes, ordered from oldest to
147    //    newest, with an additional column containing a running sum:
148    //    - Unspent notes accumulate the values of all unspent notes in that note's
149    //      account, up to itself.
150    //    - Spent notes accumulate the values of all notes in the transaction they were
151    //      spent in, up to itself.
152    //
153    // 2) Select all unspent notes in the desired account, along with their running sum.
154    //
155    // 3) Select all notes for which the running sum was less than the required value, as
156    //    well as a single note for which the sum was greater than or equal to the
157    //    required value, bringing the sum of all selected notes across the threshold.
158    let mut stmt_select_notes = conn.prepare_cached(
159        &format!(
160            "WITH eligible AS (
161                 SELECT
162                     {table_prefix}_received_notes.id AS id, txid, {index_col},
163                     diversifier, value, {note_reconstruction_cols}, commitment_tree_position,
164                     SUM(value) OVER (ROWS UNBOUNDED PRECEDING) AS so_far,
165                     accounts.ufvk as ufvk, recipient_key_scope
166                 FROM {table_prefix}_received_notes
167                 INNER JOIN accounts
168                    ON accounts.id = {table_prefix}_received_notes.account_id
169                 INNER JOIN transactions
170                    ON transactions.id_tx = {table_prefix}_received_notes.tx
171                 WHERE accounts.uuid = :account_uuid
172                 AND {table_prefix}_received_notes.account_id = accounts.id
173                 AND value > 5000 -- FIXME #1316, allow selection of dust inputs
174                 AND accounts.ufvk IS NOT NULL
175                 AND recipient_key_scope IS NOT NULL
176                 AND nf IS NOT NULL
177                 AND commitment_tree_position IS NOT NULL
178                 AND transactions.block <= :anchor_height
179                 AND {table_prefix}_received_notes.id NOT IN rarray(:exclude)
180                 AND {table_prefix}_received_notes.id NOT IN (
181                   SELECT {table_prefix}_received_note_id
182                   FROM {table_prefix}_received_note_spends
183                   JOIN transactions stx ON stx.id_tx = transaction_id
184                   WHERE stx.block IS NOT NULL -- the spending tx is mined
185                   OR stx.expiry_height IS NULL -- the spending tx will not expire
186                   OR stx.expiry_height > :anchor_height -- the spending tx is unexpired
187                 )
188                 AND NOT EXISTS (
189                    SELECT 1 FROM v_{table_prefix}_shard_unscanned_ranges unscanned
190                    -- select all the unscanned ranges involving the shard containing this note
191                    WHERE {table_prefix}_received_notes.commitment_tree_position >= unscanned.start_position
192                    AND {table_prefix}_received_notes.commitment_tree_position < unscanned.end_position_exclusive
193                    -- exclude unscanned ranges that start above the anchor height (they don't affect spendability)
194                    AND unscanned.block_range_start <= :anchor_height
195                    -- exclude unscanned ranges that end below the wallet birthday
196                    AND unscanned.block_range_end > :wallet_birthday
197                 )
198             )
199             SELECT id, txid, {index_col},
200                    diversifier, value, {note_reconstruction_cols}, commitment_tree_position,
201                    ufvk, recipient_key_scope
202             FROM eligible WHERE so_far < :target_value
203             UNION
204             SELECT id, txid, {index_col},
205                    diversifier, value, {note_reconstruction_cols}, commitment_tree_position,
206                    ufvk, recipient_key_scope
207             FROM (SELECT * from eligible WHERE so_far >= :target_value LIMIT 1)",
208        )
209    )?;
210
211    let excluded: Vec<Value> = exclude
212        .iter()
213        .filter_map(|ReceivedNoteId(p, n)| {
214            if *p == protocol {
215                Some(Value::from(*n))
216            } else {
217                None
218            }
219        })
220        .collect();
221    let excluded_ptr = Rc::new(excluded);
222
223    let notes = stmt_select_notes.query_and_then(
224        named_params![
225            ":account_uuid": account.0,
226            ":anchor_height": &u32::from(anchor_height),
227            ":target_value": &u64::from(target_value),
228            ":exclude": &excluded_ptr,
229            ":wallet_birthday": u32::from(birthday_height)
230        ],
231        |r| to_spendable_note(params, r),
232    )?;
233
234    notes
235        .filter_map(|r| r.transpose())
236        .collect::<Result<_, _>>()
237}

239pub(crate) fn spendable_notes_meta(
240    conn: &rusqlite::Connection,
241    protocol: ShieldedProtocol,
242    chain_tip_height: BlockHeight,
243    account: AccountUuid,
244    filter: &NoteFilter,
245    exclude: &[ReceivedNoteId],
246) -> Result<Option<PoolMeta>, SqliteClientError> {
247    let (table_prefix, _, _) = per_protocol_names(protocol);
248
249    let excluded: Vec<Value> = exclude
250        .iter()
251        .filter_map(|ReceivedNoteId(p, n)| {
252            if *p == protocol {
253                Some(Value::from(*n))
254            } else {
255                None
256            }
257        })
258        .collect();
259    let excluded_ptr = Rc::new(excluded);
260
261    fn zatoshis(value: i64) -> Result<Zatoshis, SqliteClientError> {
262        Zatoshis::from_nonnegative_i64(value).map_err(|_| {
263            SqliteClientError::CorruptedData(format!("Negative received note value: {}", value))
264        })
265    }
266
267    let run_selection = |min_value| {
268        conn.query_row_and_then::<_, SqliteClientError, _, _>(
269            &format!(
270                "SELECT COUNT(*), SUM(rn.value)
271                 FROM {table_prefix}_received_notes rn
272                 INNER JOIN accounts a ON a.id = rn.account_id
273                 INNER JOIN transactions ON transactions.id_tx = rn.tx
274                 WHERE a.uuid = :account_uuid
275                 AND a.ufvk IS NOT NULL
276                 AND rn.value >= :min_value
277                 AND transactions.mined_height IS NOT NULL
278                 AND rn.id NOT IN rarray(:exclude)
279                 AND rn.id NOT IN (
280                   SELECT {table_prefix}_received_note_id
281                   FROM {table_prefix}_received_note_spends rns
282                   JOIN transactions stx ON stx.id_tx = rns.transaction_id
283                   WHERE stx.block IS NOT NULL -- the spending tx is mined
284                   OR stx.expiry_height IS NULL -- the spending tx will not expire
285                   OR stx.expiry_height > :chain_tip_height -- the spending tx is unexpired
286                 )"
287            ),
288            named_params![
289                ":account_uuid": account.0,
290                ":min_value": u64::from(min_value),
291                ":exclude": &excluded_ptr,
292                ":chain_tip_height": u32::from(chain_tip_height)
293            ],
294            |row| {
295                Ok((
296                    row.get::<_, usize>(0)?,
297                    row.get::<_, Option<i64>>(1)?.map(zatoshis).transpose()?,
298                ))
299            },
300        )
301    };
302
303    // Evaluates the provided note filter conditions against the wallet database in order to
304    // determine the minimum value of notes to be produced by note splitting.
305    fn min_note_value(
306        conn: &rusqlite::Connection,
307        account: AccountUuid,
308        filter: &NoteFilter,
309        chain_tip_height: BlockHeight,
310    ) -> Result<Option<Zatoshis>, SqliteClientError> {
311        match filter {
312            NoteFilter::ExceedsMinValue(v) => Ok(Some(*v)),
313            NoteFilter::ExceedsPriorSendPercentile(n) => {
314                let mut bucket_query = conn.prepare(
315                    "WITH bucketed AS (
316                        SELECT s.value, NTILE(10) OVER (ORDER BY s.value) AS bucket_index
317                        FROM sent_notes s
318                        JOIN transactions t ON s.tx = t.id_tx
319                        JOIN accounts a on a.id = s.from_account_id
320                        WHERE a.uuid = :account_uuid
321                        -- only count mined transactions
322                        AND t.mined_height IS NOT NULL
323                        -- exclude change and account-internal sends
324                        AND (s.to_account_id IS NULL OR s.from_account_id != s.to_account_id)
325                    )
326                    SELECT MAX(value) as value
327                    FROM bucketed
328                    GROUP BY bucket_index
329                    ORDER BY bucket_index",
330                )?;
331
332                let bucket_maxima = bucket_query
333                    .query_and_then::<_, SqliteClientError, _, _>(
334                        named_params![":account_uuid": account.0],
335                        |row| {
336                            Zatoshis::from_nonnegative_i64(row.get::<_, i64>(0)?).map_err(|_| {
337                                SqliteClientError::CorruptedData(format!(
338                                    "Negative received note value: {}",
339                                    n.value()
340                                ))
341                            })
342                        },
343                    )?
344                    .collect::<Result<Vec<_>, _>>()?;
345
346                // Pick a bucket index by scaling the requested percentile to the number of buckets
347                let i = (bucket_maxima.len() * usize::from(*n) / 100).saturating_sub(1);
348                Ok(bucket_maxima.get(i).copied())
349            }
350            NoteFilter::ExceedsBalancePercentage(p) => {
351                let balance = conn.query_row_and_then::<_, SqliteClientError, _, _>(
352                    "SELECT SUM(rn.value)
353                     FROM v_received_outputs rn
354                     INNER JOIN accounts a ON a.id = rn.account_id
355                     INNER JOIN transactions ON transactions.id_tx = rn.transaction_id
356                     WHERE a.uuid = :account_uuid
357                     AND a.ufvk IS NOT NULL
358                     AND transactions.mined_height IS NOT NULL
359                     AND rn.pool != :transparent_pool
360                     AND (rn.pool, rn.id_within_pool_table) NOT IN (
361                       SELECT rns.pool, rns.received_output_id
362                       FROM v_received_output_spends rns
363                       JOIN transactions stx ON stx.id_tx = rns.transaction_id
364                       WHERE (
365                           stx.block IS NOT NULL -- the spending tx is mined
366                           OR stx.expiry_height IS NULL -- the spending tx will not expire
367                           OR stx.expiry_height > :chain_tip_height -- the spending tx is unexpired
368                       )
369                     )",
370                    named_params![
371                        ":account_uuid": account.0,
372                        ":chain_tip_height": u32::from(chain_tip_height),
373                        ":transparent_pool": pool_code(PoolType::Transparent)
374                    ],
375                    |row| row.get::<_, Option<i64>>(0)?.map(zatoshis).transpose(),
376                )?;
377
378                Ok(match balance {
379                    None => None,
380                    Some(b) => {
381                        let numerator = (b * u64::from(p.value())).ok_or(BalanceError::Overflow)?;
382                        Some(numerator / NonZeroU64::new(100).expect("Constant is nonzero."))
383                    }
384                })
385            }
386            NoteFilter::Combine(a, b) => {
387                // All the existing note selectors set lower bounds on note value, so the "and"
388                // operation is just taking the maximum of the two lower bounds.
389                let a_min_value = min_note_value(conn, account, a.as_ref(), chain_tip_height)?;
390                let b_min_value = min_note_value(conn, account, b.as_ref(), chain_tip_height)?;
391                Ok(a_min_value
392                    .zip(b_min_value)
393                    .map(|(av, bv)| std::cmp::max(av, bv))
394                    .or(a_min_value)
395                    .or(b_min_value))
396            }
397            NoteFilter::Attempt {
398                condition,
399                fallback,
400            } => {
401                let cond = min_note_value(conn, account, condition.as_ref(), chain_tip_height)?;
402                if cond.is_none() {
403                    min_note_value(conn, account, fallback, chain_tip_height)
404                } else {
405                    Ok(cond)
406                }
407            }
408        }
409    }
410
411    // TODO: Simplify the query before executing it. Not worrying about this now because queries
412    // will be developer-configured, not end-user defined.
413    if let Some(min_value) = min_note_value(conn, account, filter, chain_tip_height)? {
414        let (note_count, total_value) = run_selection(min_value)?;
415
416        Ok(Some(PoolMeta::new(
417            note_count,
418            total_value.unwrap_or(Zatoshis::ZERO),
419        )))
420    } else {
421        Ok(None)
422    }
