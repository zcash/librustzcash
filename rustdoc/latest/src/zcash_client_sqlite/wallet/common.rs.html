26pub(crate) fn per_protocol_names(
27    protocol: ShieldedProtocol,
28) -> (&'static str, &'static str, &'static str) {
29    match protocol {
30        ShieldedProtocol::Sapling => (SAPLING_TABLES_PREFIX, "output_index", "rcm"),
31        #[cfg(feature = "orchard")]
32        ShieldedProtocol::Orchard => (ORCHARD_TABLES_PREFIX, "action_index", "rho, rseed"),
33        #[cfg(not(feature = "orchard"))]
34        ShieldedProtocol::Orchard => {
35            unreachable!("Should never be called unless the `orchard` feature is enabled")
36        }
37    }
38}

40fn unscanned_tip_exists(
41    conn: &Connection,
42    anchor_height: BlockHeight,
43    table_prefix: &'static str,
44) -> Result<bool, rusqlite::Error> {
45    // v_sapling_shard_unscanned_ranges only returns ranges ending on or after wallet birthday, so
46    // we don't need to refer to the birthday in this query.
47    conn.query_row(
48        &format!(
49            "SELECT EXISTS (
50                 SELECT 1 FROM v_{table_prefix}_shard_unscanned_ranges range
51                 WHERE range.block_range_start <= :anchor_height
52                 AND :anchor_height BETWEEN
53                    range.subtree_start_height
54                    AND IFNULL(range.subtree_end_height, :anchor_height)
55             )"
56        ),
57        named_params![":anchor_height": u32::from(anchor_height),],
58        |row| row.get::<_, bool>(0),
59    )
60}

62// The `clippy::let_and_return` lint is explicitly allowed here because a bug in Clippy
63// (https://github.com/rust-lang/rust-clippy/issues/11308) means it fails to identify that the `result` temporary
64// is required in order to resolve the borrows involved in the `query_and_then` call.
65#[allow(clippy::let_and_return)]
66pub(crate) fn get_spendable_note<P: consensus::Parameters, F, Note>(
67    conn: &Connection,
68    params: &P,
69    txid: &TxId,
70    index: u32,
71    protocol: ShieldedProtocol,
72    to_spendable_note: F,
73) -> Result<Option<ReceivedNote<ReceivedNoteId, Note>>, SqliteClientError>
74where
75    F: Fn(&P, &Row) -> Result<Option<ReceivedNote<ReceivedNoteId, Note>>, SqliteClientError>,
76{
77    let (table_prefix, index_col, note_reconstruction_cols) = per_protocol_names(protocol);
78    let result = conn.query_row_and_then(
79        &format!(
80            "SELECT rn.id, txid, {index_col},
81                diversifier, value, {note_reconstruction_cols}, commitment_tree_position,
82                accounts.ufvk, recipient_key_scope
83             FROM {table_prefix}_received_notes rn
84             INNER JOIN accounts ON accounts.id = rn.account_id
85             INNER JOIN transactions ON transactions.id_tx = rn.tx
86             WHERE txid = :txid
87             AND transactions.block IS NOT NULL
88             AND {index_col} = :output_index
89             AND accounts.ufvk IS NOT NULL
90             AND recipient_key_scope IS NOT NULL
91             AND nf IS NOT NULL
92             AND commitment_tree_position IS NOT NULL
93             AND rn.id NOT IN (
94               SELECT {table_prefix}_received_note_id
95               FROM {table_prefix}_received_note_spends
96               JOIN transactions stx ON stx.id_tx = transaction_id
97               WHERE stx.block IS NOT NULL -- the spending tx is mined
98               OR stx.expiry_height IS NULL -- the spending tx will not expire
99             )"
100        ),
101        named_params![
102           ":txid": txid.as_ref(),
103           ":output_index": index,
104        ],
105        |row| to_spendable_note(params, row),
106    );
107
108    // `OptionalExtension` doesn't work here because the error type of `Result` is already
109    // `SqliteClientError`
110    match result {
111        Ok(r) => Ok(r),
112        Err(SqliteClientError::DbError(rusqlite::Error::QueryReturnedNoRows)) => Ok(None),
113        Err(e) => Err(e),
114    }
115}

118pub(crate) fn select_spendable_notes<P: consensus::Parameters, F, Note>(
119    conn: &Connection,
120    params: &P,
121    account: AccountUuid,
122    target_value: TargetValue,
123    anchor_height: BlockHeight,
124    exclude: &[ReceivedNoteId],
125    protocol: ShieldedProtocol,
126    to_spendable_note: F,
127) -> Result<Vec<ReceivedNote<ReceivedNoteId, Note>>, SqliteClientError>
128where
129    F: Fn(&P, &Row) -> Result<Option<ReceivedNote<ReceivedNoteId, Note>>, SqliteClientError>,
130{
131    match target_value {
132        TargetValue::AtLeast(zats) => select_minimum_spendable_notes(
133            conn,
134            params,
135            account,
136            zats,
137            anchor_height,
138            exclude,
139            protocol,
140            to_spendable_note,
141        ),
142    }
143}

145#[allow(clippy::too_many_arguments)]
146fn select_minimum_spendable_notes<P: consensus::Parameters, F, Note>(
147    conn: &Connection,
148    params: &P,
149    account: AccountUuid,
150    target_value: Zatoshis,
151    anchor_height: BlockHeight,
152    exclude: &[ReceivedNoteId],
153    protocol: ShieldedProtocol,
154    to_spendable_note: F,
155) -> Result<Vec<ReceivedNote<ReceivedNoteId, Note>>, SqliteClientError>
156where
157    F: Fn(&P, &Row) -> Result<Option<ReceivedNote<ReceivedNoteId, Note>>, SqliteClientError>,
158{
159    let birthday_height = match wallet_birthday(conn)? {
160        Some(birthday) => birthday,
161        None => {
162            // the wallet birthday can only be unknown if there are no accounts in the wallet; in
163            // such a case, the wallet has no notes to spend.
164            return Ok(vec![]);
165        }
166    };
167
168    let (table_prefix, index_col, note_reconstruction_cols) = per_protocol_names(protocol);
169    if unscanned_tip_exists(conn, anchor_height, table_prefix)? {
170        return Ok(vec![]);
171    }
172
173    // The goal of this SQL statement is to select the oldest notes until the required
174    // value has been reached.
175    // 1) Use a window function to create a view of all notes, ordered from oldest to
176    //    newest, with an additional column containing a running sum:
177    //    - Unspent notes accumulate the values of all unspent notes in that note's
178    //      account, up to itself.
179    //    - Spent notes accumulate the values of all notes in the transaction they were
180    //      spent in, up to itself.
181    //
182    // 2) Select all unspent notes in the desired account, along with their running sum.
183    //
184    // 3) Select all notes for which the running sum was less than the required value, as
185    //    well as a single note for which the sum was greater than or equal to the
186    //    required value, bringing the sum of all selected notes across the threshold.
187    let mut stmt_select_notes = conn.prepare_cached(
188        &format!(
189            "WITH eligible AS (
190                 SELECT
191                     {table_prefix}_received_notes.id AS id, txid, {index_col},
192                     diversifier, value, {note_reconstruction_cols}, commitment_tree_position,
193                     SUM(value) OVER (ROWS UNBOUNDED PRECEDING) AS so_far,
194                     accounts.ufvk as ufvk, recipient_key_scope
195                 FROM {table_prefix}_received_notes
196                 INNER JOIN accounts
197                    ON accounts.id = {table_prefix}_received_notes.account_id
198                 INNER JOIN transactions
199                    ON transactions.id_tx = {table_prefix}_received_notes.tx
200                 WHERE accounts.uuid = :account_uuid
201                 AND {table_prefix}_received_notes.account_id = accounts.id
202                 AND value > 5000 -- FIXME #1316, allow selection of dust inputs
203                 AND accounts.ufvk IS NOT NULL
204                 AND recipient_key_scope IS NOT NULL
205                 AND nf IS NOT NULL
206                 AND commitment_tree_position IS NOT NULL
207                 AND transactions.block <= :anchor_height
208                 AND {table_prefix}_received_notes.id NOT IN rarray(:exclude)
209                 AND {table_prefix}_received_notes.id NOT IN (
210                   SELECT {table_prefix}_received_note_id
211                   FROM {table_prefix}_received_note_spends
212                   JOIN transactions stx ON stx.id_tx = transaction_id
213                   WHERE stx.block IS NOT NULL -- the spending tx is mined
214                   OR stx.expiry_height IS NULL -- the spending tx will not expire
215                   OR stx.expiry_height > :anchor_height -- the spending tx is unexpired
216                 )
217                 AND NOT EXISTS (
218                    SELECT 1 FROM v_{table_prefix}_shard_unscanned_ranges unscanned
219                    -- select all the unscanned ranges involving the shard containing this note
220                    WHERE {table_prefix}_received_notes.commitment_tree_position >= unscanned.start_position
221                    AND {table_prefix}_received_notes.commitment_tree_position < unscanned.end_position_exclusive
222                    -- exclude unscanned ranges that start above the anchor height (they don't affect spendability)
223                    AND unscanned.block_range_start <= :anchor_height
224                    -- exclude unscanned ranges that end below the wallet birthday
225                    AND unscanned.block_range_end > :wallet_birthday
226                 )
227             )
228             SELECT id, txid, {index_col},
229                    diversifier, value, {note_reconstruction_cols}, commitment_tree_position,
230                    ufvk, recipient_key_scope
231             FROM eligible WHERE so_far < :target_value
232             UNION
233             SELECT id, txid, {index_col},
234                    diversifier, value, {note_reconstruction_cols}, commitment_tree_position,
235                    ufvk, recipient_key_scope
236             FROM (SELECT * from eligible WHERE so_far >= :target_value LIMIT 1)",
237        )
238    )?;
239
240    let excluded: Vec<Value> = exclude
241        .iter()
242        .filter_map(|ReceivedNoteId(p, n)| {
243            if *p == protocol {
244                Some(Value::from(*n))
245            } else {
246                None
247            }
248        })
249        .collect();
250    let excluded_ptr = Rc::new(excluded);
251
252    let notes = stmt_select_notes.query_and_then(
253        named_params![
254            ":account_uuid": account.0,
255            ":anchor_height": &u32::from(anchor_height),
256            ":target_value": &u64::from(target_value),
257            ":exclude": &excluded_ptr,
258            ":wallet_birthday": u32::from(birthday_height)
259        ],
260        |r| to_spendable_note(params, r),
261    )?;
262
263    notes
264        .filter_map(|r| r.transpose())
265        .collect::<Result<_, _>>()
266}

300pub(crate) fn select_unspent_note_meta(
301    conn: &rusqlite::Connection,
302    protocol: ShieldedProtocol,
303    chain_tip_height: BlockHeight,
304    wallet_birthday: BlockHeight,
305) -> Result<Vec<UnspentNoteMeta>, SqliteClientError> {
306    let (table_prefix, index_col, _) = per_protocol_names(protocol);
307    // This query is effectively the same as the internal `eligible` subquery
308    // used in `select_spendable_notes`.
309    //
310    // TODO: Deduplicate this in the future by introducing a view?
311    let mut stmt = conn.prepare_cached(&format!("
312        SELECT {table_prefix}_received_notes.id AS id, txid, {index_col},
313               commitment_tree_position, value
314        FROM {table_prefix}_received_notes
315        INNER JOIN transactions
316           ON transactions.id_tx = {table_prefix}_received_notes.tx
317        WHERE value > 5000 -- FIXME #1316, allow selection of dust inputs
318        AND recipient_key_scope IS NOT NULL
319        AND nf IS NOT NULL
320        AND commitment_tree_position IS NOT NULL
321        AND {table_prefix}_received_notes.id NOT IN (
322          SELECT {table_prefix}_received_note_id
323          FROM {table_prefix}_received_note_spends
324          JOIN transactions stx ON stx.id_tx = transaction_id
325          WHERE stx.block IS NOT NULL -- the spending tx is mined
326          OR stx.expiry_height IS NULL -- the spending tx will not expire
327          OR stx.expiry_height > :anchor_height -- the spending tx is unexpired
328        )
329        AND NOT EXISTS (
330           SELECT 1 FROM v_{table_prefix}_shard_unscanned_ranges unscanned
331           -- select all the unscanned ranges involving the shard containing this note
332           WHERE {table_prefix}_received_notes.commitment_tree_position >= unscanned.start_position
333           AND {table_prefix}_received_notes.commitment_tree_position < unscanned.end_position_exclusive
334           -- exclude unscanned ranges that start above the anchor height (they don't affect spendability)
335           AND unscanned.block_range_start <= :anchor_height
336           -- exclude unscanned ranges that end below the wallet birthday
337           AND unscanned.block_range_end > :wallet_birthday
338        )
339    "))?;
340
341    let res = stmt
342        .query_and_then::<_, SqliteClientError, _, _>(
343            named_params![
344                ":anchor_height": u32::from(chain_tip_height),
345                ":wallet_birthday": u32::from(wallet_birthday),
346            ],
347            |row| {
348                Ok(UnspentNoteMeta {
349                    note_id: row.get("id").map(|id| ReceivedNoteId(protocol, id))?,
350                    txid: row.get("txid").map(TxId::from_bytes)?,
351                    output_index: row.get(index_col)?,
352                    commitment_tree_position: row
353                        .get::<_, u64>("commitment_tree_position")
354                        .map(Position::from)?,
355                    value: Zatoshis::from_nonnegative_i64(row.get("value")?)?,
356                })
357            },
358        )?
359        .collect::<Result<Vec<_>, _>>()?;
360
361    Ok(res)
362}

364pub(crate) fn spendable_notes_meta(
365    conn: &rusqlite::Connection,
366    protocol: ShieldedProtocol,
367    chain_tip_height: BlockHeight,
368    account: AccountUuid,
369    filter: &NoteFilter,
370    exclude: &[ReceivedNoteId],
371) -> Result<Option<PoolMeta>, SqliteClientError> {
372    let (table_prefix, _, _) = per_protocol_names(protocol);
373
374    let excluded: Vec<Value> = exclude
375        .iter()
376        .filter_map(|ReceivedNoteId(p, n)| {
377            if *p == protocol {
378                Some(Value::from(*n))
379            } else {
380                None
381            }
382        })
383        .collect();
384    let excluded_ptr = Rc::new(excluded);
385
386    fn zatoshis(value: i64) -> Result<Zatoshis, SqliteClientError> {
387        Zatoshis::from_nonnegative_i64(value).map_err(|_| {
388            SqliteClientError::CorruptedData(format!("Negative received note value: {}", value))
389        })
390    }
391
392    let run_selection = |min_value| {
393        conn.query_row_and_then::<_, SqliteClientError, _, _>(
394            &format!(
395                "SELECT COUNT(*), SUM(rn.value)
396                 FROM {table_prefix}_received_notes rn
397                 INNER JOIN accounts a ON a.id = rn.account_id
398                 INNER JOIN transactions ON transactions.id_tx = rn.tx
399                 WHERE a.uuid = :account_uuid
400                 AND a.ufvk IS NOT NULL
401                 AND rn.value >= :min_value
402                 AND transactions.mined_height IS NOT NULL
403                 AND rn.id NOT IN rarray(:exclude)
404                 AND rn.id NOT IN (
405                   SELECT {table_prefix}_received_note_id
406                   FROM {table_prefix}_received_note_spends rns
407                   JOIN transactions stx ON stx.id_tx = rns.transaction_id
408                   WHERE stx.block IS NOT NULL -- the spending tx is mined
409                   OR stx.expiry_height IS NULL -- the spending tx will not expire
410                   OR stx.expiry_height > :chain_tip_height -- the spending tx is unexpired
411                 )"
412            ),
413            named_params![
414                ":account_uuid": account.0,
415                ":min_value": u64::from(min_value),
416                ":exclude": &excluded_ptr,
417                ":chain_tip_height": u32::from(chain_tip_height)
418            ],
419            |row| {
420                Ok((
421                    row.get::<_, usize>(0)?,
422                    row.get::<_, Option<i64>>(1)?.map(zatoshis).transpose()?,
423                ))
424            },
425        )
426    };
427
428    // Evaluates the provided note filter conditions against the wallet database in order to
429    // determine the minimum value of notes to be produced by note splitting.
430    fn min_note_value(
431        conn: &rusqlite::Connection,
432        account: AccountUuid,
433        filter: &NoteFilter,
434        chain_tip_height: BlockHeight,
435    ) -> Result<Option<Zatoshis>, SqliteClientError> {
436        match filter {
437            NoteFilter::ExceedsMinValue(v) => Ok(Some(*v)),
438            NoteFilter::ExceedsPriorSendPercentile(n) => {
439                let mut bucket_query = conn.prepare(
440                    "WITH bucketed AS (
441                        SELECT s.value, NTILE(10) OVER (ORDER BY s.value) AS bucket_index
442                        FROM sent_notes s
443                        JOIN transactions t ON s.tx = t.id_tx
444                        JOIN accounts a on a.id = s.from_account_id
445                        WHERE a.uuid = :account_uuid
446                        -- only count mined transactions
447                        AND t.mined_height IS NOT NULL
448                        -- exclude change and account-internal sends
449                        AND (s.to_account_id IS NULL OR s.from_account_id != s.to_account_id)
450                    )
451                    SELECT MAX(value) as value
452                    FROM bucketed
453                    GROUP BY bucket_index
454                    ORDER BY bucket_index",
455                )?;
456
457                let bucket_maxima = bucket_query
458                    .query_and_then::<_, SqliteClientError, _, _>(
459                        named_params![":account_uuid": account.0],
460                        |row| {
461                            Zatoshis::from_nonnegative_i64(row.get::<_, i64>(0)?).map_err(|_| {
462                                SqliteClientError::CorruptedData(format!(
463                                    "Negative received note value: {}",
464                                    n.value()
465                                ))
466                            })
467                        },
468                    )?
469                    .collect::<Result<Vec<_>, _>>()?;
470
471                // Pick a bucket index by scaling the requested percentile to the number of buckets
472                let i = (bucket_maxima.len() * usize::from(*n) / 100).saturating_sub(1);
473                Ok(bucket_maxima.get(i).copied())
474            }
475            NoteFilter::ExceedsBalancePercentage(p) => {
476                let balance = conn.query_row_and_then::<_, SqliteClientError, _, _>(
477                    "SELECT SUM(rn.value)
478                     FROM v_received_outputs rn
479                     INNER JOIN accounts a ON a.id = rn.account_id
480                     INNER JOIN transactions ON transactions.id_tx = rn.transaction_id
481                     WHERE a.uuid = :account_uuid
482                     AND a.ufvk IS NOT NULL
483                     AND transactions.mined_height IS NOT NULL
484                     AND rn.pool != :transparent_pool
485                     AND (rn.pool, rn.id_within_pool_table) NOT IN (
486                       SELECT rns.pool, rns.received_output_id
487                       FROM v_received_output_spends rns
488                       JOIN transactions stx ON stx.id_tx = rns.transaction_id
489                       WHERE (
490                           stx.block IS NOT NULL -- the spending tx is mined
491                           OR stx.expiry_height IS NULL -- the spending tx will not expire
492                           OR stx.expiry_height > :chain_tip_height -- the spending tx is unexpired
493                       )
494                     )",
495                    named_params![
496                        ":account_uuid": account.0,
497                        ":chain_tip_height": u32::from(chain_tip_height),
498                        ":transparent_pool": pool_code(PoolType::Transparent)
499                    ],
500                    |row| row.get::<_, Option<i64>>(0)?.map(zatoshis).transpose(),
501                )?;
502
503                Ok(match balance {
504                    None => None,
505                    Some(b) => {
506                        let numerator = (b * u64::from(p.value())).ok_or(BalanceError::Overflow)?;
507                        Some(numerator / NonZeroU64::new(100).expect("Constant is nonzero."))
508                    }
509                })
510            }
511            NoteFilter::Combine(a, b) => {
512                // All the existing note selectors set lower bounds on note value, so the "and"
513                // operation is just taking the maximum of the two lower bounds.
514                let a_min_value = min_note_value(conn, account, a.as_ref(), chain_tip_height)?;
515                let b_min_value = min_note_value(conn, account, b.as_ref(), chain_tip_height)?;
516                Ok(a_min_value
517                    .zip(b_min_value)
518                    .map(|(av, bv)| std::cmp::max(av, bv))
519                    .or(a_min_value)
520                    .or(b_min_value))
521            }
522            NoteFilter::Attempt {
523                condition,
524                fallback,
525            } => {
526                let cond = min_note_value(conn, account, condition.as_ref(), chain_tip_height)?;
527                if cond.is_none() {
528                    min_note_value(conn, account, fallback, chain_tip_height)
529                } else {
530                    Ok(cond)
531                }
532            }
533        }
534    }
535
536    // TODO: Simplify the query before executing it. Not worrying about this now because queries
537    // will be developer-configured, not end-user defined.
538    if let Some(min_value) = min_note_value(conn, account, filter, chain_tip_height)? {
539        let (note_count, total_value) = run_selection(min_value)?;
540
541        Ok(Some(PoolMeta::new(
542            note_count,
543            total_value.unwrap_or(Zatoshis::ZERO),
544        )))
545    } else {
546        Ok(None)
547    }
