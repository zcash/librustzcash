29/// This trait provides a generalization over shielded output representations.
30pub(crate) trait ReceivedSaplingOutput {
31    type AccountId;
32
33    fn index(&self) -> usize;
34    fn account_id(&self) -> Self::AccountId;
35    fn note(&self) -> &sapling::Note;
36    fn memo(&self) -> Option<&MemoBytes>;
37    fn is_change(&self) -> bool;
38    fn nullifier(&self) -> Option<&sapling::Nullifier>;
39    fn note_commitment_tree_position(&self) -> Option<Position>;
40    fn recipient_key_scope(&self) -> Option<Scope>;
41}

43impl<AccountId: Copy> ReceivedSaplingOutput for WalletSaplingOutput<AccountId> {
44    type AccountId = AccountId;
45
46    fn index(&self) -> usize {
47        self.index()
48    }
49    fn account_id(&self) -> Self::AccountId {
50        *WalletSaplingOutput::account_id(self)
51    }
52    fn note(&self) -> &sapling::Note {
53        WalletSaplingOutput::note(self)
54    }
55    fn memo(&self) -> Option<&MemoBytes> {
56        None
57    }
58    fn is_change(&self) -> bool {
59        WalletSaplingOutput::is_change(self)
60    }
61    fn nullifier(&self) -> Option<&sapling::Nullifier> {
62        self.nf()
63    }
64    fn note_commitment_tree_position(&self) -> Option<Position> {
65        Some(WalletSaplingOutput::note_commitment_tree_position(self))
66    }
67    fn recipient_key_scope(&self) -> Option<Scope> {
68        self.recipient_key_scope()
69    }
70}

72impl<AccountId: Copy> ReceivedSaplingOutput for DecryptedOutput<sapling::Note, AccountId> {
73    type AccountId = AccountId;
74
75    fn index(&self) -> usize {
76        self.index()
77    }
78    fn account_id(&self) -> Self::AccountId {
79        *self.account()
80    }
81    fn note(&self) -> &sapling::Note {
82        self.note()
83    }
84    fn memo(&self) -> Option<&MemoBytes> {
85        Some(self.memo())
86    }
87    fn is_change(&self) -> bool {
88        self.transfer_type() == TransferType::WalletInternal
89    }
90    fn nullifier(&self) -> Option<&sapling::Nullifier> {
91        None
92    }
93    fn note_commitment_tree_position(&self) -> Option<Position> {
94        None
95    }
96    fn recipient_key_scope(&self) -> Option<Scope> {
97        if self.transfer_type() == TransferType::WalletInternal {
98            Some(Scope::Internal)
99        } else {
100            Some(Scope::External)
101        }
102    }
103}

105fn to_spendable_note<P: consensus::Parameters>(
106    params: &P,
107    row: &Row,
108) -> Result<Option<ReceivedNote<ReceivedNoteId, sapling::Note>>, SqliteClientError> {
109    let note_id = ReceivedNoteId(ShieldedProtocol::Sapling, row.get("id")?);
110    let txid = row.get::<_, [u8; 32]>("txid").map(TxId::from_bytes)?;
111    let output_index = row.get("output_index")?;
112    let diversifier = {
113        let d: Vec<_> = row.get("diversifier")?;
114        if d.len() != 11 {
115            return Err(SqliteClientError::CorruptedData(
116                "Invalid diversifier length".to_string(),
117            ));
118        }
119        let mut tmp = [0; 11];
120        tmp.copy_from_slice(&d);
121        Diversifier(tmp)
122    };
123
124    let note_value: u64 = row.get::<_, i64>("value")?.try_into().map_err(|_e| {
125        SqliteClientError::CorruptedData("Note values must be nonnegative".to_string())
126    })?;
127
128    let rseed = {
129        let rcm_bytes: Vec<_> = row.get("rcm")?;
130
131        // We store rcm directly in the data DB, regardless of whether the note
132        // used a v1 or v2 note plaintext, so for the purposes of spending let's
133        // pretend this is a pre-ZIP 212 note.
134        let rcm = Option::from(jubjub::Fr::from_repr(
135            rcm_bytes[..]
136                .try_into()
137                .map_err(|_| SqliteClientError::InvalidNote)?,
138        ))
139        .ok_or(SqliteClientError::InvalidNote)?;
140        Rseed::BeforeZip212(rcm)
141    };
142
143    let note_commitment_tree_position = Position::from(
144        u64::try_from(row.get::<_, i64>("commitment_tree_position")?).map_err(|_| {
145            SqliteClientError::CorruptedData("Note commitment tree position invalid.".to_string())
146        })?,
147    );
148
149    let ufvk_str: Option<String> = row.get("ufvk")?;
150    let scope_code: Option<i64> = row.get("recipient_key_scope")?;
151
152    // If we don't have information about the recipient key scope or the ufvk we can't determine
153    // which spending key to use. This may be because the received note was associated with an
154    // imported viewing key, so we treat such notes as not spendable. Although this method is
155    // presently only called using the results of queries where both the ufvk and
156    // recipient_key_scope columns are checked to be non-null, this is method is written
157    // defensively to account for the fact that both of these are nullable columns in case it
158    // is used elsewhere in the future.
159    ufvk_str
160        .zip(scope_code)
161        .map(|(ufvk_str, scope_code)| {
162            let ufvk = UnifiedFullViewingKey::decode(params, &ufvk_str)
163                .map_err(SqliteClientError::CorruptedData)?;
164
165            let spending_key_scope = parse_scope(scope_code).ok_or_else(|| {
166                SqliteClientError::CorruptedData(format!("Invalid key scope code {}", scope_code))
167            })?;
168
169            let recipient = match spending_key_scope {
170                Scope::Internal => ufvk
171                    .sapling()
172                    .and_then(|dfvk| dfvk.diversified_change_address(diversifier)),
173                Scope::External => ufvk
174                    .sapling()
175                    .and_then(|dfvk| dfvk.diversified_address(diversifier)),
176            }
177            .ok_or_else(|| SqliteClientError::CorruptedData("Diversifier invalid.".to_owned()))?;
178
179            Ok(ReceivedNote::from_parts(
180                note_id,
181                txid,
182                output_index,
183                sapling::Note::from_parts(
184                    recipient,
185                    sapling::value::NoteValue::from_raw(note_value),
186                    rseed,
187                ),
188                spending_key_scope,
189                note_commitment_tree_position,
190            ))
191        })
192        .transpose()
193}

215/// Utility method for determining whether we have any spendable notes

217/// If the tip shard has unscanned ranges below the anchor height and greater than or equal to
218/// the wallet birthday, none of our notes can be spent because we cannot construct witnesses at
219/// the provided anchor height.

221    conn: &Connection,
222    params: &P,

224    target_value: Zatoshis,

229        conn,
230        params,
231        account,
232        target_value,
233        anchor_height,
234        exclude,
235        ShieldedProtocol::Sapling,
236        to_spendable_note,
237    )
238}

240/// Retrieves the set of nullifiers for "potentially spendable" Sapling notes that the
241/// wallet is tracking.
242///
243/// "Potentially spendable" means:
244/// - The transaction in which the note was created has been observed as mined.
245/// - No transaction in which the note's nullifier appears has been observed as mined.
246pub(crate) fn get_sapling_nullifiers(
247    conn: &Connection,
248    query: NullifierQuery,
249) -> Result<Vec<(AccountUuid, Nullifier)>, SqliteClientError> {
250    // Get the nullifiers for the notes we are tracking
251    let mut stmt_fetch_nullifiers = match query {
252        NullifierQuery::Unspent => conn.prepare(
253            "SELECT a.uuid, rn.nf
254             FROM sapling_received_notes rn
255             JOIN accounts a ON a.id = rn.account_id
256             JOIN transactions tx ON tx.id_tx = rn.tx
257             WHERE rn.nf IS NOT NULL
258             AND tx.block IS NOT NULL
259             AND rn.id NOT IN (
260               SELECT spends.sapling_received_note_id
261               FROM sapling_received_note_spends spends
262               JOIN transactions stx ON stx.id_tx = spends.transaction_id
263               WHERE stx.block IS NOT NULL  -- the spending tx is mined
264               OR stx.expiry_height IS NULL -- the spending tx will not expire
265             )",
266        ),
267        NullifierQuery::All => conn.prepare(
268            "SELECT a.uuid, rn.nf
269             FROM sapling_received_notes rn
270             JOIN accounts a ON a.id = rn.account_id
271             WHERE nf IS NOT NULL",
272        ),
273    }?;
274
275    let nullifiers = stmt_fetch_nullifiers.query_and_then([], |row| {
276        let account = AccountUuid(row.get(0)?);
277        let nf_bytes: Vec<u8> = row.get(1)?;
278        Ok::<_, rusqlite::Error>((account, sapling::Nullifier::from_slice(&nf_bytes).unwrap()))
279    })?;
280
281    let res: Vec<_> = nullifiers.collect::<Result<_, _>>()?;
282    Ok(res)
283}

285pub(crate) fn detect_spending_accounts<'a>(
286    conn: &Connection,
287    nfs: impl Iterator<Item = &'a Nullifier>,
288) -> Result<HashSet<AccountUuid>, rusqlite::Error> {
289    let mut account_q = conn.prepare_cached(
290        "SELECT accounts.uuid
291        FROM sapling_received_notes rn
292        JOIN accounts ON accounts.id = rn.account_id
293        WHERE rn.nf IN rarray(:nf_ptr)",
294    )?;
295
296    let nf_values: Vec<Value> = nfs.map(|nf| Value::Blob(nf.to_vec())).collect();
297    let nf_ptr = Rc::new(nf_values);
298    let res = account_q
299        .query_and_then(named_params![":nf_ptr": &nf_ptr], |row| {
300            row.get(0).map(AccountUuid)
301        })?
302        .collect::<Result<HashSet<_>, _>>()?;
303
304    Ok(res)
305}

307/// Marks a given nullifier as having been revealed in the construction
308/// of the specified transaction.
309///
310/// Marking a note spent in this fashion does NOT imply that the
311/// spending transaction has been mined.
312pub(crate) fn mark_sapling_note_spent(
313    conn: &Connection,
314    tx_ref: TxRef,
315    nf: &sapling::Nullifier,
316) -> Result<bool, SqliteClientError> {
317    let mut stmt_mark_sapling_note_spent = conn.prepare_cached(
318        "INSERT INTO sapling_received_note_spends (sapling_received_note_id, transaction_id)
319         SELECT id, :transaction_id FROM sapling_received_notes WHERE nf = :nf
320         ON CONFLICT (sapling_received_note_id, transaction_id) DO NOTHING",
321    )?;
322
323    match stmt_mark_sapling_note_spent.execute(named_params![
324       ":nf": &nf.0[..],
325       ":transaction_id": tx_ref.0
326    ])? {
327        0 => Ok(false),
328        1 => Ok(true),
329        _ => unreachable!("nf column is marked as UNIQUE"),
330    }
331}

333/// Records the specified shielded output as having been received.
334///
335/// This implementation relies on the facts that:
336/// - A transaction will not contain more than 2^63 shielded outputs.
337/// - A note value will never exceed 2^63 zatoshis.
338pub(crate) fn put_received_note<T: ReceivedSaplingOutput<AccountId = AccountUuid>>(
339    conn: &Transaction,
340    output: &T,
341    tx_ref: TxRef,
342    spent_in: Option<TxRef>,
343) -> Result<(), SqliteClientError> {
344    let account_id = get_account_ref(conn, output.account_id())?;
345    let mut stmt_upsert_received_note = conn.prepare_cached(
346        "INSERT INTO sapling_received_notes
347        (tx, output_index, account_id, diversifier, value, rcm, memo, nf,
348         is_change, commitment_tree_position,
349         recipient_key_scope)
350        VALUES (
351            :tx,
352            :output_index,
353            :account_id,
354            :diversifier,
355            :value,
356            :rcm,
357            :memo,
358            :nf,
359            :is_change,
360            :commitment_tree_position,
361            :recipient_key_scope
362        )
363        ON CONFLICT (tx, output_index) DO UPDATE
364        SET account_id = :account_id,
365            diversifier = :diversifier,
366            value = :value,
367            rcm = :rcm,
368            nf = IFNULL(:nf, nf),
369            memo = IFNULL(:memo, memo),
370            is_change = MAX(:is_change, is_change),
371            commitment_tree_position = IFNULL(:commitment_tree_position, commitment_tree_position),
372            recipient_key_scope = :recipient_key_scope
373        RETURNING sapling_received_notes.id",
374    )?;
375
376    let rcm = output.note().rcm().to_repr();
377    let to = output.note().recipient();
378    let diversifier = to.diversifier();
379
380    let sql_args = named_params![
381        ":tx": tx_ref.0,
382        ":output_index": i64::try_from(output.index()).expect("output indices are representable as i64"),
383        ":account_id": account_id.0,
384        ":diversifier": &diversifier.0,
385        ":value": output.note().value().inner(),
386        ":rcm": &rcm,
387        ":nf": output.nullifier().map(|nf| nf.0),
388        ":memo": memo_repr(output.memo()),
389        ":is_change": output.is_change(),
390        ":commitment_tree_position": output.note_commitment_tree_position().map(u64::from),
391        ":recipient_key_scope": output.recipient_key_scope().map(scope_code)
392    ];
393
394    let received_note_id = stmt_upsert_received_note
395        .query_row(sql_args, |row| row.get::<_, i64>(0))
396        .map_err(SqliteClientError::from)?;
397
398    if let Some(spent_in) = spent_in {
399        conn.execute(
400            "INSERT INTO sapling_received_note_spends (sapling_received_note_id, transaction_id)
401             VALUES (:sapling_received_note_id, :transaction_id)
402             ON CONFLICT (sapling_received_note_id, transaction_id) DO NOTHING",
403            named_params![
404                ":sapling_received_note_id": received_note_id,
405                ":transaction_id": spent_in.0
406            ],
407        )?;
408    }
409
410    Ok(())
411}
