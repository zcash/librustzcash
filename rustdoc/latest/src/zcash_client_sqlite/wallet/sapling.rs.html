30/// This trait provides a generalization over shielded output representations.
31pub(crate) trait ReceivedSaplingOutput {
32    type AccountId;
33
34    fn index(&self) -> usize;
35    fn account_id(&self) -> Self::AccountId;
36    fn note(&self) -> &sapling::Note;
37    fn memo(&self) -> Option<&MemoBytes>;
38    fn is_change(&self) -> bool;
39    fn nullifier(&self) -> Option<&sapling::Nullifier>;
40    fn note_commitment_tree_position(&self) -> Option<Position>;
41    fn recipient_key_scope(&self) -> Option<Scope>;
42}

44impl<AccountId: Copy> ReceivedSaplingOutput for WalletSaplingOutput<AccountId> {
45    type AccountId = AccountId;
46
47    fn index(&self) -> usize {
48        self.index()
49    }
50    fn account_id(&self) -> Self::AccountId {
51        *WalletSaplingOutput::account_id(self)
52    }
53    fn note(&self) -> &sapling::Note {
54        WalletSaplingOutput::note(self)
55    }
56    fn memo(&self) -> Option<&MemoBytes> {
57        None
58    }
59    fn is_change(&self) -> bool {
60        WalletSaplingOutput::is_change(self)
61    }
62    fn nullifier(&self) -> Option<&sapling::Nullifier> {
63        self.nf()
64    }
65    fn note_commitment_tree_position(&self) -> Option<Position> {
66        Some(WalletSaplingOutput::note_commitment_tree_position(self))
67    }
68    fn recipient_key_scope(&self) -> Option<Scope> {
69        self.recipient_key_scope()
70    }
71}

73impl<AccountId: Copy> ReceivedSaplingOutput for DecryptedOutput<sapling::Note, AccountId> {
74    type AccountId = AccountId;
75
76    fn index(&self) -> usize {
77        self.index()
78    }
79    fn account_id(&self) -> Self::AccountId {
80        *self.account()
81    }
82    fn note(&self) -> &sapling::Note {
83        self.note()
84    }
85    fn memo(&self) -> Option<&MemoBytes> {
86        Some(self.memo())
87    }
88    fn is_change(&self) -> bool {
89        self.transfer_type() == TransferType::WalletInternal
90    }
91    fn nullifier(&self) -> Option<&sapling::Nullifier> {
92        None
93    }
94    fn note_commitment_tree_position(&self) -> Option<Position> {
95        None
96    }
97    fn recipient_key_scope(&self) -> Option<Scope> {
98        if self.transfer_type() == TransferType::WalletInternal {
99            Some(Scope::Internal)
100        } else {
101            Some(Scope::External)
102        }
103    }
104}

106fn to_spendable_note<P: consensus::Parameters>(
107    params: &P,
108    row: &Row,
109) -> Result<Option<ReceivedNote<ReceivedNoteId, sapling::Note>>, SqliteClientError> {
110    let note_id = ReceivedNoteId(ShieldedProtocol::Sapling, row.get("id")?);
111    let txid = row.get::<_, [u8; 32]>("txid").map(TxId::from_bytes)?;
112    let output_index = row.get("output_index")?;
113    let diversifier = {
114        let d: Vec<_> = row.get("diversifier")?;
115        if d.len() != 11 {
116            return Err(SqliteClientError::CorruptedData(
117                "Invalid diversifier length".to_string(),
118            ));
119        }
120        let mut tmp = [0; 11];
121        tmp.copy_from_slice(&d);
122        Diversifier(tmp)
123    };
124
125    let note_value: u64 = row.get::<_, i64>("value")?.try_into().map_err(|_e| {
126        SqliteClientError::CorruptedData("Note values must be nonnegative".to_string())
127    })?;
128
129    let rseed = {
130        let rcm_bytes: Vec<_> = row.get("rcm")?;
131
132        // We store rcm directly in the data DB, regardless of whether the note
133        // used a v1 or v2 note plaintext, so for the purposes of spending let's
134        // pretend this is a pre-ZIP 212 note.
135        let rcm = Option::from(jubjub::Fr::from_repr(
136            rcm_bytes[..]
137                .try_into()
138                .map_err(|_| SqliteClientError::InvalidNote)?,
139        ))
140        .ok_or(SqliteClientError::InvalidNote)?;
141        Rseed::BeforeZip212(rcm)
142    };
143
144    let note_commitment_tree_position = Position::from(
145        u64::try_from(row.get::<_, i64>("commitment_tree_position")?).map_err(|_| {
146            SqliteClientError::CorruptedData("Note commitment tree position invalid.".to_string())
147        })?,
148    );
149
150    let ufvk_str: Option<String> = row.get("ufvk")?;
151    let scope_code: Option<i64> = row.get("recipient_key_scope")?;
152
153    // If we don't have information about the recipient key scope or the ufvk we can't determine
154    // which spending key to use. This may be because the received note was associated with an
155    // imported viewing key, so we treat such notes as not spendable. Although this method is
156    // presently only called using the results of queries where both the ufvk and
157    // recipient_key_scope columns are checked to be non-null, this is method is written
158    // defensively to account for the fact that both of these are nullable columns in case it
159    // is used elsewhere in the future.
160    ufvk_str
161        .zip(scope_code)
162        .map(|(ufvk_str, scope_code)| {
163            let ufvk = UnifiedFullViewingKey::decode(params, &ufvk_str)
164                .map_err(SqliteClientError::CorruptedData)?;
165
166            let spending_key_scope = zip32::Scope::try_from(KeyScope::decode(scope_code)?)
167                .map_err(|_| {
168                    SqliteClientError::CorruptedData(format!(
169                        "Invalid key scope code {}",
170                        scope_code
171                    ))
172                })?;
173
174            let recipient = match spending_key_scope {
175                Scope::Internal => ufvk
176                    .sapling()
177                    .and_then(|dfvk| dfvk.diversified_change_address(diversifier)),
178                Scope::External => ufvk
179                    .sapling()
180                    .and_then(|dfvk| dfvk.diversified_address(diversifier)),
181            }
182            .ok_or_else(|| SqliteClientError::CorruptedData("Diversifier invalid.".to_owned()))?;
183
184            Ok(ReceivedNote::from_parts(
185                note_id,
186                txid,
187                output_index,
188                sapling::Note::from_parts(
189                    recipient,
190                    sapling::value::NoteValue::from_raw(note_value),
191                    rseed,
192                ),
193                spending_key_scope,
194                note_commitment_tree_position,
195            ))
196        })
197        .transpose()
198}

258/// Retrieves the set of nullifiers for "potentially spendable" Sapling notes that the
259/// wallet is tracking.
260///
261/// "Potentially spendable" means:
262/// - The transaction in which the note was created has been observed as mined.
263/// - No transaction in which the note's nullifier appears has been observed as mined.
264pub(crate) fn get_sapling_nullifiers(
265    conn: &Connection,
266    query: NullifierQuery,
267) -> Result<Vec<(AccountUuid, Nullifier)>, SqliteClientError> {
268    // Get the nullifiers for the notes we are tracking
269    let mut stmt_fetch_nullifiers = match query {
270        NullifierQuery::Unspent => conn.prepare(
271            "SELECT a.uuid, rn.nf
272             FROM sapling_received_notes rn
273             JOIN accounts a ON a.id = rn.account_id
274             JOIN transactions tx ON tx.id_tx = rn.tx
275             WHERE rn.nf IS NOT NULL
276             AND tx.block IS NOT NULL
277             AND rn.id NOT IN (
278               SELECT spends.sapling_received_note_id
279               FROM sapling_received_note_spends spends
280               JOIN transactions stx ON stx.id_tx = spends.transaction_id
281               WHERE stx.block IS NOT NULL  -- the spending tx is mined
282               OR stx.expiry_height IS NULL -- the spending tx will not expire
283             )",
284        ),
285        NullifierQuery::All => conn.prepare(
286            "SELECT a.uuid, rn.nf
287             FROM sapling_received_notes rn
288             JOIN accounts a ON a.id = rn.account_id
289             WHERE nf IS NOT NULL",
290        ),
291    }?;
292
293    let nullifiers = stmt_fetch_nullifiers.query_and_then([], |row| {
294        let account = AccountUuid(row.get(0)?);
295        let nf_bytes: Vec<u8> = row.get(1)?;
296        Ok::<_, rusqlite::Error>((account, sapling::Nullifier::from_slice(&nf_bytes).unwrap()))
297    })?;
298
299    let res: Vec<_> = nullifiers.collect::<Result<_, _>>()?;
300    Ok(res)
301}

303pub(crate) fn detect_spending_accounts<'a>(
304    conn: &Connection,
305    nfs: impl Iterator<Item = &'a Nullifier>,
306) -> Result<HashSet<AccountUuid>, rusqlite::Error> {
307    let mut account_q = conn.prepare_cached(
308        "SELECT accounts.uuid
309        FROM sapling_received_notes rn
310        JOIN accounts ON accounts.id = rn.account_id
311        WHERE rn.nf IN rarray(:nf_ptr)",
312    )?;
313
314    let nf_values: Vec<Value> = nfs.map(|nf| Value::Blob(nf.to_vec())).collect();
315    let nf_ptr = Rc::new(nf_values);
316    let res = account_q
317        .query_and_then(named_params![":nf_ptr": &nf_ptr], |row| {
318            row.get(0).map(AccountUuid)
319        })?
320        .collect::<Result<HashSet<_>, _>>()?;
321
322    Ok(res)
323}

325/// Marks a given nullifier as having been revealed in the construction
326/// of the specified transaction.
327///
328/// Marking a note spent in this fashion does NOT imply that the
329/// spending transaction has been mined.
330pub(crate) fn mark_sapling_note_spent(
331    conn: &Connection,
332    tx_ref: TxRef,
333    nf: &sapling::Nullifier,
334) -> Result<bool, SqliteClientError> {
335    let mut stmt_mark_sapling_note_spent = conn.prepare_cached(
336        "INSERT INTO sapling_received_note_spends (sapling_received_note_id, transaction_id)
337         SELECT id, :transaction_id FROM sapling_received_notes WHERE nf = :nf
338         ON CONFLICT (sapling_received_note_id, transaction_id) DO NOTHING",
339    )?;
340
341    match stmt_mark_sapling_note_spent.execute(named_params![
342       ":nf": &nf.0[..],
343       ":transaction_id": tx_ref.0
344    ])? {
345        0 => Ok(false),
346        1 => Ok(true),
347        _ => unreachable!("nf column is marked as UNIQUE"),
348    }
349}

351pub(crate) fn ensure_address<
352    T: ReceivedSaplingOutput<AccountId = AccountUuid>,
353    P: consensus::Parameters,
354>(
355    conn: &rusqlite::Transaction,
356    params: &P,
357    output: &T,
358    exposure_height: Option<BlockHeight>,
359) -> Result<Option<AddressRef>, SqliteClientError> {
360    if output.recipient_key_scope() != Some(Scope::Internal) {
361        let account = get_account(conn, params, output.account_id())?
362            .ok_or(SqliteClientError::AccountUnknown)?;
363
364        let uivk = account.uivk();
365        let ivk = uivk
366            .sapling()
367            .as_ref()
368            .expect("uivk decrypted this output.");
369        let to = output.note().recipient();
370        let diversifier_index = ivk
371            .decrypt_diversifier(&to)
372            .expect("address corresponds to account");
373
374        let ua = account
375            .uivk()
376            .address(diversifier_index, UnifiedAddressRequest::ALLOW_ALL)?;
377
378        upsert_address(
379            conn,
380            params,
381            account.internal_id(),
382            diversifier_index,
383            &ua,
384            exposure_height,
385            false,
386        )
387        .map(Some)
388    } else {
389        Ok(None)
390    }
391}

393/// Records the specified shielded output as having been received.
394///
395/// This implementation relies on the facts that:
396/// - A transaction will not contain more than 2^63 shielded outputs.
397/// - A note value will never exceed 2^63 zatoshis.
398///
399/// Returns the internal account identifier of the account that received the output.
400pub(crate) fn put_received_note<
401    T: ReceivedSaplingOutput<AccountId = AccountUuid>,
402    P: consensus::Parameters,
403>(
404    conn: &rusqlite::Transaction,
405    params: &P,
406    output: &T,
407    tx_ref: TxRef,
408    target_or_mined_height: Option<BlockHeight>,
409    spent_in: Option<TxRef>,
410) -> Result<AccountRef, SqliteClientError> {
411    let account_id = get_account_ref(conn, output.account_id())?;
412    let address_id = ensure_address(conn, params, output, target_or_mined_height)?;
413    let mut stmt_upsert_received_note = conn.prepare_cached(
414        "INSERT INTO sapling_received_notes (
415            tx, output_index, account_id, address_id,
416            diversifier, value, rcm, memo, nf,
417            is_change, commitment_tree_position,
418            recipient_key_scope
419        )
420        VALUES (
421            :tx,
422            :output_index,
423            :account_id,
424            :address_id,
425            :diversifier,
426            :value,
427            :rcm,
428            :memo,
429            :nf,
430            :is_change,
431            :commitment_tree_position,
432            :recipient_key_scope
433        )
434        ON CONFLICT (tx, output_index) DO UPDATE
435        SET account_id = :account_id,
436            address_id = :address_id,
437            diversifier = :diversifier,
438            value = :value,
439            rcm = :rcm,
440            nf = IFNULL(:nf, nf),
441            memo = IFNULL(:memo, memo),
442            is_change = MAX(:is_change, is_change),
443            commitment_tree_position = IFNULL(:commitment_tree_position, commitment_tree_position),
444            recipient_key_scope = :recipient_key_scope
445        RETURNING sapling_received_notes.id",
446    )?;
447
448    let rcm = output.note().rcm().to_repr();
449    let to = output.note().recipient();
450    let diversifier = to.diversifier();
451
452    let sql_args = named_params![
453        ":tx": tx_ref.0,
454        ":output_index": i64::try_from(output.index()).expect("output indices are representable as i64"),
455        ":account_id": account_id.0,
456        ":address_id": address_id.map(|a| a.0),
457        ":diversifier": &diversifier.0,
458        ":value": output.note().value().inner(),
459        ":rcm": &rcm,
460        ":nf": output.nullifier().map(|nf| nf.0),
461        ":memo": memo_repr(output.memo()),
462        ":is_change": output.is_change(),
463        ":commitment_tree_position": output.note_commitment_tree_position().map(u64::from),
464        ":recipient_key_scope": output.recipient_key_scope().map(|s| KeyScope::from(s).encode()),
465    ];
466
467    let received_note_id = stmt_upsert_received_note
468        .query_row(sql_args, |row| row.get::<_, i64>(0))
469        .map_err(SqliteClientError::from)?;
470
471    if let Some(spent_in) = spent_in {
472        conn.execute(
473            "INSERT INTO sapling_received_note_spends (sapling_received_note_id, transaction_id)
474             VALUES (:sapling_received_note_id, :transaction_id)
475             ON CONFLICT (sapling_received_note_id, transaction_id) DO NOTHING",
476            named_params![
477                ":sapling_received_note_id": received_note_id,
478                ":transaction_id": spent_in.0
479            ],
480        )?;
481    }
482
483    Ok(account_id)
484}
