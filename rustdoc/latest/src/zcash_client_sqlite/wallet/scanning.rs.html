34pub(crate) fn priority_code(priority: &ScanPriority) -> i64 {
35    use ScanPriority::*;
36    match priority {
37        Ignored => 0,
38        Scanned => 10,
39        Historic => 20,
40        OpenAdjacent => 30,
41        FoundNote => 40,
42        ChainTip => 50,
43        Verify => 60,
44    }
45}

61pub(crate) fn suggest_scan_ranges(
62    conn: &rusqlite::Connection,
63    min_priority: ScanPriority,
64) -> Result<Vec<ScanRange>, SqliteClientError> {
65    let mut stmt_scan_ranges = conn.prepare_cached(
66        "SELECT block_range_start, block_range_end, priority
67         FROM scan_queue
68         WHERE priority >= :min_priority
69         ORDER BY priority DESC, block_range_end DESC",
70    )?;
71
72    let mut rows =
73        stmt_scan_ranges.query(named_params![":min_priority": priority_code(&min_priority)])?;
74
75    let mut result = vec![];
76    while let Some(row) = rows.next()? {
77        let range = Range {
78            start: row.get::<_, u32>(0).map(BlockHeight::from)?,
79            end: row.get::<_, u32>(1).map(BlockHeight::from)?,
80        };
81        let code = row.get::<_, i64>(2)?;
82        let priority = parse_priority_code(code).ok_or_else(|| {
83            SqliteClientError::CorruptedData(format!("scan priority not recognized: {code}"))
84        })?;
85
86        result.push(ScanRange::from_parts(range, priority));
87    }
88
89    Ok(result)
90}

92pub(crate) fn insert_queue_entries<'a>(
93    conn: &rusqlite::Connection,
94    entries: impl Iterator<Item = &'a ScanRange>,
95) -> Result<(), rusqlite::Error> {
96    let mut stmt = conn.prepare_cached(
97        "INSERT INTO scan_queue (block_range_start, block_range_end, priority)
98        VALUES (:block_range_start, :block_range_end, :priority)",
99    )?;
100
101    for entry in entries {
102        trace!("Inserting queue entry {}", entry);
103        if !entry.is_empty() {
104            stmt.execute(named_params![
105                ":block_range_start": u32::from(entry.block_range().start),
106                ":block_range_end": u32::from(entry.block_range().end),
107                ":priority": priority_code(&entry.priority())
108            ])?;
109        }
110    }
111
112    Ok(())
113}

145pub(crate) fn replace_queue_entries<E: WalletError>(
146    conn: &rusqlite::Transaction<'_>,
147    query_range: &Range<BlockHeight>,
148    entries: impl Iterator<Item = ScanRange>,
149    force_rescans: bool,
150) -> Result<(), E> {
151    let (to_create, to_delete_ends) = {
152        let mut suggested_stmt = conn
153            .prepare_cached(
154                "SELECT block_range_start, block_range_end, priority
155                 FROM scan_queue
156                 -- Ignore ranges that do not overlap and are not adjacent to the query range.
157                 WHERE NOT (block_range_start > :end OR :start > block_range_end)
158                 ORDER BY block_range_end",
159            )
160            .map_err(E::db_error)?;
161
162        let mut rows = suggested_stmt
163            .query(named_params![
164                ":start": u32::from(query_range.start),
165                ":end": u32::from(query_range.end),
166            ])
167            .map_err(E::db_error)?;
168
169        // Iterate over the ranges in the scan queue that overlap the range that we have
170        // identified as needing to be fully scanned. For each such range add it to the
171        // spanning tree (these should all be nonoverlapping ranges, but we might coalesce
172        // some in the process).
173        let mut to_create: Option<SpanningTree> = None;
174        let mut to_delete_ends: Vec<Value> = vec![];
175        while let Some(row) = rows.next().map_err(E::db_error)? {
176            let entry = ScanRange::from_parts(
177                Range {
178                    start: BlockHeight::from(row.get::<_, u32>(0).map_err(E::db_error)?),
179                    end: BlockHeight::from(row.get::<_, u32>(1).map_err(E::db_error)?),
180                },
181                {
182                    let code = row.get::<_, i64>(2).map_err(E::db_error)?;
183                    parse_priority_code(code).ok_or_else(|| {
184                        E::corrupt(format!("scan priority not recognized: {code}"))
185                    })?
186                },
187            );
188            to_delete_ends.push(Value::from(u32::from(entry.block_range().end)));
189            to_create = if let Some(cur) = to_create {
190                Some(cur.insert(entry, force_rescans))
191            } else {
192                Some(SpanningTree::Leaf(entry))
193            };
194        }
195
196        // Update the tree that we read from the database, or if we didn't find any ranges
197        // start with the scanned range.
198        for entry in entries {
199            to_create = if let Some(cur) = to_create {
200                Some(cur.insert(entry, force_rescans))
201            } else {
202                Some(SpanningTree::Leaf(entry))
203            };
204        }
205
206        (to_create, to_delete_ends)
207    };
208
209    if let Some(tree) = to_create {
210        let ends_ptr = Rc::new(to_delete_ends);
211        conn.execute(
212            "DELETE FROM scan_queue WHERE block_range_end IN rarray(:ends)",
213            named_params![":ends": ends_ptr],
214        )
215        .map_err(E::db_error)?;
216
217        let scan_ranges = tree.into_vec();
218        insert_queue_entries(conn, scan_ranges.iter()).map_err(E::db_error)?;
219    }
220
221    Ok(())
222}

224fn extend_range(
225    conn: &rusqlite::Transaction<'_>,
226    range: &Range<BlockHeight>,
227    required_subtree_indices: BTreeSet<u64>,
228    protocol: ShieldedProtocol,
229    fallback_start_height: Option<BlockHeight>,
230    birthday_height: Option<BlockHeight>,
231) -> Result<Option<Range<BlockHeight>>, SqliteClientError> {
232    let TableConstants { table_prefix, .. } = table_constants::<SqliteClientError>(protocol)?;
233
234    // we'll either have both min and max bounds, or we'll have neither
235    let subtree_index_bounds = required_subtree_indices
236        .iter()
237        .min()
238        .zip(required_subtree_indices.iter().max());
239
240    let mut shard_end_stmt = conn.prepare_cached(&format!(
241        "SELECT subtree_end_height
242                FROM {table_prefix}_tree_shards
243                WHERE shard_index = :shard_index"
244    ))?;
245
246    let mut shard_end = |index: u64| -> Result<Option<BlockHeight>, rusqlite::Error> {
247        Ok(shard_end_stmt
248            .query_row(named_params![":shard_index": index], |row| {
249                row.get::<_, Option<u32>>(0)
250                    .map(|opt| opt.map(BlockHeight::from))
251            })
252            .optional()?
253            .flatten())
254    };
255
256    // If no notes belonging to the wallet were found, we don't need to extend the scanning
257    // range suggestions to include the associated subtrees, and our bounds are just the
258    // scanned range. Otherwise, ensure that all shard ranges starting from the wallet
259    // birthday are included.
260    subtree_index_bounds
261        .map(|(min_idx, max_idx)| {
262            let range_min = if *min_idx > 0 {
263                // get the block height of the end of the previous shard
264                shard_end(*min_idx - 1)?
265            } else {
266                // our lower bound is going to be the fallback height
267                fallback_start_height
268            };
269
270            // bound the minimum to the wallet birthday
271            let range_min = range_min.map(|h| birthday_height.map_or(h, |b| std::cmp::max(b, h)));
272
273            // Get the block height for the end of the current shard, and make it an
274            // exclusive end bound.
275            let range_max = shard_end(*max_idx)?.map(|end| end + 1);
276
277            Ok::<Range<BlockHeight>, rusqlite::Error>(Range {
278                start: range.start.min(range_min.unwrap_or(range.start)),
279                end: range.end.max(range_max.unwrap_or(range.end)),
280            })
281        })
282        .transpose()
283        .map_err(SqliteClientError::from)
284}

286pub(crate) fn scan_complete<P: consensus::Parameters>(
287    conn: &rusqlite::Transaction<'_>,
288    params: &P,
289    range: Range<BlockHeight>,
290    wallet_note_positions: &[(ShieldedProtocol, Position)],
291) -> Result<(), SqliteClientError> {
292    // Read the wallet birthday (if known).
293    // TODO: use per-pool birthdays?
294    let wallet_birthday = wallet_birthday(conn)?;
295
296    // Determine the range of block heights for which we will be updating the scan queue.
297    let extended_range = {
298        // If notes have been detected in the scan, we need to extend any adjacent un-scanned
299        // ranges starting from the wallet birthday to include the blocks needed to complete
300        // the note commitment tree subtrees containing the positions of the discovered notes.
301        // We will query by subtree index to find these bounds.
302        let mut required_sapling_subtrees = BTreeSet::new();
303        #[cfg(feature = "orchard")]
304        let mut required_orchard_subtrees = BTreeSet::new();
305        for (protocol, position) in wallet_note_positions {
306            match protocol {
307                ShieldedProtocol::Sapling => {
308                    required_sapling_subtrees.insert(
309                        Address::above_position(SAPLING_SHARD_HEIGHT.into(), *position).index(),
310                    );
311                }
312                ShieldedProtocol::Orchard => {
313                    #[cfg(feature = "orchard")]
314                    required_orchard_subtrees.insert(
315                        Address::above_position(ORCHARD_SHARD_HEIGHT.into(), *position).index(),
316                    );
317
318                    #[cfg(not(feature = "orchard"))]
319                    return Err(SqliteClientError::UnsupportedPoolType(PoolType::Shielded(
320                        *protocol,
321                    )));
322                }
323            }
324        }
325
326        let extended_range = extend_range(
327            conn,
328            &range,
329            required_sapling_subtrees,
330            ShieldedProtocol::Sapling,
331            params.activation_height(NetworkUpgrade::Sapling),
332            wallet_birthday,
333        )?;
334
335        #[cfg(feature = "orchard")]
336        let extended_range = extend_range(
337            conn,
338            extended_range.as_ref().unwrap_or(&range),
339            required_orchard_subtrees,
340            ShieldedProtocol::Orchard,
341            params.activation_height(NetworkUpgrade::Nu5),
342            wallet_birthday,
343        )?
344        .or(extended_range);
345
346        #[allow(clippy::let_and_return)]
347        extended_range
348    };
349
350    let query_range = extended_range.clone().unwrap_or_else(|| range.clone());
351
352    let scanned = ScanRange::from_parts(range.clone(), ScanPriority::Scanned);
353
354    // If any of the extended range actually extends beyond the scanned range, we need to
355    // scan that extension in order to make the found note(s) spendable. We need to avoid
356    // creating empty ranges here, as that acts as an optimization barrier preventing
357    // `SpanningTree` from merging non-empty scanned ranges on either side.
358    let extended_before = extended_range
359        .as_ref()
360        .map(|extended| ScanRange::from_parts(extended.start..range.start, ScanPriority::FoundNote))
361        .filter(|range| !range.is_empty());
362    let extended_after = extended_range
363        .map(|extended| ScanRange::from_parts(range.end..extended.end, ScanPriority::FoundNote))
364        .filter(|range| !range.is_empty());
365
366    let replacement = Some(scanned)
367        .into_iter()
368        .chain(extended_before)
369        .chain(extended_after);
370
371    replace_queue_entries::<SqliteClientError>(conn, &query_range, replacement, false)?;
372
373    Ok(())
374}

389pub(crate) fn update_chain_tip<P: consensus::Parameters>(
390    conn: &rusqlite::Transaction<'_>,
391    params: &P,
392    new_tip: BlockHeight,
393) -> Result<(), SqliteClientError> {
394    // If the caller provided a chain tip that is before Sapling activation, do nothing.
395    let sapling_activation = match params.activation_height(NetworkUpgrade::Sapling) {
396        Some(h) if h <= new_tip => h,
397        _ => return Ok(()),
398    };
399
400    // Read the previous max scanned height from the blocks table
401    let max_scanned = block_height_extrema(conn)?.map(|range| *range.end());
402
403    // Read the wallet birthday (if known).
404    let wallet_birthday = wallet_birthday(conn)?;
405
406    // If the chain tip is below the prior max scanned height, then the caller has caught
407    // the chain in the middle of a reorg. Do nothing; the caller will continue using the
408    // old scan ranges and either:
409    // - encounter an error trying to fetch the blocks (and thus trigger the same handling
410    //   logic as if this happened with the old linear scanning code); or
411    // - encounter a discontinuity error in `scan_cached_blocks`, at which point they will
412    //   call `WalletDb::truncate_to_height` as part of their reorg handling which will
413    //   resolve the problem.
414    //
415    // We don't check the shard height, as normal usage would have the caller update the
416    // shard state prior to this call, so it is possible and expected to be in a situation
417    // where we should update the tip-related scan ranges but not the shard-related ones.
418    match max_scanned {
419        Some(h) if new_tip < h => return Ok(()),
420        _ => (),
421    };
422
423    // `ScanRange` uses an exclusive upper bound.
424    let chain_end = new_tip + 1;
425
426    // Read the maximum height from each of the shards tables. The minimum of the two
427    // gives the start of a height range that covers the last incomplete shard of both the
428    // Sapling and Orchard pools.
429    let sapling_shard_tip = tip_shard_end_height(conn, ShieldedProtocol::Sapling)?;
430    #[cfg(feature = "orchard")]
431    let orchard_shard_tip = tip_shard_end_height(conn, ShieldedProtocol::Orchard)?;
432
433    #[cfg(feature = "orchard")]
434    let min_shard_tip = match (sapling_shard_tip, orchard_shard_tip) {
435        (None, None) => None,
436        (None, Some(o)) => Some(o),
437        (Some(s), None) => Some(s),
438        (Some(s), Some(o)) => Some(std::cmp::min(s, o)),
439    };
440    #[cfg(not(feature = "orchard"))]
441    let min_shard_tip = sapling_shard_tip;
442
443    // Create a scanning range for the fragment of the last shard leading up to new tip.
444    // We set a lower bound at the wallet birthday (if known), because account creation
445    // requires specifying a tree frontier that ensures we don't need tree information
446    // prior to the birthday.
447    let tip_shard_entry = min_shard_tip.filter(|h| h < &chain_end).map(|h| {
448        let min_to_scan = wallet_birthday.filter(|b| b > &h).unwrap_or(h);
449        ScanRange::from_parts(min_to_scan..chain_end, ScanPriority::ChainTip)
450    });
451
452    // Create scan ranges to either validate potentially invalid blocks at the wallet's
453    // view of the chain tip, or connect the prior tip to the new tip.
454    let tip_entry = max_scanned.map_or_else(
455        || {
456            // No blocks have been scanned, so we need to anchor the start of the new scan
457            // range to something else.
458            wallet_birthday.map_or_else(
459                // We don't have a wallet birthday, which means we have no accounts yet.
460                // We can therefore ignore all blocks up to the chain tip.
461                || ScanRange::from_parts(sapling_activation..chain_end, ScanPriority::Ignored),
462                // We have a wallet birthday, so mark all blocks between that and the
463                // chain tip as `Historic` (performing wallet recovery).
464                |wallet_birthday| {
465                    ScanRange::from_parts(wallet_birthday..chain_end, ScanPriority::Historic)
466                },
467            )
468        },
469        |max_scanned| {
470            // The scan range starts at the block after the max scanned height. Since
471            // `scan_cached_blocks` retrieves the metadata for the block being connected to
472            // (if it exists), the connectivity of the scan range to the max scanned block
473            // will always be checked if relevant.
474            let min_unscanned = max_scanned + 1;
475
476            // If we don't have shard metadata, this means we're doing linear scanning, so
477            // create a scan range from the prior tip to the current tip with `Historic`
478            // priority.
479            if tip_shard_entry.is_none() {
480                ScanRange::from_parts(min_unscanned..chain_end, ScanPriority::Historic)
481            } else {
482                // Determine the height to which we expect new blocks retrieved from the
483                // block source to be stable and not subject to being reorg'ed.
484                let stable_height = new_tip.saturating_sub(PRUNING_DEPTH);
485
486                // If the wallet's max scanned height is above the stable height,
487                // prioritize the range between it and the new tip as `ChainTip`.
488                if max_scanned > stable_height {
489                    // We are in the steady-state case, where a wallet is close to the
490                    // chain tip and just needs to catch up.
491                    //
492                    // This overlaps the `tip_shard_entry` range and so will be coalesced
493                    // with it.
494                    ScanRange::from_parts(min_unscanned..chain_end, ScanPriority::ChainTip)
495                } else {
496                    // In this case, the max scanned height is considered stable relative
497                    // to the chain tip. However, it may be stable or unstable relative to
498                    // the prior chain tip, which we could determine by looking up the
499                    // prior chain tip height from the scan queue. For simplicity we merge
500                    // these two cases together, and proceed as though the max scanned
501                    // block is unstable relative to the prior chain tip.
502                    //
503                    // To confirm its stability, prioritize the `VERIFY_LOOKAHEAD` blocks
504                    // above the max scanned height as `Verify`:
505                    //
506                    // - We use `Verify` to ensure that a connectivity check is performed,
507                    //   along with any required rewinds, before any `ChainTip` ranges
508                    //   (from this or any prior `update_chain_tip` call) are scanned.
509                    //
510                    // - We prioritize `VERIFY_LOOKAHEAD` blocks because this is expected
511                    //   to be 12.5 minutes, within which it is reasonable for a user to
512                    //   have potentially received a transaction (if they opened their
513                    //   wallet to provide an address to someone else, or spent their own
514                    //   funds creating a change output), without necessarily having left
515                    //   their wallet open long enough for the transaction to be mined and
516                    //   the corresponding block to be scanned.
517                    //
518                    // - We limit the range to at most the stable region, to prevent any
519                    //   `Verify` ranges from being susceptible to reorgs, and potentially
520                    //   interfering with subsequent `Verify` ranges defined by future
521                    //   calls to `update_chain_tip`. Any gap between `stable_height` and
522                    //   `shard_start_height` will be filled by the scan range merging
523                    //   logic with a `Historic` range.
524                    //
525                    // If `max_scanned == stable_height` then this is a zero-length range.
526                    // In this case, any non-empty `(stable_height+1)..shard_start_height`
527                    // will be marked `Historic`, minimising the prioritised blocks at the
528                    // chain tip and allowing for other ranges (for example, `FoundNote`)
529                    // to take priority.
530                    ScanRange::from_parts(
531                        min_unscanned..min(stable_height + 1, min_unscanned + VERIFY_LOOKAHEAD),
532                        ScanPriority::Verify,
533                    )
534                }
535            }
536        },
537    );
538    if let Some(entry) = &tip_shard_entry {
539        debug!("{} will update latest shard", entry);
540    }
541    debug!("{} will connect prior scanned state to new tip", tip_entry);
542
543    let query_range = match tip_shard_entry.as_ref() {
544        Some(se) => Range {
545            start: min(se.block_range().start, tip_entry.block_range().start),
546            end: max(se.block_range().end, tip_entry.block_range().end),
547        },
548        None => tip_entry.block_range().clone(),
549    };
550
551    // persist the updated scan queue entries
552    replace_queue_entries::<SqliteClientError>(
553        conn,
554        &query_range,
555        tip_shard_entry.into_iter().chain(Some(tip_entry)),
556        false,
557    )?;
558
559    Ok(())
560}

782    /// * Shielded chain history begins at NU5 activation
783    /// * Both the Sapling and the Orchard note commitment trees have the following structure:
784    /// * The initial 2^16 shard of the note commitment tree covers `initial_shard_blocks` blocks.
785    ///   If `insert_prior_roots` is set, the root of the initial shard is inserted into each note

787    ///   roots have been inserted prior to scanning.

789    /// * The note commitment tree contains 2^16+1235 notes at the end of the block prior to the
790    ///   wallet birthday.

795        insert_prior_roots: bool,

804            .with_block_cache(BlockCache::new())
805            .with_initial_chain_state(|rng, network| {
806                // We set the Sapling and Orchard frontiers at the birthday height to be
807                // 1234 notes into the second shard.
808                let frontier_position = Position::from((0x1 << 16) + 1234);
809                let initial_shard_end =
810                    network.activation_height(NetworkUpgrade::Nu5).unwrap() + initial_shard_blocks;
811                let birthday_height = initial_shard_end + birthday_offset;
812
813                // Construct a fake chain state for the end of the block with the given
814                // birthday_offset from the end of the last shard.
815                let (prior_sapling_roots, sapling_initial_tree) =

817                        rng,

823                    Frontier::random_with_prior_subtree_roots(
824                        rng,
825                        (frontier_position + 1).into(),
826                        NonZeroU8::new(16).unwrap(),
827                    );
828
829                InitialChainState {
830                    chain_state: ChainState::new(
831                        birthday_height,
832                        prior_block_hash,
833                        sapling_initial_tree,
834                        #[cfg(feature = "orchard")]
835                        orchard_initial_tree,
836                    ),
837                    prior_sapling_roots: if insert_prior_roots {
838                        prior_sapling_roots
839                            .into_iter()
840                            .map(|root| CommitmentTreeRoot::from_parts(initial_shard_end, root))
841                            .collect()
842                    } else {
843                        vec![]
844                    },
845                    #[cfg(feature = "orchard")]
846                    prior_orchard_roots: if insert_prior_roots {
847                        prior_orchard_roots
848                            .into_iter()
849                            .map(|root| CommitmentTreeRoot::from_parts(initial_shard_end, root))
850                            .collect()
851                    } else {
852                        vec![]
853                    },
854                }
855            })
856            .with_account_having_current_birthday()
857            .build();
858
859        let birthday = st.test_account().unwrap().birthday().clone();
860        let dfvk = T::test_account_fvk(&st);
861        let sap_active = st.sapling_activation_height();
862
863        (st, dfvk, birthday, sap_active.into())
864    }
