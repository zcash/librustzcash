32pub(crate) fn priority_code(priority: &ScanPriority) -> i64 {
33    use ScanPriority::*;
34    match priority {
35        Ignored => 0,
36        Scanned => 10,
37        Historic => 20,
38        OpenAdjacent => 30,
39        FoundNote => 40,
40        ChainTip => 50,
41        Verify => 60,
42    }
43}

59pub(crate) fn suggest_scan_ranges(
60    conn: &rusqlite::Connection,
61    min_priority: ScanPriority,
62) -> Result<Vec<ScanRange>, SqliteClientError> {
63    let mut stmt_scan_ranges = conn.prepare_cached(
64        "SELECT block_range_start, block_range_end, priority
65         FROM scan_queue
66         WHERE priority >= :min_priority
67         ORDER BY priority DESC, block_range_end DESC",
68    )?;
69
70    let mut rows =
71        stmt_scan_ranges.query(named_params![":min_priority": priority_code(&min_priority)])?;
72
73    let mut result = vec![];
74    while let Some(row) = rows.next()? {
75        let range = Range {
76            start: row.get::<_, u32>(0).map(BlockHeight::from)?,
77            end: row.get::<_, u32>(1).map(BlockHeight::from)?,
78        };
79        let code = row.get::<_, i64>(2)?;
80        let priority = parse_priority_code(code).ok_or_else(|| {
81            SqliteClientError::CorruptedData(format!("scan priority not recognized: {}", code))
82        })?;
83
84        result.push(ScanRange::from_parts(range, priority));
85    }
86
87    Ok(result)
88}

90pub(crate) fn insert_queue_entries<'a>(
91    conn: &rusqlite::Connection,
92    entries: impl Iterator<Item = &'a ScanRange>,
93) -> Result<(), rusqlite::Error> {
94    let mut stmt = conn.prepare_cached(
95        "INSERT INTO scan_queue (block_range_start, block_range_end, priority)
96        VALUES (:block_range_start, :block_range_end, :priority)",
97    )?;
98
99    for entry in entries {
100        trace!("Inserting queue entry {}", entry);
101        if !entry.is_empty() {
102            stmt.execute(named_params![
103                ":block_range_start": u32::from(entry.block_range().start),
104                ":block_range_end": u32::from(entry.block_range().end),
105                ":priority": priority_code(&entry.priority())
106            ])?;
107        }
108    }
109
110    Ok(())
111}

143pub(crate) fn replace_queue_entries<E: WalletError>(
144    conn: &rusqlite::Transaction<'_>,
145    query_range: &Range<BlockHeight>,
146    entries: impl Iterator<Item = ScanRange>,
147    force_rescans: bool,
148) -> Result<(), E> {
149    let (to_create, to_delete_ends) = {
150        let mut suggested_stmt = conn
151            .prepare_cached(
152                "SELECT block_range_start, block_range_end, priority
153                 FROM scan_queue
154                 -- Ignore ranges that do not overlap and are not adjacent to the query range.
155                 WHERE NOT (block_range_start > :end OR :start > block_range_end)
156                 ORDER BY block_range_end",
157            )
158            .map_err(E::db_error)?;
159
160        let mut rows = suggested_stmt
161            .query(named_params![
162                ":start": u32::from(query_range.start),
163                ":end": u32::from(query_range.end),
164            ])
165            .map_err(E::db_error)?;
166
167        // Iterate over the ranges in the scan queue that overlap the range that we have
168        // identified as needing to be fully scanned. For each such range add it to the
169        // spanning tree (these should all be nonoverlapping ranges, but we might coalesce
170        // some in the process).
171        let mut to_create: Option<SpanningTree> = None;
172        let mut to_delete_ends: Vec<Value> = vec![];
173        while let Some(row) = rows.next().map_err(E::db_error)? {
174            let entry = ScanRange::from_parts(
175                Range {
176                    start: BlockHeight::from(row.get::<_, u32>(0).map_err(E::db_error)?),
177                    end: BlockHeight::from(row.get::<_, u32>(1).map_err(E::db_error)?),
178                },
179                {
180                    let code = row.get::<_, i64>(2).map_err(E::db_error)?;
181                    parse_priority_code(code).ok_or_else(|| {
182                        E::corrupt(format!("scan priority not recognized: {}", code))
183                    })?
184                },
185            );
186            to_delete_ends.push(Value::from(u32::from(entry.block_range().end)));
187            to_create = if let Some(cur) = to_create {
188                Some(cur.insert(entry, force_rescans))
189            } else {
190                Some(SpanningTree::Leaf(entry))
191            };
192        }
193
194        // Update the tree that we read from the database, or if we didn't find any ranges
195        // start with the scanned range.
196        for entry in entries {
197            to_create = if let Some(cur) = to_create {
198                Some(cur.insert(entry, force_rescans))
199            } else {
200                Some(SpanningTree::Leaf(entry))
201            };
202        }
203
204        (to_create, to_delete_ends)
205    };
206
207    if let Some(tree) = to_create {
208        let ends_ptr = Rc::new(to_delete_ends);
209        conn.execute(
210            "DELETE FROM scan_queue WHERE block_range_end IN rarray(:ends)",
211            named_params![":ends": ends_ptr],
212        )
213        .map_err(E::db_error)?;
214
215        let scan_ranges = tree.into_vec();
216        insert_queue_entries(conn, scan_ranges.iter()).map_err(E::db_error)?;
217    }
218
219    Ok(())
220}

222fn extend_range(
223    conn: &rusqlite::Transaction<'_>,
224    range: &Range<BlockHeight>,
225    required_subtree_indices: BTreeSet<u64>,
226    table_prefix: &'static str,
227    fallback_start_height: Option<BlockHeight>,
228    birthday_height: Option<BlockHeight>,
229) -> Result<Option<Range<BlockHeight>>, SqliteClientError> {
230    // we'll either have both min and max bounds, or we'll have neither
231    let subtree_index_bounds = required_subtree_indices
232        .iter()
233        .min()
234        .zip(required_subtree_indices.iter().max());
235
236    let mut shard_end_stmt = conn.prepare_cached(&format!(
237        "SELECT subtree_end_height
238                FROM {}_tree_shards
239                WHERE shard_index = :shard_index",
240        table_prefix
241    ))?;
242
243    let mut shard_end = |index: u64| -> Result<Option<BlockHeight>, rusqlite::Error> {
244        Ok(shard_end_stmt
245            .query_row(named_params![":shard_index": index], |row| {
246                row.get::<_, Option<u32>>(0)
247                    .map(|opt| opt.map(BlockHeight::from))
248            })
249            .optional()?
250            .flatten())
251    };
252
253    // If no notes belonging to the wallet were found, we don't need to extend the scanning
254    // range suggestions to include the associated subtrees, and our bounds are just the
255    // scanned range. Otherwise, ensure that all shard ranges starting from the wallet
256    // birthday are included.
257    subtree_index_bounds
258        .map(|(min_idx, max_idx)| {
259            let range_min = if *min_idx > 0 {
260                // get the block height of the end of the previous shard
261                shard_end(*min_idx - 1)?
262            } else {
263                // our lower bound is going to be the fallback height
264                fallback_start_height
265            };
266
267            // bound the minimum to the wallet birthday
268            let range_min = range_min.map(|h| birthday_height.map_or(h, |b| std::cmp::max(b, h)));
269
270            // Get the block height for the end of the current shard, and make it an
271            // exclusive end bound.
272            let range_max = shard_end(*max_idx)?.map(|end| end + 1);
273
274            Ok::<Range<BlockHeight>, rusqlite::Error>(Range {
275                start: range.start.min(range_min.unwrap_or(range.start)),
276                end: range.end.max(range_max.unwrap_or(range.end)),
277            })
278        })
279        .transpose()
280        .map_err(SqliteClientError::from)
281}

283pub(crate) fn scan_complete<P: consensus::Parameters>(
284    conn: &rusqlite::Transaction<'_>,
285    params: &P,
286    range: Range<BlockHeight>,
287    wallet_note_positions: &[(ShieldedProtocol, Position)],
288) -> Result<(), SqliteClientError> {
289    // Read the wallet birthday (if known).
290    // TODO: use per-pool birthdays?
291    let wallet_birthday = wallet_birthday(conn)?;
292
293    // Determine the range of block heights for which we will be updating the scan queue.
294    let extended_range = {
295        // If notes have been detected in the scan, we need to extend any adjacent un-scanned
296        // ranges starting from the wallet birthday to include the blocks needed to complete
297        // the note commitment tree subtrees containing the positions of the discovered notes.
298        // We will query by subtree index to find these bounds.
299        let mut required_sapling_subtrees = BTreeSet::new();
300        #[cfg(feature = "orchard")]
301        let mut required_orchard_subtrees = BTreeSet::new();
302        for (protocol, position) in wallet_note_positions {
303            match protocol {
304                ShieldedProtocol::Sapling => {
305                    required_sapling_subtrees.insert(
306                        Address::above_position(SAPLING_SHARD_HEIGHT.into(), *position).index(),
307                    );
308                }
309                ShieldedProtocol::Orchard => {
310                    #[cfg(feature = "orchard")]
311                    required_orchard_subtrees.insert(
312                        Address::above_position(ORCHARD_SHARD_HEIGHT.into(), *position).index(),
313                    );
314
315                    #[cfg(not(feature = "orchard"))]
316                    return Err(SqliteClientError::UnsupportedPoolType(PoolType::Shielded(
317                        *protocol,
318                    )));
319                }
320            }
321        }
322
323        let extended_range = extend_range(
324            conn,
325            &range,
326            required_sapling_subtrees,
327            SAPLING_TABLES_PREFIX,
328            params.activation_height(NetworkUpgrade::Sapling),
329            wallet_birthday,
330        )?;
331
332        #[cfg(feature = "orchard")]
333        let extended_range = extend_range(
334            conn,
335            extended_range.as_ref().unwrap_or(&range),
336            required_orchard_subtrees,
337            ORCHARD_TABLES_PREFIX,
338            params.activation_height(NetworkUpgrade::Nu5),
339            wallet_birthday,
340        )?
341        .or(extended_range);
342
343        #[allow(clippy::let_and_return)]
344        extended_range
345    };
346
347    let query_range = extended_range.clone().unwrap_or_else(|| range.clone());
348
349    let scanned = ScanRange::from_parts(range.clone(), ScanPriority::Scanned);
350
351    // If any of the extended range actually extends beyond the scanned range, we need to
352    // scan that extension in order to make the found note(s) spendable. We need to avoid
353    // creating empty ranges here, as that acts as an optimization barrier preventing
354    // `SpanningTree` from merging non-empty scanned ranges on either side.
355    let extended_before = extended_range
356        .as_ref()
357        .map(|extended| ScanRange::from_parts(extended.start..range.start, ScanPriority::FoundNote))
358        .filter(|range| !range.is_empty());
359    let extended_after = extended_range
360        .map(|extended| ScanRange::from_parts(range.end..extended.end, ScanPriority::FoundNote))
361        .filter(|range| !range.is_empty());
362
363    let replacement = Some(scanned)
364        .into_iter()
365        .chain(extended_before)
366        .chain(extended_after);
367
368    replace_queue_entries::<SqliteClientError>(conn, &query_range, replacement, false)?;
369
370    Ok(())
371}

387pub(crate) fn update_chain_tip<P: consensus::Parameters>(
388    conn: &rusqlite::Transaction<'_>,
389    params: &P,
390    new_tip: BlockHeight,
391) -> Result<(), SqliteClientError> {
392    // If the caller provided a chain tip that is before Sapling activation, do nothing.
393    let sapling_activation = match params.activation_height(NetworkUpgrade::Sapling) {
394        Some(h) if h <= new_tip => h,
395        _ => return Ok(()),
396    };
397
398    // Read the previous max scanned height from the blocks table
399    let max_scanned = block_height_extrema(conn)?.map(|range| *range.end());
400
401    // Read the wallet birthday (if known).
402    let wallet_birthday = wallet_birthday(conn)?;
403
404    // If the chain tip is below the prior max scanned height, then the caller has caught
405    // the chain in the middle of a reorg. Do nothing; the caller will continue using the
406    // old scan ranges and either:
407    // - encounter an error trying to fetch the blocks (and thus trigger the same handling
408    //   logic as if this happened with the old linear scanning code); or
409    // - encounter a discontinuity error in `scan_cached_blocks`, at which point they will
410    //   call `WalletDb::truncate_to_height` as part of their reorg handling which will
411    //   resolve the problem.
412    //
413    // We don't check the shard height, as normal usage would have the caller update the
414    // shard state prior to this call, so it is possible and expected to be in a situation
415    // where we should update the tip-related scan ranges but not the shard-related ones.
416    match max_scanned {
417        Some(h) if new_tip < h => return Ok(()),
418        _ => (),
419    };
420
421    // `ScanRange` uses an exclusive upper bound.
422    let chain_end = new_tip + 1;
423
424    // Read the maximum height from each of the shards tables. The minimum of the two
425    // gives the start of a height range that covers the last incomplete shard of both the
426    // Sapling and Orchard pools.
427    let sapling_shard_tip = tip_shard_end_height(conn, SAPLING_TABLES_PREFIX)?;
428    #[cfg(feature = "orchard")]
429    let orchard_shard_tip = tip_shard_end_height(conn, ORCHARD_TABLES_PREFIX)?;
430
431    #[cfg(feature = "orchard")]
432    let min_shard_tip = match (sapling_shard_tip, orchard_shard_tip) {
433        (None, None) => None,
434        (None, Some(o)) => Some(o),
435        (Some(s), None) => Some(s),
436        (Some(s), Some(o)) => Some(std::cmp::min(s, o)),
437    };
438    #[cfg(not(feature = "orchard"))]
439    let min_shard_tip = sapling_shard_tip;
440
441    // Create a scanning range for the fragment of the last shard leading up to new tip.
442    // We set a lower bound at the wallet birthday (if known), because account creation
443    // requires specifying a tree frontier that ensures we don't need tree information
444    // prior to the birthday.
445    let tip_shard_entry = min_shard_tip.filter(|h| h < &chain_end).map(|h| {
446        let min_to_scan = wallet_birthday.filter(|b| b > &h).unwrap_or(h);
447        ScanRange::from_parts(min_to_scan..chain_end, ScanPriority::ChainTip)
448    });
449
450    // Create scan ranges to either validate potentially invalid blocks at the wallet's
451    // view of the chain tip, or connect the prior tip to the new tip.
452    let tip_entry = max_scanned.map_or_else(
453        || {
454            // No blocks have been scanned, so we need to anchor the start of the new scan
455            // range to something else.
456            wallet_birthday.map_or_else(
457                // We don't have a wallet birthday, which means we have no accounts yet.
458                // We can therefore ignore all blocks up to the chain tip.
459                || ScanRange::from_parts(sapling_activation..chain_end, ScanPriority::Ignored),
460                // We have a wallet birthday, so mark all blocks between that and the
461                // chain tip as `Historic` (performing wallet recovery).
462                |wallet_birthday| {
463                    ScanRange::from_parts(wallet_birthday..chain_end, ScanPriority::Historic)
464                },
465            )
466        },
467        |max_scanned| {
468            // The scan range starts at the block after the max scanned height. Since
469            // `scan_cached_blocks` retrieves the metadata for the block being connected to
470            // (if it exists), the connectivity of the scan range to the max scanned block
471            // will always be checked if relevant.
472            let min_unscanned = max_scanned + 1;
473
474            // If we don't have shard metadata, this means we're doing linear scanning, so
475            // create a scan range from the prior tip to the current tip with `Historic`
476            // priority.
477            if tip_shard_entry.is_none() {
478                ScanRange::from_parts(min_unscanned..chain_end, ScanPriority::Historic)
479            } else {
480                // Determine the height to which we expect new blocks retrieved from the
481                // block source to be stable and not subject to being reorg'ed.
482                let stable_height = new_tip.saturating_sub(PRUNING_DEPTH);
483
484                // If the wallet's max scanned height is above the stable height,
485                // prioritize the range between it and the new tip as `ChainTip`.
486                if max_scanned > stable_height {
487                    // We are in the steady-state case, where a wallet is close to the
488                    // chain tip and just needs to catch up.
489                    //
490                    // This overlaps the `tip_shard_entry` range and so will be coalesced
491                    // with it.
492                    ScanRange::from_parts(min_unscanned..chain_end, ScanPriority::ChainTip)
493                } else {
494                    // In this case, the max scanned height is considered stable relative
495                    // to the chain tip. However, it may be stable or unstable relative to
496                    // the prior chain tip, which we could determine by looking up the
497                    // prior chain tip height from the scan queue. For simplicity we merge
498                    // these two cases together, and proceed as though the max scanned
499                    // block is unstable relative to the prior chain tip.
500                    //
501                    // To confirm its stability, prioritize the `VERIFY_LOOKAHEAD` blocks
502                    // above the max scanned height as `Verify`:
503                    //
504                    // - We use `Verify` to ensure that a connectivity check is performed,
505                    //   along with any required rewinds, before any `ChainTip` ranges
506                    //   (from this or any prior `update_chain_tip` call) are scanned.
507                    //
508                    // - We prioritize `VERIFY_LOOKAHEAD` blocks because this is expected
509                    //   to be 12.5 minutes, within which it is reasonable for a user to
510                    //   have potentially received a transaction (if they opened their
511                    //   wallet to provide an address to someone else, or spent their own
512                    //   funds creating a change output), without necessarily having left
513                    //   their wallet open long enough for the transaction to be mined and
514                    //   the corresponding block to be scanned.
515                    //
516                    // - We limit the range to at most the stable region, to prevent any
517                    //   `Verify` ranges from being susceptible to reorgs, and potentially
518                    //   interfering with subsequent `Verify` ranges defined by future
519                    //   calls to `update_chain_tip`. Any gap between `stable_height` and
520                    //   `shard_start_height` will be filled by the scan range merging
521                    //   logic with a `Historic` range.
522                    //
523                    // If `max_scanned == stable_height` then this is a zero-length range.
524                    // In this case, any non-empty `(stable_height+1)..shard_start_height`
525                    // will be marked `Historic`, minimising the prioritised blocks at the
526                    // chain tip and allowing for other ranges (for example, `FoundNote`)
527                    // to take priority.
528                    ScanRange::from_parts(
529                        min_unscanned..min(stable_height + 1, min_unscanned + VERIFY_LOOKAHEAD),
530                        ScanPriority::Verify,
531                    )
532                }
533            }
534        },
535    );
536    if let Some(entry) = &tip_shard_entry {
537        debug!("{} will update latest shard", entry);
538    }
539    debug!("{} will connect prior scanned state to new tip", tip_entry);
540
541    let query_range = match tip_shard_entry.as_ref() {
542        Some(se) => Range {
543            start: min(se.block_range().start, tip_entry.block_range().start),
544            end: max(se.block_range().end, tip_entry.block_range().end),
545        },
546        None => tip_entry.block_range().clone(),
547    };
548
549    replace_queue_entries::<SqliteClientError>(
550        conn,
551        &query_range,
552        tip_shard_entry.into_iter().chain(Some(tip_entry)),
553        false,
554    )?;
555
556    Ok(())
557}

1553    /// * The wallet birthday is in the shard prior to the chain tip

1559    ///   is pruned.
1560    ///
1561    /// The diagram below shows the arrangement. the position of the X indicates the
1562    /// note commitment for the note belonging to our wallet.
1563    /// ```
1564    /// blocks:      |<---- 5000 ---->|<----- 10 ---->|<--- 11 --->|<- 1  ->|<- 1 ->|<----- 110 ----->|
1565    ///       nu5_activation                       birthday                                       chain_tip
1566    /// commitments: |<---- 2^16 ---->|<--(2^16-50)-->|<--- 44 --->|<-___X->|<- 4 ->|<----- 110 ------|

1568    /// ```
1569    ///
1570    /// # Parameters:
1571    /// - `with_birthday_subtree_root`: When this is set to `true`, the wallet state will be
1572    ///   initialized such that the subtree root containing the wallet birthday has been inserted
1573    ///   into the note commitment tree.

1576        with_birthday_subtree_root: bool,

1591                    Frontier::random_with_prior_subtree_roots(
