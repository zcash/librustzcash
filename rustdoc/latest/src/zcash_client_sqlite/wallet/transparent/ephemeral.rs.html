27/// Returns the first unstored ephemeral address index in the given account.
28pub(crate) fn first_unstored_index(
29    conn: &rusqlite::Connection,
30    account_id: AccountRef,
31) -> Result<u32, SqliteClientError> {
32    match conn
33        .query_row(
34            "SELECT address_index FROM ephemeral_addresses
35             WHERE account_id = :account_id
36             ORDER BY address_index DESC
37             LIMIT 1",
38            named_params![":account_id": account_id.0],
39            |row| row.get::<_, u32>(0),
40        )
41        .optional()?
42    {
43        Some(i) if i >= (1 << 31) + GAP_LIMIT => {
44            unreachable!("violates constraint index_range_and_address_nullity")
45        }
46        Some(i) => Ok(i.checked_add(1).unwrap()),
47        None => Ok(0),
48    }
49}

51/// Returns the first unreserved ephemeral address index in the given account.
52pub(crate) fn first_unreserved_index(
53    conn: &rusqlite::Connection,
54    account_id: AccountRef,
55) -> Result<u32, SqliteClientError> {
56    first_unstored_index(conn, account_id)?
57        .checked_sub(GAP_LIMIT)
58        .ok_or(SqliteClientError::CorruptedData(
59            "ephemeral_addresses table has not been initialized".to_owned(),
60        ))
61}

63/// Returns the first ephemeral address index in the given account that
64/// would violate the gap invariant if used.
65pub(crate) fn first_unsafe_index(
66    conn: &rusqlite::Connection,
67    account_id: AccountRef,
68) -> Result<u32, SqliteClientError> {
69    // The inner join with `transactions` excludes addresses for which
70    // `seen_in_tx` is NULL. The query also excludes addresses observed
71    // to have been mined in a transaction that we currently see as unmined.
72    // This is conservative in terms of avoiding violation of the gap
73    // invariant: it can only cause us to get to the end of the gap sooner.
74    //
75    // TODO: do we want to only consider transactions with a minimum number
76    // of confirmations here?
77    let first_unmined_index: u32 = match conn
78        .query_row(
79            "SELECT address_index FROM ephemeral_addresses
80             JOIN transactions t ON t.id_tx = seen_in_tx
81             WHERE account_id = :account_id AND t.mined_height IS NOT NULL
82             ORDER BY address_index DESC
83             LIMIT 1",
84            named_params![":account_id": account_id.0],
85            |row| row.get::<_, u32>(0),
86        )
87        .optional()?
88    {
89        Some(i) if i >= 1 << 31 => {
90            unreachable!("violates constraint index_range_and_address_nullity")
91        }
92        Some(i) => i.checked_add(1).unwrap(),
93        None => 0,
94    };
95    Ok(min(
96        1 << 31,
97        first_unmined_index.checked_add(GAP_LIMIT).unwrap(),
98    ))
99}

101/// Utility function to return an `Range<u32>` that starts at `i`
102/// and is of length up to `n`. The range is truncated if necessary
103/// so that it contains no elements beyond the maximum valid address
104/// index, `(1 << 31) - 1`.
105pub(crate) fn range_from(i: u32, n: u32) -> Range<u32> {
106    let first = min(1 << 31, i);
107    let last = min(1 << 31, i.saturating_add(n));
108    first..last
109}

111/// Returns the ephemeral transparent IVK for a given account ID.
112pub(crate) fn get_ephemeral_ivk<P: consensus::Parameters>(
113    conn: &rusqlite::Connection,
114    params: &P,
115    account_id: AccountRef,
116) -> Result<Option<EphemeralIvk>, SqliteClientError> {
117    let ufvk = conn
118        .query_row(
119            "SELECT ufvk FROM accounts WHERE id = :account_id",
120            named_params![":account_id": account_id.0],
121            |row| {
122                let ufvk_str: Option<String> = row.get("ufvk")?;
123                Ok(ufvk_str.map(|s| {
124                    UnifiedFullViewingKey::decode(params, &s[..])
125                        .map_err(SqliteClientError::BadAccountData)
126                }))
127            },
128        )
129        .optional()?
130        .ok_or(SqliteClientError::AccountUnknown)?
131        .transpose()?;
132
133    let eivk = ufvk
134        .as_ref()
135        .and_then(|ufvk| ufvk.transparent())
136        .map(|t| t.derive_ephemeral_ivk())
137        .transpose()?;
138
139    Ok(eivk)
140}

142/// Returns a vector of ephemeral transparent addresses associated with the given
143/// account controlled by this wallet, along with their metadata. The result includes
144/// reserved addresses, and addresses for `GAP_LIMIT` additional indices (capped to
145/// the maximum index).
146///
147/// If `index_range` is some `Range`, it limits the result to addresses with indices
148/// in that range.
149pub(crate) fn get_known_ephemeral_addresses<P: consensus::Parameters>(
150    conn: &rusqlite::Connection,
151    params: &P,
152    account_id: AccountRef,
153    index_range: Option<Range<u32>>,
154) -> Result<Vec<(TransparentAddress, TransparentAddressMetadata)>, SqliteClientError> {
155    let index_range = index_range.unwrap_or(0..(1 << 31));
156
157    let mut stmt = conn.prepare(
158        "SELECT address, address_index 
159         FROM ephemeral_addresses ea
160         WHERE ea.account_id = :account_id
161         AND address_index >= :start 
162         AND address_index < :end
163         ORDER BY address_index",
164    )?;
165    let mut rows = stmt.query(named_params![
166        ":account_id": account_id.0,
167        ":start": index_range.start,
168        ":end": min(1 << 31, index_range.end),
169    ])?;
170
171    let mut result = vec![];
172
173    while let Some(row) = rows.next()? {
174        let addr_str: String = row.get(0)?;
175        let raw_index: u32 = row.get(1)?;
176        let address_index = NonHardenedChildIndex::from_index(raw_index)
177            .expect("where clause ensures this is in range");
178        let address = TransparentAddress::decode(params, &addr_str)?;
179        result.push((address, metadata(address_index)));
180    }
181    Ok(result)
182}

184/// If this is a known ephemeral address in any account, return its account id.
185pub(crate) fn find_account_for_ephemeral_address_str(
186    conn: &rusqlite::Connection,
187    address_str: &str,
188) -> Result<Option<AccountUuid>, SqliteClientError> {
189    Ok(conn
190        .query_row(
191            "SELECT accounts.uuid 
192             FROM ephemeral_addresses ea
193             JOIN accounts ON accounts.id = ea.account_id
194             WHERE address = :address",
195            named_params![":address": &address_str],
196            |row| Ok(AccountUuid(row.get(0)?)),
197        )
198        .optional()?)
199}

201/// If this is a known ephemeral address in the given account, return its index.
202pub(crate) fn find_index_for_ephemeral_address_str(
203    conn: &rusqlite::Connection,
204    account_uuid: AccountUuid,
205    address_str: &str,
206) -> Result<Option<NonHardenedChildIndex>, SqliteClientError> {
207    let account_id = get_account_ref(conn, account_uuid)?;
208    Ok(conn
209        .query_row(
210            "SELECT address_index FROM ephemeral_addresses
211            WHERE account_id = :account_id AND address = :address",
212            named_params![":account_id": account_id.0, ":address": &address_str],
213            |row| row.get::<_, u32>(0),
214        )
215        .optional()?
216        .map(|index| {
217            NonHardenedChildIndex::from_index(index)
218                .expect("valid by constraint index_range_and_address_nullity")
219        }))
220}

222/// Returns a vector with the next `n` previously unreserved ephemeral addresses for
223/// the given account.
224///
225/// # Errors
226///
227/// * `SqliteClientError::AccountUnknown`, if there is no account with the given id.
228/// * `SqliteClientError::UnknownZip32Derivation`, if the account is imported and
229///   it is not possible to derive new addresses for it.
230/// * `SqliteClientError::ReachedGapLimit`, if it is not possible to reserve `n` addresses
231///   within the gap limit after the last address in this account that is known to have an
232///   output in a mined transaction.
233/// * `SqliteClientError::AddressGeneration(AddressGenerationError::DiversifierSpaceExhausted)`,
234///   if the limit on transparent address indices has been reached.
235pub(crate) fn reserve_next_n_ephemeral_addresses<P: consensus::Parameters>(
236    conn: &rusqlite::Transaction,
237    params: &P,
238    account_id: AccountRef,
239    n: usize,
240) -> Result<Vec<(TransparentAddress, TransparentAddressMetadata)>, SqliteClientError> {
241    if n == 0 {
242        return Ok(vec![]);
243    }
244
245    let first_unreserved = first_unreserved_index(conn, account_id)?;
246    let first_unsafe = first_unsafe_index(conn, account_id)?;
247    let allocation = range_from(
248        first_unreserved,
249        u32::try_from(n).map_err(|_| AddressGenerationError::DiversifierSpaceExhausted)?,
250    );
251
252    if allocation.len() < n {
253        return Err(AddressGenerationError::DiversifierSpaceExhausted.into());
254    }
255    if allocation.end > first_unsafe {
256        let account_uuid = wallet::get_account_uuid(conn, account_id)?;
257        return Err(SqliteClientError::ReachedGapLimit(
258            account_uuid,
259            max(first_unreserved, first_unsafe),
260        ));
261    }
262    reserve_until(conn, params, account_id, allocation.end)?;
263    get_known_ephemeral_addresses(conn, params, account_id, Some(allocation))
264}

276/// Extend the range of stored addresses in an account if necessary so that the index of the next
277/// address to reserve will be *at least* `next_to_reserve`. If no transparent key exists for the
278/// given account or it would already have been at least `next_to_reserve`, then do nothing.
279///
280/// Note that this is called from database migration code.
281///
282/// # Panics
283///
284/// Panics if the precondition `next_to_reserve <= (1 << 31)` does not hold.
285fn reserve_until<P: consensus::Parameters>(
286    conn: &rusqlite::Transaction,
287    params: &P,
288    account_id: AccountRef,
289    next_to_reserve: u32,
290) -> Result<(), SqliteClientError> {
291    assert!(next_to_reserve <= 1 << 31);
292
293    if let Some(ephemeral_ivk) = get_ephemeral_ivk(conn, params, account_id)? {
294        let first_unstored = first_unstored_index(conn, account_id)?;
295        let range_to_store = first_unstored..(next_to_reserve.checked_add(GAP_LIMIT).unwrap());
296        if range_to_store.is_empty() {
297            return Ok(());
298        }
299
300        // used_in_tx and seen_in_tx are initially NULL
301        let mut stmt_insert_ephemeral_address = conn.prepare_cached(
302            "INSERT INTO ephemeral_addresses (account_id, address_index, address)
303             VALUES (:account_id, :address_index, :address)",
304        )?;
305
306        for raw_index in range_to_store {
307            // The range to store may contain indicies that are out of the valid range of non hardened
308            // child indices; we still store explicit rows in the ephemeral_addresses table for these
309            // so that it's possible to find the first unused address using dead reckoning with the gap
310            // limit.
311            let address_str_opt = NonHardenedChildIndex::from_index(raw_index)
312                .map(|address_index| {
313                    ephemeral_ivk
314                        .derive_ephemeral_address(address_index)
315                        .map(|addr| addr.encode(params))
316                })
317                .transpose()?;
318
319            stmt_insert_ephemeral_address.execute(named_params![
320                ":account_id": account_id.0,
321                ":address_index": raw_index,
322                ":address": address_str_opt,
323            ])?;
324        }
325    }
326
327    Ok(())
328}

330/// Returns a `SqliteClientError::EphemeralAddressReuse` error if the address was
331/// already used.
332fn ephemeral_address_reuse_check(
333    conn: &rusqlite::Transaction,
334    address_str: &str,
335) -> Result<(), SqliteClientError> {
336    // It is intentional that we don't require `t.mined_height` to be non-null.
337    // That is, we conservatively treat an ephemeral address as potentially
338    // reused even if we think that the transaction where we had evidence of
339    // its use is at present unmined. This should never occur in supported
340    // situations where only a single correctly operating wallet instance is
341    // using a given seed, because such a wallet will not reuse an address that
342    // it ever reserved.
343    //
344    // `COALESCE(used_in_tx, seen_in_tx)` can only differ from `used_in_tx`
345    // if the address was reserved, an error occurred in transaction creation
346    // before calling `mark_ephemeral_address_as_used`, and then we saw the
347    // address in another transaction (presumably created by another wallet
348    // instance, or as a result of a bug) anyway.
349    let res = conn
350        .query_row(
351            "SELECT t.txid FROM ephemeral_addresses
352             LEFT OUTER JOIN transactions t
353             ON t.id_tx = COALESCE(used_in_tx, seen_in_tx)
354             WHERE address = :address",
355            named_params![":address": address_str],
356            |row| row.get::<_, Option<Vec<u8>>>(0),
357        )
358        .optional()?
359        .flatten();
360
361    if let Some(txid_bytes) = res {
362        let txid = TxId::from_bytes(
363            txid_bytes
364                .try_into()
365                .map_err(|_| SqliteClientError::CorruptedData("invalid txid".to_owned()))?,
366        );
367        Err(SqliteClientError::EphemeralAddressReuse(
368            address_str.to_owned(),
369            txid,
370        ))
371    } else {
372        Ok(())
373    }
374}

376/// If `address` is one of our ephemeral addresses, mark it as having an output
377/// in a transaction that we have just created. This has no effect if `address` is
378/// not one of our ephemeral addresses.
379///
380/// Returns a `SqliteClientError::EphemeralAddressReuse` error if the address was
381/// already used.
382pub(crate) fn mark_ephemeral_address_as_used<P: consensus::Parameters>(
383    conn: &rusqlite::Transaction,
384    params: &P,
385    ephemeral_address: &TransparentAddress,
386    tx_ref: TxRef,
387) -> Result<(), SqliteClientError> {
388    let address_str = ephemeral_address.encode(params);
389    ephemeral_address_reuse_check(conn, &address_str)?;
390
391    // We update both `used_in_tx` and `seen_in_tx` here, because a used address has
392    // necessarily been seen in a transaction. We will not treat this as extending the
393    // range of addresses that are safe to reserve unless and until the transaction is
394    // observed as mined.
395    let update_result = conn
396        .query_row(
397            "UPDATE ephemeral_addresses
398             SET used_in_tx = :tx_ref, seen_in_tx = :tx_ref
399             WHERE address = :address
400             RETURNING account_id, address_index",
401            named_params![":tx_ref": tx_ref.0, ":address": address_str],
402            |row| Ok((AccountRef(row.get::<_, u32>(0)?), row.get::<_, u32>(1)?)),
403        )
404        .optional()?;
405
406    // Maintain the invariant that the last `GAP_LIMIT` addresses are unused and unseen.
407    if let Some((account_id, address_index)) = update_result {
408        let next_to_reserve = address_index.checked_add(1).expect("ensured by constraint");
409        reserve_until(conn, params, account_id, next_to_reserve)?;
410    }
411    Ok(())
412}

414/// If `address` is one of our ephemeral addresses, mark it as having an output
415/// in the given mined transaction (which may or may not be a transaction we sent).
416///
417/// `tx_ref` must be a valid transaction reference. This call has no effect if
418/// `address` is not one of our ephemeral addresses.
419pub(crate) fn mark_ephemeral_address_as_seen<P: consensus::Parameters>(
420    conn: &rusqlite::Transaction,
421    params: &P,
422    address: &TransparentAddress,
423    tx_ref: TxRef,
424) -> Result<(), SqliteClientError> {
425    let address_str = address.encode(params);
426
427    // Figure out which transaction was mined earlier: `tx_ref`, or any existing
428    // tx referenced by `seen_in_tx` for the given address. Prefer the existing
429    // reference in case of a tie or if both transactions are unmined.
430    // This slightly reduces the chance of unnecessarily reaching the gap limit
431    // too early in some corner cases (because the earlier transaction is less
432    // likely to be unmined).
433    //
434    // The query should always return a value if `tx_ref` is valid.
435    let earlier_ref = conn.query_row(
436        "SELECT id_tx FROM transactions
437         LEFT OUTER JOIN ephemeral_addresses e
438         ON id_tx = e.seen_in_tx
439         WHERE id_tx = :tx_ref OR e.address = :address
440         ORDER BY mined_height ASC NULLS LAST,
441                  tx_index ASC NULLS LAST,
442                  e.seen_in_tx ASC NULLS LAST
443         LIMIT 1",
444        named_params![":tx_ref": tx_ref.0, ":address": address_str],
445        |row| row.get::<_, i64>(0),
446    )?;
447
448    let update_result = conn
449        .query_row(
450            "UPDATE ephemeral_addresses
451             SET seen_in_tx = :seen_in_tx
452             WHERE address = :address
453             RETURNING account_id, address_index",
454            named_params![":seen_in_tx": &earlier_ref, ":address": address_str],
455            |row| Ok((AccountRef(row.get::<_, u32>(0)?), row.get::<_, u32>(1)?)),
456        )
457        .optional()?;
458
459    // Maintain the invariant that the last `GAP_LIMIT` addresses are unused and unseen.
460    if let Some((account_id, address_index)) = update_result {
461        let next_to_reserve = address_index.checked_add(1).expect("ensured by constraint");
462        reserve_until(conn, params, account_id, next_to_reserve)?;
463    }
464    Ok(())
