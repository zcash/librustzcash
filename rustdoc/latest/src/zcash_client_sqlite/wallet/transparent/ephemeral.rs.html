29// Returns `TransparentAddressMetadata` in the ephemeral scope for the
30// given address index.
31pub(crate) fn metadata(
32    address_index: NonHardenedChildIndex,
33    exposure: Exposure,
34    next_check_time: Option<SystemTime>,
35) -> TransparentAddressMetadata {
36    TransparentAddressMetadata::derived(
37        TransparentKeyScope::EPHEMERAL,
38        address_index,
39        exposure,
40        next_check_time,
41    )
42}

44/// Returns a vector of ephemeral transparent addresses associated with the given account
45/// controlled by this wallet, along with their metadata. The result includes reserved addresses,
46/// and addresses for the wallet's configured ephemeral address gap limit of additional indices
47/// (capped to the maximum index).
48///
49/// If `index_range` is some `Range`, it limits the result to addresses with indices in that range.
50#[cfg(any(test, feature = "test-dependencies"))]
51pub(crate) fn get_known_ephemeral_addresses<P: consensus::Parameters>(
52    conn: &rusqlite::Connection,
53    params: &P,
54    gap_limits: &GapLimits,
55    account_id: AccountRef,
56    index_range: Option<Range<NonHardenedChildIndex>>,
57) -> Result<Vec<(TransparentAddress, TransparentAddressMetadata)>, SqliteClientError> {
58    use crate::wallet::transparent::find_gap_start;
59
60    let gap_start = find_gap_start(
61        conn,
62        account_id,
63        TransparentKeyScope::EPHEMERAL,
64        gap_limits.ephemeral(),
65    )?;
66
67    let mut stmt = conn.prepare(
68        "SELECT
69            cached_transparent_receiver_address,
70            transparent_child_index,
71            exposed_at_height,
72            transparent_receiver_next_check_time
73         FROM addresses
74         WHERE account_id = :account_id
75         AND transparent_child_index >= :start
76         AND transparent_child_index < :end
77         AND key_scope = :key_scope
78         ORDER BY transparent_child_index",
79    )?;
80
81    let results = stmt
82        .query_and_then(
83            named_params! {
84                ":account_id": account_id.0,
85                ":start": index_range.as_ref().map_or(NonHardenedChildIndex::ZERO, |i| i.start).index(),
86                ":end": index_range.as_ref().map_or(NonHardenedChildIndex::MAX, |i| i.end).index(),
87                ":key_scope": KeyScope::Ephemeral.encode()
88            },
89            |row| {
90                use zcash_client_backend::wallet::GapMetadata;
91
92                let addr_str: String = row.get("cached_transparent_receiver_address")?;
93
94                let raw_index: u32 = row.get("transparent_child_index")?;
95                let address_index = NonHardenedChildIndex::from_index(raw_index)
96                    .expect("where clause ensures this is in range");
97
98                let exposure = row.get::<_, Option<u32>>("exposed_at_height")?.map_or(
99                    Exposure::Unknown,
100                    |h| Exposure::Exposed {
101                        at_height: BlockHeight::from(h),
102                        gap_metadata: gap_start.map_or(
103                            GapMetadata::DerivationUnknown,
104                            |gap_start| {
105                                if let Some(gap_position) = address_index.index().checked_sub(gap_start.index()) {
106                                    GapMetadata::InGap {
107                                        gap_position,
108                                        gap_limit: gap_limits.ephemeral(),
109                                    }
110                                } else {
111                                    GapMetadata::GapRecoverable { gap_limit: gap_limits.ephemeral() }
112                                }
113                            }
114                        )
115                    }
116                );
117
118                let next_check_time = row
119                    .get::<_, Option<i64>>("transparent_receiver_next_check_time")?
120                    .map(decode_epoch_seconds)
121                    .transpose()?;
122
123                Ok::<_, SqliteClientError>((
124                    TransparentAddress::decode(params, &addr_str)?,
125                    metadata(address_index, exposure, next_check_time)
126                ))
127            },
128        )?
129        .collect::<Result<Vec<_>, _>>()?;
130
131    Ok(results)
132}

134/// If this is a known ephemeral address in any account, return its account id.
135pub(crate) fn find_account_for_ephemeral_address_str(
136    conn: &rusqlite::Connection,
137    address_str: &str,
138) -> Result<Option<AccountUuid>, SqliteClientError> {
139    Ok(conn
140        .query_row(
141            "SELECT accounts.uuid
142             FROM addresses
143             JOIN accounts ON accounts.id = account_id
144             WHERE cached_transparent_receiver_address = :address
145             AND key_scope = :key_scope",
146            named_params![
147                ":address": &address_str,
148                ":key_scope": KeyScope::Ephemeral.encode()
149            ],
150            |row| Ok(AccountUuid(row.get(0)?)),
151        )
152        .optional()?)
153}

155pub(crate) fn schedule_ephemeral_address_checks<C: Clock, R: RngCore>(
156    conn: &rusqlite::Transaction,
157    clock: C,
158    mut rng: R,
159) -> Result<(), SqliteClientError> {
160    let mut addr_check_times = conn.prepare(
161        "SELECT id, transparent_receiver_next_check_time
162         FROM addresses
163         WHERE key_scope = :ephemeral_key_scope
164         ORDER BY transparent_receiver_next_check_time NULLS FIRST",
165    )?;
166    let mut rows = addr_check_times
167        .query_and_then(
168            named_params! {
169                ":ephemeral_key_scope": KeyScope::Ephemeral.encode()
170            },
171            |row| {
172                let id: i64 = row.get("id")?;
173                let next_check = row
174                    .get::<_, Option<i64>>("transparent_receiver_next_check_time")?
175                    .map(decode_epoch_seconds)
176                    .transpose()?;
177                Ok::<_, SqliteClientError>((id, next_check))
178            },
179        )?
180        .collect::<Result<Vec<_>, _>>()?;
181
182    if let Some((_, max_check_time)) = rows.last().as_ref() {
183        // Updating the next check time should not result in an already-scheduled check being
184        // further deferred.
185        let mut set_check_time = conn.prepare(
186            "UPDATE addresses
187             SET transparent_receiver_next_check_time = CASE
188                WHEN transparent_receiver_next_check_time < :current_time THEN :next_check
189                WHEN :next_check <= IFNULL(transparent_receiver_next_check_time, :next_check) THEN :next_check
190                ELSE IFNULL(transparent_receiver_next_check_time, :next_check)
191             END
192             WHERE id = :address_id",
193        )?;
194
195        // Set the expected value of the check time such that each ephemeral address will be
196        // checked once per day.
197        let check_interval =
198            (24 * 60 * 60) / u32::try_from(rows.len()).expect("number of addresses fits in a u32");
199        let now = clock.now();
200        let mut check_time = max_check_time.map_or(now, |t| std::cmp::max(t, now));
201
202        // Shuffle the addresses so that we don't always check them in the same order.
203        rows.shuffle(&mut rng);
204        for (address_id, addr_check_time) in rows {
205            // if the check time for this address is absent or in the past, schedule a check.
206            if addr_check_time.iter().all(|t| *t < now) {
207                check_time = next_check_time(&mut rng, check_time, check_interval)?;
208                set_check_time.execute(named_params! {
209                    ":current_time": epoch_seconds(now)?,
210                    ":next_check": epoch_seconds(check_time)?,
211                    ":address_id": address_id
212                })?;
213            }
214        }
215    }
216
217    Ok(())
