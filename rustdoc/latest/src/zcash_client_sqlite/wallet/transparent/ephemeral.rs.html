27// Returns `TransparentAddressMetadata` in the ephemeral scope for the
28// given address index.
29pub(crate) fn metadata(address_index: NonHardenedChildIndex) -> TransparentAddressMetadata {
30    TransparentAddressMetadata::new(TransparentKeyScope::EPHEMERAL, address_index)
31}

33/// Returns a vector of ephemeral transparent addresses associated with the given account
34/// controlled by this wallet, along with their metadata. The result includes reserved addresses,
35/// and addresses for the wallet's configured ephemeral address gap limit of additional indices
36/// (capped to the maximum index).
37///
38/// If `index_range` is some `Range`, it limits the result to addresses with indices in that range.
39pub(crate) fn get_known_ephemeral_addresses<P: consensus::Parameters>(
40    conn: &rusqlite::Connection,
41    params: &P,
42    account_id: AccountRef,
43    index_range: Option<Range<NonHardenedChildIndex>>,
44) -> Result<Vec<(TransparentAddress, TransparentAddressMetadata)>, SqliteClientError> {
45    let mut stmt = conn.prepare(
46        "SELECT cached_transparent_receiver_address, transparent_child_index 
47         FROM addresses
48         WHERE account_id = :account_id
49         AND transparent_child_index >= :start 
50         AND transparent_child_index < :end
51         AND key_scope = :key_scope
52         ORDER BY transparent_child_index",
53    )?;
54
55    let results = stmt
56        .query_and_then(
57            named_params![
58                ":account_id": account_id.0,
59                ":start": index_range.as_ref().map_or(NonHardenedChildIndex::ZERO, |i| i.start).index(),
60                ":end": index_range.as_ref().map_or(NonHardenedChildIndex::MAX, |i| i.end).index(),
61                ":key_scope": KeyScope::Ephemeral.encode()
62            ],
63            |row| {
64                let addr_str: String = row.get(0)?;
65                let raw_index: u32 = row.get(1)?;
66                let address_index = NonHardenedChildIndex::from_index(raw_index)
67                    .expect("where clause ensures this is in range");
68                Ok::<_, SqliteClientError>((
69                    TransparentAddress::decode(params, &addr_str)?,
70                    metadata(address_index)
71                ))
72            },
73        )?
74        .collect::<Result<Vec<_>, _>>()?;
75
76    Ok(results)
77}

79/// If this is a known ephemeral address in any account, return its account id.
80pub(crate) fn find_account_for_ephemeral_address_str(
81    conn: &rusqlite::Connection,
82    address_str: &str,
83) -> Result<Option<AccountUuid>, SqliteClientError> {
84    Ok(conn
85        .query_row(
86            "SELECT accounts.uuid 
87             FROM addresses
88             JOIN accounts ON accounts.id = account_id
89             WHERE cached_transparent_receiver_address = :address
90             AND key_scope = :key_scope",
91            named_params![
92                ":address": &address_str,
93                ":key_scope": KeyScope::Ephemeral.encode()
94            ],
95            |row| Ok(AccountUuid(row.get(0)?)),
96        )
97        .optional()?)
98}

100pub(crate) fn schedule_ephemeral_address_checks<C: Clock, R: RngCore>(
101    conn: &rusqlite::Transaction,
102    clock: C,
103    mut rng: R,
104) -> Result<(), SqliteClientError> {
105    let mut addr_check_times = conn.prepare(
106        "SELECT id, transparent_receiver_next_check_time
107         FROM addresses
108         WHERE key_scope = :ephemeral_key_scope
109         ORDER BY transparent_receiver_next_check_time NULLS FIRST",
110    )?;
111    let mut rows = addr_check_times
112        .query_and_then(
113            named_params! {
114                ":ephemeral_key_scope": KeyScope::Ephemeral.encode()
115            },
116            |row| {
117                let id: i64 = row.get("id")?;
118                let next_check = row
119                    .get::<_, Option<i64>>("transparent_receiver_next_check_time")?
120                    .map(decode_epoch_seconds)
121                    .transpose()?;
122                Ok::<_, SqliteClientError>((id, next_check))
123            },
124        )?
125        .collect::<Result<Vec<_>, _>>()?;
126
127    if let Some((_, max_check_time)) = rows.last().as_ref() {
128        let mut set_check_time = conn.prepare(
129            "UPDATE addresses
130             SET transparent_receiver_next_check_time = :next_check
131             WHERE id = :address_id",
132        )?;
133
134        // Set the expected value of the check time such that each ephemeral address will be
135        // checked once per day.
136        let check_interval =
137            (24 * 60 * 60) / u32::try_from(rows.len()).expect("number of addresses fits in a u32");
138        let start_time = clock.now();
139        let mut check_time = max_check_time.map_or(start_time, |t| std::cmp::max(t, start_time));
140
141        // Shuffle the addresses so that we don't always check them in the same order.
142        rows.shuffle(&mut rng);
143        for (address_id, addr_check_time) in rows {
144            // if the check time for this address is absent or in the past, schedule a check.
145            if addr_check_time.iter().all(|t| *t < start_time) {
146                check_time = next_check_time(&mut rng, check_time, check_interval)?;
147                set_check_time.execute(named_params! {
148                    ":next_check": epoch_seconds(check_time)?,
149                    ":address_id": address_id
150                })?;
151            }
152        }
153    }
154
155    Ok(())
156}
