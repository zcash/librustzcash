1//! Documentation about the wallet database structure.

4//! applies migrations (defined in `crate::wallet::init::migrations`) that produce the
5//! current structure.

8//! represented internally by SQLite. We do not use these constants at runtime; instead we
9//! check the output of the migrations in `crate::wallet::init::tests::verify_schema`, to
10//! pin the expected database structure.

21/// Stores information about the accounts that the wallet is tracking.
22///
23/// An account corresponds to a logical "bucket of funds" that has its own balance within the
24/// wallet and for which spending operations should treat received value as interchangeable,
25/// excepting situations where care must be taken to avoid publicly linking addresses within the
26/// account or where turnstile-crossings may have privacy implications.
27///
28/// ### Columns
29///
30/// - `id`: Internal primary key for the account record.
31/// - `name`: A human-readable reference for the account. This column is present merely as a
32///   convenience for front-ends and debugging; it has no stable semantics and values are not
33///   required to be unique.
34/// - `uuid`: A wallet-instance-specific identifier for the account. This identifier will remain
35///   stable for the lifetime of the wallet database, but is not expected or required to be
36///   stable across wallet restores and it should not be stored in external backup formats.
37/// - `account_kind`: 0 for accounts derived from a mnemonic seed, 1 for imported accounts
38///   for which derivation path information may not be available. This column may be removed in the
39///   future; the distinction between whether an account is derived or imported is better
40///   represented by the presence or absence of HD seed fingerprint and HD account index data.
41/// - `hd_seed_fingerprint`: If this account contains funds in keys obtained via HD derivation,
42///   the ZIP 32 fingerprint of the root HD seed. If this column is non-null, `hd_account_index`
43///   must also be non-null.
44/// - `hd_account_index`: If this account contains funds in keys obtained via HD derivation,
45///   the BIP 44 account-level component of the HD derivation path. If this column is non-null,
46///   `hd_seed_fingerprint` must also be non-null.
47/// - `ufvk`: The unified full viewing key for the account, if known.
48/// - `uivk`: The unified incoming viewing key for the account.
49/// - `orchard_fvk_item_cache`: The serialized representation of the Orchard item of the `ufvk`,
50///   if any.
51/// - `sapling_fvk_item_cache`: The serialized representation of the Sapling item of the `ufvk`,
52///   if any.
53/// - `p2pkh_fvk_item_cache`: The serialized representation of the P2PKH item of the `ufvk`,
54///   if any.
55/// - `birthday_height`: The minimum block height among blocks that may potentially contain
56///   shielded funds belonging to the account.
57/// - `birthday_sapling_tree_size`: A cache of the size of the Sapling note commitment tree
58///   as of the start of the birthday block.
59/// - `birthday_orchard_tree_size`: A cache of the size of the Orchard note commitment tree
60///   as of the start of the birthday block.
61/// - `recover_until_height`: The boundary between recovery and regular scanning for this account.
62///   Unscanned blocks up to and excluding this height are counted towards recovery progress. It
63///   is initially set via the `AccountBirthday` parameter of the `WalletWrite::import_account_*`
64///   methods (usually to the chain tip height at which account recovery was initiated), and may
65///   in future be automatically updated by the backend if the wallet is offline for an extended
66///   period (to keep the scan progress percentage accurate to what actually needs scanning).
67/// - `has_spend_key`: A boolean flag (0 or 1) indicating whether the application that embeds
68///   this wallet database has access to spending key(s) for the account.
69/// - `zcash_legacy_address_index`: This column is only potentially populated for wallets imported
70///   from a `zcashd` `wallet.dat` file, for "standalone" Sapling addresses (each of which
71///   corresponds to an independent account) derived after the introduction of mnemonic seed
72///   derivation in the `4.7.0` `zcashd` release. This column will only be non-negative in
73///   the case that the `hd_account_index` column has the value `0x7FFFFFFF`, in accordance with
74///   how post-v4.7.0 Sapling addresses were produced by the `z_getnewaddress` RPC method.
75///   This relationship is not currently enforced by a CHECK constraint; such a constraint should
76///   be added the next time that the `accounts` table is deleted and re-created to support a
77///   SQLite-breaking change to the columns of the table.
78pub(super) const TABLE_ACCOUNTS: &str = r#"
79CREATE TABLE "accounts" (
80    id INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
81    name TEXT,
82    uuid BLOB NOT NULL,
83    account_kind INTEGER NOT NULL DEFAULT 0,
84    key_source TEXT,
85    hd_seed_fingerprint BLOB,
86    hd_account_index INTEGER,
87    ufvk TEXT,
88    uivk TEXT NOT NULL,
89    orchard_fvk_item_cache BLOB,
90    sapling_fvk_item_cache BLOB,
91    p2pkh_fvk_item_cache BLOB,
92    birthday_height INTEGER NOT NULL,
93    birthday_sapling_tree_size INTEGER,
94    birthday_orchard_tree_size INTEGER,
95    recover_until_height INTEGER,
96    has_spend_key INTEGER NOT NULL DEFAULT 1,
97    zcashd_legacy_address_index INTEGER NOT NULL DEFAULT -1,
98    CHECK (
99      (
100        account_kind = 0
101        AND hd_seed_fingerprint IS NOT NULL
102        AND hd_account_index IS NOT NULL
103        AND ufvk IS NOT NULL
104      )
105      OR
106      (
107        account_kind = 1
108        AND (hd_seed_fingerprint IS NULL) = (hd_account_index IS NULL)
109      )
110    )
111)"#;

120/// Stores addresses that have been generated from accounts in the wallet.
121///
122/// ### Columns
123///
124/// - `account_id`: the account whose IVK was used to derive this address.
125/// - `diversifier_index_be`: the diversifier index at which this address was derived.
126///   This may be null for imported standalone addresses.
127/// - `key_scope`: the BIP 44 change-level index at which this address was derived, or `-1`
128///   for imported transparent pubkeys.
129/// - `address`: The Unified, Sapling, or transparent address. For Unified and Sapling addresses,
130///   only external-key scoped addresses should be stored in this table; for purely transparent
131///   addresses, this may be an internal-scope (change) address, so that we can provide
132///   compatibility with HD-derived change addresses produced by transparent-only wallets.
133/// - `transparent_child_index`: the diversifier index in integer form, if it is in the range of a  `u31`
134///   (i.e. a non-hardened transparent address index). It is used for gap limit handling, and is set
135///   whenever a transparent address at a given index should be scanned at receive time. This
136///   includes:
137///   - Unified Addresses with transparent receivers (at any valid index).
138///   - Unified Addresses without transparent receivers, but within the gap limit of potential
139///     sequential transparent addresses.
140///   - Transparent change addresses.
141///   - ZIP 320 ephemeral addresses.
142///
143///   This column exists because the diversifier index is stored as a byte array, meaning that we
144///   cannot use SQL integer operations on it for gap limit calculations, and thus need it as an
145///   integer as well.
146/// - `cached_transparent_receiver_address`: the transparent address derived from the same
147///   viewing key and at the same diversifier index as `address`. This may be the same as `address`
148///   in the case of an internal-scope transparent change address or a ZIP 320 interstitial
149///   address, and it may be a receiver within `address` in the case of a Unified Address with
150///   transparent receiver. It is cached directly in the table to make account lookups for
151///   transparent outputs more efficient, enabling joins to [`TABLE_TRANSPARENT_RECEIVED_OUTPUTS`].
152/// - `exposed_at_height`: Our best knowledge as to when this address was first exposed to the
153///   wider ecosystem.
154///   - For user-generated addresses, this is the chain tip height at the time that the address was
155///     generated by an explicit request by the user or reserved for use in a ZIP 320 transaction.
156///     These heights are not recoverable from chain.
157///   - In the case of an address with its first use discovered in a transaction obtained by scanning
158///     the chain, this will be set to the mined height of that transaction. In recover from seed
159///     cases, this is what user-generated addresses will be assigned.
160/// - `receiver_flags`: A set of bitflags that describes which receiver types are included in
161///   `address`. See the documentation of [`ReceiverFlags`] for details.
162/// - `transparent_receiver_next_check_time`: The Unix epoch time at which a client should next
163///   check to determine whether any new UTXOs have been received by the cached transparent receiver
164///   address. At present, this will ordinarily be populated only for ZIP 320 ephemeral addresses.
165//  - `imported_transparent_receiver_pubkey`: The 33-byte pubkey corresponding to the
166//    `cached_transparent_receiver_address` value, for imported transparent addresses that were not
167//    obtained via derivation from an HD seed associated with the account. In cases that
168//    `cached_transparent_receiver_address` is non-null, either this column or
169//    `transparent_child_index` must also be non-null.
170///
171/// [`ReceiverFlags`]: crate::wallet::encoding::ReceiverFlags
172pub(super) const TABLE_ADDRESSES: &str = r#"
173CREATE TABLE "addresses" (
174    id INTEGER NOT NULL PRIMARY KEY,
175    account_id INTEGER NOT NULL
176        REFERENCES accounts(id) ON DELETE CASCADE,
177    key_scope INTEGER NOT NULL,
178    diversifier_index_be BLOB,
179    address TEXT NOT NULL,
180    transparent_child_index INTEGER,
181    cached_transparent_receiver_address TEXT,
182    exposed_at_height INTEGER,
183    receiver_flags INTEGER NOT NULL,
184    transparent_receiver_next_check_time INTEGER,
185    imported_transparent_receiver_pubkey BLOB,
186    UNIQUE (account_id, key_scope, diversifier_index_be),
187    UNIQUE (imported_transparent_receiver_pubkey),
188    CONSTRAINT ck_addr_transparent_index_consistency CHECK (
189        (transparent_child_index IS NULL OR diversifier_index_be < x'0000000F00000000000000')
190        AND (
191            (
192                cached_transparent_receiver_address IS NULL
193                AND transparent_child_index IS NULL
194                AND imported_transparent_receiver_pubkey IS NULL
195            )
196            OR (
197                cached_transparent_receiver_address IS NOT NULL
198                AND (transparent_child_index IS NULL) == (imported_transparent_receiver_pubkey IS NOT NULL)
199            )
200        )
201    ),
202    CONSTRAINT ck_addr_foreign_or_diversified CHECK (
203        (diversifier_index_be IS NULL) == (key_scope = -1)
204    )
205)"#;
206pub(super) const INDEX_ADDRESSES_ACCOUNTS: &str = r#"
207CREATE INDEX idx_addresses_accounts ON addresses (
208    account_id ASC
209)"#;
210pub(super) const INDEX_ADDRESSES_INDICES: &str = r#"
211CREATE INDEX idx_addresses_indices ON addresses (

213)"#;
214pub(super) const INDEX_ADDRESSES_PUBKEYS: &str = r#"

239/// Stores the wallet's transactions.
240///
241/// Any transactions that the wallet observes as being associated with one of the accounts in
242/// [`TABLE_ACCOUNTS`] may be tracked in this table. As a result, this table may contain
243/// data that is not recoverable from the chain (for example, transactions created by the
244/// wallet that expired before being mined).
245///
246/// When an account is deleted, all transactions that are associated with that account in some way
247/// that are not associated with any *other* account in the wallet must be first be deleted before
248/// the account deletion operation is allowed to proceed.
249///
250/// ### Columns
251/// - `created`: The time at which the transaction was created as a string in the format
252///   `yyyy-MM-dd HH:mm:ss.fffffffzzz`.
253/// - `block`: stores the height (in the wallet's chain view) of the mined block containing the
254///   transaction. It is `NULL` for transactions that have not yet been observed in scanned blocks,
255///   including transactions in the mempool or that have expired.
256/// - `mined_height`: stores the height (in the wallet's chain view) of the mined block containing
257///   the transaction. It is present to allow the block height for a retrieved transaction to be
258///   stored without requiring that the entire block containing the transaction be scanned; the
259///   foreign key constraint on `block` prevents that column from being populated prior to complete
260///   scanning of the block. This is constrained to be equal to the `block` column if `block` is
261///   non-null.
262/// - `tx_index`: the index of the transaction within the block.
263/// - `expiry_height`: stores the maximum height at which the transaction may be mined, if known.
264/// - `raw`: the original serialized byte representation of the transaction, if it has been
265///   retrieved.
266/// - `fee`: the fee paid to send the transaction, if known. This should be present for all
267///   transactions constructed by this wallet.
268/// - `target_height`: stores the target height for which the transaction was constructed, if
269///   known. This will ordinarily be null for transactions discovered via chain scanning; it
270///   will only be set for transactions created using this wallet specifically, and not any
271///   other wallet that uses the same seed (including previous installations of the same
272///   wallet application.)
273/// - `min_observed_height`: the mempool height at the time that the wallet observed the
274///   transaction, or the mined height of the transaction, whichever is less.
275/// - `confirmed_unmined_at_height`: the maximum block height at which the wallet has observed
276///   positive proof that the transaction has not been mined in a block. Must be NULL if
277///   `mined_height` is not null.
278/// - `trust_status`: A flag indicating whether the transaction should be considered "trusted".
279///   When set to `1`, outputs of this transaction will be considered spendable with `trusted`
280///   confirmations instead of `untrusted` confirmations.
281pub(super) const TABLE_TRANSACTIONS: &str = r#"
282CREATE TABLE "transactions" (
283    id_tx INTEGER PRIMARY KEY,
284    txid BLOB NOT NULL UNIQUE,
285    created TEXT,
286    block INTEGER,
287    mined_height INTEGER,
288    tx_index INTEGER,
289    expiry_height INTEGER,
290    raw BLOB,
291    fee INTEGER,
292    target_height INTEGER,
293    min_observed_height INTEGER NOT NULL,
294    confirmed_unmined_at_height INTEGER,
295    trust_status INTEGER,
296    FOREIGN KEY (block) REFERENCES blocks(height),
297    CONSTRAINT height_consistency CHECK (
298        block IS NULL OR mined_height = block
299    ),
300    CONSTRAINT min_observed_consistency CHECK (
301        mined_height IS NULL OR min_observed_height <= mined_height
302    ),
303    CONSTRAINT confirmed_unmined_consistency CHECK (
304        confirmed_unmined_at_height IS NULL OR mined_height IS NULL
305    )
306)"#;

467/// Stores the transparent outputs received by the wallet.
468///
469/// Originally this table only stored the current UTXO set (as of latest refresh), and the
470/// table was cleared prior to loading in the latest UTXO set. We now upsert instead of
471/// insert into the database, meaning that spent outputs are left in the database. This
472/// makes it similar to the `*_received_notes` tables in that it can store history.
473/// Depending upon how transparent TXOs for the wallet are discovered, the following
474/// may be true:
475/// - The table may have incomplete contents for recovered-from-seed wallets.
476/// - The table may have inconsistent contents for seeds loaded into multiple wallets
477///   simultaneously.
478/// - The wallet's transparent balance may be incorrect prior to "transaction enhancement"
479///   (downloading the full transaction containing the transparent output spend).
480///
481/// ### Columns:
482/// - `id`: Primary key
483/// - `transaction_id`: Reference to the transaction in which this TXO was created
484/// - `output_index`: The output index of this TXO in the transaction referred to by `transaction_id`
485/// - `account_id`: The account that controls spend authority for this TXO
486/// - `address`: The address to which this TXO was sent. We store this address to make querying
487///   for UTXOs for a single address easier, because when shielding we always select UTXOs
488///   for only a single address at a time to prevent linking addresses in the shielding
489///   transaction.
490/// - `script`: The full txout script
491/// - `value_zat`: The value of the TXO in zatoshis
492/// - `max_observed_unspent_height`: The maximum block height at which this TXO was observed to be

829/// A map from nullifiers to the transaction they were observed in.

848    block_height INTEGER NOT NULL,

851        FOREIGN KEY (block_height, tx_index)
852        REFERENCES tx_locator_map(block_height, tx_index)

955WITH
956notes AS (
957    -- Outputs received in this transaction
958    SELECT ro.account_id              AS account_id,
959           ro.transaction_id          AS transaction_id,
960           ro.pool                    AS pool,
961           id_within_pool_table,
962           ro.value                   AS value,
963           ro.value                   AS received_value,
964           0                          AS spent_value,
965           0                          AS spent_note_count,
966           CASE
967                WHEN ro.is_change THEN 1
968                ELSE 0
969           END AS change_note_count,
970           CASE
971                WHEN ro.is_change THEN 0
972                ELSE 1
973           END AS received_count,
974           CASE
975             WHEN (ro.memo IS NULL OR ro.memo = X'F6')
976               THEN 0
977             ELSE 1
978           END AS memo_present,
979           -- The wallet cannot receive transparent outputs in shielding transactions.
980           CASE
981             WHEN ro.pool = 0
982               THEN 1
983             ELSE 0
984           END AS does_not_match_shielding
985    FROM v_received_outputs ro
986    UNION
987    -- Outputs spent in this transaction
988    SELECT ro.account_id              AS account_id,
989           ros.transaction_id         AS transaction_id,
990           ro.pool                    AS pool,
991           id_within_pool_table,
992           -ro.value                  AS value,
993           0                          AS received_value,
994           ro.value                   AS spent_value,
995           1                          AS spent_note_count,
996           0                          AS change_note_count,
997           0                          AS received_count,
998           0                          AS memo_present,
999           -- The wallet cannot spend shielded outputs in shielding transactions.
1000           CASE
1001             WHEN ro.pool != 0
1002               THEN 1
1003             ELSE 0
1004           END AS does_not_match_shielding
1005    FROM v_received_outputs ro
1006    JOIN v_received_output_spends ros
1007         ON ros.pool = ro.pool
1008         AND ros.received_output_id = ro.id_within_pool_table

1010-- Obtain a count of the notes that the wallet created in each transaction,
1011-- not counting change notes.
1012sent_note_counts AS (
1013    SELECT sent_notes.from_account_id     AS account_id,
1014           sent_notes.transaction_id      AS transaction_id,
1015           COUNT(DISTINCT sent_notes.id)  AS sent_notes,
1016           SUM(
1017             CASE
1018               WHEN (sent_notes.memo IS NULL OR sent_notes.memo = X'F6' OR ro.transaction_id IS NOT NULL)
1019                 THEN 0
1020               ELSE 1
1021             END
1022           ) AS memo_count
1023    FROM sent_notes
1024    LEFT JOIN v_received_outputs ro ON sent_notes.id = ro.sent_note_id
1025    WHERE COALESCE(ro.is_change, 0) = 0
1026    GROUP BY account_id, sent_notes.transaction_id

1028blocks_max_height AS (
1029    SELECT MAX(blocks.height) AS max_height FROM blocks

1032       transactions.mined_height    AS mined_height,
1033       transactions.txid            AS txid,
1034       transactions.tx_index        AS tx_index,
1035       transactions.expiry_height   AS expiry_height,
1036       transactions.raw             AS raw,
1037       SUM(notes.value)             AS account_balance_delta,
1038       SUM(notes.spent_value)       AS total_spent,
1039       SUM(notes.received_value)    AS total_received,
1040       transactions.fee             AS fee_paid,
1041       SUM(notes.change_note_count) > 0  AS has_change,
1042       MAX(COALESCE(sent_note_counts.sent_notes, 0))  AS sent_note_count,
1043       SUM(notes.received_count)         AS received_note_count,
1044       SUM(notes.memo_present) + MAX(COALESCE(sent_note_counts.memo_count, 0)) AS memo_count,
1045       blocks.time                       AS block_time,
1046       (
1047            transactions.mined_height IS NULL
1048            AND transactions.expiry_height BETWEEN 1 AND blocks_max_height.max_height
1049       ) AS expired_unmined,
1050       SUM(notes.spent_note_count) AS spent_note_count,
1051       (
1052            -- All of the wallet-spent and wallet-received notes are consistent with a
1053            -- shielding transaction.
1054            SUM(notes.does_not_match_shielding) = 0
1055            -- The transaction contains at least one wallet-spent output.
1056            AND SUM(notes.spent_note_count) > 0
1057            -- The transaction contains at least one wallet-received note.
1058            AND (SUM(notes.received_count) + SUM(notes.change_note_count)) > 0
1059            -- We do not know about any external outputs of the transaction.
1060            AND MAX(COALESCE(sent_note_counts.sent_notes, 0)) = 0
1061       ) AS is_shielding,
1062       transactions.trust_status
1063FROM notes
1064JOIN accounts ON accounts.id = notes.account_id
1065JOIN transactions ON transactions.id_tx = notes.transaction_id
1066LEFT JOIN blocks_max_height
1067LEFT JOIN blocks ON blocks.height = transactions.mined_height
1068LEFT JOIN sent_note_counts
1069     ON sent_note_counts.account_id = notes.account_id
1070     AND sent_note_counts.transaction_id = notes.transaction_id
1071GROUP BY notes.account_id, notes.transaction_id";
1072
1073/// Selects all outputs received by the wallet, plus any outputs sent from the wallet to
1074/// external recipients.
1075///
1076/// This will contain:
1077/// * Outputs received from external recipients
1078/// * Outputs sent to external recipients
1079/// * Outputs received as part of a wallet-internal operation, including
1080///   both outputs received as a consequence of wallet-internal transfers
1081///   and as change.
1082///
1083/// # Columns
1084/// - `txid`: The id of the transaction in which the output was sent or received.
1085/// - `output_pool`: The value pool for the transaction; valid values for this are:
1086///   - 0: Transparent
1087///   - 2: Sapling
1088///   - 3: Orchard
1089/// - `output_index`: The index of the output within the transaction bundle associated with
1090///   the `output_pool` value; that is, within `vout` for transparent, the vector of
1091///   Sapling `OutputDescription` values, or the vector of Orchard actions.
1092/// - `from_account_uuid`: The UUID of the wallet account that created the output, if the wallet
1093///   spent notes in creating the transaction. Note that if multiple accounts in the wallet
1094///   contributed funds in creating the associated transaction, redundant rows will exist in the
1095///   output of this view, one for each such account.
1096/// - `to_account_uuid`: The UUID of the wallet account that received the output, if any; for
1097///   outgoing transaction outputs this will be `NULL`.
1098/// - `address`: The address to which the output was sent; for received outputs, this is the
1099///   address at which the output was received, or `NULL` for wallet-internal outputs.
1100/// - `diversifier_index_be`: The big-endian representation of the diversifier index (or, for
1101///   transparent addresses, the BIP 44 change-level index of the derivation path) of the receiving
1102///   address. This will be `NULL` for outgoing transaction outputs.
1103/// - `value`: The value of the output, in zatoshis.
1104/// - `is_change`: `0` for outgoing outputs and outputs received at external-facing addresses, `1`
1105///   for outputs received at wallet-internal addresses. This represents a best-effort judgement
1106///   for whether or not the output should be considered change, and may not be correct for
1107///   cross-account internal transactions, shielding transactions, or outputs explicitly sent from
1108///   the wallet to itself. The determination of what counts as change is somewhat subjective and
1109///   the value of this column should be used with caution.
1110/// - `memo`: The binary content of the memo associated with the output, if the output is a
1111///   shielded output and the memo was received by the wallet, sent by the wallet or was able to be
1112///   decrypted with the wallet's outgoing viewing key.
1113pub(super) const VIEW_TX_OUTPUTS: &str = "
1114CREATE VIEW v_tx_outputs AS
1115WITH unioned AS (
1116    -- select all outputs received by the wallet
1117    SELECT transactions.txid            AS txid,
1118           ro.pool                      AS output_pool,
1119           ro.output_index              AS output_index,
1120           from_account.uuid            AS from_account_uuid,
1121           to_account.uuid              AS to_account_uuid,
1122           a.address                    AS to_address,
1123           a.diversifier_index_be       AS diversifier_index_be,
1124           ro.value                     AS value,
1125           ro.is_change                 AS is_change,
1126           ro.memo                      AS memo
1127    FROM v_received_outputs ro
1128    JOIN transactions
1129        ON transactions.id_tx = ro.transaction_id
1130    LEFT JOIN addresses a ON a.id = ro.address_id
1131    -- join to the sent_notes table to obtain `from_account_id`
1132    LEFT JOIN sent_notes ON sent_notes.id = ro.sent_note_id
1133    -- join on the accounts table to obtain account UUIDs
1134    LEFT JOIN accounts from_account ON from_account.id = sent_notes.from_account_id
1135    LEFT JOIN accounts to_account ON to_account.id = ro.account_id
1136    UNION ALL
1137    -- select all outputs sent from the wallet to external recipients
1138    SELECT transactions.txid            AS txid,
1139           sent_notes.output_pool       AS output_pool,
1140           sent_notes.output_index      AS output_index,
1141           from_account.uuid            AS from_account_uuid,
1142           NULL                         AS to_account_uuid,
1143           sent_notes.to_address        AS to_address,
1144           NULL                         AS diversifier_index_be,
1145           sent_notes.value             AS value,
1146           0                            AS is_change,
1147           sent_notes.memo              AS memo
1148    FROM sent_notes
1149    JOIN transactions
1150        ON transactions.id_tx = sent_notes.transaction_id
1151    LEFT JOIN v_received_outputs ro ON ro.sent_note_id = sent_notes.id
1152    -- join on the accounts table to obtain account UUIDs
1153    LEFT JOIN accounts from_account ON from_account.id = sent_notes.from_account_id
1154)
1155-- merge duplicate rows while retaining maximum information
1156SELECT
1157    txid,
1158    output_pool,
1159    output_index,
1160    max(from_account_uuid) AS from_account_uuid,
1161    max(to_account_uuid) AS to_account_uuid,
1162    max(to_address) AS to_address,
1163    max(value) AS value,
1164    max(is_change) AS is_change,
1165    max(memo) AS memo
1166FROM unioned
1167GROUP BY txid, output_pool, output_index";

1170    format!(
1171        "CREATE VIEW v_sapling_shard_scan_ranges AS
1172        SELECT
1173            shard.shard_index,
1174            shard.shard_index << 16 AS start_position,
1175            (shard.shard_index + 1) << 16 AS end_position_exclusive,
1176            IFNULL(prev_shard.subtree_end_height, {}) AS subtree_start_height,
1177            shard.subtree_end_height,
1178            shard.contains_marked,
1179            scan_queue.block_range_start,
1180            scan_queue.block_range_end,
1181            scan_queue.priority

1183        LEFT OUTER JOIN sapling_tree_shards prev_shard
1184            ON shard.shard_index = prev_shard.shard_index + 1
1185        -- Join with scan ranges that overlap with the subtree's involved blocks.
1186        INNER JOIN scan_queue ON (
1187            subtree_start_height < scan_queue.block_range_end AND
1188            (
1189                scan_queue.block_range_start <= shard.subtree_end_height OR
1190                shard.subtree_end_height IS NULL
1191            )
1192        )",
1193        u32::from(params.activation_height(NetworkUpgrade::Sapling).unwrap()),
1194    )
