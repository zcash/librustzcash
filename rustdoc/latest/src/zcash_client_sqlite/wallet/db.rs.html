1//! Documentation about the wallet database structure.
2//!
3//! The database structure is managed by [`crate::wallet::init::WalletMigrator`], which
4//! applies migrations (defined in `crate::wallet::init::migrations`) that produce the
5//! current structure.
6//!
7//! The SQL code in this module's constants encodes the current database structure, as
8//! represented internally by SQLite. We do not use these constants at runtime; instead we
9//! check the output of the migrations in `crate::wallet::init::tests::verify_schema`, to
10//! pin the expected database structure.

64/// Stores addresses that have been generated from accounts in the wallet.
65///
66/// ### Columns
67///
68/// - `account_id`: the account whose IVK was used to derive this address.
69/// - `diversifier_index_be`: the diversifier index at which this address was derived.
70/// - `key_scope`: the key scope for which this address was derived.
71/// - `address`: The Unified, Sapling, or transparent address. For Unified and Sapling addresses,
72///   only external-key scoped addresses should be stored in this table; for purely transparent
73///   addresses, this may be an internal-scope (change) address, so that we can provide
74///   compatibility with HD-derived change addresses produced by transparent-only wallets.
75/// - `transparent_child_index`: the diversifier index in integer form, if it is in the range of a  `u31`
76///   (i.e. a non-hardened transparent address index). It is used for gap limit handling, and is set
77///   whenever a transparent address at a given index should be scanned at receive time. This
78///   includes:
79///   - Unified Addresses with transparent receivers (at any valid index).
80///   - Unified Addresses without transparent receivers, but within the gap limit of potential
81///     sequential transparent addresses.
82///   - Transparent change addresses.
83///   - ZIP 320 ephemeral addresses.
84///
85///   This column exists because the diversifier index is stored as a byte array, meaning that we
86///   cannot use SQL integer operations on it for gap limit calculations, and thus need it as an
87///   integer as well.
88/// - `cached_transparent_receiver_address`: the transparent address derived from the same
89///   viewing key and at the same diversifier index as `address`. This may be the same as `address`
90///   in the case of an internal-scope transparent change address or a ZIP 320 interstitial
91///   address, and it may be a receiver within `address` in the case of a Unified Address with
92///   transparent receiver. It is cached directly in the table to make account lookups for
93///   transparent outputs more efficient, enabling joins to [`TABLE_TRANSPARENT_RECEIVED_OUTPUTS`].
94/// - `exposed_at_height`: Our best knowledge as to when this address was first exposed to the
95///   wider ecosystem.
96///   - For user-generated addresses, this is the chain tip height at the time that the address was
97///     generated by an explicit request by the user or reserved for use in a ZIP 320 transaction.
98///     These heights are not recoverable from chain.
99///   - In the case of an address with its first use discovered in a transaction obtained by scanning
100///     the chain, this will be set to the mined height of that transaction. In recover from seed
101///     cases, this is what user-generated addresses will be assigned.
102/// - `receiver_flags`: A set of bitflags that describes which receiver types are included in
103///   `address`. See the documentation of [`ReceiverFlags`] for details.
104/// - `transparent_receiver_next_check_time`: The Unix epoch time at which a client should next
105///   check to determine whether any new UTXOs have been received by the cached transparent receiver
106///   address. At present, this will ordinarily be populated only for ZIP 320 ephemeral addresses.
107///
108/// [`ReceiverFlags`]: crate::wallet::encoding::ReceiverFlags
109pub(super) const TABLE_ADDRESSES: &str = r#"
110CREATE TABLE "addresses" (
111    id INTEGER NOT NULL PRIMARY KEY,
112    account_id INTEGER NOT NULL,
113    key_scope INTEGER NOT NULL,
114    diversifier_index_be BLOB NOT NULL,
115    address TEXT NOT NULL,
116    transparent_child_index INTEGER,
117    cached_transparent_receiver_address TEXT,
118    exposed_at_height INTEGER,
119    receiver_flags INTEGER NOT NULL,
120    transparent_receiver_next_check_time INTEGER,
121    FOREIGN KEY (account_id) REFERENCES accounts(id),
122    CONSTRAINT diversification UNIQUE (account_id, key_scope, diversifier_index_be),
123    CONSTRAINT transparent_index_consistency CHECK (
124        (transparent_child_index IS NOT NULL) == (cached_transparent_receiver_address IS NOT NULL)
125    )
126)"#;
127pub(super) const INDEX_ADDRESSES_ACCOUNTS: &str = r#"
128CREATE INDEX idx_addresses_accounts ON addresses (
129    account_id ASC
130)"#;
131pub(super) const INDEX_ADDRESSES_INDICES: &str = r#"
132CREATE INDEX idx_addresses_indices ON addresses (
133    diversifier_index_be ASC
134)"#;
135pub(super) const INDEX_ADDRESSES_T_INDICES: &str = r#"
136CREATE INDEX idx_addresses_t_indices ON addresses (
137    transparent_child_index ASC
138)"#;

156/// Stores the wallet's transactions.
157///
158/// Any transactions that the wallet observes as "belonging to" one of the accounts in
159/// [`TABLE_ACCOUNTS`] may be tracked in this table. As a result, this table may contain
160/// data that is not recoverable from the chain (for example, transactions created by the
161/// wallet that expired before being mined).
162///
163/// ### Columns
164/// - `created`: The time at which the transaction was created as a string in the format
165///   `yyyy-MM-dd HH:mm:ss.fffffffzzz`.
166/// - `block`: stores the height (in the wallet's chain view) of the mined block containing the
167///   transaction. It is `NULL` for transactions that have not yet been observed in scanned blocks,
168///   including transactions in the mempool or that have expired.
169/// - `mined_height`: stores the height (in the wallet's chain view) of the mined block containing
170///   the transaction. It is present to allow the block height for a retrieved transaction to be
171///   stored without requiring that the entire block containing the transaction be scanned; the
172///   foreign key constraint on `block` prevents that column from being populated prior to complete
173///   scanning of the block. This is constrained to be equal to the `block` column if `block` is
174///   non-null.
175/// - `target_height`: stores the target height for which the transaction was constructed, if
176///   known. This will ordinarily be null for transactions discovered via chain scanning; it
177///   will only be set for transactions created using this wallet specifically, and not any
178///   other wallet that uses the same seed (including previous installations of the same
179///   wallet application.)
180pub(super) const TABLE_TRANSACTIONS: &str = r#"
181CREATE TABLE "transactions" (
182    id_tx INTEGER PRIMARY KEY,
183    txid BLOB NOT NULL UNIQUE,
184    created TEXT,
185    block INTEGER,
186    mined_height INTEGER,
187    tx_index INTEGER,
188    expiry_height INTEGER,
189    raw BLOB,
190    fee INTEGER,
191    target_height INTEGER,
192    FOREIGN KEY (block) REFERENCES blocks(height),
193    CONSTRAINT height_consistency CHECK (block IS NULL OR mined_height = block)
194)"#;

206/// - `rcm`: the random commitment trapdoor for the note
207/// - `nf`: the nullifier that will be exposed when the note is spent

246/// A junction table between received Sapling notes and the transactions that spend them.
247///
248/// Only one mined transaction can spend a note. However, transactions created by the
249/// wallet may expire before being mined, and the wallet still tracks the fact that the
250/// user created the transaction. The junction table enables the "spent-in" relationship
251/// between notes and expired transactions to be preserved; note spent-ness is determined
252/// by joining this table with [`TABLE_TRANSACTIONS`] and then filtering out transactions
253/// where either `transactions.block` is non-null, or `transactions.expiry_height` is not
254/// greater than the wallet's view of the chain tip.
255pub(super) const TABLE_SAPLING_RECEIVED_NOTE_SPENDS: &str = "
256CREATE TABLE sapling_received_note_spends (
257    sapling_received_note_id INTEGER NOT NULL,
258    transaction_id INTEGER NOT NULL,
259    FOREIGN KEY (sapling_received_note_id)
260        REFERENCES sapling_received_notes(id)
261        ON DELETE CASCADE,
262    FOREIGN KEY (transaction_id)
263        -- We do not delete transactions, so this does not cascade
264        REFERENCES transactions(id_tx),
265    UNIQUE (sapling_received_note_id, transaction_id)
266)";

337/// Stores the transparent outputs received by the wallet.
338///
339/// Originally this table only stored the current UTXO set (as of latest refresh), and the
340/// table was cleared prior to loading in the latest UTXO set. We now upsert instead of
341/// insert into the database, meaning that spent outputs are left in the database. This
342/// makes it similar to the `*_received_notes` tables in that it can store history.
343/// Depending upon how transparent TXOs for the wallet are discovered, the following
344/// may be true:
345/// - The table may have incomplete contents for recovered-from-seed wallets.
346/// - The table may have inconsistent contents for seeds loaded into multiple wallets
347///   simultaneously.
348/// - The wallet's transparent balance may be incorrect prior to "transaction enhancement"
349///   (downloading the full transaction containing the transparent output spend).
350///
351/// ### Columns:
352/// - `id`: Primary key
353/// - `transaction_id`: Reference to the transaction in which this TXO was created
354/// - `output_index`: The output index of this TXO in the transaction referred to by `transaction_id`
355/// - `account_id`: The account that controls spend authority for this TXO
356/// - `address`: The address to which this TXO was sent. We store this address to make querying
357///   for UTXOs for a single address easier, because when shielding we always select UTXOs
358///   for only a single address at a time to prevent linking addresses in the shielding
359///   transaction.
360/// - `script`: The full txout script
361/// - `value_zat`: The value of the TXO in zatoshis
362/// - `max_observed_unspent_height`: The maximum block height at which this TXO was either
363///   observed to be a member of the UTXO set at the start of the block, or observed
364///   to be an output of a transaction mined in the block. This is intended to be used to
365///   determine when the TXO is no longer a part of the UTXO set, in the case that the
366///   transaction that spends it is not detected by the wallet.
367/// - `address_id`: a foreign key to the address that this note was sent to; non-null because
368///   we can only find transparent outputs for known addresses (and therefore we must record
369///   both internal and external addresses in the `addresses` table).
370pub(super) const TABLE_TRANSPARENT_RECEIVED_OUTPUTS: &str = r#"
371CREATE TABLE "transparent_received_outputs" (
372    id INTEGER PRIMARY KEY,
373    transaction_id INTEGER NOT NULL,
374    output_index INTEGER NOT NULL,
375    account_id INTEGER NOT NULL,
376    address TEXT NOT NULL,
377    script BLOB NOT NULL,
378    value_zat INTEGER NOT NULL,
379    max_observed_unspent_height INTEGER,
380    address_id INTEGER NOT NULL REFERENCES addresses(id),
381    FOREIGN KEY (transaction_id) REFERENCES transactions(id_tx),
382    FOREIGN KEY (account_id) REFERENCES accounts(id),
383    CONSTRAINT transparent_output_unique UNIQUE (transaction_id, output_index)
384)"#;
385pub(super) const INDEX_TRANSPARENT_RECEIVED_OUTPUTS_ACCOUNT_ID: &str = r#"
386CREATE INDEX idx_transparent_received_outputs_account_id
387ON "transparent_received_outputs" (account_id)"#;

433/// Stores the outputs of transactions created by the wallet.
434///
435/// Unlike with outputs received by the wallet, we store sent outputs for all pools in
436/// this table, distinguished by the `output_pool` column. The information we want to
437/// record for sent outputs is the same across all pools, whereas for received outputs we
438/// want to cache pool-specific data.
439///
440/// ### Columns
441/// - `(tx, output_pool, output_index)` collectively identify a transaction output.
442/// - `from_account_id`: the ID of the account that created the transaction.
443///   - On recover-from-seed or when scanning by UFVK, this will be either the account
444///     that decrypted the output, or one of the accounts that funded the transaction.
445/// - `to_address`: the address of the external recipient of this output, or `NULL` if the
446///   output was received by the wallet.
447/// - `to_account_id`: the ID of the account that received this output, or `NULL` if the
448///   output was for an external recipient.
449/// - `value`: the value of the output in zatoshis.
450/// - `memo`: the memo bytes associated with this output, if known.
451///   - This is always `NULL` for transparent outputs.
452///   - This will be set for all shielded outputs of transactions created by the wallet.
453///   - On recover-from-seed or when scanning by UFVK, this will only be set for shielded
454///     outputs after post-scanning transaction enhancement. For shielded notes sent to
455///     external recipients, the transaction needs to have been created with an
456///     [`OvkPolicy`] using a known OVK.
457///
458/// [`OvkPolicy`]: zcash_client_backend::wallet::OvkPolicy
459pub(super) const TABLE_SENT_NOTES: &str = r#"
460CREATE TABLE "sent_notes" (
461    id INTEGER PRIMARY KEY,
462    tx INTEGER NOT NULL,
463    output_pool INTEGER NOT NULL,
464    output_index INTEGER NOT NULL,
465    from_account_id INTEGER NOT NULL,
466    to_address TEXT,
467    to_account_id INTEGER,
468    value INTEGER NOT NULL,
469    memo BLOB,
470    FOREIGN KEY (tx) REFERENCES transactions(id_tx),
471    FOREIGN KEY (from_account_id) REFERENCES accounts(id),
472    FOREIGN KEY (to_account_id) REFERENCES accounts(id),
473    CONSTRAINT tx_output UNIQUE (tx, output_pool, output_index),
474    CONSTRAINT note_recipient CHECK (
475        (to_address IS NOT NULL) OR (to_account_id IS NOT NULL)
476    )
477)"#;

503/// Stores the set of transaction outputs received by the wallet for which spend information
504/// (if any) should be retrieved.
505///
506/// This table is populated in the process of wallet recovery when a deshielding transaction

673/// A map from nullifiers to the transaction they were observed in.
674///
675/// The purpose of this map is to allow non-linear scanning. If the wallet scans a block
676/// range `Y..Z` that leaves a gap between the wallet's birthday height and `Y`, then the
677/// wallet must assume that any nullifier observed in `Y..Z` might be spending one of its
678/// notes (that it has not yet observed), otherwise it will fail to detect those spends
679/// and report a too-large balance. Once the wallet has scanned every block between its
680/// birthday height and `Y`, the nullifier map contents up to `Z` is no longer necessary
681/// and can be dropped.
682///
683/// The map stores transaction locators instead of transaction IDs for efficiency. SQLite
684/// will represent the transaction locator in at most 6 bytes, so a transaction that only
685/// spends one shielded note will incur a 12-byte overhead (across both this table and
686/// [`TABLE_TX_LOCATOR_MAP`]), but each additional spent note in a transaction saves 26
687/// bytes.
688pub(super) const TABLE_NULLIFIER_MAP: &str = "
689CREATE TABLE nullifier_map (
690    spend_pool INTEGER NOT NULL,
691    nf BLOB NOT NULL,
692    block_height INTEGER NOT NULL,
693    tx_index INTEGER NOT NULL,
694    CONSTRAINT tx_locator
695        FOREIGN KEY (block_height, tx_index)
696        REFERENCES tx_locator_map(block_height, tx_index)
697        ON DELETE CASCADE
698        ON UPDATE RESTRICT,
699    CONSTRAINT nf_uniq UNIQUE (spend_pool, nf)
700)";
701pub(super) const INDEX_NF_MAP_LOCATOR_IDX: &str =
702    r#"CREATE INDEX nf_map_locator_idx ON nullifier_map(block_height, tx_index)"#;

791pub(super) const VIEW_TRANSACTIONS: &str = "
792CREATE VIEW v_transactions AS
793WITH
794notes AS (
795    -- Outputs received in this transaction
796    SELECT ro.account_id              AS account_id,
797           transactions.mined_height  AS mined_height,
798           transactions.txid          AS txid,
799           ro.pool                    AS pool,
800           id_within_pool_table,
801           ro.value                   AS value,
802           ro.value                   AS received_value,
803           0                          AS spent_value,
804           0                          AS spent_note_count,
805           CASE
806                WHEN ro.is_change THEN 1
807                ELSE 0
808           END AS change_note_count,
809           CASE
810                WHEN ro.is_change THEN 0
811                ELSE 1
812           END AS received_count,
813           CASE
814             WHEN (ro.memo IS NULL OR ro.memo = X'F6')
815               THEN 0
816             ELSE 1
817           END AS memo_present,
818           -- The wallet cannot receive transparent outputs in shielding transactions.
819           CASE
820             WHEN ro.pool = 0
821               THEN 1
822             ELSE 0
823           END AS does_not_match_shielding
824    FROM v_received_outputs ro
825    JOIN transactions
826         ON transactions.id_tx = ro.transaction_id
827    UNION
828    -- Outputs spent in this transaction
829    SELECT ro.account_id              AS account_id,
830           transactions.mined_height  AS mined_height,
831           transactions.txid          AS txid,
832           ro.pool                    AS pool,
833           id_within_pool_table,
834           -ro.value                  AS value,
835           0                          AS received_value,
836           ro.value                   AS spent_value,
837           1                          AS spent_note_count,
838           0                          AS change_note_count,
839           0                          AS received_count,
840           0                          AS memo_present,
841           -- The wallet cannot spend shielded outputs in shielding transactions.
842           CASE
843             WHEN ro.pool != 0
844               THEN 1
845             ELSE 0
846           END AS does_not_match_shielding
847    FROM v_received_outputs ro
848    JOIN v_received_output_spends ros
849         ON ros.pool = ro.pool
850         AND ros.received_output_id = ro.id_within_pool_table
851    JOIN transactions
852         ON transactions.id_tx = ros.transaction_id
853),
854-- Obtain a count of the notes that the wallet created in each transaction,
855-- not counting change notes.
856sent_note_counts AS (
857    SELECT sent_notes.from_account_id     AS account_id,
858           transactions.txid              AS txid,
859           COUNT(DISTINCT sent_notes.id)  AS sent_notes,
860           SUM(
861             CASE
862               WHEN (sent_notes.memo IS NULL OR sent_notes.memo = X'F6' OR ro.transaction_id IS NOT NULL)
863                 THEN 0
864               ELSE 1
865             END
866           ) AS memo_count
867    FROM sent_notes
868    JOIN transactions
869         ON transactions.id_tx = sent_notes.tx
870    LEFT JOIN v_received_outputs ro
871         ON sent_notes.id = ro.sent_note_id
872    WHERE COALESCE(ro.is_change, 0) = 0
873    GROUP BY account_id, txid
874),
875blocks_max_height AS (
876    SELECT MAX(blocks.height) AS max_height FROM blocks
877)
878SELECT accounts.uuid                AS account_uuid,
879       notes.mined_height           AS mined_height,
880       notes.txid                   AS txid,
881       transactions.tx_index        AS tx_index,
882       transactions.expiry_height   AS expiry_height,
883       transactions.raw             AS raw,
884       SUM(notes.value)             AS account_balance_delta,
885       SUM(notes.spent_value)       AS total_spent,
886       SUM(notes.received_value)    AS total_received,
887       transactions.fee             AS fee_paid,
888       SUM(notes.change_note_count) > 0  AS has_change,
889       MAX(COALESCE(sent_note_counts.sent_notes, 0))  AS sent_note_count,
890       SUM(notes.received_count)         AS received_note_count,
891       SUM(notes.memo_present) + MAX(COALESCE(sent_note_counts.memo_count, 0)) AS memo_count,
892       blocks.time                       AS block_time,
893       (
894            blocks.height IS NULL
895            AND transactions.expiry_height BETWEEN 1 AND blocks_max_height.max_height
896       ) AS expired_unmined,
897       SUM(notes.spent_note_count) AS spent_note_count,
898       (
899            -- All of the wallet-spent and wallet-received notes are consistent with a
900            -- shielding transaction.
901            SUM(notes.does_not_match_shielding) = 0
902            -- The transaction contains at least one wallet-spent output.
903            AND SUM(notes.spent_note_count) > 0
904            -- The transaction contains at least one wallet-received note.
905            AND (SUM(notes.received_count) + SUM(notes.change_note_count)) > 0
906            -- We do not know about any external outputs of the transaction.
907            AND MAX(COALESCE(sent_note_counts.sent_notes, 0)) = 0
908       ) AS is_shielding
909FROM notes
910LEFT JOIN accounts ON accounts.id = notes.account_id
911LEFT JOIN transactions
912     ON notes.txid = transactions.txid
913JOIN blocks_max_height
914LEFT JOIN blocks ON blocks.height = notes.mined_height
915LEFT JOIN sent_note_counts
916     ON sent_note_counts.account_id = notes.account_id
917     AND sent_note_counts.txid = notes.txid
918GROUP BY notes.account_id, notes.txid";

920/// Selects all outputs received by the wallet, plus any outputs sent from the wallet to
921/// external recipients.
922///
923/// This will contain:
924/// * Outputs received from external recipients
925/// * Outputs sent to external recipients
926/// * Outputs received as part of a wallet-internal operation, including
927///   both outputs received as a consequence of wallet-internal transfers
928///   and as change.
929///
930/// The `to_address` column will only contain an address when the recipient is
931/// external. In all other cases, the recipient account id indicates the account
932/// that controls the output.
933pub(super) const VIEW_TX_OUTPUTS: &str = "
934CREATE VIEW v_tx_outputs AS
935WITH unioned AS (
936    -- select all outputs received by the wallet
937    SELECT transactions.txid            AS txid,
938           ro.pool                      AS output_pool,
939           ro.output_index              AS output_index,
940           from_account.uuid            AS from_account_uuid,
941           to_account.uuid              AS to_account_uuid,
942           NULL                         AS to_address,
943           ro.value                     AS value,
944           ro.is_change                 AS is_change,
945           ro.memo                      AS memo
946    FROM v_received_outputs ro
947    JOIN transactions
948        ON transactions.id_tx = ro.transaction_id
949    -- join to the sent_notes table to obtain `from_account_id`
950    LEFT JOIN sent_notes ON sent_notes.id = ro.sent_note_id
951    -- join on the accounts table to obtain account UUIDs
952    LEFT JOIN accounts from_account ON from_account.id = sent_notes.from_account_id
953    LEFT JOIN accounts to_account ON to_account.id = ro.account_id
954    UNION ALL
955    -- select all outputs sent from the wallet to external recipients
956    SELECT transactions.txid            AS txid,
957           sent_notes.output_pool       AS output_pool,
958           sent_notes.output_index      AS output_index,
959           from_account.uuid            AS from_account_uuid,
960           NULL                         AS to_account_uuid,
961           sent_notes.to_address        AS to_address,
962           sent_notes.value             AS value,
963           0                            AS is_change,
964           sent_notes.memo              AS memo
965    FROM sent_notes
966    JOIN transactions
967        ON transactions.id_tx = sent_notes.tx
968    LEFT JOIN v_received_outputs ro ON ro.sent_note_id = sent_notes.id
969    -- join on the accounts table to obtain account UUIDs
970    LEFT JOIN accounts from_account ON from_account.id = sent_notes.from_account_id
971)
972-- merge duplicate rows while retaining maximum information
973SELECT
974    txid,
975    output_pool,
976    output_index,
977    max(from_account_uuid) AS from_account_uuid,
978    max(to_account_uuid) AS to_account_uuid,
979    max(to_address) AS to_address,
980    max(value) AS value,
981    max(is_change) AS is_change,
982    max(memo) AS memo
983FROM unioned
984GROUP BY txid, output_pool, output_index";

988        "CREATE VIEW v_sapling_shard_scan_ranges AS
989        SELECT
990            shard.shard_index,
991            shard.shard_index << 16 AS start_position,
992            (shard.shard_index + 1) << 16 AS end_position_exclusive,
993            IFNULL(prev_shard.subtree_end_height, {}) AS subtree_start_height,

1007                shard.subtree_end_height IS NULL
