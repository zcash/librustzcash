26pub(crate) fn pool_code(pool_type: PoolType) -> i64 {
27    // These constants are *incidentally* shared with the typecodes
28    // for unified addresses, but this is exclusively an internal
29    // implementation detail.
30    match pool_type {
31        PoolType::Transparent => 0i64,
32        PoolType::Shielded(ShieldedProtocol::Sapling) => 2i64,
33        PoolType::Shielded(ShieldedProtocol::Orchard) => 3i64,
34    }
35}

44pub(crate) fn encode_diversifier_index_be(idx: DiversifierIndex) -> [u8; 11] {
45    let mut di_be = *idx.as_bytes();
46    di_be.reverse();
47    di_be
48}
49
50pub(crate) fn decode_diversifier_index_be(
51    di_be: &[u8],
52) -> Result<DiversifierIndex, SqliteClientError> {
53    let mut di_be: [u8; 11] = di_be.try_into().map_err(|_| {
54        SqliteClientError::CorruptedData("Diversifier index is not an 11-byte value".to_owned())
55    })?;
56    di_be.reverse();
57
58    Ok(DiversifierIndex::from(di_be))
59}

61pub(crate) fn memo_repr(memo: Option<&MemoBytes>) -> Option<&[u8]> {
62    memo.map(|m| {
63        if m == &MemoBytes::empty() {
64            // we store the empty memo as a single 0xf6 byte
65            &[0xf6]
66        } else {
67            m.as_slice()
68        }
69    })
70}

73pub(crate) fn epoch_seconds(t: SystemTime) -> Result<i64, SchedulingError> {
74    let integer_seconds_since_epoch =
75        i64::try_from(t.duration_since(SystemTime::UNIX_EPOCH)?.as_secs())?;

77    Ok(integer_seconds_since_epoch)

81pub(crate) fn decode_epoch_seconds(i: i64) -> Result<SystemTime, SchedulingError> {
82    use std::time::Duration;
83
84    Ok(SystemTime::UNIX_EPOCH + Duration::from_secs(u64::try_from(i)?))

87/// An enumeration of the scopes of keys that are generated by the `zcash_client_sqlite`
88/// implementation of the `WalletWrite` trait.
89///
90/// This extends the [`zip32::Scope`] type to include the custom scope used to generate keys for
91/// ephemeral transparent addresses.
92#[derive(Clone, Copy, Debug, PartialEq, Eq)]
93pub(crate) enum KeyScope {
94    /// A key scope corresponding to a [`zip32::Scope`].
95    Zip32(zip32::Scope),
96    /// An ephemeral transparent address, which is derived from an account's transparent
97    /// [`AccountPubKey`] with the BIP 44 path `change` level index set to the value `2`.
98    ///
99    /// [`AccountPubKey`]: zcash_primitives::legacy::keys::AccountPubKey
100    Ephemeral,
101}
102
103impl KeyScope {
104    pub(crate) const EXTERNAL: KeyScope = KeyScope::Zip32(zip32::Scope::External);
105    pub(crate) const INTERNAL: KeyScope = KeyScope::Zip32(zip32::Scope::Internal);
106
107    pub(crate) fn encode(&self) -> i64 {
108        match self {
109            KeyScope::Zip32(zip32::Scope::External) => 0i64,
110            KeyScope::Zip32(zip32::Scope::Internal) => 1i64,
111            KeyScope::Ephemeral => 2i64,
112        }
113    }
114
115    pub(crate) fn decode(code: i64) -> Result<Self, SqliteClientError> {
116        match code {
117            0i64 => Ok(KeyScope::EXTERNAL),
118            1i64 => Ok(KeyScope::INTERNAL),
119            2i64 => Ok(KeyScope::Ephemeral),
120            other => Err(SqliteClientError::CorruptedData(format!(
121                "Invalid key scope code: {}",
122                other
123            ))),
124        }
125    }
126}
127
128impl From<zip32::Scope> for KeyScope {
129    fn from(value: zip32::Scope) -> Self {
130        KeyScope::Zip32(value)
131    }
132}
133
134#[cfg(feature = "transparent-inputs")]
135impl From<KeyScope> for TransparentKeyScope {
136    fn from(value: KeyScope) -> Self {
137        match value {
138            KeyScope::Zip32(scope) => scope.into(),
139            KeyScope::Ephemeral => TransparentKeyScope::custom(2).expect("valid scope"),
140        }
141    }
142}
143
144impl TryFrom<KeyScope> for zip32::Scope {
145    type Error = ();
146
147    fn try_from(value: KeyScope) -> Result<Self, Self::Error> {
148        match value {
149            KeyScope::Zip32(scope) => Ok(scope),
150            KeyScope::Ephemeral => Err(()),
151        }
152    }
153}

155bitflags! {
156    /// A set of flags describing the type(s) of outputs that a Zcash address can receive.
157    #[derive(Clone, Copy, Debug, PartialEq, Eq)]
158    pub(crate) struct ReceiverFlags: i64 {
159        /// The address did not contain any recognized receiver types.
160        const UNKNOWN = 0b00000000;
161        /// The associated address can receive transparent p2pkh outputs.
162        const P2PKH = 0b00000001;
163        /// The associated address can receive transparent p2sh outputs.
164        const P2SH = 0b00000010;
165        /// The associated address can receive Sapling outputs.
166        const SAPLING = 0b00000100;
167        /// The associated address can receive Orchard outputs.
168        const ORCHARD = 0b00001000;
169    }
170}
171
172impl ReceiverFlags {
173    pub(crate) fn required(request: ReceiverRequirements) -> Self {
174        let mut flags = ReceiverFlags::UNKNOWN;
175        if matches!(request.orchard(), ReceiverRequirement::Require) {
176            flags |= ReceiverFlags::ORCHARD;
177        }
178        if matches!(request.sapling(), ReceiverRequirement::Require) {
179            flags |= ReceiverFlags::SAPLING;
180        }
181        if matches!(request.p2pkh(), ReceiverRequirement::Require) {
182            flags |= ReceiverFlags::P2PKH;
183        }
184        flags
185    }
186
187    pub(crate) fn omitted(request: ReceiverRequirements) -> Self {
188        let mut flags = ReceiverFlags::UNKNOWN;
189        if matches!(request.orchard(), ReceiverRequirement::Omit) {
190            flags |= ReceiverFlags::ORCHARD;
191        }
192        if matches!(request.sapling(), ReceiverRequirement::Omit) {
193            flags |= ReceiverFlags::SAPLING;
194        }
195        if matches!(request.p2pkh(), ReceiverRequirement::Omit) {
196            flags |= ReceiverFlags::P2PKH;
197        }
198        flags
199    }
200}
201
202/// Computes the [`ReceiverFlags`] describing the types of outputs that the provided
203/// [`UnifiedAddress`] can receive.
204impl From<&UnifiedAddress> for ReceiverFlags {
205    fn from(value: &UnifiedAddress) -> Self {
206        let mut flags = ReceiverFlags::UNKNOWN;
207        match value.transparent() {
208            Some(PublicKeyHash(_)) => {
209                flags |= ReceiverFlags::P2PKH;
210            }
211            Some(ScriptHash(_)) => {
212                flags |= ReceiverFlags::P2SH;
213            }
214            _ => {}
215        }
216        if value.has_sapling() {
217            flags |= ReceiverFlags::SAPLING;
218        }
219        if value.has_orchard() {
220            flags |= ReceiverFlags::ORCHARD;
221        }
222        flags
223    }
224}
225
226/// Computes the [`ReceiverFlags`] describing the types of outputs that the provided
227/// [`Address`] can receive.
228impl From<&Address> for ReceiverFlags {
229    fn from(address: &Address) -> Self {
230        match address {
231            Address::Sapling(_) => ReceiverFlags::SAPLING,
232            Address::Transparent(addr) => match addr {
233                PublicKeyHash(_) => ReceiverFlags::P2PKH,
234                ScriptHash(_) => ReceiverFlags::P2SH,
235            },
236            Address::Unified(ua) => ReceiverFlags::from(ua),
237            Address::Tex(_) => ReceiverFlags::P2PKH,
238        }
239    }
240}

242impl TryFromAddress for ReceiverFlags {
243    type Error = ();
244
245    fn try_from_sapling(
246        _net: NetworkType,
247        _data: [u8; 43],
248    ) -> Result<Self, ConversionError<Self::Error>> {
249        Ok(ReceiverFlags::SAPLING)
250    }
251
252    fn try_from_unified(
253        _net: NetworkType,
254        data: zcash_address::unified::Address,
255    ) -> Result<Self, ConversionError<Self::Error>> {
256        let mut result = ReceiverFlags::UNKNOWN;
257        for i in data.items() {
258            match i {
259                Receiver::Orchard(_) => result |= ReceiverFlags::ORCHARD,
260                Receiver::Sapling(_) => result |= ReceiverFlags::SAPLING,
261                Receiver::P2pkh(_) => result |= ReceiverFlags::P2PKH,
262                Receiver::P2sh(_) => result |= ReceiverFlags::P2SH,
263                Receiver::Unknown { .. } => {}
264            }
265        }
266
267        Ok(result)
268    }
269
270    fn try_from_transparent_p2pkh(
271        _net: NetworkType,
272        _data: [u8; 20],
273    ) -> Result<Self, ConversionError<Self::Error>> {
274        Ok(ReceiverFlags::P2PKH)
275    }
276
277    fn try_from_transparent_p2sh(
278        _net: NetworkType,
279        _data: [u8; 20],
280    ) -> Result<Self, ConversionError<Self::Error>> {
281        Ok(ReceiverFlags::P2SH)
282    }
283
284    fn try_from_tex(
285        _net: NetworkType,
286        _data: [u8; 20],
287    ) -> Result<Self, ConversionError<Self::Error>> {
288        Ok(ReceiverFlags::P2PKH)
289    }
