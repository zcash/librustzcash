31pub(crate) fn detect_spending_accounts<'a>(
32    conn: &Connection,
33    spent: impl Iterator<Item = &'a OutPoint>,
34) -> Result<HashSet<AccountUuid>, rusqlite::Error> {
35    let mut account_q = conn.prepare_cached(
36        "SELECT accounts.uuid
37        FROM transparent_received_outputs o
38        JOIN accounts ON accounts.id = o.account_id
39        JOIN transactions t ON t.id_tx = o.transaction_id
40        WHERE t.txid = :prevout_txid
41        AND o.output_index = :prevout_idx",
42    )?;
43
44    let mut acc = HashSet::new();
45    for prevout in spent {
46        for account in account_q.query_and_then(
47            named_params![
48                ":prevout_txid": prevout.hash(),
49                ":prevout_idx": prevout.n()
50            ],
51            |row| row.get(0).map(AccountUuid),
52        )? {
53            acc.insert(account?);
54        }
55    }
56
57    Ok(acc)
58}

60/// Returns the `NonHardenedChildIndex` corresponding to a diversifier index
61/// given as bytes in big-endian order (the reverse of the usual order).
62fn address_index_from_diversifier_index_be(
63    diversifier_index_be: &[u8],
64) -> Result<NonHardenedChildIndex, SqliteClientError> {
65    let mut di: [u8; 11] = diversifier_index_be.try_into().map_err(|_| {
66        SqliteClientError::CorruptedData("Diversifier index is not an 11-byte value".to_owned())
67    })?;
68    di.reverse(); // BE -> LE conversion
69
70    NonHardenedChildIndex::from_index(DiversifierIndex::from(di).try_into().map_err(|_| {
71        SqliteClientError::CorruptedData(
72            "Unable to get diversifier for transparent address.".to_string(),
73        )
74    })?)
75    .ok_or_else(|| {
76        SqliteClientError::CorruptedData(
77            "Unexpected hardened index for transparent address.".to_string(),
78        )
79    })
80}

82pub(crate) fn get_transparent_receivers<P: consensus::Parameters>(
83    conn: &rusqlite::Connection,
84    params: &P,
85    account_uuid: AccountUuid,
86) -> Result<HashMap<TransparentAddress, Option<TransparentAddressMetadata>>, SqliteClientError> {
87    let mut ret: HashMap<TransparentAddress, Option<TransparentAddressMetadata>> = HashMap::new();
88
89    // Get all UAs derived
90    let mut ua_query = conn.prepare(
91        "SELECT address, diversifier_index_be 
92         FROM addresses 
93         JOIN accounts ON accounts.id = addresses.account_id
94         WHERE accounts.uuid = :account_uuid",
95    )?;
96    let mut rows = ua_query.query(named_params![":account_uuid": account_uuid.0])?;
97
98    while let Some(row) = rows.next()? {
99        let ua_str: String = row.get(0)?;
100        let di_vec: Vec<u8> = row.get(1)?;
101
102        let ua = Address::decode(params, &ua_str)
103            .ok_or_else(|| {
104                SqliteClientError::CorruptedData("Not a valid Zcash recipient address".to_owned())
105            })
106            .and_then(|addr| match addr {
107                Address::Unified(ua) => Ok(ua),
108                _ => Err(SqliteClientError::CorruptedData(format!(
109                    "Addresses table contains {} which is not a unified address",
110                    ua_str,
111                ))),
112            })?;
113
114        if let Some(taddr) = ua.transparent() {
115            let address_index = address_index_from_diversifier_index_be(&di_vec)?;
116            let metadata = TransparentAddressMetadata::new(Scope::External.into(), address_index);
117            ret.insert(*taddr, Some(metadata));
118        }
119    }
120
121    if let Some((taddr, address_index)) =
122        get_legacy_transparent_address(params, conn, account_uuid)?
123    {
124        let metadata = TransparentAddressMetadata::new(Scope::External.into(), address_index);
125        ret.insert(taddr, Some(metadata));
126    }
127
128    Ok(ret)
129}

131pub(crate) fn uivk_legacy_transparent_address<P: consensus::Parameters>(
132    params: &P,
133    uivk_str: &str,
134) -> Result<Option<(TransparentAddress, NonHardenedChildIndex)>, SqliteClientError> {
135    use ::transparent::keys::ExternalIvk;
136    use zcash_address::unified::{Container as _, Encoding as _};
137
138    let (network, uivk) = Uivk::decode(uivk_str)
139        .map_err(|e| SqliteClientError::CorruptedData(format!("Unable to parse UIVK: {e}")))?;
140
141    if params.network_type() != network {
142        let network_name = |n| match n {
143            consensus::NetworkType::Main => "mainnet",
144            consensus::NetworkType::Test => "testnet",
145            consensus::NetworkType::Regtest => "regtest",
146        };
147        return Err(SqliteClientError::CorruptedData(format!(
148            "Network type mismatch: account UIVK is for {} but a {} address was requested.",
149            network_name(network),
150            network_name(params.network_type())
151        )));
152    }
153
154    // Derive the default transparent address (if it wasn't already part of a derived UA).
155    for item in uivk.items() {
156        if let Ivk::P2pkh(tivk_bytes) = item {
157            let tivk = ExternalIvk::deserialize(&tivk_bytes)?;
158            return Ok(Some(tivk.default_address()));
159        }
160    }
161
162    Ok(None)
163}

165pub(crate) fn get_legacy_transparent_address<P: consensus::Parameters>(
166    params: &P,
167    conn: &rusqlite::Connection,
168    account_uuid: AccountUuid,
169) -> Result<Option<(TransparentAddress, NonHardenedChildIndex)>, SqliteClientError> {
170    // Get the UIVK for the account.
171    let uivk_str: Option<String> = conn
172        .query_row(
173            "SELECT uivk FROM accounts WHERE uuid = :account_uuid",
174            named_params![":account_uuid": account_uuid.0],
175            |row| row.get(0),
176        )
177        .optional()?;
178
179    if let Some(uivk_str) = uivk_str {
180        return uivk_legacy_transparent_address(params, &uivk_str);
181    }
182
183    Ok(None)
184}

186fn to_unspent_transparent_output(row: &Row) -> Result<WalletTransparentOutput, SqliteClientError> {
187    let txid: Vec<u8> = row.get("txid")?;
188    let mut txid_bytes = [0u8; 32];
189    txid_bytes.copy_from_slice(&txid);
190
191    let index: u32 = row.get("output_index")?;
192    let script_pubkey = Script(row.get("script")?);
193    let raw_value: i64 = row.get("value_zat")?;
194    let value = Zatoshis::from_nonnegative_i64(raw_value).map_err(|_| {
195        SqliteClientError::CorruptedData(format!("Invalid UTXO value: {}", raw_value))
196    })?;
197    let height: Option<u32> = row.get("received_height")?;
198
199    let outpoint = OutPoint::new(txid_bytes, index);
200    WalletTransparentOutput::from_parts(
201        outpoint,
202        TxOut {
203            value,
204            script_pubkey,
205        },
206        height.map(BlockHeight::from),
207    )
208    .ok_or_else(|| {
209        SqliteClientError::CorruptedData(
210            "Txout script_pubkey value did not correspond to a P2PKH or P2SH address".to_string(),
211        )
212    })
213}

215/// Select an output to fund a new transaction that is targeting at least `chain_tip_height + 1`.
216pub(crate) fn get_wallet_transparent_output(
217    conn: &rusqlite::Connection,
218    outpoint: &OutPoint,
219    allow_unspendable: bool,
220) -> Result<Option<WalletTransparentOutput>, SqliteClientError> {
221    let chain_tip_height = chain_tip_height(conn)?;
222
223    // This could return as unspent outputs that are actually not spendable, if they are the
224    // outputs of deshielding transactions where the spend anchors have been invalidated by a
225    // rewind or spent in a transaction that has not been observed by this wallet. There isn't a
226    // way to detect the circumstance related to anchor invalidation at present, but it should be
227    // vanishingly rare as the vast majority of rewinds are of a single block.
228    let mut stmt_select_utxo = conn.prepare_cached(
229        "SELECT t.txid, u.output_index, u.script,
230                u.value_zat, t.mined_height AS received_height
231         FROM transparent_received_outputs u
232         JOIN transactions t ON t.id_tx = u.transaction_id
233         WHERE t.txid = :txid
234         AND u.output_index = :output_index
235         -- the transaction that created the output is mined or is definitely unexpired
236         AND (
237             :allow_unspendable
238             OR (
239                 (
240                    t.mined_height IS NOT NULL -- tx is mined
241                    -- TODO: uncomment the following two lines in order to enable zero-conf spends
242                    -- OR t.expiry_height = 0 -- tx will not expire
243                    -- OR t.expiry_height >= :mempool_height -- tx has not yet expired
244                 )
245                 -- and the output is unspent
246                 AND u.id NOT IN (
247                    SELECT txo_spends.transparent_received_output_id
248                    FROM transparent_received_output_spends txo_spends
249                    JOIN transactions tx ON tx.id_tx = txo_spends.transaction_id
250                    WHERE tx.mined_height IS NOT NULL  -- the spending tx is mined
251                    OR tx.expiry_height = 0 -- the spending tx will not expire
252                    OR tx.expiry_height >= :mempool_height -- the spending tx has not yet expired
253                 )
254             )
255         )",
256    )?;
257
258    let result: Result<Option<WalletTransparentOutput>, SqliteClientError> = stmt_select_utxo
259        .query_and_then(
260            named_params![
261                ":txid": outpoint.hash(),
262                ":output_index": outpoint.n(),
263                ":mempool_height": chain_tip_height.map(|h| u32::from(h) + 1),
264                ":allow_unspendable": allow_unspendable
265            ],
266            to_unspent_transparent_output,
267        )?
268        .next()
269        .transpose();
270
271    result
272}

274/// Returns the list of spendable transparent outputs received by this wallet at `address`
275/// such that, at height `target_height`:
276/// * the transaction that produced the output had or will have at least `min_confirmations`
277///   confirmations; and
278/// * the output is unspent as of the current chain tip.
279///
280/// An output that is potentially spent by an unmined transaction in the mempool is excluded
281/// iff the spending transaction will not be expired at `target_height`.
282///
283/// This could, in very rare circumstances, return as unspent outputs that are actually not
284/// spendable, if they are the outputs of deshielding transactions where the spend anchors have
285/// been invalidated by a rewind. There isn't a way to detect this circumstance at present, but
286/// it should be vanishingly rare as the vast majority of rewinds are of a single block.
287pub(crate) fn get_spendable_transparent_outputs<P: consensus::Parameters>(
288    conn: &rusqlite::Connection,
289    params: &P,
290    address: &TransparentAddress,
291    target_height: BlockHeight,
292    min_confirmations: u32,
293) -> Result<Vec<WalletTransparentOutput>, SqliteClientError> {
294    let confirmed_height = target_height - min_confirmations;
295
296    let mut stmt_utxos = conn.prepare(
297        "SELECT t.txid, u.output_index, u.script,
298                u.value_zat, t.mined_height AS received_height
299         FROM transparent_received_outputs u
300         JOIN transactions t ON t.id_tx = u.transaction_id
301         WHERE u.address = :address
302         -- the transaction that created the output is mined or unexpired as of `confirmed_height`
303         AND (
304            t.mined_height <= :confirmed_height -- tx is mined
305            -- TODO: uncomment the following lines in order to enable zero-conf spends
306            -- OR (
307            --     :min_confirmations = 0
308            --     AND (
309            --         t.expiry_height = 0 -- tx will not expire
310            --         OR t.expiry_height >= :target_height
311            --     )
312            -- )
313         )
314         -- and the output is unspent
315         AND u.id NOT IN (
316            SELECT txo_spends.transparent_received_output_id
317            FROM transparent_received_output_spends txo_spends
318            JOIN transactions tx ON tx.id_tx = txo_spends.transaction_id
319            WHERE tx.mined_height IS NOT NULL -- the spending transaction is mined
320            OR tx.expiry_height = 0 -- the spending tx will not expire
321            OR tx.expiry_height >= :target_height -- the spending tx has not yet expired
322            -- we are intentionally conservative and exclude outputs that are potentially spent
323            -- as of the target height, even if they might actually be spendable due to expiry
324            -- of the spending transaction as of the chain tip
325         )",
326    )?;
327
328    let addr_str = address.encode(params);
329    let mut rows = stmt_utxos.query(named_params![
330        ":address": addr_str,
331        ":confirmed_height": u32::from(confirmed_height),
332        ":target_height": u32::from(target_height),
333        //":min_confirmations": min_confirmations
334    ])?;
335
336    let mut utxos = Vec::<WalletTransparentOutput>::new();
337    while let Some(row) = rows.next()? {
338        let output = to_unspent_transparent_output(row)?;
339        utxos.push(output);
340    }
341
342    Ok(utxos)
343}

345/// Returns a mapping from each transparent receiver associated with the specified account
346/// to its not-yet-shielded UTXO balance, including only the effects of transactions mined
347/// at a block height less than or equal to `summary_height`.
348///
349/// Only non-ephemeral transparent receivers with a non-zero balance at the summary height
350/// will be included.
351pub(crate) fn get_transparent_balances<P: consensus::Parameters>(
352    conn: &rusqlite::Connection,
353    params: &P,
354    account_uuid: AccountUuid,
355    summary_height: BlockHeight,
356) -> Result<HashMap<TransparentAddress, Zatoshis>, SqliteClientError> {
357    let chain_tip_height = chain_tip_height(conn)?.ok_or(SqliteClientError::ChainHeightUnknown)?;
358
359    let mut stmt_address_balances = conn.prepare(
360        "SELECT u.address, SUM(u.value_zat)
361         FROM transparent_received_outputs u
362         JOIN accounts ON accounts.id = u.account_id
363         JOIN transactions t ON t.id_tx = u.transaction_id
364         WHERE accounts.uuid = :account_uuid
365         -- the transaction that created the output is mined or is definitely unexpired
366         AND (
367            t.mined_height <= :summary_height -- tx is mined
368            OR ( -- or the caller has requested to include zero-conf funds that are not expired
369                :summary_height > :chain_tip_height
370                AND (
371                    t.expiry_height = 0 -- tx will not expire
372                    OR t.expiry_height >= :summary_height
373                )
374            )
375         )
376         -- and the output is unspent
377         AND u.id NOT IN (
378            SELECT txo_spends.transparent_received_output_id
379            FROM transparent_received_output_spends txo_spends
380            JOIN transactions tx ON tx.id_tx = txo_spends.transaction_id
381            WHERE tx.mined_height IS NOT NULL -- the spending tx is mined
382            OR tx.expiry_height = 0 -- the spending tx will not expire
383            OR tx.expiry_height >= :spend_expiry_height -- the spending tx is unexpired
384         )
385         GROUP BY u.address",
386    )?;
387
388    let mut res = HashMap::new();
389    let mut rows = stmt_address_balances.query(named_params![
390        ":account_uuid": account_uuid.0,
391        ":summary_height": u32::from(summary_height),
392        ":chain_tip_height": u32::from(chain_tip_height),
393        ":spend_expiry_height": u32::from(std::cmp::min(summary_height, chain_tip_height + 1)),
394    ])?;
395    while let Some(row) = rows.next()? {
396        let taddr_str: String = row.get(0)?;
397        let taddr = TransparentAddress::decode(params, &taddr_str)?;
398        let value = Zatoshis::from_nonnegative_i64(row.get(1)?)?;
399
400        res.insert(taddr, value);
401    }
402
403    Ok(res)
404}

406#[tracing::instrument(skip(conn, account_balances))]
407pub(crate) fn add_transparent_account_balances(
408    conn: &rusqlite::Connection,
409    mempool_height: BlockHeight,
410    min_confirmations: u32,
411    account_balances: &mut HashMap<AccountUuid, AccountBalance>,
412) -> Result<(), SqliteClientError> {
413    // TODO (#1592): Ability to distinguish between Transparent pending change and pending non-change
414    let mut stmt_account_spendable_balances = conn.prepare(
415        "SELECT a.uuid, SUM(u.value_zat)
416         FROM transparent_received_outputs u
417         JOIN accounts a ON a.id = u.account_id
418         JOIN transactions t ON t.id_tx = u.transaction_id
419         -- the transaction that created the output is mined and with enough confirmations
420         WHERE (
421            t.mined_height < :mempool_height -- tx is mined
422            AND :mempool_height - t.mined_height >= :min_confirmations -- has at least min_confirmations
423         )
424         -- and the received txo is unspent
425         AND u.id NOT IN (
426           SELECT transparent_received_output_id
427           FROM transparent_received_output_spends txo_spends
428           JOIN transactions tx
429             ON tx.id_tx = txo_spends.transaction_id
430           WHERE tx.mined_height IS NOT NULL -- the spending tx is mined
431           OR tx.expiry_height = 0 -- the spending tx will not expire
432           OR tx.expiry_height >= :mempool_height -- the spending tx is unexpired
433         )
434         GROUP BY a.uuid",
435    )?;
436    let mut rows = stmt_account_spendable_balances.query(named_params![
437        ":mempool_height": u32::from(mempool_height),
438        ":min_confirmations": min_confirmations,
439    ])?;
440
441    while let Some(row) = rows.next()? {
442        let account = AccountUuid(row.get(0)?);
443        let raw_value = row.get(1)?;
444        let value = Zatoshis::from_nonnegative_i64(raw_value).map_err(|_| {
445            SqliteClientError::CorruptedData(format!("Negative UTXO value {:?}", raw_value))
446        })?;
447
448        account_balances
449            .entry(account)
450            .or_insert(AccountBalance::ZERO)
451            .with_unshielded_balance_mut(|bal| bal.add_spendable_value(value))?;
452    }
453
454    let mut stmt_account_unconfirmed_balances = conn.prepare(
455        "SELECT a.uuid, SUM(u.value_zat)
456         FROM transparent_received_outputs u
457         JOIN accounts a ON a.id = u.account_id
458         JOIN transactions t ON t.id_tx = u.transaction_id
459         -- the transaction that created the output is mined with not enough confirmations or is definitely unexpired
460         WHERE (
461            t.mined_height < :mempool_height
462            AND :mempool_height - t.mined_height < :min_confirmations -- tx is mined but not confirmed
463            OR t.expiry_height = 0 -- tx will not expire
464            OR t.expiry_height >= :mempool_height
465         )
466         -- and the received txo is unspent
467         AND u.id NOT IN (
468           SELECT transparent_received_output_id
469           FROM transparent_received_output_spends txo_spends
470           JOIN transactions tx
471             ON tx.id_tx = txo_spends.transaction_id
472           WHERE tx.mined_height IS NOT NULL -- the spending tx is mined
473           OR tx.expiry_height = 0 -- the spending tx will not expire
474           OR tx.expiry_height >= :mempool_height -- the spending tx is unexpired
475         )
476         GROUP BY a.uuid",
477    )?;
478
479    let mut rows = stmt_account_unconfirmed_balances.query(named_params![
480        ":mempool_height": u32::from(mempool_height),
481        ":min_confirmations": min_confirmations,
482    ])?;
483
484    while let Some(row) = rows.next()? {
485        let account = AccountUuid(row.get(0)?);
486        let raw_value = row.get(1)?;
487        let value = Zatoshis::from_nonnegative_i64(raw_value).map_err(|_| {
488            SqliteClientError::CorruptedData(format!("Negative UTXO value {:?}", raw_value))
489        })?;
490
491        account_balances
492            .entry(account)
493            .or_insert(AccountBalance::ZERO)
494            .with_unshielded_balance_mut(|bal| bal.add_pending_spendable_value(value))?;
495    }
496    Ok(())
497}

499/// Marks the given UTXO as having been spent.
500///
501/// Returns `true` if the UTXO was known to the wallet.
502pub(crate) fn mark_transparent_utxo_spent(
503    conn: &rusqlite::Connection,
504    spent_in_tx: TxRef,
505    outpoint: &OutPoint,
506) -> Result<bool, SqliteClientError> {
507    let spend_params = named_params![
508        ":spent_in_tx": spent_in_tx.0,
509        ":prevout_txid": outpoint.hash(),
510        ":prevout_idx": outpoint.n(),
511    ];
512    let mut stmt_mark_transparent_utxo_spent = conn.prepare_cached(
513        "INSERT INTO transparent_received_output_spends (transparent_received_output_id, transaction_id)
514         SELECT txo.id, :spent_in_tx
515         FROM transparent_received_outputs txo
516         JOIN transactions t ON t.id_tx = txo.transaction_id
517         WHERE t.txid = :prevout_txid
518         AND txo.output_index = :prevout_idx
519         ON CONFLICT (transparent_received_output_id, transaction_id)
520         -- The following UPDATE is effectively a no-op, but we perform it anyway so that the
521         -- number of affected rows can be used to determine whether a record existed.
522         DO UPDATE SET transaction_id = :spent_in_tx",
523    )?;
524    let affected_rows = stmt_mark_transparent_utxo_spent.execute(spend_params)?;
525
526    // Since we know that the output is spent, we no longer need to search for
527    // it to find out if it has been spent.
528    let mut stmt_remove_spend_detection = conn.prepare_cached(
529        "DELETE FROM transparent_spend_search_queue
530         WHERE output_index = :prevout_idx
531         AND transaction_id IN (
532            SELECT id_tx FROM transactions WHERE txid = :prevout_txid
533         )",
534    )?;
535    stmt_remove_spend_detection.execute(named_params![
536        ":prevout_txid": outpoint.hash(),
537        ":prevout_idx": outpoint.n(),
538    ])?;
539
540    // If no rows were affected, we know that we don't actually have the output in
541    // `transparent_received_outputs` yet, so we have to record the output as spent
542    // so that when we eventually detect the output, we can create the spend record.
543    if affected_rows == 0 {
544        conn.execute(
545            "INSERT INTO transparent_spend_map (
546                spending_transaction_id,
547                prevout_txid,
548                prevout_output_index
549            )
550            VALUES (:spent_in_tx, :prevout_txid, :prevout_idx)
551            ON CONFLICT (spending_transaction_id, prevout_txid, prevout_output_index)
552            DO NOTHING",
553            spend_params,
554        )?;
555    }
556
557    Ok(affected_rows > 0)
558}

560/// Adds the given received UTXO to the datastore.
561pub(crate) fn put_received_transparent_utxo<P: consensus::Parameters>(
562    conn: &rusqlite::Connection,
563    params: &P,
564    output: &WalletTransparentOutput,
565) -> Result<UtxoId, SqliteClientError> {
566    let address = output.recipient_address();
567    if let Some(receiving_account) =
568        find_account_uuid_for_transparent_address(conn, params, address)?
569    {
570        put_transparent_output(
571            conn,
572            params,
573            output.outpoint(),
574            output.txout(),
575            output.mined_height(),
576            address,
577            receiving_account,
578            true,
579        )
580    } else {
581        // The UTXO was not for any of our transparent addresses.
582        Err(SqliteClientError::AddressNotRecognized(*address))
583    }

586/// Returns the vector of [`TransactionDataRequest`]s that represents the information needed by the
587/// wallet backend in order to be able to present a complete view of wallet history and memo data.
588pub(crate) fn transaction_data_requests<P: consensus::Parameters>(
589    conn: &rusqlite::Connection,
590    params: &P,
591) -> Result<Vec<TransactionDataRequest>, SqliteClientError> {
592    // `lightwalletd` will return an error for `GetTaddressTxids` requests having an end height
593    // greater than the current chain tip height, so we take the chain tip height into account
594    // here in order to make this pothole easier for clients of the API to avoid.
595    let chain_tip_height = super::chain_tip_height(conn)?;
596
597    // We cannot construct address-based transaction data requests for the case where we cannot
598    // determine the height at which to begin, so we require that either the target height or mined
599    // height be set.
600    let mut address_request_stmt = conn.prepare_cached(
601        "SELECT ssq.address, IFNULL(t.target_height, t.mined_height)
602         FROM transparent_spend_search_queue ssq
603         JOIN transactions t ON t.id_tx = ssq.transaction_id
604         WHERE t.target_height IS NOT NULL
605         OR t.mined_height IS NOT NULL",
606    )?;
607
608    let result = address_request_stmt
609        .query_and_then([], |row| {
610            let address = TransparentAddress::decode(params, &row.get::<_, String>(0)?)?;
611            let block_range_start = BlockHeight::from(row.get::<_, u32>(1)?);
612            let max_end_height = block_range_start + DEFAULT_TX_EXPIRY_DELTA + 1;
613
614            Ok::<TransactionDataRequest, SqliteClientError>(
615                TransactionDataRequest::SpendsFromAddress {
616                    address,
617                    block_range_start,
618                    block_range_end: Some(
619                        chain_tip_height
620                            .map_or(max_end_height, |h| std::cmp::min(h + 1, max_end_height)),
621                    ),
622                },
623            )
624        })?
625        .collect::<Result<Vec<_>, _>>()?;
626
627    Ok(result)
628}

630pub(crate) fn get_transparent_address_metadata<P: consensus::Parameters>(
631    conn: &rusqlite::Connection,
632    params: &P,
633    account_uuid: AccountUuid,
634    address: &TransparentAddress,
635) -> Result<Option<TransparentAddressMetadata>, SqliteClientError> {
636    let address_str = address.encode(params);
637
638    if let Some(di_vec) = conn
639        .query_row(
640            "SELECT diversifier_index_be FROM addresses
641             JOIN accounts ON addresses.account_id = accounts.id
642             WHERE accounts.uuid = :account_uuid 
643             AND cached_transparent_receiver_address = :address",
644            named_params![":account_uuid": account_uuid.0, ":address": &address_str],
645            |row| row.get::<_, Vec<u8>>(0),
646        )
647        .optional()?
648    {
649        let address_index = address_index_from_diversifier_index_be(&di_vec)?;
650        let metadata = TransparentAddressMetadata::new(Scope::External.into(), address_index);
651        return Ok(Some(metadata));
652    }
653
654    if let Some((legacy_taddr, address_index)) =
655        get_legacy_transparent_address(params, conn, account_uuid)?
656    {
657        if &legacy_taddr == address {
658            let metadata = TransparentAddressMetadata::new(Scope::External.into(), address_index);
659            return Ok(Some(metadata));
660        }
661    }
662
663    // Search known ephemeral addresses.
664    if let Some(address_index) =
665        ephemeral::find_index_for_ephemeral_address_str(conn, account_uuid, &address_str)?
666    {
667        return Ok(Some(ephemeral::metadata(address_index)));
668    }
669
670    Ok(None)
671}

673/// Attempts to determine the account that received the given transparent output.
674///
675/// The following three locations in the wallet's key tree are searched:
676/// - Transparent receivers that have been generated as part of a Unified Address.
677/// - Transparent ephemeral addresses that have been reserved or are within
678///   the gap limit from the last reserved address.
679/// - "Legacy transparent addresses" (at BIP 44 address index 0 within an account).
680///
681/// Returns `Ok(None)` if the transparent output's recipient address is not in any of the
682/// above locations. This means the wallet considers the output "not interesting".
683pub(crate) fn find_account_uuid_for_transparent_address<P: consensus::Parameters>(
684    conn: &rusqlite::Connection,
685    params: &P,
686    address: &TransparentAddress,
687) -> Result<Option<AccountUuid>, SqliteClientError> {
688    let address_str = address.encode(params);
689
690    if let Some(account_id) = conn
691        .query_row(
692            "SELECT accounts.uuid 
693             FROM addresses 
694             JOIN accounts ON accounts.id = addresses.account_id
695             WHERE cached_transparent_receiver_address = :address",
696            named_params![":address": &address_str],
697            |row| Ok(AccountUuid(row.get(0)?)),
698        )
699        .optional()?
700    {
701        return Ok(Some(account_id));
702    }
703
704    // Search known ephemeral addresses.
705    if let Some(account_id) = ephemeral::find_account_for_ephemeral_address_str(conn, &address_str)?
706    {
707        return Ok(Some(account_id));
708    }
709
710    let account_ids = get_account_ids(conn)?;
711
712    // If the UTXO is received at the legacy transparent address (at BIP 44 address
713    // index 0 within its particular account, which we specifically ensure is returned
714    // from `get_transparent_receivers`), there may be no entry in the addresses table
715    // that can be used to tie the address to a particular account. In this case, we
716    // look up the legacy address for each account in the wallet, and check whether it
717    // matches the address for the received UTXO.
718    for &account_id in account_ids.iter() {
719        if let Some((legacy_taddr, _)) = get_legacy_transparent_address(params, conn, account_id)? {
720            if &legacy_taddr == address {
721                return Ok(Some(account_id));
722            }
723        }
724    }
725
726    Ok(None)
727}

729/// Add a transparent output relevant to this wallet to the database.
730///
731/// `output_height` may be None if this is an ephemeral output from a
732/// transaction we created, that we do not yet know to have been mined.
733#[allow(clippy::too_many_arguments)]
734pub(crate) fn put_transparent_output<P: consensus::Parameters>(
735    conn: &rusqlite::Connection,
736    params: &P,
737    outpoint: &OutPoint,
738    txout: &TxOut,
739    output_height: Option<BlockHeight>,
740    address: &TransparentAddress,
741    receiving_account_uuid: AccountUuid,
742    known_unspent: bool,
743) -> Result<UtxoId, SqliteClientError> {
744    let output_height = output_height.map(u32::from);
745    let receiving_account_id = super::get_account_ref(conn, receiving_account_uuid)?;
746
747    // Check whether we have an entry in the blocks table for the output height;
748    // if not, the transaction will be updated with its mined height when the
749    // associated block is scanned.
750    let block = match output_height {
751        Some(height) => conn
752            .query_row(
753                "SELECT height FROM blocks WHERE height = :height",
754                named_params![":height": height],
755                |row| row.get::<_, u32>(0),
756            )
757            .optional()?,
758        None => None,
759    };
760
761    let id_tx = conn.query_row(
762        "INSERT INTO transactions (txid, block, mined_height)
763         VALUES (:txid, :block, :mined_height)
764         ON CONFLICT (txid) DO UPDATE
765         SET block = IFNULL(block, :block),
766             mined_height = :mined_height
767         RETURNING id_tx",
768        named_params![
769           ":txid": &outpoint.hash().to_vec(),
770           ":block": block,
771           ":mined_height": output_height
772        ],
773        |row| row.get::<_, i64>(0),
774    )?;
775
776    let spent_height = conn
777        .query_row(
778            "SELECT t.mined_height
779             FROM transactions t
780             JOIN transparent_received_output_spends ts ON ts.transaction_id = t.id_tx
781             JOIN transparent_received_outputs tro ON tro.id = ts.transparent_received_output_id
782             WHERE tro.transaction_id = :transaction_id
783             AND tro.output_index = :output_index",
784            named_params![
785                ":transaction_id": id_tx,
786                ":output_index": &outpoint.n(),
787            ],
788            |row| {
789                row.get::<_, Option<u32>>(0)
790                    .map(|o| o.map(BlockHeight::from))
791            },
792        )
793        .optional()?
794        .flatten();
795
796    // The max observed unspent height is either the spending transaction's mined height - 1, or
797    // the current chain tip height if the UTXO was received via a path that confirmed that it was
798    // unspent, such as by querying the UTXO set of the network.
799    let max_observed_unspent = match spent_height {
800        Some(h) => Some(h - 1),
801        None => {
802            if known_unspent {
803                chain_tip_height(conn)?
804            } else {
805                None
806            }
807        }
808    };
809
810    let mut stmt_upsert_transparent_output = conn.prepare_cached(
811        "INSERT INTO transparent_received_outputs (
812            transaction_id, output_index,
813            account_id, address, script,
814            value_zat, max_observed_unspent_height
815        )
816        VALUES (
817            :transaction_id, :output_index,
818            :account_id, :address, :script,
819            :value_zat, :max_observed_unspent_height
820        )
821        ON CONFLICT (transaction_id, output_index) DO UPDATE
822        SET account_id = :account_id,
823            address = :address,
824            script = :script,
825            value_zat = :value_zat,
826            max_observed_unspent_height = IFNULL(:max_observed_unspent_height, max_observed_unspent_height)
827        RETURNING id",
828    )?;
829
830    let sql_args = named_params![
831        ":transaction_id": id_tx,
832        ":output_index": &outpoint.n(),
833        ":account_id": receiving_account_id.0,
834        ":address": &address.encode(params),
835        ":script": &txout.script_pubkey.0,
836        ":value_zat": &i64::from(ZatBalance::from(txout.value)),
837        ":max_observed_unspent_height": max_observed_unspent.map(u32::from),
838    ];
839
840    let utxo_id = stmt_upsert_transparent_output
841        .query_row(sql_args, |row| row.get::<_, i64>(0).map(UtxoId))?;
842
843    // If we have a record of the output already having been spent, then mark it as spent using the
844    // stored reference to the spending transaction.
845    let spending_tx_ref = conn
846        .query_row(
847            "SELECT ts.spending_transaction_id
848             FROM transparent_spend_map ts
849             JOIN transactions t ON t.id_tx = ts.spending_transaction_id
850             WHERE ts.prevout_txid = :prevout_txid
851             AND ts.prevout_output_index = :prevout_idx
852             ORDER BY t.block NULLS LAST LIMIT 1",
853            named_params![
854                ":prevout_txid": outpoint.txid().as_ref(),
855                ":prevout_idx": outpoint.n()
856            ],
857            |row| row.get::<_, i64>(0).map(TxRef),
858        )
859        .optional()?;
860
861    if let Some(spending_transaction_id) = spending_tx_ref {
862        mark_transparent_utxo_spent(conn, spending_transaction_id, outpoint)?;
863    }
864
865    Ok(utxo_id)
866}

868/// Adds a request to retrieve transactions involving the specified address to the transparent
869/// spend search queue. Note that such requests are _not_ for data related to `tx_ref`, but instead
870/// a request to find where the UTXO with the outpoint `(tx_ref, output_index)` is spent.
871///
872/// ### Parameters
873/// - `receiving_address`: The address that received the UTXO.
874/// - `tx_ref`: The transaction in which the UTXO was received.
875/// - `output_index`: The index of the output within `vout` of the specified transaction.
876pub(crate) fn queue_transparent_spend_detection<P: consensus::Parameters>(
877    conn: &rusqlite::Transaction<'_>,
878    params: &P,
879    receiving_address: TransparentAddress,
880    tx_ref: TxRef,
881    output_index: u32,
882) -> Result<(), SqliteClientError> {
883    let mut stmt = conn.prepare_cached(
884        "INSERT INTO transparent_spend_search_queue
885         (address, transaction_id, output_index)
886         VALUES
887         (:address, :transaction_id, :output_index)
888         ON CONFLICT (transaction_id, output_index) DO NOTHING",
889    )?;
890
891    let addr_str = receiving_address.encode(params);
892    stmt.execute(named_params! {
893        ":address": addr_str,
894        ":transaction_id": tx_ref.0,
895        ":output_index": output_index
896    })?;
897
898    Ok(())
899}
