55pub(crate) fn detect_spending_accounts<'a>(
56    conn: &Connection,
57    spent: impl Iterator<Item = &'a OutPoint>,
58) -> Result<HashSet<AccountUuid>, rusqlite::Error> {
59    let mut account_q = conn.prepare_cached(
60        "SELECT accounts.uuid
61        FROM transparent_received_outputs o
62        JOIN accounts ON accounts.id = o.account_id
63        JOIN transactions t ON t.id_tx = o.transaction_id
64        WHERE t.txid = :prevout_txid
65        AND o.output_index = :prevout_idx",
66    )?;
67
68    let mut acc = HashSet::new();
69    for prevout in spent {
70        for account in account_q.query_and_then(
71            named_params![
72                ":prevout_txid": prevout.hash(),
73                ":prevout_idx": prevout.n()
74            ],
75            |row| row.get(0).map(AccountUuid),
76        )? {
77            acc.insert(account?);
78        }
79    }
80
81    Ok(acc)
82}

98pub(crate) fn get_transparent_receivers<P: consensus::Parameters>(
99    conn: &rusqlite::Connection,
100    params: &P,
101    account_uuid: AccountUuid,
102    scopes: &[KeyScope],
103) -> Result<HashMap<TransparentAddress, Option<TransparentAddressMetadata>>, SqliteClientError> {
104    let mut ret: HashMap<TransparentAddress, Option<TransparentAddressMetadata>> = HashMap::new();
105
106    // Get all addresses with the provided scopes.
107    let mut addr_query = conn.prepare(
108        "SELECT cached_transparent_receiver_address, transparent_child_index, key_scope
109         FROM addresses
110         JOIN accounts ON accounts.id = addresses.account_id
111         WHERE accounts.uuid = :account_uuid
112         AND cached_transparent_receiver_address IS NOT NULL
113         AND key_scope IN rarray(:scopes_ptr)",
114    )?;
115
116    let scope_values: Vec<Value> = scopes.iter().map(|s| Value::Integer(s.encode())).collect();
117    let scopes_ptr = Rc::new(scope_values);
118    let mut rows = addr_query.query(named_params![
119        ":account_uuid": account_uuid.0,
120        ":scopes_ptr": &scopes_ptr
121    ])?;
122
123    while let Some(row) = rows.next()? {
124        let addr_str: String = row.get(0)?;
125        let address_index: u32 = row.get(1)?;
126        let address_index = NonHardenedChildIndex::from_index(address_index).ok_or(
127            SqliteClientError::CorruptedData(format!(
128                "{} is not a valid transparent child index",
129                address_index
130            )),
131        )?;
132        let scope = KeyScope::decode(row.get(2)?)?;
133
134        let taddr = Address::decode(params, &addr_str)
135            .ok_or_else(|| {
136                SqliteClientError::CorruptedData("Not a valid Zcash recipient address".to_owned())
137            })?
138            .to_transparent_address();
139
140        if let Some(taddr) = taddr {
141            let metadata = TransparentAddressMetadata::new(scope.into(), address_index);
142            ret.insert(taddr, Some(metadata));
143        }
144    }
145
146    Ok(ret)
147}

149pub(crate) fn uivk_legacy_transparent_address<P: consensus::Parameters>(
150    params: &P,
151    uivk_str: &str,
152) -> Result<Option<(TransparentAddress, NonHardenedChildIndex)>, SqliteClientError> {
153    use transparent::keys::ExternalIvk;
154    use zcash_address::unified::{Container as _, Encoding as _};
155
156    let (network, uivk) = Uivk::decode(uivk_str)
157        .map_err(|e| SqliteClientError::CorruptedData(format!("Unable to parse UIVK: {e}")))?;
158
159    if params.network_type() != network {
160        let network_name = |n| match n {
161            consensus::NetworkType::Main => "mainnet",
162            consensus::NetworkType::Test => "testnet",
163            consensus::NetworkType::Regtest => "regtest",
164        };
165        return Err(SqliteClientError::CorruptedData(format!(
166            "Network type mismatch: account UIVK is for {} but a {} address was requested.",
167            network_name(network),
168            network_name(params.network_type())
169        )));
170    }
171
172    // Derive the default transparent address (if it wasn't already part of a derived UA).
173    for item in uivk.items() {
174        if let Ivk::P2pkh(tivk_bytes) = item {
175            let tivk = ExternalIvk::deserialize(&tivk_bytes)?;
176            return Ok(Some(tivk.default_address()));
177        }
178    }
179
180    Ok(None)
181}

204/// Returns the transparent address index at the start of the first gap of at least `gap_limit`
205/// indices in the given account, considering only addresses derived for the specified key scope.
206///
207/// Returns `Ok(None)` if the gap would start at an index greater than the maximum valid
208/// non-hardened transparent child index.
209pub(crate) fn find_gap_start(
210    conn: &rusqlite::Connection,
211    account_id: AccountRef,
212    key_scope: KeyScope,
213    gap_limit: u32,
214) -> Result<Option<NonHardenedChildIndex>, SqliteClientError> {
215    match conn
216        .query_row(
217            r#"
218            WITH offsets AS (
219                SELECT
220                    a.transparent_child_index,
221                    LEAD(a.transparent_child_index)
222                        OVER (ORDER BY a.transparent_child_index)
223                        AS next_child_index
224                FROM v_address_first_use a
225                WHERE a.account_id = :account_id
226                AND a.key_scope = :key_scope
227                AND a.transparent_child_index IS NOT NULL
228                AND a.first_use_height IS NOT NULL
229            )
230            SELECT
231                transparent_child_index + 1,
232                -- both next_child_index and transparent_child_index are used indices,
233                -- so the gap between them is one less than their difference
234                next_child_index - transparent_child_index - 1 AS gap_len
235            FROM offsets
236            -- if gap_len is at least the gap limit, then we have found a gap.
237            -- if next_child_index is NULL, then we have reached the end of
238            -- the allocated indices (the remainder of the index space is a gap).
239            WHERE gap_len >= :gap_limit OR next_child_index IS NULL
240            ORDER BY transparent_child_index
241            LIMIT 1
242            "#,
243            named_params![
244                ":account_id": account_id.0,
245                ":key_scope": key_scope.encode(),
246                ":gap_limit": gap_limit
247            ],
248            |row| row.get::<_, u32>(0),
249        )
250        .optional()?
251    {
252        Some(i) => Ok(NonHardenedChildIndex::from_index(i)),
253        None => Ok(Some(NonHardenedChildIndex::ZERO)),
254    }
255}

268/// Returns the current gap start, along with a vector with at most the next `n` previously
269/// unreserved transparent addresses for the given account. These addresses must have been
270/// previously generated using `generate_gap_addresses`.
271///
272/// WARNING: the addresses returned by this method have not been marked as exposed; it is the
273/// responsibility of the caller to correctly update the `exposed_at_height` value for each
274/// returned address before such an address is exposed to a user.
275///
276/// # Errors
277///
278/// * `SqliteClientError::AccountUnknown`, if there is no account with the given id.
279/// * `SqliteClientError::AddressGeneration(AddressGenerationError::DiversifierSpaceExhausted)`,
280///   if the limit on transparent address indices has been reached.
281#[allow(clippy::type_complexity)]
282pub(crate) fn select_addrs_to_reserve<P: consensus::Parameters>(
283    conn: &rusqlite::Transaction,
284    params: &P,
285    account_id: AccountRef,
286    key_scope: KeyScope,
287    gap_limit: u32,
288    n: usize,
289) -> Result<
290    (
291        NonHardenedChildIndex,
292        Vec<(AddressRef, TransparentAddress, TransparentAddressMetadata)>,
293    ),
294    SqliteClientError,
295> {
296    let gap_start = find_gap_start(conn, account_id, key_scope, gap_limit)?.ok_or(
297        SqliteClientError::AddressGeneration(AddressGenerationError::DiversifierSpaceExhausted),
298    )?;
299
300    let mut stmt_addrs_to_reserve = conn.prepare(
301        "SELECT id, transparent_child_index, cached_transparent_receiver_address
302         FROM addresses
303         WHERE account_id = :account_id
304         AND key_scope = :key_scope
305         AND transparent_child_index >= :gap_start
306         AND transparent_child_index < :gap_end
307         AND exposed_at_height IS NULL
308         ORDER BY transparent_child_index
309         LIMIT :n",
310    )?;
311
312    let addresses_to_reserve = stmt_addrs_to_reserve
313        .query_and_then(
314            named_params! {
315                ":account_id": account_id.0,
316                ":key_scope": key_scope.encode(),
317                ":gap_start": gap_start.index(),
318                // NOTE: this approach means that the address at index 2^31 - 1 will never be
319                // allocated. I think that's fine.
320                ":gap_end": gap_start.saturating_add(gap_limit).index(),
321                ":n": n
322            },
323            |row| {
324                let address_id = row.get("id").map(AddressRef)?;
325                let transparent_child_index = row
326                    .get::<_, Option<i64>>("transparent_child_index")?
327                    .map(decode_transparent_child_index)
328                    .transpose()?;
329                let address = row
330                    .get::<_, Option<String>>("cached_transparent_receiver_address")?
331                    .map(|addr_str| TransparentAddress::decode(params, &addr_str))
332                    .transpose()?;
333
334                Ok::<_, SqliteClientError>(transparent_child_index.zip(address).map(|(i, a)| {
335                    (
336                        address_id,
337                        a,
338                        TransparentAddressMetadata::new(key_scope.into(), i),
339                    )
340                }))
341            },
342        )?
343        .filter_map(|r| r.transpose())
344        .collect::<Result<Vec<_>, _>>()?;
345
346    Ok((gap_start, addresses_to_reserve))
347}

349/// Returns a vector with the next `n` previously unreserved transparent addresses for the given
350/// account, having marked each address as having been exposed at the current chain-tip height.
351/// These addresses must have been previously generated using `generate_gap_addresses`.
352///
353/// # Errors
354///
355/// * [`SqliteClientError::AccountUnknown`], if there is no account with the given id.
356/// * [`SqliteClientError::ReachedGapLimit`], if it is not possible to reserve `n` addresses
357///   within the gap limit after the last address in this account that is known to have an
358///   output in a mined transaction.
359/// * [`SqliteClientError::AddressGeneration(AddressGenerationError::DiversifierSpaceExhausted)`]
360///   if the limit on transparent address indices has been reached.
361///
362/// [`SqliteClientError::AddressGeneration(AddressGenerationError::DiversifierSpaceExhausted)`]:
363/// SqliteClientError::AddressGeneration
364pub(crate) fn reserve_next_n_addresses<P: consensus::Parameters>(
365    conn: &rusqlite::Transaction,
366    params: &P,
367    account_id: AccountRef,
368    key_scope: KeyScope,
369    gap_limit: u32,
370    n: usize,
371) -> Result<Vec<(AddressRef, TransparentAddress, TransparentAddressMetadata)>, SqliteClientError> {
372    if n == 0 {
373        return Ok(vec![]);
374    }
375
376    let (gap_start, addresses_to_reserve) =
377        select_addrs_to_reserve(conn, params, account_id, key_scope, gap_limit, n)?;
378
379    if addresses_to_reserve.len() < n {
380        return Err(SqliteClientError::ReachedGapLimit(
381            key_scope.into(),
382            gap_start.index() + gap_limit,
383        ));
384    }
385
386    let current_chain_tip = chain_tip_height(conn)?.ok_or(SqliteClientError::ChainHeightUnknown)?;
387
388    let reserve_id_values: Vec<Value> = addresses_to_reserve
389        .iter()
390        .map(|(id, _, _)| Value::Integer(id.0))
391        .collect();
392    let reserved_ptr = Rc::new(reserve_id_values);
393    conn.execute(
394        "UPDATE addresses
395         SET exposed_at_height = :chain_tip_height
396         WHERE id IN rarray(:reserved_ptr)",
397        named_params! {
398            ":chain_tip_height": u32::from(current_chain_tip),
399            ":reserved_ptr": &reserved_ptr
400        },
401    )?;
402
403    Ok(addresses_to_reserve)
404}

406pub(crate) fn generate_external_address(
407    uivk: &UnifiedIncomingViewingKey,
408    ua_request: UnifiedAddressRequest,
409    index: NonHardenedChildIndex,
410) -> Result<(Address, TransparentAddress), AddressGenerationError> {
411    let ua = uivk.address(index.into(), ua_request);
412    let transparent_address = uivk
413        .transparent()
414        .as_ref()
415        .ok_or(AddressGenerationError::KeyNotAvailable(Typecode::P2pkh))?
416        .derive_address(index)
417        .map_err(|_| {
418            AddressGenerationError::InvalidTransparentChildIndex(DiversifierIndex::from(index))
419        })?;
420    Ok((
421        ua.map_or_else(
422            |e| {
423                if matches!(e, AddressGenerationError::ShieldedReceiverRequired) {
424                    // fall back to the transparent-only address
425                    Ok(Address::from(transparent_address))
426                } else {
427                    // other address generation errors are allowed to propagate
428                    Err(e)
429                }
430            },
431            |addr| Ok(Address::from(addr)),
432        )?,
433        transparent_address,
434    ))
435}

437/// Generates addresses to fill the specified non-hardened child index range.
438///
439/// The provided [`UnifiedAddressRequest`] is used to pre-generate unified addresses that correspond
440/// to each transparent address index in question; such unified addresses need not internally
441/// contain a transparent receiver, and may be overwritten when these addresses are exposed via the
442/// [`WalletWrite::get_next_available_address`] or [`WalletWrite::get_address_for_index`] methods.
443/// If no request is provided, each address so generated will contain a receiver for each possible
444/// pool: i.e., a recevier for each data item in the account's UFVK or UIVK where the transparent
445/// child index is valid.
446///
447/// [`WalletWrite::get_next_available_address`]: zcash_client_backend::data_api::WalletWrite::get_next_available_address
448/// [`WalletWrite::get_address_for_index`]: zcash_client_backend::data_api::WalletWrite::get_address_for_index
449pub(crate) fn generate_address_range<P: consensus::Parameters>(
450    conn: &rusqlite::Transaction,
451    params: &P,
452    account_id: AccountRef,
453    key_scope: KeyScope,
454    request: UnifiedAddressRequest,
455    range_to_store: Range<NonHardenedChildIndex>,
456    require_key: bool,
457) -> Result<(), SqliteClientError> {
458    let account = get_account_internal(conn, params, account_id)?
459        .ok_or_else(|| SqliteClientError::AccountUnknown)?;
460
461    if !account.uivk().has_transparent() {
462        if require_key {
463            return Err(SqliteClientError::AddressGeneration(
464                AddressGenerationError::KeyNotAvailable(Typecode::P2pkh),
465            ));
466        } else {
467            return Ok(());
468        }
469    }
470
471    let gen_addrs = |key_scope: KeyScope, index: NonHardenedChildIndex| {
472        Ok::<_, SqliteClientError>(match key_scope {
473            KeyScope::Zip32(zip32::Scope::External) => {
474                generate_external_address(&account.uivk(), request, index)?
475            }
476            KeyScope::Zip32(zip32::Scope::Internal) => {
477                let internal_address = account
478                    .ufvk()
479                    .and_then(|k| k.transparent())
480                    .expect("presence of transparent key was checked above.")
481                    .derive_internal_ivk()?
482                    .derive_address(index)?;
483                (Address::from(internal_address), internal_address)
484            }
485            KeyScope::Ephemeral => {
486                let ephemeral_address = account
487                    .ufvk()
488                    .and_then(|k| k.transparent())
489                    .expect("presence of transparent key was checked above.")
490                    .derive_ephemeral_ivk()?
491                    .derive_ephemeral_address(index)?;
492                (Address::from(ephemeral_address), ephemeral_address)
493            }
494        })
495    };
496
497    // exposed_at_height is initially NULL
498    let mut stmt_insert_address = conn.prepare_cached(
499        "INSERT INTO addresses (
500            account_id, diversifier_index_be, key_scope, address,
501            transparent_child_index, cached_transparent_receiver_address,
502            receiver_flags
503         )
504         VALUES (
505            :account_id, :diversifier_index_be, :key_scope, :address,
506            :transparent_child_index, :transparent_address,
507            :receiver_flags
508         )
509         ON CONFLICT (account_id, diversifier_index_be, key_scope) DO NOTHING",
510    )?;
511
512    for transparent_child_index in NonHardenedChildRange::from(range_to_store) {
513        let (address, transparent_address) = gen_addrs(key_scope, transparent_child_index)?;
514        let zcash_address = address.to_zcash_address(params);
515        let receiver_flags: ReceiverFlags = zcash_address
516            .clone()
517            .convert::<ReceiverFlags>()
518            .expect("address is valid");
519
520        stmt_insert_address.execute(named_params![
521            ":account_id": account_id.0,
522            ":diversifier_index_be": encode_diversifier_index_be(transparent_child_index.into()),
523            ":key_scope": key_scope.encode(),
524            ":address": zcash_address.encode(),
525            ":transparent_child_index": transparent_child_index.index(),
526            ":transparent_address": transparent_address.encode(params),
527            ":receiver_flags": receiver_flags.bits()
528        ])?;
529    }
530    Ok(())
531}

533/// Extend the range of preallocated addresses in an account to ensure that a full `gap_limit` of
534/// transparent addresses is available from the first gap in existing indices of addresses at which
535/// a received transaction has been observed on the chain, for each key scope.
536///
537/// The provided [`UnifiedAddressRequest`] is used to pre-generate unified addresses that correspond
538/// to the transparent address index in question; such unified addresses need not internally
539/// contain a transparent receiver, and may be overwritten when these addresses are exposed via the
540/// [`WalletWrite::get_next_available_address`] or [`WalletWrite::get_address_for_index`] methods.
541/// If no request is provided, each address so generated will contain a receiver for each possible
542/// pool: i.e., a recevier for each data item in the account's UFVK or UIVK where the transparent
543/// child index is valid.
544///
545/// [`WalletWrite::get_next_available_address`]: zcash_client_backend::data_api::WalletWrite::get_next_available_address
546/// [`WalletWrite::get_address_for_index`]: zcash_client_backend::data_api::WalletWrite::get_address_for_index
547pub(crate) fn generate_gap_addresses<P: consensus::Parameters>(
548    conn: &rusqlite::Transaction,
549    params: &P,
550    account_id: AccountRef,
551    key_scope: KeyScope,
552    gap_limits: &GapLimits,
553    request: UnifiedAddressRequest,
554    require_key: bool,
555) -> Result<(), SqliteClientError> {
556    let gap_limit = match key_scope {
557        KeyScope::Zip32(zip32::Scope::External) => gap_limits.external(),
558        KeyScope::Zip32(zip32::Scope::Internal) => gap_limits.internal(),
559        KeyScope::Ephemeral => gap_limits.ephemeral(),
560    };
561
562    if let Some(gap_start) = find_gap_start(conn, account_id, key_scope, gap_limit)? {
563        generate_address_range(
564            conn,
565            params,
566            account_id,
567            key_scope,
568            request,
569            gap_start..gap_start.saturating_add(gap_limit),
570            require_key,
571        )?;
572    }
573
574    Ok(())
575}

577/// Check whether `address` has previously been used as the recipient address for any previously
578/// received output. This is intended primarily for use in ensuring that the wallet does not create
579/// ZIP 320 transactions that reuse the same ephemeral address, although it is written in such a
580/// way that it may be used for detection of transparent address reuse more generally.
581///
582/// If the address was already used in an output we received, this method will return
583/// [`SqliteClientError::AddressReuse`].
584pub(crate) fn check_ephemeral_address_reuse<P: consensus::Parameters>(
585    conn: &rusqlite::Transaction,
586    params: &P,
587    address: &TransparentAddress,
588) -> Result<(), SqliteClientError> {
589    let taddr_str = address.encode(params);
590    let mut stmt = conn.prepare_cached(
591        "SELECT t.txid
592         FROM transactions t
593         JOIN v_received_outputs vro ON vro.transaction_id = t.id_tx
594         JOIN addresses a ON a.id = vro.address_id
595         WHERE a.cached_transparent_receiver_address = :transparent_address",
596    )?;
597
598    let txids = stmt
599        .query_and_then(
600            named_params![
601                ":transparent_address": taddr_str,
602            ],
603            |row| Ok(TxId::from_bytes(row.get::<_, [u8; 32]>(0)?)),
604        )?
605        .collect::<Result<Vec<_>, SqliteClientError>>()?;
606
607    if let Some(txids) = NonEmpty::from_vec(txids) {
608        return Err(SqliteClientError::AddressReuse(taddr_str, txids));
609    }
610
611    Ok(())
612}

615///
616/// We must start looking for UTXOs for addresses within the current gap limit as of the block
617/// height at which they might have first been revealed. This would have occurred when the gap
618/// advanced as a consequence of a transaction being mined. The address at the start of the current
619/// gap was potentially first revealed after the address at index `gap_start - (gap_limit + 1)`
620/// received an output in a mined transaction; therefore, we take that height to be where we should
621/// start searching for UTXOs.
622pub(crate) fn utxo_query_height(
623    conn: &rusqlite::Connection,
624    account_ref: AccountRef,
625    gap_limits: &GapLimits,
626) -> Result<BlockHeight, SqliteClientError> {
627    let mut stmt = conn.prepare_cached(
628        "SELECT MIN(au.mined_height)
629         FROM v_address_uses au
630         JOIN addresses a ON a.id = au.address_id
631         WHERE a.account_id = :account_id
632         AND au.key_scope = :key_scope
633         AND au.transparent_child_index >= :transparent_child_index",
634    )?;
635
636    let mut get_height = |key_scope: KeyScope, gap_limit: u32| {
637        if let Some(gap_start) = find_gap_start(conn, account_ref, key_scope, gap_limit)? {
638            stmt.query_row(
639                named_params! {
640                    ":account_id": account_ref.0,
641                    ":key_scope": key_scope.encode(),
642                    ":transparent_child_index": gap_start.index().saturating_sub(gap_limit + 1)
643                },
644                |row| {
645                    row.get::<_, Option<u32>>(0)
646                        .map(|opt| opt.map(BlockHeight::from))
647                },
648            )
649            .optional()
650            .map(|opt| opt.flatten())
651            .map_err(SqliteClientError::from)
652        } else {
653            Ok(None)
654        }
655    };
656
657    let h_external = get_height(KeyScope::EXTERNAL, gap_limits.external())?;
658    let h_internal = get_height(KeyScope::INTERNAL, gap_limits.internal())?;
659
660    match (h_external, h_internal) {
661        (Some(ext), Some(int)) => Ok(std::cmp::min(ext, int)),
662        (Some(h), None) | (None, Some(h)) => Ok(h),
663        (None, None) => account_birthday_internal(conn, account_ref),
664    }
665}

667fn to_unspent_transparent_output(row: &Row) -> Result<WalletTransparentOutput, SqliteClientError> {
668    let txid: Vec<u8> = row.get("txid")?;
669    let mut txid_bytes = [0u8; 32];
670    txid_bytes.copy_from_slice(&txid);
671
672    let index: u32 = row.get("output_index")?;
673    let script_pubkey = Script(row.get("script")?);
674    let raw_value: i64 = row.get("value_zat")?;
675    let value = Zatoshis::from_nonnegative_i64(raw_value).map_err(|_| {
676        SqliteClientError::CorruptedData(format!("Invalid UTXO value: {}", raw_value))
677    })?;
678    let height: Option<u32> = row.get("received_height")?;
679
680    let outpoint = OutPoint::new(txid_bytes, index);
681    WalletTransparentOutput::from_parts(
682        outpoint,
683        TxOut {
684            value,
685            script_pubkey,
686        },
687        height.map(BlockHeight::from),
688    )
689    .ok_or_else(|| {
690        SqliteClientError::CorruptedData(
691            "Txout script_pubkey value did not correspond to a P2PKH or P2SH address".to_string(),
692        )
693    })

696/// Select an output to fund a new transaction that is targeting at least `chain_tip_height + 1`.
697pub(crate) fn get_wallet_transparent_output(
698    conn: &rusqlite::Connection,
699    outpoint: &OutPoint,
700    allow_unspendable: bool,
701) -> Result<Option<WalletTransparentOutput>, SqliteClientError> {
702    let chain_tip_height = chain_tip_height(conn)?;
703
704    // This could return as unspent outputs that are actually not spendable, if they are the
705    // outputs of deshielding transactions where the spend anchors have been invalidated by a
706    // rewind or spent in a transaction that has not been observed by this wallet. There isn't a
707    // way to detect the circumstance related to anchor invalidation at present, but it should be
708    // vanishingly rare as the vast majority of rewinds are of a single block.
709    let mut stmt_select_utxo = conn.prepare_cached(
710        "SELECT t.txid, u.output_index, u.script,
711                u.value_zat, t.mined_height AS received_height
712         FROM transparent_received_outputs u
713         JOIN transactions t ON t.id_tx = u.transaction_id
714         WHERE t.txid = :txid
715         AND u.output_index = :output_index
716         -- the transaction that created the output is mined or is definitely unexpired
717         AND (
718             :allow_unspendable
719             OR (
720                 (
721                    t.mined_height IS NOT NULL -- tx is mined
722                    -- TODO: uncomment the following two lines in order to enable zero-conf spends
723                    -- OR t.expiry_height = 0 -- tx will not expire
724                    -- OR t.expiry_height >= :mempool_height -- tx has not yet expired
725                 )
726                 -- and the output is unspent
727                 AND u.id NOT IN (
728                    SELECT txo_spends.transparent_received_output_id
729                    FROM transparent_received_output_spends txo_spends
730                    JOIN transactions tx ON tx.id_tx = txo_spends.transaction_id
731                    WHERE tx.mined_height IS NOT NULL  -- the spending tx is mined
732                    OR tx.expiry_height = 0 -- the spending tx will not expire
733                    OR tx.expiry_height >= :mempool_height -- the spending tx has not yet expired
734                 )
735             )
736         )",
737    )?;
738
739    let result: Result<Option<WalletTransparentOutput>, SqliteClientError> = stmt_select_utxo
740        .query_and_then(
741            named_params![
742                ":txid": outpoint.hash(),
743                ":output_index": outpoint.n(),
744                ":mempool_height": chain_tip_height.map(|h| u32::from(h) + 1),
745                ":allow_unspendable": allow_unspendable
746            ],
747            to_unspent_transparent_output,
748        )?
749        .next()
750        .transpose();
751
752    result
753}

755/// Returns the list of spendable transparent outputs received by this wallet at `address`
756/// such that, at height `target_height`:
757/// * the transaction that produced the output had or will have at least `min_confirmations`
758///   confirmations; and
759/// * the output is unspent as of the current chain tip.
760///
761/// An output that is potentially spent by an unmined transaction in the mempool is excluded
762/// iff the spending transaction will not be expired at `target_height`.
763///
764/// This could, in very rare circumstances, return as unspent outputs that are actually not
765/// spendable, if they are the outputs of deshielding transactions where the spend anchors have
766/// been invalidated by a rewind. There isn't a way to detect this circumstance at present, but
767/// it should be vanishingly rare as the vast majority of rewinds are of a single block.
768pub(crate) fn get_spendable_transparent_outputs<P: consensus::Parameters>(
769    conn: &rusqlite::Connection,
770    params: &P,
771    address: &TransparentAddress,
772    target_height: BlockHeight,
773    min_confirmations: u32,
774) -> Result<Vec<WalletTransparentOutput>, SqliteClientError> {
775    let confirmed_height = target_height - min_confirmations;
776
777    let mut stmt_utxos = conn.prepare(
778        "SELECT t.txid, u.output_index, u.script,
779                u.value_zat, t.mined_height AS received_height
780         FROM transparent_received_outputs u
781         JOIN transactions t ON t.id_tx = u.transaction_id
782         WHERE u.address = :address
783         -- the transaction that created the output is mined or unexpired as of `confirmed_height`
784         AND (
785            t.mined_height <= :confirmed_height -- tx is mined
786            -- TODO: uncomment the following lines in order to enable zero-conf spends
787            -- OR (
788            --     :min_confirmations = 0
789            --     AND (
790            --         t.expiry_height = 0 -- tx will not expire
791            --         OR t.expiry_height >= :target_height
792            --     )
793            -- )
794         )
795         -- and the output is unspent
796         AND u.id NOT IN (
797            SELECT txo_spends.transparent_received_output_id
798            FROM transparent_received_output_spends txo_spends
799            JOIN transactions tx ON tx.id_tx = txo_spends.transaction_id
800            WHERE tx.mined_height IS NOT NULL -- the spending transaction is mined
801            OR tx.expiry_height = 0 -- the spending tx will not expire
802            OR tx.expiry_height >= :target_height -- the spending tx has not yet expired
803            -- we are intentionally conservative and exclude outputs that are potentially spent
804            -- as of the target height, even if they might actually be spendable due to expiry
805            -- of the spending transaction as of the chain tip
806         )",
807    )?;
808
809    let addr_str = address.encode(params);
810    let mut rows = stmt_utxos.query(named_params![
811        ":address": addr_str,
812        ":confirmed_height": u32::from(confirmed_height),
813        ":target_height": u32::from(target_height),
814        //":min_confirmations": min_confirmations
815    ])?;
816
817    let mut utxos = Vec::<WalletTransparentOutput>::new();
818    while let Some(row) = rows.next()? {
819        let output = to_unspent_transparent_output(row)?;
820        utxos.push(output);
821    }
822
823    Ok(utxos)
824}

826/// Returns a mapping from each transparent receiver associated with the specified account
827/// to its not-yet-shielded UTXO balance, including only the effects of transactions mined
828/// at a block height less than or equal to `summary_height`.
829///
830/// Only non-ephemeral transparent receivers with a non-zero balance at the summary height
831/// will be included.
832pub(crate) fn get_transparent_balances<P: consensus::Parameters>(
833    conn: &rusqlite::Connection,
834    params: &P,
835    account_uuid: AccountUuid,
836    summary_height: BlockHeight,
837) -> Result<HashMap<TransparentAddress, Zatoshis>, SqliteClientError> {
838    let chain_tip_height = chain_tip_height(conn)?.ok_or(SqliteClientError::ChainHeightUnknown)?;
839
840    let mut stmt_address_balances = conn.prepare(
841        "SELECT u.address, SUM(u.value_zat)
842         FROM transparent_received_outputs u
843         JOIN accounts ON accounts.id = u.account_id
844         JOIN transactions t ON t.id_tx = u.transaction_id
845         WHERE accounts.uuid = :account_uuid
846         -- the transaction that created the output is mined or is definitely unexpired
847         AND (
848            t.mined_height <= :summary_height -- tx is mined
849            OR ( -- or the caller has requested to include zero-conf funds that are not expired
850                :summary_height > :chain_tip_height
851                AND (
852                    t.expiry_height = 0 -- tx will not expire
853                    OR t.expiry_height >= :summary_height
854                )
855            )
856         )
857         -- and the output is unspent
858         AND u.id NOT IN (
859            SELECT txo_spends.transparent_received_output_id
860            FROM transparent_received_output_spends txo_spends
861            JOIN transactions tx ON tx.id_tx = txo_spends.transaction_id
862            WHERE tx.mined_height IS NOT NULL -- the spending tx is mined
863            OR tx.expiry_height = 0 -- the spending tx will not expire
864            OR tx.expiry_height >= :spend_expiry_height -- the spending tx is unexpired
865         )
866         GROUP BY u.address",
867    )?;
868
869    let mut res = HashMap::new();
870    let mut rows = stmt_address_balances.query(named_params![
871        ":account_uuid": account_uuid.0,
872        ":summary_height": u32::from(summary_height),
873        ":chain_tip_height": u32::from(chain_tip_height),
874        ":spend_expiry_height": u32::from(std::cmp::min(summary_height, chain_tip_height + 1)),
875    ])?;
876    while let Some(row) = rows.next()? {
877        let taddr_str: String = row.get(0)?;
878        let taddr = TransparentAddress::decode(params, &taddr_str)?;
879        let value = Zatoshis::from_nonnegative_i64(row.get(1)?)?;
880
881        res.insert(taddr, value);
882    }
883
884    Ok(res)
885}

970        })?;

972        account_balances

980/// Marks the given UTXO as having been spent.
981///
982/// Returns `true` if the UTXO was known to the wallet.
983pub(crate) fn mark_transparent_utxo_spent(
984    conn: &rusqlite::Connection,
985    spent_in_tx: TxRef,
986    outpoint: &OutPoint,
987) -> Result<bool, SqliteClientError> {
988    let spend_params = named_params![
989        ":spent_in_tx": spent_in_tx.0,
990        ":prevout_txid": outpoint.hash(),
991        ":prevout_idx": outpoint.n(),
992    ];
993    let mut stmt_mark_transparent_utxo_spent = conn.prepare_cached(
994        "INSERT INTO transparent_received_output_spends (transparent_received_output_id, transaction_id)
995         SELECT txo.id, :spent_in_tx
996         FROM transparent_received_outputs txo
997         JOIN transactions t ON t.id_tx = txo.transaction_id
998         WHERE t.txid = :prevout_txid
999         AND txo.output_index = :prevout_idx
1000         ON CONFLICT (transparent_received_output_id, transaction_id)
1001         -- The following UPDATE is effectively a no-op, but we perform it anyway so that the
1002         -- number of affected rows can be used to determine whether a record existed.
1003         DO UPDATE SET transaction_id = :spent_in_tx",
1004    )?;
1005    let affected_rows = stmt_mark_transparent_utxo_spent.execute(spend_params)?;
1006
1007    // Since we know that the output is spent, we no longer need to search for
1008    // it to find out if it has been spent.
1009    let mut stmt_remove_spend_detection = conn.prepare_cached(
1010        "DELETE FROM transparent_spend_search_queue
1011         WHERE output_index = :prevout_idx
1012         AND transaction_id IN (
1013            SELECT id_tx FROM transactions WHERE txid = :prevout_txid
1014         )",
1015    )?;
1016    stmt_remove_spend_detection.execute(named_params![
1017        ":prevout_txid": outpoint.hash(),
1018        ":prevout_idx": outpoint.n(),
1019    ])?;
1020
1021    // If no rows were affected, we know that we don't actually have the output in
1022    // `transparent_received_outputs` yet, so we have to record the output as spent
1023    // so that when we eventually detect the output, we can create the spend record.
1024    if affected_rows == 0 {
1025        conn.execute(
1026            "INSERT INTO transparent_spend_map (
1027                spending_transaction_id,
1028                prevout_txid,
1029                prevout_output_index
1030            )
1031            VALUES (:spent_in_tx, :prevout_txid, :prevout_idx)
1032            ON CONFLICT (spending_transaction_id, prevout_txid, prevout_output_index)
1033            DO NOTHING",
1034            spend_params,
1035        )?;
1036    }
1037
1038    Ok(affected_rows > 0)
1039}

1050/// An enumeration of the types of errors that can occur when scheduling an event to happen at a
1051/// specific time.
1052#[derive(Debug, Clone)]
1053pub enum SchedulingError {
1054    /// An error occurred in sampling a time offset using an exponential distribution.
1055    Distribution(rand_distr::ExpError),
1056    /// The system attempted to generate an invalid timestamp.
1057    Time(SystemTimeError),
1058    /// A generated duration was out of the range of valid integer values for durations.
1059    OutOfRange(TryFromIntError),
1060}
1061
1062impl std::fmt::Display for SchedulingError {
1063    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
1064        match &self {
1065            SchedulingError::Distribution(e) => {
1066                write!(f, "Failure in sampling scheduling time: {}", e)
1067            }
1068            SchedulingError::Time(t) => write!(f, "Invalid system time: {}", t),
1069            SchedulingError::OutOfRange(t) => write!(f, "Not a valid timestamp or duration: {}", t),
1070        }
1071    }
1072}
1073
1074impl std::error::Error for SchedulingError {
1075    fn source(&self) -> Option<&(dyn std::error::Error + 'static)> {
1076        match &self {
1077            SchedulingError::Distribution(_) => None,
1078            SchedulingError::Time(t) => Some(t),
1079            SchedulingError::OutOfRange(i) => Some(i),
1080        }
1081    }
1082}
1083
1084impl From<rand_distr::ExpError> for SchedulingError {
1085    fn from(value: rand_distr::ExpError) -> Self {
1086        SchedulingError::Distribution(value)
1087    }
1088}
1089
1090impl From<SystemTimeError> for SchedulingError {
1091    fn from(value: SystemTimeError) -> Self {
1092        SchedulingError::Time(value)
1093    }
1094}
1095
1096impl From<TryFromIntError> for SchedulingError {
1097    fn from(value: TryFromIntError) -> Self {
1098        SchedulingError::OutOfRange(value)
1099    }
1100}

1102/// Sample a random timestamp from an exponential distribution such that the expected value of the
1103/// generated timestamp is `check_interval_seconds` after the provided `from_event` time.
1104pub(crate) fn next_check_time<R: RngCore, D: DerefMut<Target = R>>(
1105    mut rng: D,
1106    from_event: SystemTime,
1107    check_interval_seconds: u32,
1108) -> Result<SystemTime, SchedulingError> {
1109    // A  parameter of 1/check_interval_seconds will result in a distribution with an expected
1110    // value of `check_interval_seconds`.
1111    let dist = rand_distr::Exp::new(1.0 / f64::from(check_interval_seconds))?;
1112    let event_delay = dist.sample(rng.deref_mut()).round() as u64;
1113
1114    Ok(from_event + Duration::new(event_delay, 0))
1115}

1117/// Returns the vector of [`TransactionDataRequest`]s that represents the information needed by the
1118/// wallet backend in order to be able to present a complete view of wallet history and memo data.
1119pub(crate) fn transaction_data_requests<P: consensus::Parameters>(
1120    conn: &rusqlite::Connection,
1121    params: &P,
1122) -> Result<Vec<TransactionDataRequest>, SqliteClientError> {
1123    // `lightwalletd` will return an error for `GetTaddressTxids` requests having an end height
1124    // greater than the current chain tip height, so we take the chain tip height into account
1125    // here in order to make this pothole easier for clients of the API to avoid.
1126    let chain_tip_height =
1127        super::chain_tip_height(conn)?.ok_or(SqliteClientError::ChainHeightUnknown)?;
1128
1129    // We cannot construct address-based transaction data requests for the case where we cannot
1130    // determine the height at which to begin, so we require that either the target height or mined
1131    // height be set.
1132    let mut spend_requests_stmt = conn.prepare_cached(
1133        "SELECT
1134            ssq.address,
1135            IFNULL(t.target_height, t.mined_height)
1136         FROM transparent_spend_search_queue ssq
1137         JOIN transactions t ON t.id_tx = ssq.transaction_id
1138         WHERE t.target_height IS NOT NULL
1139         OR t.mined_height IS NOT NULL",
1140    )?;
1141
1142    let spend_search_rows = spend_requests_stmt.query_and_then([], |row| {
1143        let address = TransparentAddress::decode(params, &row.get::<_, String>(0)?)?;
1144        let block_range_start = BlockHeight::from(row.get::<_, u32>(1)?);
1145        let max_end_height = block_range_start + DEFAULT_TX_EXPIRY_DELTA + 1;
1146        Ok::<TransactionDataRequest, SqliteClientError>(
1147            TransactionDataRequest::TransactionsInvolvingAddress {
1148                address,
1149                block_range_start,
1150                block_range_end: Some(std::cmp::min(chain_tip_height + 1, max_end_height)),
1151                request_at: None,
1152                tx_status_filter: TransactionStatusFilter::Mined,
1153                output_status_filter: OutputStatusFilter::All,
1154            },
1155        )
1156    })?;
1157
1158    // Since we don't want to interpret funds that are temporarily held by an ephemeral address in
1159    // the course of creating ZIP 320 transaction pair as belonging to the wallet, we will perform
1160    // ephemeral address checks only for addresses that do not have an unexpired transaction
1161    // associated with them in the database. If, for some reason, the second transaction in a ZIP
1162    // 320 pair fails to be mined after the first transaction in the pair succeeded, we will begin
1163    // including the associated ephemeral address in the set to be checked for funds only after
1164    // the transaction that spends from it has expired.
1165    let mut ephemeral_check_stmt = conn.prepare_cached(
1166        "SELECT
1167            cached_transparent_receiver_address,
1168            transparent_receiver_next_check_time
1169         FROM addresses
1170         WHERE key_scope = :ephemeral_key_scope
1171         AND NOT EXISTS (
1172            SELECT 'x'
1173            FROM transparent_received_outputs tro
1174            JOIN transactions t ON t.id_tx = tro.transaction_id
1175            WHERE tro.address_id = addresses.id
1176            AND t.expiry_height > :chain_tip_height
1177         )",
1178    )?;
1179
1180    let ephemeral_check_rows = ephemeral_check_stmt.query_and_then(
1181        named_params! {
1182            ":ephemeral_key_scope": KeyScope::Ephemeral.encode(),
1183            ":chain_tip_height": u32::from(chain_tip_height)
1184        },
1185        |row| {
1186            let address = TransparentAddress::decode(params, &row.get::<_, String>(0)?)?;
1187            let request_at = row
1188                .get::<_, Option<i64>>(1)?
1189                .map(decode_epoch_seconds)
1190                .transpose()?;
1191
1192            Ok::<TransactionDataRequest, SqliteClientError>(
1193                TransactionDataRequest::TransactionsInvolvingAddress {
1194                    address,
1195                    // We don't want these queries to leak anything about when the wallet created
1196                    // or exposed the address, so we just query for all UTXOs for the address.
1197                    block_range_start: BlockHeight::from(0),
1198                    block_range_end: None,
1199                    request_at,
1200                    tx_status_filter: TransactionStatusFilter::All,
1201                    output_status_filter: OutputStatusFilter::Unspent,
1202                },
1203            )
1204        },
1205    )?;
1206
1207    spend_search_rows
1208        .chain(ephemeral_check_rows)
1209        .collect::<Result<Vec<_>, _>>()
1210}

1212pub(crate) fn get_transparent_address_metadata<P: consensus::Parameters>(
1213    conn: &rusqlite::Connection,
1214    params: &P,
1215    account_uuid: AccountUuid,
1216    address: &TransparentAddress,
1217) -> Result<Option<TransparentAddressMetadata>, SqliteClientError> {
1218    let address_str = address.encode(params);
1219    let addr_meta = conn
1220        .query_row(
1221            "SELECT diversifier_index_be, key_scope
1222             FROM addresses
1223             JOIN accounts ON addresses.account_id = accounts.id
1224             WHERE accounts.uuid = :account_uuid
1225             AND cached_transparent_receiver_address = :address",
1226            named_params![":account_uuid": account_uuid.0, ":address": &address_str],
1227            |row| {
1228                let di_be: Vec<u8> = row.get(0)?;
1229                let scope_code = row.get(1)?;
1230                Ok(KeyScope::decode(scope_code).and_then(|key_scope| {
1231                    address_index_from_diversifier_index_be(&di_be).map(|address_index| {
1232                        TransparentAddressMetadata::new(key_scope.into(), address_index)
1233                    })
1234                }))
1235            },
1236        )
1237        .optional()?
1238        .transpose()?;
1239
1240    if addr_meta.is_some() {
1241        return Ok(addr_meta);
1242    }
1243
1244    if let Some((legacy_taddr, address_index)) =
1245        get_legacy_transparent_address(params, conn, account_uuid)?
1246    {
1247        if &legacy_taddr == address {
1248            let metadata = TransparentAddressMetadata::new(Scope::External.into(), address_index);
1249            return Ok(Some(metadata));
1250        }
1251    }
1252
1253    Ok(None)
1254}

1256/// Attempts to determine the account that received the given transparent output.
1257///
1258/// The following three locations in the wallet's key tree are searched:
1259/// - Transparent receivers that have been generated as part of a Unified Address.
1260/// - Transparent ephemeral addresses that have been reserved or are within
1261///   the gap limit from the last reserved address.
1262/// - "Legacy transparent addresses" (at BIP 44 address index 0 within an account).
1263///
1264/// Returns `Ok(None)` if the transparent output's recipient address is not in any of the
1265/// above locations. This means the wallet considers the output "not interesting".
1266pub(crate) fn find_account_uuid_for_transparent_address<P: consensus::Parameters>(
1267    conn: &rusqlite::Connection,
1268    params: &P,
1269    address: &TransparentAddress,
1270) -> Result<Option<(AccountUuid, KeyScope)>, SqliteClientError> {
1271    let address_str = address.encode(params);
1272
1273    if let Some((account_id, key_scope_code)) = conn
1274        .query_row(
1275            "SELECT accounts.uuid, addresses.key_scope
1276             FROM addresses
1277             JOIN accounts ON accounts.id = addresses.account_id
1278             WHERE cached_transparent_receiver_address = :address",
1279            named_params![":address": &address_str],
1280            |row| Ok((AccountUuid(row.get(0)?), row.get(1)?)),
1281        )
1282        .optional()?
1283    {
1284        return Ok(Some((account_id, KeyScope::decode(key_scope_code)?)));
1285    }
1286
1287    let account_ids = get_account_ids(conn)?;
1288
1289    // If the UTXO is received at the legacy transparent address (at BIP 44 address
1290    // index 0 within its particular account, which we specifically ensure is returned
1291    // from `get_transparent_receivers`), there may be no entry in the addresses table
1292    // that can be used to tie the address to a particular account. In this case, we
1293    // look up the legacy address for each account in the wallet, and check whether it
1294    // matches the address for the received UTXO.
1295    for &account_id in account_ids.iter() {
1296        if let Some((legacy_taddr, _)) = get_legacy_transparent_address(params, conn, account_id)? {
1297            if &legacy_taddr == address {
1298                return Ok(Some((account_id, KeyScope::EXTERNAL)));
1299            }
1300        }
1301    }
1302
1303    Ok(None)
1304}

1306/// Add a transparent output relevant to this wallet to the database.
1307///
1308/// `output_height` may be None if this is an ephemeral output from a
1309/// transaction we created, that we do not yet know to have been mined.
1310#[allow(clippy::too_many_arguments)]
1311pub(crate) fn put_transparent_output<P: consensus::Parameters>(
1312    conn: &rusqlite::Connection,
1313    params: &P,
1314    output: &WalletTransparentOutput,
1315    known_unspent: bool,
1316) -> Result<(AccountRef, KeyScope, UtxoId), SqliteClientError> {
1317    let addr_str = output.recipient_address().encode(params);
1318
1319    // Unlike the shielded pools, we only can receive transparent outputs on addresses for which we
1320    // have an `addresses` table entry, so we can just query for that here.
1321    let (address_id, account_id, key_scope_code) = conn
1322        .query_row(
1323            "SELECT id, account_id, key_scope
1324             FROM addresses
1325             WHERE cached_transparent_receiver_address = :transparent_address",
1326            named_params! {":transparent_address": addr_str},
1327            |row| {
1328                Ok((
1329                    row.get("id").map(AddressRef)?,
1330                    row.get("account_id").map(AccountRef)?,
1331                    row.get("key_scope")?,
1332                ))
1333            },
1334        )
1335        .optional()?
1336        .ok_or(SqliteClientError::AddressNotRecognized(
1337            *output.recipient_address(),
1338        ))?;
1339
1340    let key_scope = KeyScope::decode(key_scope_code)?;
1341
1342    let output_height = output.mined_height().map(u32::from);
1343
1344    // Check whether we have an entry in the blocks table for the output height;
1345    // if not, the transaction will be updated with its mined height when the
1346    // associated block is scanned.
1347    let block = match output_height {
1348        Some(height) => conn
1349            .query_row(
1350                "SELECT height FROM blocks WHERE height = :height",
1351                named_params![":height": height],
1352                |row| row.get::<_, u32>(0),
1353            )
1354            .optional()?,
1355        None => None,
1356    };
1357
1358    let id_tx = conn.query_row(
1359        "INSERT INTO transactions (txid, block, mined_height)
1360         VALUES (:txid, :block, :mined_height)
1361         ON CONFLICT (txid) DO UPDATE
1362         SET block = IFNULL(block, :block),
1363             mined_height = :mined_height
1364         RETURNING id_tx",
1365        named_params![
1366           ":txid": &output.outpoint().hash().to_vec(),
1367           ":block": block,
1368           ":mined_height": output_height
1369        ],
1370        |row| row.get::<_, i64>(0),
1371    )?;
1372
1373    let spent_height = conn
1374        .query_row(
1375            "SELECT t.mined_height
1376             FROM transactions t
1377             JOIN transparent_received_output_spends ts ON ts.transaction_id = t.id_tx
1378             JOIN transparent_received_outputs tro ON tro.id = ts.transparent_received_output_id
1379             WHERE tro.transaction_id = :transaction_id
1380             AND tro.output_index = :output_index",
1381            named_params![
1382                ":transaction_id": id_tx,
1383                ":output_index": output.outpoint().n(),
1384            ],
1385            |row| {
1386                row.get::<_, Option<u32>>(0)
1387                    .map(|o| o.map(BlockHeight::from))
1388            },
1389        )
1390        .optional()?
1391        .flatten();
1392
1393    // The max observed unspent height is either the spending transaction's mined height - 1, or
1394    // the current chain tip height if the UTXO was received via a path that confirmed that it was
1395    // unspent, such as by querying the UTXO set of the network.
1396    let max_observed_unspent = match spent_height {
1397        Some(h) => Some(h - 1),
1398        None => {
1399            if known_unspent {
1400                chain_tip_height(conn)?
1401            } else {
1402                None
1403            }
1404        }
1405    };
1406
1407    let mut stmt_upsert_transparent_output = conn.prepare_cached(
1408        "INSERT INTO transparent_received_outputs (
1409            transaction_id, output_index,
1410            account_id, address_id, address, script,
1411            value_zat, max_observed_unspent_height
1412        )
1413        VALUES (
1414            :transaction_id, :output_index,
1415            :account_id, :address_id, :address, :script,
1416            :value_zat, :max_observed_unspent_height
1417        )
1418        ON CONFLICT (transaction_id, output_index) DO UPDATE
1419        SET account_id = :account_id,
1420            address_id = :address_id,
1421            address = :address,
1422            script = :script,
1423            value_zat = :value_zat,
1424            max_observed_unspent_height = IFNULL(:max_observed_unspent_height, max_observed_unspent_height)
1425        RETURNING id",
1426    )?;
1427
1428    let sql_args = named_params![
1429        ":transaction_id": id_tx,
1430        ":output_index": output.outpoint().n(),
1431        ":account_id": account_id.0,
1432        ":address_id": address_id.0,
1433        ":address": output.recipient_address().encode(params),
1434        ":script": &output.txout().script_pubkey.0,
1435        ":value_zat": &i64::from(ZatBalance::from(output.txout().value)),
1436        ":max_observed_unspent_height": max_observed_unspent.map(u32::from),
1437    ];
1438
1439    let utxo_id = stmt_upsert_transparent_output
1440        .query_row(sql_args, |row| row.get::<_, i64>(0).map(UtxoId))?;
1441
1442    // If we have a record of the output already having been spent, then mark it as spent using the
1443    // stored reference to the spending transaction.
1444    let spending_tx_ref = conn
1445        .query_row(
1446            "SELECT ts.spending_transaction_id
1447             FROM transparent_spend_map ts
1448             JOIN transactions t ON t.id_tx = ts.spending_transaction_id
1449             WHERE ts.prevout_txid = :prevout_txid
1450             AND ts.prevout_output_index = :prevout_idx
1451             ORDER BY t.block NULLS LAST LIMIT 1",
1452            named_params![
1453                ":prevout_txid": output.outpoint().txid().as_ref(),
1454                ":prevout_idx": output.outpoint().n()
1455            ],
1456            |row| row.get::<_, i64>(0).map(TxRef),
1457        )
1458        .optional()?;
1459
1460    if let Some(spending_transaction_id) = spending_tx_ref {
1461        mark_transparent_utxo_spent(conn, spending_transaction_id, output.outpoint())?;
1462    }
1463
1464    Ok((account_id, key_scope, utxo_id))
1465}

1467/// Adds a request to retrieve transactions involving the specified address to the transparent
1468/// spend search queue. Note that such requests are _not_ for data related to `tx_ref`, but instead
1469/// a request to find where the UTXO with the outpoint `(tx_ref, output_index)` is spent.
1470///
1471/// ### Parameters
1472/// - `receiving_address`: The address that received the UTXO.
1473/// - `tx_ref`: The transaction in which the UTXO was received.
1474/// - `output_index`: The index of the output within `vout` of the specified transaction.
1475pub(crate) fn queue_transparent_spend_detection<P: consensus::Parameters>(
1476    conn: &rusqlite::Transaction<'_>,
1477    params: &P,
1478    receiving_address: TransparentAddress,
1479    tx_ref: TxRef,
1480    output_index: u32,
1481) -> Result<(), SqliteClientError> {
1482    let mut stmt = conn.prepare_cached(
1483        "INSERT INTO transparent_spend_search_queue
1484         (address, transaction_id, output_index)
1485         VALUES
1486         (:address, :transaction_id, :output_index)
1487         ON CONFLICT (transaction_id, output_index) DO NOTHING",
1488    )?;
1489
1490    let addr_str = receiving_address.encode(params);
1491    stmt.execute(named_params! {
1492        ":address": addr_str,
1493        ":transaction_id": tx_ref.0,
1494        ":output_index": output_index
1495    })?;
1496
1497    Ok(())
1498}
