1//! Functions for querying information in the wallet database.
2//!
3//! These functions should generally not be used directly; instead,
4//! their functionality is available via the [`WalletRead`] and
5//! [`WalletWrite`] traits.
6//!
7//! [`WalletRead`]: zcash_client_backend::data_api::WalletRead
8//! [`WalletWrite`]: zcash_client_backend::data_api::WalletWrite
9//!
10//! # Views
11//!
12//! The wallet database exposes the following views as part of its public API:
13//!
14//! ## `v_transactions`
15//!
16//! This view exposes the history of transactions that affect the balance of each account in the
17//! wallet. A transaction may be represented by multiple rows in this view, one for each account in
18//! the wallet that contributes funds to or receives funds from the transaction in question. Each
19//! row of the view contains:
20//! - `account_balance_delta`: the net effect of the transaction on the associated account's
21//!   balance. This value is positive when funds are received by the account, and negative when the
22//!   balance of the account decreases due to a spend.
23//! - `fee_paid`: the total fee paid to send the transaction, as a positive value. This fee is
24//!   associated with the transaction (similar to e.g. `txid` or `mined_height`), and not with any
25//!   specific account involved with that transaction. ` If multiple rows exist for a single
26//!   transaction, this fee amount will be repeated for each such row. Therefore, if more than one
27//!   of the wallet's accounts is involved with the transaction, this fee should be considered only
28//!   once in determining the total value sent from the wallet as a whole.
29//!
30//! ### Seed Phrase with Single Account
31//!
32//! In the case that the seed phrase for in this wallet has only been used to create a single
33//! account, this view will contain one row per transaction, in the case that
34//! `account_balance_delta` is negative, it is usually safe to add `fee_paid` back to the
35//! `account_balance_delta` value to determine the amount sent to addresses outside the wallet.
36//!
37//! ### Seed Phrase with Multiple Accounts
38//!
39//! In the case that the seed phrase for in this wallet has been used to create multiple accounts,
40//! this view may contain multiple rows per transaction, one for each account involved. In this
41//! case, the total amount sent to addresses outside the wallet can usually be calculated by
42//! grouping rows by `id_tx` and then using `SUM(account_balance_delta) + MAX(fee_paid)`.
43//!
44//! ### Imported Seed Phrases
45//!
46//! If a seed phrase is imported, and not every account associated with it is loaded into the
47//! wallet, this view may show partial information about some transactions. In particular, any
48//! computation that involves both `account_balance_delta` and `fee_paid` is likely to be
49//! inaccurate.
50//!
51//! ## `v_tx_outputs`
52//!
53//! This view exposes the history of transaction outputs received by and sent from the wallet,
54//! keyed by transaction ID, pool type, and output index. The contents of this view are useful for
55//! producing a detailed report of the effects of a transaction. Each row of this view contains:
56//! - `from_account_id` for sent outputs, the account from which the value was sent.
57//! - `to_account_id` in the case that the output was received by an account in the wallet, the
58//!   identifier for the account receiving the funds.
59//! - `to_address` the address to which an output was sent, or the address at which value was
60//!   received in the case of received transparent funds.
61//! - `value` the value of the output. This is always a positive number, for both sent and received
62//!   outputs.
63//! - `is_change` a boolean flag indicating whether this is a change output belonging to the
64//!   wallet.
65//! - `memo` the shielded memo associated with the output, if any.

165fn parse_account_source(
166    account_kind: u32,
167    hd_seed_fingerprint: Option<[u8; 32]>,
168    hd_account_index: Option<u32>,
169    spending_key_available: bool,
170    key_source: Option<String>,
171) -> Result<AccountSource, SqliteClientError> {
172    let derivation = hd_seed_fingerprint
173        .zip(hd_account_index)
174        .map(|(seed_fp, idx)| {
175            zip32::AccountId::try_from(idx)
176                .map_err(|_| {
177                    SqliteClientError::CorruptedData(
178                        "ZIP-32 account ID from wallet DB is out of range.".to_string(),
179                    )
180                })
181                .map(|idx| Zip32Derivation::new(SeedFingerprint::from_bytes(seed_fp), idx))
182        })
183        .transpose()?;
184
185    match (account_kind, derivation) {
186        (0, Some(derivation)) => Ok(AccountSource::Derived {
187            derivation,
188            key_source,
189        }),
190        (1, derivation) => Ok(AccountSource::Imported {
191            purpose: if spending_key_available {
192                AccountPurpose::Spending { derivation }
193            } else {
194                AccountPurpose::ViewOnly
195            },
196            key_source,
197        }),
198        (0, None) => Err(SqliteClientError::CorruptedData(
199            "Wallet DB account_kind constraint violated".to_string(),
200        )),
201        (_, _) => Err(SqliteClientError::CorruptedData(
202            "Unrecognized account_kind".to_string(),
203        )),
204    }
205}

296pub(crate) fn seed_matches_derived_account<P: consensus::Parameters>(
297    params: &P,
298    seed: &SecretVec<u8>,
299    seed_fingerprint: &SeedFingerprint,
300    account_index: zip32::AccountId,
301    uivk: &UnifiedIncomingViewingKey,
302) -> Result<bool, SqliteClientError> {
303    let seed_fingerprint_match =
304        &SeedFingerprint::from_seed(seed.expose_secret()).ok_or_else(|| {
305            SqliteClientError::BadAccountData(
306                "Seed must be between 32 and 252 bytes in length.".to_owned(),
307            )
308        })? == seed_fingerprint;
309
310    // `UnifiedIncomingViewingKey`s are not comparable with `Eq`, but Unified Address
311    // components are, so we derive corresponding addresses for each key and use
312    // those to check whether any components match.
313    let uivk_match = {
314        let usk = UnifiedSpendingKey::from_seed(params, &seed.expose_secret()[..], account_index)
315            .map_err(|_| SqliteClientError::KeyDerivationError(account_index))?;
316
317        let (seed_addr, _) = usk
318            .to_unified_full_viewing_key()
319            .default_address(UnifiedAddressRequest::AllAvailableKeys)?;
320        let (uivk_addr, _) = uivk.default_address(UnifiedAddressRequest::AllAvailableKeys)?;
321
322        #[cfg(not(feature = "orchard"))]
323        let orchard_match = false;
324        #[cfg(feature = "orchard")]
325        let orchard_match = seed_addr
326            .orchard()
327            .zip(uivk_addr.orchard())
328            .map(|(a, b)| a == b)
329            == Some(true);
330
331        let sapling_match = seed_addr
332            .sapling()
333            .zip(uivk_addr.sapling())
334            .map(|(a, b)| a == b)
335            == Some(true);
336
337        let p2pkh_match = seed_addr
338            .transparent()
339            .zip(uivk_addr.transparent())
340            .map(|(a, b)| a == b)
341            == Some(true);
342
343        orchard_match || sapling_match || p2pkh_match
344    };
345
346    if seed_fingerprint_match != uivk_match {
347        // If these mismatch, it suggests database corruption.
348        Err(SqliteClientError::CorruptedData(format!(
349            "Seed fingerprint match: {seed_fingerprint_match}, uivk match: {uivk_match}"
350        )))
351    } else {
352        Ok(seed_fingerprint_match && uivk_match)
353    }
354}

356// Returns the highest used account index for a given seed.
357pub(crate) fn max_zip32_account_index(
358    conn: &rusqlite::Connection,
359    seed_id: &SeedFingerprint,
360) -> Result<Option<zip32::AccountId>, SqliteClientError> {
361    conn.query_row_and_then(
362        "SELECT MAX(hd_account_index) FROM accounts WHERE hd_seed_fingerprint = :hd_seed",
363        [seed_id.to_bytes()],
364        |row| {
365            row.get::<_, Option<u32>>(0)?
366                .map(zip32::AccountId::try_from)
367                .transpose()
368                .map_err(|_| SqliteClientError::Zip32AccountIndexOutOfRange)
369        },
370    )
371}

373pub(crate) fn add_account<P: consensus::Parameters>(
374    conn: &rusqlite::Transaction,
375    params: &P,
376    account_name: &str,
377    kind: &AccountSource,
378    viewing_key: ViewingKey,
379    birthday: &AccountBirthday,
380    #[cfg(feature = "transparent-inputs")] gap_limits: &GapLimits,
381) -> Result<Account, SqliteClientError> {
382    if let Some(ufvk) = viewing_key.ufvk() {
383        // Check whether any component of this UFVK collides with an existing imported or derived FVK.
384        if let Some(existing_account) = get_account_for_ufvk(conn, params, ufvk)? {
385            return Err(SqliteClientError::AccountCollision(existing_account.id()));
386        }
387    }
388    // TODO(#1490): check for IVK collisions.
389
390    let account_uuid = AccountUuid(Uuid::new_v4());
391
392    let (derivation, spending_key_available, key_source) = match kind {
393        AccountSource::Derived {
394            derivation,
395            key_source,
396        } => (Some(derivation), true, key_source),
397        AccountSource::Imported {
398            purpose: AccountPurpose::Spending { derivation },
399            key_source,
400        } => (derivation.as_ref(), true, key_source),
401        AccountSource::Imported {
402            purpose: AccountPurpose::ViewOnly,
403            key_source,
404        } => (None, false, key_source),
405    };
406
407    #[cfg(feature = "orchard")]
408    let orchard_item = viewing_key
409        .ufvk()
410        .and_then(|ufvk| ufvk.orchard().map(|k| k.to_bytes()));
411    #[cfg(not(feature = "orchard"))]
412    let orchard_item: Option<Vec<u8>> = None;
413
414    let sapling_item = viewing_key
415        .ufvk()
416        .and_then(|ufvk| ufvk.sapling().map(|k| k.to_bytes()));
417
418    #[cfg(feature = "transparent-inputs")]
419    let transparent_item = viewing_key
420        .ufvk()
421        .and_then(|ufvk| ufvk.transparent().map(|k| k.serialize()));
422    #[cfg(not(feature = "transparent-inputs"))]
423    let transparent_item: Option<Vec<u8>> = None;
424
425    let birthday_sapling_tree_size = Some(birthday.sapling_frontier().tree_size());
426    #[cfg(feature = "orchard")]
427    let birthday_orchard_tree_size = Some(birthday.orchard_frontier().tree_size());
428    #[cfg(not(feature = "orchard"))]
429    let birthday_orchard_tree_size: Option<u64> = None;
430
431    let ufvk_encoded = viewing_key.ufvk().map(|ufvk| ufvk.encode(params));
432    let account_id = conn
433        .query_row(
434            r#"
435            INSERT INTO accounts (
436                name,
437                uuid,
438                account_kind, hd_seed_fingerprint, hd_account_index, key_source,
439                ufvk, uivk,
440                orchard_fvk_item_cache, sapling_fvk_item_cache, p2pkh_fvk_item_cache,
441                birthday_height, birthday_sapling_tree_size, birthday_orchard_tree_size,
442                recover_until_height,
443                has_spend_key
444            )
445            VALUES (
446                :account_name,
447                :uuid,
448                :account_kind, :hd_seed_fingerprint, :hd_account_index, :key_source,
449                :ufvk, :uivk,
450                :orchard_fvk_item_cache, :sapling_fvk_item_cache, :p2pkh_fvk_item_cache,
451                :birthday_height, :birthday_sapling_tree_size, :birthday_orchard_tree_size,
452                :recover_until_height,
453                :has_spend_key
454            )
455            RETURNING id
456            "#,
457            named_params![
458                ":account_name": account_name,
459                ":uuid": account_uuid.0,
460                ":account_kind": account_kind_code(kind),
461                ":hd_seed_fingerprint": derivation.map(|d| d.seed_fingerprint().to_bytes()),
462                ":hd_account_index": derivation.map(|d| u32::from(d.account_index())),
463                ":key_source": key_source,
464                ":ufvk": ufvk_encoded,
465                ":uivk": viewing_key.uivk().encode(params),
466                ":orchard_fvk_item_cache": orchard_item,
467                ":sapling_fvk_item_cache": sapling_item,
468                ":p2pkh_fvk_item_cache": transparent_item,
469                ":birthday_height": u32::from(birthday.height()),
470                ":birthday_sapling_tree_size": birthday_sapling_tree_size,
471                ":birthday_orchard_tree_size": birthday_orchard_tree_size,
472                ":recover_until_height": birthday.recover_until().map(u32::from),
473                ":has_spend_key": spending_key_available as i64,
474            ],
475            |row| row.get(0).map(AccountRef),
476        )
477        .map_err(|e| match e {
478            rusqlite::Error::SqliteFailure(f, s)
479                if f.code == rusqlite::ErrorCode::ConstraintViolation =>
480            {
481                // An account conflict occurred. This should already have been caught by
482                // the check using `get_account_for_ufvk` above, but in case it wasn't,
483                // make a best effort to determine the AccountRef of the pre-existing row
484                // and provide that to our caller.
485                if let Ok(colliding_uuid) = conn.query_row(
486                    "SELECT uuid FROM accounts WHERE ufvk = ?",
487                    params![ufvk_encoded],
488                    |row| Ok(AccountUuid(row.get(0)?)),
489                ) {
490                    return SqliteClientError::AccountCollision(colliding_uuid);
491                }
492
493                SqliteClientError::from(rusqlite::Error::SqliteFailure(f, s))
494            }
495            _ => SqliteClientError::from(e),
496        })?;
497
498    let account = Account {
499        id: account_id,
500        name: Some(account_name.to_owned()),
501        uuid: account_uuid,
502        kind: kind.clone(),
503        viewing_key,
504        birthday: birthday.height(),
505    };
506
507    // If a birthday frontier is available, insert it into the note commitment tree. If the
508    // birthday frontier is the empty frontier, we don't need to do anything.
509    if let Some(frontier) = birthday.sapling_frontier().value() {
510        debug!("Inserting Sapling frontier into ShardTree: {:?}", frontier);
511        let shard_store =
512            SqliteShardStore::<_, ::sapling::Node, SAPLING_SHARD_HEIGHT>::from_connection(
513                conn,
514                SAPLING_TABLES_PREFIX,
515            )?;
516        let mut shard_tree: ShardTree<
517            _,
518            { ::sapling::NOTE_COMMITMENT_TREE_DEPTH },
519            SAPLING_SHARD_HEIGHT,
520        > = ShardTree::new(shard_store, PRUNING_DEPTH.try_into().unwrap());
521        shard_tree.insert_frontier_nodes(
522            frontier.clone(),
523            Retention::Checkpoint {
524                // This subtraction is safe, because all leaves in the tree appear in blocks, and
525                // the invariant that birthday.height() always corresponds to the block for which
526                // `frontier` is the tree state at the start of the block. Together, this means
527                // there exists a prior block for which frontier is the tree state at the end of
528                // the block.
529                id: birthday.height() - 1,
530                marking: Marking::Reference,
531            },
532        )?;
533    }
534
535    #[cfg(feature = "orchard")]
536    if let Some(frontier) = birthday.orchard_frontier().value() {
537        debug!("Inserting Orchard frontier into ShardTree: {:?}", frontier);
538        let shard_store = SqliteShardStore::<
539            _,
540            ::orchard::tree::MerkleHashOrchard,
541            ORCHARD_SHARD_HEIGHT,
542        >::from_connection(conn, ORCHARD_TABLES_PREFIX)?;
543        let mut shard_tree: ShardTree<
544            _,
545            { ::orchard::NOTE_COMMITMENT_TREE_DEPTH as u8 },
546            ORCHARD_SHARD_HEIGHT,
547        > = ShardTree::new(shard_store, PRUNING_DEPTH.try_into().unwrap());
548        shard_tree.insert_frontier_nodes(
549            frontier.clone(),
550            Retention::Checkpoint {
551                // This subtraction is safe, because all leaves in the tree appear in blocks, and
552                // the invariant that birthday.height() always corresponds to the block for which
553                // `frontier` is the tree state at the start of the block. Together, this means
554                // there exists a prior block for which frontier is the tree state at the end of
555                // the block.
556                id: birthday.height() - 1,
557                marking: Marking::Reference,
558            },
559        )?;
560    }
561
562    // The ignored range always starts at Sapling activation
563    let sapling_activation_height = params
564        .activation_height(NetworkUpgrade::Sapling)
565        .expect("Sapling activation height must be available.");
566
567    // Add the ignored range up to the birthday height.
568    if sapling_activation_height < birthday.height() {
569        let ignored_range = sapling_activation_height..birthday.height();
570
571        replace_queue_entries::<SqliteClientError>(
572            conn,
573            &ignored_range,
574            Some(ScanRange::from_parts(
575                ignored_range.clone(),
576                ScanPriority::Ignored,
577            ))
578            .into_iter(),
579            false,
580        )?;
581    };
582
583    // Rewrite the scan ranges from the birthday height up to the chain tip so that we'll ensure we
584    // re-scan to find any notes that might belong to the newly added account.
585    if let Some(t) = chain_tip_height(conn)? {
586        let rescan_range = birthday.height()..(t + 1);
587
588        replace_queue_entries::<SqliteClientError>(
589            conn,
590            &rescan_range,
591            Some(ScanRange::from_parts(
592                rescan_range.clone(),
593                ScanPriority::Historic,
594            ))
595            .into_iter(),
596            true, // force rescan
597        )?;
598    }
599
600    // Always derive the default Unified Address for the account. If the account's viewing
601    // key has fewer components than the wallet supports (most likely due to this being an
602    // imported viewing key), derive an address containing the common subset of receivers.
603    let (address, d_idx) = account.default_address(UnifiedAddressRequest::AllAvailableKeys)?;
604    upsert_address(
605        conn,
606        params,
607        account_id,
608        d_idx,
609        &address,
610        Some(birthday.height()),
611        false,
612    )?;
613
614    // Pre-generate external transparent addresses prior to the index of the default address.
615    #[cfg(feature = "transparent-inputs")]
616    if let Ok(default_addr_idx) = NonHardenedChildIndex::try_from(d_idx) {
617        transparent::generate_address_range(
618            conn,
619            params,
620            account_id,
621            KeyScope::EXTERNAL,
622            UnifiedAddressRequest::ALLOW_ALL,
623            NonHardenedChildIndex::const_from_index(0)..default_addr_idx,
624            false,
625        )?
626    }
627
628    // Pre-generate transparent addresses up to the gap limits for the external, internal,
629    // and ephemeral key scopes.
630    #[cfg(feature = "transparent-inputs")]
631    for key_scope in [KeyScope::EXTERNAL, KeyScope::INTERNAL, KeyScope::Ephemeral] {
632        use ReceiverRequirement::*;
633        transparent::generate_gap_addresses(
634            conn,
635            params,
636            account_id,
637            key_scope,
638            gap_limits,
639            UnifiedAddressRequest::unsafe_custom(Allow, Allow, Require),
640            false,
641        )?;
642    }
643
644    Ok(account)
645}

647pub(crate) fn get_next_available_address<P: consensus::Parameters, C: Clock>(
648    conn: &rusqlite::Transaction,
649    params: &P,
650    clock: &C,
651    account_uuid: AccountUuid,
652    request: UnifiedAddressRequest,
653    #[cfg(feature = "transparent-inputs")] gap_limits: &GapLimits,
654) -> Result<Option<(UnifiedAddress, DiversifierIndex)>, SqliteClientError> {
655    let account: Account = match get_account(conn, params, account_uuid)? {
656        Some(account) => account,
657        None => {
658            return Ok(None);
659        }
660    };
661
662    // This will also ensure that the provided request can be satisfied by the account's UIVK
663    let requirements = account.uivk().receiver_requirements(request)?;
664
665    let (addr, diversifier_index) = if requirements.p2pkh() == ReceiverRequirement::Require {
666        #[cfg(not(feature = "transparent-inputs"))]
667        {
668            return Err(SqliteClientError::AddressGeneration(
669                AddressGenerationError::ReceiverTypeNotSupported(
670                    zcash_address::unified::Typecode::P2pkh,
671                ),
672            ));
673        }
674
675        // If a p2pkh receiver is required, return the first un-exposed address from within the
676        // transparent gap limit.
677        #[cfg(feature = "transparent-inputs")]
678        {
679            use ReceiverRequirement::*;
680            // First, ensure that we have pre-generated as many addresses as we can.
681            transparent::generate_gap_addresses(
682                conn,
683                params,
684                account.internal_id(),
685                KeyScope::EXTERNAL,
686                gap_limits,
687                UnifiedAddressRequest::unsafe_custom(Allow, Allow, Require),
688                true,
689            )?;
690
691            // Select indices from the transparent gap limit that are available for use as
692            // diversifier indices.
693            let (gap_start, addrs) = transparent::select_addrs_to_reserve(
694                conn,
695                params,
696                account.internal_id(),
697                KeyScope::EXTERNAL,
698                gap_limits.external(),
699                gap_limits
700                    .external()
701                    .try_into()
702                    .expect("gap limit fits in usize"),
703            )?;
704
705            // Find the first index that generates an address conforming to the request.
706            addrs
707                .iter()
708                .find_map(|(_, _, meta)| {
709                    let j = DiversifierIndex::from(meta.address_index());
710                    account.uivk().address(j, request).ok().map(|ua| (ua, j))
711                })
712                .ok_or(SqliteClientError::ReachedGapLimit(
713                    TransparentKeyScope::EXTERNAL,
714                    gap_start.index() + gap_limits.external(),
715                ))?
716        }
717    } else {
718        // compute a base diversifier index from the timestamp
719        let mut j = DiversifierIndex::from(
720            clock
721                .now()
722                .duration_since(SystemTime::UNIX_EPOCH)
723                .expect("system time is valid")
724                .as_secs()
725                .saturating_add(MIN_SHIELDED_DIVERSIFIER_OFFSET),
726        );
727
728        let mut find_collision = conn.prepare(
729            "SELECT exposed_at_height
730             FROM addresses
731             WHERE account_id = :account_id
732             AND key_scope = :key_scope
733             AND diversifier_index_be = :diversifier_index_be",
734        )?;
735
736        // search the diversifier space for a diversifier index that creates a valid address
737        // satisfying the request and is currently not used in an exposed address
738        loop {
739            let found_addr = account.uivk().find_address(j, request)?;
740            let collision = find_collision
741                .query_row(
742                    named_params! {
743                        ":account_id": account.internal_id().0,
744                        ":key_scope": KeyScope::EXTERNAL.encode(),
745                        ":diversifier_index_be": &encode_diversifier_index_be(found_addr.1)
746                    },
747                    |row| row.get::<_, Option<u32>>(0),
748                )
749                .optional()?
750                .flatten();
751
752            if collision.is_none() {
753                break found_addr;
754            } else {
755                j.increment().map_err(|_| {
756                    SqliteClientError::AddressGeneration(
757                        AddressGenerationError::DiversifierSpaceExhausted,
758                    )
759                })?;
760            }
761        }
762    };
763
764    let chain_tip_height = chain_tip_height(conn)?.ok_or(SqliteClientError::ChainHeightUnknown)?;
765    upsert_address(
766        conn,
767        params,
768        account.internal_id(),
769        diversifier_index,
770        &addr,
771        Some(chain_tip_height),
772        true,
773    )?;
774
775    Ok(Some((addr, diversifier_index)))
776}

778pub(crate) fn list_addresses<P: consensus::Parameters>(
779    conn: &rusqlite::Connection,
780    params: &P,
781    account_uuid: AccountUuid,
782) -> Result<Vec<AddressInfo>, SqliteClientError> {
783    let mut addrs = vec![];
784
785    let mut stmt_addrs = conn.prepare(
786        "SELECT address, diversifier_index_be, key_scope
787         FROM addresses
788         JOIN accounts ON accounts.id = addresses.account_id
789         WHERE accounts.uuid = :account_uuid
790         AND exposed_at_height IS NOT NULL
791         ORDER BY exposed_at_height ASC, diversifier_index_be ASC",
792    )?;
793
794    let mut rows = stmt_addrs.query(named_params![
795        ":account_uuid": account_uuid.0,
796    ])?;
797
798    while let Some(row) = rows.next()? {
799        let addr_str: String = row.get(0)?;
800        let di_vec: Vec<u8> = row.get(1)?;
801        let _scope = KeyScope::decode(row.get(2)?)?;
802
803        let addr = Address::decode(params, &addr_str).ok_or_else(|| {
804            SqliteClientError::CorruptedData("Not a valid Zcash recipient address".to_owned())
805        })?;
806        let diversifier_index = decode_diversifier_index_be(&di_vec)?;
807        // Sapling and Unified addresses always have external scope.
808        #[cfg(feature = "transparent-inputs")]
809        let transparent_scope =
810            matches!(addr, Address::Transparent(_) | Address::Tex(_)).then(|| _scope.into());
811
812        addrs.push(
813            AddressInfo::from_parts(
814                addr,
815                diversifier_index,
816                #[cfg(feature = "transparent-inputs")]
817                transparent_scope,
818            )
819            .expect("valid"),
820        );
821    }
822
823    Ok(addrs)
824}

826pub(crate) fn get_last_generated_address_matching<P: consensus::Parameters>(
827    conn: &rusqlite::Connection,
828    params: &P,
829    account_uuid: AccountUuid,
830    address_filter: UnifiedAddressRequest,
831) -> Result<Option<(UnifiedAddress, DiversifierIndex)>, SqliteClientError> {
832    let account: Account =
833        get_account(conn, params, account_uuid)?.ok_or(SqliteClientError::AccountUnknown)?;
834
835    let requirements = account
836        .uivk()
837        .receiver_requirements(address_filter)
838        .map_err(|_| {
839            SqliteClientError::BadAccountData(
840                "Could not generate UnifiedAddressRequest for UIVK".to_string(),
841            )
842        })?;
843    let require_flags = ReceiverFlags::required(requirements);
844    let omit_flags = ReceiverFlags::omitted(requirements);
845    // This returns the most recently exposed external-scope address (the address that was exposed
846    // at the greatest block height, using the largest diversifier index to break ties)
847    // that conforms to the specified requirements.
848    let addr: Option<(String, Vec<u8>)> = conn
849        .query_row(
850            "SELECT address, diversifier_index_be
851             FROM addresses
852             WHERE account_id = :account_id
853             AND key_scope = :key_scope
854             AND (receiver_flags & :require_flags) = :require_flags
855             AND (receiver_flags & :omit_flags) = 0
856             AND exposed_at_height IS NOT NULL
857             ORDER BY exposed_at_height DESC, diversifier_index_be DESC
858             LIMIT 1",
859            named_params![
860                ":account_id": account.internal_id().0,
861                ":key_scope": KeyScope::EXTERNAL.encode(),
862                ":require_flags": require_flags.bits(),
863                ":omit_flags": omit_flags.bits(),
864            ],
865            |row| Ok((row.get(0)?, row.get(1)?)),
866        )
867        .optional()?;
868
869    addr.map(|(addr_str, di_vec)| {
870        let diversifier_index = decode_diversifier_index_be(&di_vec)?;
871        Address::decode(params, &addr_str)
872            .ok_or_else(|| {
873                SqliteClientError::CorruptedData("Not a valid Zcash recipient address".to_owned())
874            })
875            .and_then(|addr| match addr {
876                Address::Unified(ua) => Ok(ua),
877                _ => Err(SqliteClientError::CorruptedData(format!(
878                    "Addresses table contains {} which is not a unified address",
879                    addr_str,
880                ))),
881            })
882            .map(|addr| (addr, diversifier_index))
883    })
884    .transpose()
885}

887/// Adds the given external address and diversifier index to the addresses table.
888///
889/// Returns the primary key identifier for the newly-inserted address.
890///
891/// ## Parameters
892/// - `account_id`: The account that the address was generated for.
893/// - `diversifier_index`: The diversifier index used to generate the address.
894/// - `address`: The unified address itself.
895/// - `exposed_at_height`: The block height at the earliest time that the address may have been
896///   exposed to a user, assuming a single generator of addresses.
897/// - `force_update_address`: If this argument is set to `true`, an address has already been
898///   inserted for the given account and diversifier index, and the `exposed_at_height` column
899///   is currently `NULL` (i.e. the address at this diversifier index has not yet been exposed)
900///   then the value of the `address` column will be replaced with the provided address.
901pub(crate) fn upsert_address<P: consensus::Parameters>(
902    conn: &rusqlite::Connection,
903    params: &P,
904    account_id: AccountRef,
905    diversifier_index: DiversifierIndex,
906    address: &UnifiedAddress,
907    exposed_at_height: Option<BlockHeight>,
908    force_update_address: bool,
909) -> Result<AddressRef, SqliteClientError> {
910    // the diversifier index is stored in big-endian order to allow sorting
911    let di_be = encode_diversifier_index_be(diversifier_index);
912
913    // If a force-update was requested, check whether an address has previously been exposed for
914    // this diversifier index. If so, and if that address differs from the given address, return an
915    // error.
916    if force_update_address {
917        let previously_exposed_as = conn
918            .query_row(
919                "SELECT address, exposed_at_height
920                 FROM addresses
921                 WHERE account_id = :account_id
922                 AND diversifier_index_be = :diversifier_index_be
923                 AND key_scope = :key_scope",
924                named_params![
925                    ":account_id": account_id.0,
926                    ":diversifier_index_be": di_be,
927                    ":key_scope": KeyScope::EXTERNAL.encode(),
928                ],
929                |row| {
930                    let address = row.get::<_, String>("address")?;
931                    let exposed_at = row.get::<_, Option<u32>>("exposed_at_height")?;
932                    Ok(exposed_at.map(|_| address))
933                },
934            )
935            .optional()?
936            .flatten()
937            .map(|addr_str| UnifiedAddress::decode(params, &addr_str))
938            .transpose()
939            .map_err(SqliteClientError::CorruptedData)?;
940
941        match previously_exposed_as {
942            Some(addr) if &addr != address => {
943                return Err(SqliteClientError::DiversifierIndexReuse(
944                    diversifier_index,
945                    Box::new(addr),
946                ));
947            }
948            _ => (),
949        }
950    }
951
952    let mut stmt = conn.prepare_cached(
953        "INSERT INTO addresses (
954            account_id,
955            diversifier_index_be,
956            key_scope,
957            address,
958            transparent_child_index,
959            cached_transparent_receiver_address,
960            exposed_at_height,
961            receiver_flags
962        )
963        VALUES (
964            :account_id,
965            :diversifier_index_be,
966            :key_scope,
967            :address,
968            :transparent_child_index,
969            :cached_transparent_receiver_address,
970            :exposed_at_height,
971            :receiver_flags
972        )
973        ON CONFLICT (account_id, diversifier_index_be, key_scope) DO UPDATE
974        SET exposed_at_height = COALESCE(
975                MIN(exposed_at_height, :exposed_at_height),
976                exposed_at_height,
977                :exposed_at_height
978            ),
979            address = IIF(
980                exposed_at_height IS NULL AND :force_update_address,
981                :address,
982                address
983            ),
984            receiver_flags = IIF(
985                exposed_at_height IS NULL AND :force_update_address,
986                :receiver_flags,
987                receiver_flags
988            )
989        RETURNING id",
990    )?;
991
992    #[cfg(feature = "transparent-inputs")]
993    let (transparent_child_index, cached_taddr) = {
994        let idx = NonHardenedChildIndex::try_from(diversifier_index)
995            .ok()
996            .map(|i| i.index());
997
998        // This upholds the `transparent_index_consistency` check on the `addresses` table.
999        match (idx, address.transparent()) {
1000            (Some(idx), Some(r)) => Ok((Some(idx), Some(r.encode(params)))),
1001            (_, None) => Ok((None, None)),
1002            (None, Some(addr)) => Err(SqliteClientError::AddressNotRecognized(*addr)),
1003        }
1004    }?;
1005
1006    #[cfg(not(feature = "transparent-inputs"))]
1007    let (transparent_child_index, cached_taddr): (Option<u32>, Option<String>) = (None, None);
1008
1009    stmt.query_row(
1010        named_params![
1011            ":account_id": account_id.0,
1012            // the diversifier index is stored in big-endian order to allow sorting
1013            ":diversifier_index_be": &di_be,
1014            ":key_scope": KeyScope::EXTERNAL.encode(),
1015            ":address": &address.encode(params),
1016            ":transparent_child_index": transparent_child_index,
1017            ":cached_transparent_receiver_address": &cached_taddr,
1018            ":exposed_at_height": exposed_at_height.map(u32::from),
1019            ":force_update_address": force_update_address,
1020            ":receiver_flags": ReceiverFlags::from(address).bits()
1021        ],
1022        |row| row.get(0).map(AddressRef),
1023    )
1024    .map_err(SqliteClientError::from)
1025}

1240    account_index: zip32::AccountId,
1241) -> Result<Option<Account>, SqliteClientError> {
1242    let mut stmt = conn.prepare(
1243        "SELECT id, name, key_source, uuid, ufvk, birthday_height
1244         FROM accounts
1245         WHERE hd_seed_fingerprint = :hd_seed_fingerprint
1246         AND hd_account_index = :hd_account_index",
1247    )?;
1248
1249    let mut accounts = stmt.query_and_then::<_, SqliteClientError, _, _>(
1250        named_params![
1251            ":hd_seed_fingerprint": seed_fp.to_bytes(),
1252            ":hd_account_index": u32::from(account_index),
1253        ],
1254        |row| {
1255            let account_id = AccountRef(row.get("id")?);
1256            let account_name = row.get("name")?;
1257            let key_source = row.get("key_source")?;
1258            let account_uuid = AccountUuid(row.get("uuid")?);
1259            let ufvk = match row.get::<_, Option<String>>("ufvk")? {
1260                None => Err(SqliteClientError::CorruptedData(format!(
1261                    "Missing unified full viewing key for derived account {}",
1262                    account_uuid.0,
1263                ))),
1264                Some(ufvk_str) => UnifiedFullViewingKey::decode(params, &ufvk_str).map_err(|e| {
1265                    SqliteClientError::CorruptedData(format!(
1266                        "Could not decode unified full viewing key for account {}: {}",
1267                        account_uuid.0, e
1268                    ))
1269                }),
1270            }?;
1271            let birthday = BlockHeight::from(row.get::<_, u32>("birthday_height")?);
1272
1273            Ok(Account {
1274                id: account_id,
1275                name: account_name,
1276                uuid: account_uuid,
1277                kind: AccountSource::Derived {
1278                    derivation: Zip32Derivation::new(*seed_fp, account_index),
1279                    key_source,
1280                },
1281                viewing_key: ViewingKey::Full(Box::new(ufvk)),
1282                birthday,
1283            })
1284        },
1285    )?;
1286
1287    accounts.next().transpose()
1288}

1336fn estimate_tree_size<P: consensus::Parameters>(
1337    conn: &rusqlite::Connection,
1338    params: &P,
1339    shielded_protocol: ShieldedProtocol,
1340    pool_activation_height: BlockHeight,
1341    chain_tip_height: BlockHeight,
1342) -> Result<Option<u64>, SqliteClientError> {
1343    let (table_prefix, _, shard_height) = table_constants(shielded_protocol)?;
1344
1345    // Estimate the size of the tree by linear extrapolation from available
1346    // data closest to the chain tip.
1347    //
1348    // - If we have scanned blocks within the incomplete subtree, and we know
1349    //   the tree size for the end of the most recent scanned range, then we
1350    //   extrapolate from the start of the incomplete subtree:
1351    //
1352    //         subtree
1353    //         /     \
1354    //       /         \
1355    //     /             \
1356    //   /                 \
1357    //   |<--------->|  |
1358    //     | scanned |  tip
1359    //           last_scanned
1360    //
1361    //
1362    //             subtree
1363    //             /     \
1364    //           /         \
1365    //         /             \
1366    //       /                 \
1367    //       |<------->|    |
1368    //   |   scanned   |    tip
1369    //             last_scanned
1370    //
1371    // - If we don't have scanned blocks within the incomplete subtree, or we
1372    //   don't know the tree size, then we extrapolate from the block-width of
1373    //   the last complete subtree.
1374    //
1375    // This avoids having a sharp discontinuity in the progress percentages
1376    // shown to users, and gets more accurate the closer to the chain tip we
1377    // have scanned.
1378    //
1379    // TODO: it would be nice to be able to reliably have the size of the
1380    // commitment tree at the chain tip without having to have scanned that
1381    // block.
1382
1383    // Get the tree size at the last scanned height, if known.
1384    let last_scanned = block_max_scanned(conn, params)?.and_then(|last_scanned| {
1385        match shielded_protocol {
1386            ShieldedProtocol::Sapling => last_scanned.sapling_tree_size(),
1387            #[cfg(feature = "orchard")]
1388            ShieldedProtocol::Orchard => last_scanned.orchard_tree_size(),
1389            #[cfg(not(feature = "orchard"))]
1390            ShieldedProtocol::Orchard => None,
1391        }
1392        .map(|tree_size| (last_scanned.block_height(), u64::from(tree_size)))
1393    });
1394
1395    // Get the last completed subtree.
1396    let last_completed_subtree = conn
1397        .query_row(
1398            &format!(
1399                "SELECT shard_index, subtree_end_height
1400                 FROM {table_prefix}_tree_shards
1401                 WHERE subtree_end_height IS NOT NULL
1402                 ORDER BY shard_index DESC
1403                 LIMIT 1"
1404            ),
1405            [],
1406            |row| {
1407                Ok((
1408                    incrementalmerkletree::Address::from_parts(
1409                        incrementalmerkletree::Level::new(shard_height),
1410                        row.get(0)?,
1411                    ),
1412                    BlockHeight::from_u32(row.get(1)?),
1413                ))
1414            },
1415        )
1416        // `None` if we have no subtree roots yet.
1417        .optional()?;
1418
1419    let result = if let Some((last_completed_subtree, last_completed_subtree_end)) =
1420        last_completed_subtree
1421    {
1422        // If we know the tree size at the last scanned height, and that
1423        // height is within the incomplete subtree, extrapolate.
1424        let tip_tree_size = last_scanned.and_then(|(last_scanned, last_scanned_tree_size)| {
1425            (last_scanned > last_completed_subtree_end)
1426                .then(|| {
1427                    let scanned_notes = last_scanned_tree_size
1428                        - u64::from(last_completed_subtree.position_range_end());
1429                    let scanned_range = u64::from(last_scanned - last_completed_subtree_end);
1430                    let unscanned_range = u64::from(chain_tip_height - last_scanned);
1431
1432                    (scanned_notes * unscanned_range)
1433                        .checked_div(scanned_range)
1434                        .map(|extrapolated_unscanned_notes| {
1435                            last_scanned_tree_size + extrapolated_unscanned_notes
1436                        })
1437                })
1438                .flatten()
1439        });
1440
1441        if let Some(tree_size) = tip_tree_size {
1442            Some(tree_size)
1443        } else if let Some(second_to_last_completed_subtree_end) = last_completed_subtree
1444            .index()
1445            .checked_sub(1)
1446            .and_then(|subtree_index| {
1447                conn.query_row(
1448                    &format!(
1449                        "SELECT subtree_end_height
1450                         FROM {table_prefix}_tree_shards
1451                         WHERE shard_index = :shard_index"
1452                    ),
1453                    named_params! {":shard_index": subtree_index},
1454                    |row| Ok(row.get::<_, Option<_>>(0)?.map(BlockHeight::from_u32)),
1455                )
1456                .transpose()
1457            })
1458            .transpose()?
1459        {
1460            let notes_in_complete_subtrees = u64::from(last_completed_subtree.position_range_end());
1461
1462            let subtree_notes = 1 << shard_height;
1463            let subtree_range =
1464                u64::from(last_completed_subtree_end - second_to_last_completed_subtree_end);
1465            let unscanned_range = u64::from(chain_tip_height - last_completed_subtree_end);
1466
1467            (subtree_notes * unscanned_range)
1468                .checked_div(subtree_range)
1469                .map(|extrapolated_incomplete_subtree_notes| {
1470                    notes_in_complete_subtrees + extrapolated_incomplete_subtree_notes
1471                })
1472        } else {
1473            // There's only one completed subtree; its start height must
1474            // be the activation height for this shielded protocol.
1475            let subtree_notes = 1 << shard_height;
1476
1477            let subtree_range = u64::from(last_completed_subtree_end - pool_activation_height);
1478            let unscanned_range = u64::from(chain_tip_height - last_completed_subtree_end);
1479
1480            (subtree_notes * unscanned_range)
1481                .checked_div(subtree_range)
1482                .map(|extrapolated_incomplete_subtree_notes| {
1483                    subtree_notes + extrapolated_incomplete_subtree_notes
1484                })
1485        }
1486    } else {
1487        // If there are no completed subtrees, but we have scanned some blocks, we can still
1488        // interpolate based upon the tree size as of the last scanned block. Here, since we
1489        // don't have any subtree data to draw on, we will interpolate based on the number of
1490        // blocks since the pool activation height
1491        last_scanned.and_then(|(last_scanned_height, last_scanned_tree_size)| {
1492            let subtree_range = u64::from(last_scanned_height - pool_activation_height);
1493            let unscanned_range = u64::from(chain_tip_height - last_scanned_height);
1494
1495            (last_scanned_tree_size * unscanned_range)
1496                .checked_div(subtree_range)
1497                .map(|extrapolated_incomplete_subtree_notes| {
1498                    last_scanned_tree_size + extrapolated_incomplete_subtree_notes
1499                })
1500        })
1501    };
1502
1503    Ok(result)
1504}

1506#[allow(clippy::too_many_arguments)]
1507fn subtree_scan_progress<P: consensus::Parameters>(
1508    conn: &rusqlite::Connection,
1509    params: &P,
1510    shielded_protocol: ShieldedProtocol,
1511    pool_activation_height: BlockHeight,
1512    birthday_height: BlockHeight,
1513    recover_until_height: Option<BlockHeight>,
1514    fully_scanned_height: Option<BlockHeight>,
1515    chain_tip_height: BlockHeight,
1516) -> Result<Option<Progress>, SqliteClientError> {
1517    let (table_prefix, output_count_col, shard_height) = table_constants(shielded_protocol)?;
1518
1519    let mut stmt_scanned_count_until = conn.prepare_cached(&format!(
1520        "SELECT SUM({output_count_col})
1521        FROM blocks
1522        WHERE :start_height <= height AND height < :end_height",
1523    ))?;
1524    let mut stmt_scanned_count_from = conn.prepare_cached(&format!(
1525        "SELECT SUM({output_count_col})
1526        FROM blocks
1527        WHERE :start_height <= height",
1528    ))?;
1529    let mut stmt_start_tree_size = conn.prepare_cached(&format!(
1530        "SELECT MAX({table_prefix}_commitment_tree_size - {output_count_col})
1531        FROM blocks
1532        WHERE height <= :start_height",
1533    ))?;
1534    let mut stmt_start_tree_size_at = conn.prepare_cached(&format!(
1535        "SELECT {table_prefix}_commitment_tree_size - {output_count_col}
1536        FROM blocks
1537        WHERE height = :start_height",
1538    ))?;
1539    let mut stmt_end_tree_size_at = conn.prepare_cached(&format!(
1540        "SELECT {table_prefix}_commitment_tree_size
1541        FROM blocks
1542        WHERE height = :height",
1543    ))?;
1544
1545    if fully_scanned_height == Some(chain_tip_height) {
1546        // Compute the total blocks scanned since the wallet birthday on either side of
1547        // the recover-until height.
1548        let recover = match recover_until_height {
1549            Some(end_height) => stmt_scanned_count_until.query_row(
1550                named_params! {
1551                    ":start_height": u32::from(birthday_height),
1552                    ":end_height": u32::from(end_height),
1553                },
1554                |row| {
1555                    let recovered = row.get::<_, Option<u64>>(0)?;
1556                    Ok(recovered.map(|n| Ratio::new(n, n)))
1557                },
1558            )?,
1559            None => {
1560                // If none of the wallet's accounts have a recover-until height, then there
1561                // is no recovery phase for the wallet, and therefore the denominator in the
1562                // resulting ratio (the number of notes in the recovery range) is zero.
1563                Some(Ratio::new(0, 0))
1564            }
1565        };
1566
1567        let scan = stmt_scanned_count_from.query_row(
1568            named_params! {
1569                ":start_height": u32::from(
1570                    recover_until_height.unwrap_or(birthday_height)
1571                ),
1572            },
1573            |row| {
1574                let scanned = row.get::<_, Option<u64>>(0)?;
1575                Ok(scanned.map(|n| Ratio::new(n, n)))
1576            },
1577        )?;
1578
1579        Ok(scan.map(|scan| Progress::new(scan, recover)))
1580    } else {
1581        // In case we didn't have information about the tree size at the birthday height,
1582        // get the tree size from a nearby subtree. It's fine for this to be approximate;
1583        // it just alters the magnitude of recovery progress a bit.
1584        let mut get_tree_size_near = |as_of: BlockHeight| {
1585            let size_from_blocks = stmt_start_tree_size
1586                .query_row(named_params![":start_height": u32::from(as_of)], |row| {
1587                    row.get::<_, Option<u64>>(0)
1588                })
1589                .optional()?
1590                .flatten();
1591
1592            let size_from_subtree_roots = || {
1593                conn.query_row(
1594                    &format!(
1595                        "SELECT MIN(shard_index)
1596                             FROM {table_prefix}_tree_shards
1597                             WHERE subtree_end_height >= :start_height
1598                             OR subtree_end_height IS NULL",
1599                    ),
1600                    named_params! {
1601                        ":start_height": u32::from(as_of),
1602                    },
1603                    |row| {
1604                        let min_tree_size = row
1605                            .get::<_, Option<u64>>(0)?
1606                            .map(|min_idx| min_idx << shard_height);
1607                        Ok(min_tree_size)
1608                    },
1609                )
1610                .optional()
1611                .map(|opt| opt.flatten())
1612            };
1613
1614            match size_from_blocks {
1615                Some(size) => Ok(Some(size)),
1616                None => size_from_subtree_roots(),
1617            }
1618        };
1619
1620        // Get the starting note commitment tree size from the wallet birthday, or failing that
1621        // from the blocks table.
1622        let birthday_size = match conn
1623            .query_row(
1624                &format!(
1625                    "SELECT birthday_{table_prefix}_tree_size
1626                     FROM accounts
1627                     WHERE birthday_height = :birthday_height",
1628                ),
1629                named_params![":birthday_height": u32::from(birthday_height)],
1630                |row| row.get::<_, Option<u64>>(0),
1631            )
1632            .optional()?
1633            .flatten()
1634        {
1635            Some(tree_size) => Some(tree_size),
1636            // If we don't have an explicit birthday tree size, find something nearby.
1637            None => get_tree_size_near(birthday_height)?,
1638        };
1639
1640        // If we've scanned the block at the chain tip, we know how many notes are currently in the
1641        // tree.
1642        let tip_tree_size = match stmt_end_tree_size_at
1643            .query_row(
1644                named_params! {":height": u32::from(chain_tip_height)},
1645                |row| row.get::<_, Option<u64>>(0),
1646            )
1647            .optional()?
1648            .flatten()
1649        {
1650            Some(tree_size) => Some(tree_size),
1651            None => estimate_tree_size(
1652                conn,
1653                params,
1654                shielded_protocol,
1655                pool_activation_height,
1656                chain_tip_height,
1657            )?,
1658        };
1659
1660        // Get the note commitment tree size as of the start of the recover-until height.
1661        // The outer option indicates whether or not we have recover-until height information;
1662        // the inner option indicates whether or not we were able to obtain a tree size given
1663        // the recover-until height.
1664        let recover_until_size: Option<Option<u64>> =
1665            recover_until_height
1666                .map(|h| {
1667                    let size_from_blocks = stmt_start_tree_size_at
1668                        .query_row(named_params![":start_height": u32::from(h)], |row| {
1669                            row.get::<_, Option<u64>>(0)
1670                        })
1671                        .optional()?
1672                        .flatten();
1673
1674                    match size_from_blocks {
1675                        // We know the tree size as of the start of the recover-until height.
1676                        Some(size) => Ok::<_, SqliteClientError>(Some(size)),
1677
1678                        // If the recover-until height is equal to the chain tip height,
1679                        // then this is almost certainly a newly-recovered wallet, and all
1680                        // progress can count as recovery progress. Approximate the size
1681                        // of the tree at the start of the block as equal to the size of
1682                        // the tree at the end of the block; the scan progress will show
1683                        // as 0/0 which is fine.
1684                        None if h == chain_tip_height => Ok(tip_tree_size),
1685
1686                        // Linearly extrapolate a tree size between the nearest two bounds
1687                        // we have.
1688                        // TODO: Use a closer lower bound if available.
1689                        None => Ok(birthday_size.zip(tip_tree_size).and_then(
1690                            |(lower_size, upper_size)| {
1691                                let total_notes = upper_size - lower_size;
1692                                let total_range = u64::from(chain_tip_height - birthday_height);
1693                                let recovery_range = u64::from(h - birthday_height);
1694
1695                                (total_notes * recovery_range).checked_div(total_range).map(
1696                                    |extrapolated_recovery_notes| {
1697                                        lower_size + extrapolated_recovery_notes
1698                                    },
1699                                )
1700                            },
1701                        )),
1702                    }
1703                })
1704                .transpose()?;
1705
1706        // Count the total outputs scanned so far on the birthday side of the recover-until height.
1707        let recovered_count = recover_until_height
1708            .map(|end_height| {
1709                stmt_scanned_count_until.query_row(
1710                    named_params! {
1711                        ":start_height": u32::from(birthday_height),
1712                        ":end_height": u32::from(end_height),
1713                    },
1714                    |row| row.get::<_, Option<u64>>(0),
1715                )
1716            })
1717            .transpose()?;
1718
1719        let recover = recovered_count
1720            .zip(recover_until_size)
1721            .map(|(recovered, end_size)| {
1722                birthday_size.zip(end_size).map(|(start_size, end_size)| {
1723                    Ratio::new(recovered.unwrap_or(0), end_size - start_size)
1724                })
1725            })
1726            // If none of the wallet's accounts have a recover-until height, then there
1727            // is no recovery phase for the wallet, and therefore the denominator in the
1728            // resulting ratio (the number of notes in the recovery range) is zero.
1729            .unwrap_or_else(|| Some(Ratio::new(0, 0)));
1730
1731        let scan = {
1732            // Count the total outputs scanned so far on the chain tip side of the
1733            // recover-until height.
1734            let scanned_count = stmt_scanned_count_from.query_row(
1735                named_params![":start_height": u32::from(recover_until_height.unwrap_or(birthday_height))],
1736                |row| row.get::<_, Option<u64>>(0),
1737            )?;
1738
1739            recover_until_size
1740                .unwrap_or(birthday_size)
1741                .zip(tip_tree_size)
1742                .map(|(start_size, tip_tree_size)| {
1743                    Ratio::new(scanned_count.unwrap_or(0), tip_tree_size - start_size)
1744                })
1745        };
1746
1747        Ok(scan.map(|scan| Progress::new(scan, recover)))
1748    }
1749}

1802/// Returns the spendable balance for the account at the specified height.
1803///
1804/// This may be used to obtain a balance that ignores notes that have been detected so recently
1805/// that they are not yet spendable, or for which it is not yet possible to construct witnesses.
1806///
1807/// `min_confirmations` can be 0, but that case is currently treated identically to
1808/// `min_confirmations == 1` for both shielded and transparent TXOs. This behaviour
1809/// may change in the future.
1810#[tracing::instrument(skip(tx, params, progress))]
1811pub(crate) fn get_wallet_summary<P: consensus::Parameters>(
1812    tx: &rusqlite::Transaction,
1813    params: &P,
1814    min_confirmations: u32,
1815    progress: &impl ProgressEstimator,
1816) -> Result<Option<WalletSummary<AccountUuid>>, SqliteClientError> {
1817    let chain_tip_height = match chain_tip_height(tx)? {
1818        Some(h) => h,
1819        None => {
1820            return Ok(None);
1821        }
1822    };
1823
1824    let birthday_height = match wallet_birthday(tx)? {
1825        Some(h) => h,
1826        None => {
1827            return Ok(None);
1828        }
1829    };
1830
1831    let recover_until_height = recover_until_height(tx)?;
1832
1833    let fully_scanned_height = block_fully_scanned(tx, params)?.map(|m| m.block_height());
1834    let summary_height = (chain_tip_height + 1).saturating_sub(std::cmp::max(min_confirmations, 1));
1835
1836    let sapling_progress = progress.sapling_scan_progress(
1837        tx,
1838        params,
1839        birthday_height,
1840        recover_until_height,
1841        fully_scanned_height,
1842        chain_tip_height,
1843    )?;
1844
1845    #[cfg(feature = "orchard")]
1846    let orchard_progress = progress.orchard_scan_progress(
1847        tx,
1848        params,
1849        birthday_height,
1850        recover_until_height,
1851        fully_scanned_height,
1852        chain_tip_height,
1853    )?;
1854    #[cfg(not(feature = "orchard"))]
1855    let orchard_progress: Option<Progress> = None;
1856
1857    // Treat Sapling and Orchard outputs as having the same cost to scan.
1858    let progress = sapling_progress
1859        .as_ref()
1860        .zip(orchard_progress.as_ref())
1861        .map(|(s, o)| {
1862            Progress::new(
1863                Ratio::new(
1864                    s.scan().numerator() + o.scan().numerator(),
1865                    s.scan().denominator() + o.scan().denominator(),
1866                ),
1867                s.recovery()
1868                    .zip(o.recovery())
1869                    .map(|(s, o)| {
1870                        Ratio::new(
1871                            s.numerator() + o.numerator(),
1872                            s.denominator() + o.denominator(),
1873                        )
1874                    })
1875                    .or_else(|| s.recovery())
1876                    .or_else(|| o.recovery()),
1877            )
1878        })
1879        .or(sapling_progress)
1880        .or(orchard_progress);
1881
1882    let progress = match progress {
1883        Some(p) => p,
1884        None => return Ok(None),
1885    };
1886
1887    let mut stmt_accounts = tx.prepare_cached("SELECT uuid FROM accounts")?;
1888    let mut account_balances = stmt_accounts
1889        .query([])?
1890        .and_then(|row| {
1891            Ok::<_, SqliteClientError>((AccountUuid(row.get::<_, Uuid>(0)?), AccountBalance::ZERO))
1892        })
1893        .collect::<Result<HashMap<AccountUuid, AccountBalance>, _>>()?;
1894
1895    fn count_notes<F>(
1896        tx: &rusqlite::Transaction,
1897        summary_height: BlockHeight,
1898        account_balances: &mut HashMap<AccountUuid, AccountBalance>,
1899        table_prefix: &'static str,
1900        with_pool_balance: F,
1901    ) -> Result<(), SqliteClientError>
1902    where
1903        F: Fn(&mut AccountBalance, Zatoshis, Zatoshis, Zatoshis) -> Result<(), SqliteClientError>,
1904    {
1905        // If the shard containing the summary height contains any unscanned ranges that start below or
1906        // including that height, none of our shielded balance is currently spendable.
1907        #[tracing::instrument(skip_all)]
1908        fn is_any_spendable(
1909            conn: &rusqlite::Connection,
1910            summary_height: BlockHeight,
1911            table_prefix: &'static str,
1912        ) -> Result<bool, SqliteClientError> {
1913            conn.query_row(
1914                &format!(
1915                    "SELECT NOT EXISTS(
1916                         SELECT 1 FROM v_{table_prefix}_shard_unscanned_ranges
1917                         WHERE :summary_height
1918                            BETWEEN subtree_start_height
1919                            AND IFNULL(subtree_end_height, :summary_height)
1920                         AND block_range_start <= :summary_height
1921                     )"
1922                ),
1923                named_params![":summary_height": u32::from(summary_height)],
1924                |row| row.get::<_, bool>(0),
1925            )
1926            .map_err(|e| e.into())
1927        }
1928
1929        let any_spendable = is_any_spendable(tx, summary_height, table_prefix)?;
1930        let mut stmt_select_notes = tx.prepare_cached(&format!(
1931            "SELECT a.uuid, n.value, n.is_change, scan_state.max_priority, t.block
1932             FROM {table_prefix}_received_notes n
1933             JOIN accounts a ON a.id = n.account_id
1934             JOIN transactions t ON t.id_tx = n.tx
1935             LEFT OUTER JOIN v_{table_prefix}_shards_scan_state scan_state
1936                ON n.commitment_tree_position >= scan_state.start_position
1937                AND n.commitment_tree_position < scan_state.end_position_exclusive
1938             WHERE (
1939                t.block IS NOT NULL -- the receiving tx is mined
1940                OR t.expiry_height IS NULL -- the receiving tx will not expire
1941                OR t.expiry_height >= :summary_height -- the receiving tx is unexpired
1942             )
1943             -- and the received note is unspent
1944             AND n.id NOT IN (
1945               SELECT {table_prefix}_received_note_id
1946               FROM {table_prefix}_received_note_spends
1947               JOIN transactions t ON t.id_tx = transaction_id
1948               WHERE t.block IS NOT NULL -- the spending transaction is mined
1949               OR t.expiry_height IS NULL -- the spending tx will not expire
1950               OR t.expiry_height > :summary_height -- the spending tx is unexpired
1951             )"
1952        ))?;
1953
1954        let mut rows =
1955            stmt_select_notes.query(named_params![":summary_height": u32::from(summary_height)])?;
1956        while let Some(row) = rows.next()? {
1957            let account = AccountUuid(row.get::<_, Uuid>(0)?);
1958
1959            let value_raw = row.get::<_, i64>(1)?;
1960            let value = Zatoshis::from_nonnegative_i64(value_raw).map_err(|_| {
1961                SqliteClientError::CorruptedData(format!(
1962                    "Negative received note value: {}",
1963                    value_raw
1964                ))
1965            })?;
1966
1967            let is_change = row.get::<_, bool>(2)?;
1968
1969            // If `max_priority` is null, this means that the note is not positioned; the note
1970            // will not be spendable, so we assign the scan priority to `ChainTip` as a priority
1971            // that is greater than `Scanned`
1972            let max_priority_raw = row.get::<_, Option<i64>>(3)?;
1973            let max_priority = max_priority_raw.map_or_else(
1974                || Ok(ScanPriority::ChainTip),
1975                |raw| {
1976                    parse_priority_code(raw).ok_or_else(|| {
1977                        SqliteClientError::CorruptedData(format!(
1978                            "Priority code {} not recognized.",
1979                            raw
1980                        ))
1981                    })
1982                },
1983            )?;
1984
1985            let received_height = row.get::<_, Option<u32>>(4)?.map(BlockHeight::from);
1986
1987            let is_spendable = any_spendable
1988                && received_height.iter().any(|h| h <= &summary_height)
1989                && max_priority <= ScanPriority::Scanned;
1990
1991            let is_pending_change =
1992                is_change && received_height.iter().all(|h| h > &summary_height);
1993
1994            let (spendable_value, change_pending_confirmation, value_pending_spendability) = {
1995                let zero = Zatoshis::ZERO;
1996                if is_spendable {
1997                    (value, zero, zero)
1998                } else if is_pending_change {
1999                    (zero, value, zero)
2000                } else {
2001                    (zero, zero, value)
2002                }
2003            };
2004
2005            if let Some(balances) = account_balances.get_mut(&account) {
2006                with_pool_balance(
2007                    balances,
2008                    spendable_value,
2009                    change_pending_confirmation,
2010                    value_pending_spendability,
2011                )?;
2012            }
2013        }
2014        Ok(())
2015    }
2016
2017    #[cfg(feature = "orchard")]
2018    {
2019        let orchard_trace = tracing::info_span!("orchard_balances").entered();
2020        count_notes(
2021            tx,
2022            summary_height,
2023            &mut account_balances,
2024            ORCHARD_TABLES_PREFIX,
2025            |balances, spendable_value, change_pending_confirmation, value_pending_spendability| {
2026                balances.with_orchard_balance_mut::<_, SqliteClientError>(|bal| {
2027                    bal.add_spendable_value(spendable_value)?;
2028                    bal.add_pending_change_value(change_pending_confirmation)?;
2029                    bal.add_pending_spendable_value(value_pending_spendability)?;
2030                    Ok(())
2031                })
2032            },
2033        )?;
2034        drop(orchard_trace);
2035    }
2036
2037    let sapling_trace = tracing::info_span!("sapling_balances").entered();
2038    count_notes(
2039        tx,
2040        summary_height,
2041        &mut account_balances,
2042        SAPLING_TABLES_PREFIX,
2043        |balances, spendable_value, change_pending_confirmation, value_pending_spendability| {
2044            balances.with_sapling_balance_mut::<_, SqliteClientError>(|bal| {
2045                bal.add_spendable_value(spendable_value)?;
2046                bal.add_pending_change_value(change_pending_confirmation)?;
2047                bal.add_pending_spendable_value(value_pending_spendability)?;
2048                Ok(())
2049            })
2050        },
2051    )?;
2052    drop(sapling_trace);
2053
2054    #[cfg(feature = "transparent-inputs")]
2055    transparent::add_transparent_account_balances(
2056        tx,
2057        chain_tip_height + 1,
2058        min_confirmations,
2059        &mut account_balances,
2060    )?;
2061
2062    // The approach used here for Sapling and Orchard subtree indexing was a quick hack
2063    // that has not yet been replaced. TODO: Make less hacky.
2064    // https://github.com/zcash/librustzcash/issues/1249
2065    let next_sapling_subtree_index = {
2066        let shard_store =
2067            SqliteShardStore::<_, ::sapling::Node, SAPLING_SHARD_HEIGHT>::from_connection(
2068                tx,
2069                SAPLING_TABLES_PREFIX,
2070            )?;
2071
2072        // The last shard will be incomplete, and we want the next range to overlap with
2073        // the last complete shard, so return the index of the second-to-last shard root.
2074        shard_store
2075            .get_shard_roots()
2076            .map_err(ShardTreeError::Storage)?
2077            .iter()
2078            .rev()
2079            .nth(1)
2080            .map(|addr| addr.index())
2081            .unwrap_or(0)
2082    };
2083
2084    #[cfg(feature = "orchard")]
2085    let next_orchard_subtree_index = {
2086        let shard_store = SqliteShardStore::<
2087            _,
2088            ::orchard::tree::MerkleHashOrchard,
2089            ORCHARD_SHARD_HEIGHT,
2090        >::from_connection(tx, ORCHARD_TABLES_PREFIX)?;
2091
2092        // The last shard will be incomplete, and we want the next range to overlap with
2093        // the last complete shard, so return the index of the second-to-last shard root.
2094        shard_store
2095            .get_shard_roots()
2096            .map_err(ShardTreeError::Storage)?
2097            .iter()
2098            .rev()
2099            .nth(1)
2100            .map(|addr| addr.index())
2101            .unwrap_or(0)
2102    };
2103
2104    let summary = WalletSummary::new(
2105        account_balances,
2106        chain_tip_height,
2107        fully_scanned_height.unwrap_or(birthday_height - 1),
2108        progress,
2109        next_sapling_subtree_index,
2110        #[cfg(feature = "orchard")]
2111        next_orchard_subtree_index,
2112    );
2113
2114    Ok(Some(summary))
2115}

2158/// Looks up a transaction by its [`TxId`].
2159///
2160/// Returns the decoded transaction, along with the block height that was used in its decoding.
2161/// This is either the block height at which the transaction was mined, or the expiry height if the
2162/// wallet created the transaction but the transaction has not yet been mined from the perspective
2163/// of the wallet.
2164pub(crate) fn get_transaction<P: Parameters>(
2165    conn: &rusqlite::Connection,
2166    params: &P,
2167    txid: TxId,
2168) -> Result<Option<(BlockHeight, Transaction)>, SqliteClientError> {
2169    conn.query_row(
2170        "SELECT raw, block, expiry_height FROM transactions
2171        WHERE txid = ?",
2172        [txid.as_ref()],
2173        |row| {
2174            let h: Option<u32> = row.get(1)?;
2175            let expiry: Option<u32> = row.get(2)?;
2176            Ok((
2177                row.get::<_, Vec<u8>>(0)?,
2178                h.map(BlockHeight::from),
2179                expiry.map(BlockHeight::from),
2180            ))
2181        },
2182    )
2183    .optional()?
2184    .map(|(tx_bytes, block_height, expiry_height)| {
2185        // We need to provide a consensus branch ID so that pre-v5 `Transaction` structs
2186        // (which don't commit directly to one) can store it internally.
2187        // - If the transaction is mined, we use the block height to get the correct one.
2188        // - If the transaction is unmined and has a cached non-zero expiry height, we use
2189        //   that (relying on the invariant that a transaction can't be mined across a network
2190        //   upgrade boundary, so the expiry height must be in the same epoch).
2191        // - Otherwise, we use a placeholder for the initial transaction parse (as the
2192        //   consensus branch ID is not used there), and then either use its non-zero expiry
2193        //   height or return an error.
2194        if let Some(height) =
2195            block_height.or_else(|| expiry_height.filter(|h| h > &BlockHeight::from(0)))
2196        {
2197            Transaction::read(&tx_bytes[..], BranchId::for_height(params, height))
2198                .map(|t| (height, t))
2199                .map_err(SqliteClientError::from)
2200        } else {
2201            let tx_data = Transaction::read(&tx_bytes[..], BranchId::Sprout)
2202                .map_err(SqliteClientError::from)?
2203                .into_data();
2204
2205            let expiry_height = tx_data.expiry_height();
2206            if expiry_height > BlockHeight::from(0) {
2207                TransactionData::from_parts(
2208                    tx_data.version(),
2209                    BranchId::for_height(params, expiry_height),
2210                    tx_data.lock_time(),
2211                    expiry_height,
2212                    tx_data.transparent_bundle().cloned(),
2213                    tx_data.sprout_bundle().cloned(),
2214                    tx_data.sapling_bundle().cloned(),
2215                    tx_data.orchard_bundle().cloned(),
2216                )
2217                .freeze()
2218                .map(|t| (expiry_height, t))
2219                .map_err(SqliteClientError::from)
2220            } else {
2221                Err(SqliteClientError::CorruptedData(
2222                    "Consensus branch ID not known, cannot parse this transaction until it is mined"
2223                        .to_string(),
2224                ))
2225            }
2226        }
2227    })
2228    .transpose()
2229}

2400pub(crate) fn get_target_and_anchor_heights(
2401    conn: &rusqlite::Connection,
2402    min_confirmations: NonZeroU32,
2403) -> Result<Option<(BlockHeight, BlockHeight)>, rusqlite::Error> {
2404    match chain_tip_height(conn)? {
2405        Some(chain_tip_height) => {
2406            let sapling_anchor_height = get_max_checkpointed_height(
2407                conn,
2408                SAPLING_TABLES_PREFIX,
2409                chain_tip_height,
2410                min_confirmations,
2411            )?;
2412
2413            #[cfg(feature = "orchard")]
2414            let orchard_anchor_height = get_max_checkpointed_height(
2415                conn,
2416                ORCHARD_TABLES_PREFIX,
2417                chain_tip_height,
2418                min_confirmations,
2419            )?;
2420
2421            #[cfg(not(feature = "orchard"))]
2422            let orchard_anchor_height: Option<BlockHeight> = None;
2423
2424            let anchor_height = sapling_anchor_height
2425                .zip(orchard_anchor_height)
2426                .map(|(s, o)| std::cmp::min(s, o))
2427                .or(sapling_anchor_height)
2428                .or(orchard_anchor_height);
2429
2430            Ok(anchor_height.map(|h| (chain_tip_height + 1, h)))
2431        }
2432        None => Ok(None),
2433    }
2434}

2436fn parse_block_metadata<P: consensus::Parameters>(
2437    _params: &P,
2438    row: (BlockHeight, Vec<u8>, Option<u32>, Vec<u8>, Option<u32>),
2439) -> Result<BlockMetadata, SqliteClientError> {
2440    let (block_height, hash_data, sapling_tree_size_opt, sapling_tree, _orchard_tree_size_opt) =
2441        row;
2442    let sapling_tree_size = sapling_tree_size_opt.map_or_else(|| {
2443        if sapling_tree == BLOCK_SAPLING_FRONTIER_ABSENT {
2444            Err(SqliteClientError::CorruptedData("One of either the Sapling tree size or the legacy Sapling commitment tree must be present.".to_owned()))
2445        } else {
2446            // parse the legacy commitment tree data
2447            read_commitment_tree::<
2448                ::sapling::Node,
2449                _,
2450                { ::sapling::NOTE_COMMITMENT_TREE_DEPTH },
2451            >(Cursor::new(sapling_tree))
2452            .map(|tree| tree.size().try_into().unwrap())
2453            .map_err(SqliteClientError::from)

2457    let block_hash = BlockHash::try_from_slice(&hash_data).ok_or_else(|| {
2458        SqliteClientError::from(io::Error::new(
2459            io::ErrorKind::InvalidData,
2460            format!("Invalid block hash length: {}", hash_data.len()),

2462    })?;

2465        block_height,
2466        block_hash,
2467        Some(sapling_tree_size),

2469        if _params
2470            .activation_height(NetworkUpgrade::Nu5)
2471            .iter()
2472            .any(|nu5_activation| &block_height >= nu5_activation)
2473        {
2474            _orchard_tree_size_opt
2475        } else {
2476            Some(0)
2477        },
2478    ))
2479}

2512#[tracing::instrument(skip_all)]
2513pub(crate) fn block_fully_scanned<P: consensus::Parameters>(
2514    conn: &rusqlite::Connection,
2515    params: &P,
2516) -> Result<Option<BlockMetadata>, SqliteClientError> {
2517    if let Some(birthday_height) = wallet_birthday(conn)? {
2518        // We assume that the only way we get a contiguous range of block heights in the `blocks` table
2519        // starting with the birthday block, is if all scanning operations have been performed on those
2520        // blocks. This holds because the `blocks` table is only altered by `WalletDb::put_blocks` via
2521        // `put_block`, and the effective combination of intra-range linear scanning and the nullifier
2522        // map ensures that we discover all wallet-related information within the contiguous range.
2523        //
2524        // We also assume that every contiguous range of block heights in the `blocks` table has a
2525        // single matching entry in the `scan_queue` table with priority "Scanned". This requires no
2526        // bugs in the scan queue update logic, which we have had before. However, a bug here would
2527        // mean that we return a more conservative fully-scanned height, which likely just causes a
2528        // performance regression.
2529        //
2530        // The fully-scanned height is therefore the last height that falls within the first range in
2531        // the scan queue with priority "Scanned".
2532        let calc_fully_scanned_height = |row: &rusqlite::Row| {
2533            let block_range_start = BlockHeight::from_u32(row.get(0)?);
2534            let block_range_end = BlockHeight::from_u32(row.get(1)?);
2535
2536            // If the start of the earliest scanned range is greater than
2537            // the birthday height, then there is an unscanned range between
2538            // the wallet birthday and that range, so there is no fully
2539            // scanned height.
2540            Ok(if block_range_start <= birthday_height {
2541                // Scan ranges are end-exclusive.
2542                Some(block_range_end - 1)
2543            } else {
2544                None
2545            })
2546        };
2547        let fully_scanned_height = match conn
2548            .query_row(
2549                "SELECT block_range_start, block_range_end
2550                FROM scan_queue
2551                WHERE priority = :priority
2552                ORDER BY block_range_start ASC
2553                LIMIT 1",
2554                named_params![":priority": priority_code(&ScanPriority::Scanned)],
2555                calc_fully_scanned_height,
2556            )
2557            .optional()?
2558        {
2559            Some(Some(h)) => h,
2560            _ => return Ok(None),
2561        };
2562
2563        block_metadata(conn, params, fully_scanned_height)
2564    } else {
2565        Ok(None)
2566    }
2567}

2646pub(crate) fn store_transaction_to_be_sent<P: consensus::Parameters>(
2647    conn: &rusqlite::Transaction,
2648    params: &P,
2649    sent_tx: &SentTransaction<AccountUuid>,
2650) -> Result<(), SqliteClientError> {
2651    let tx_ref = put_tx_data(
2652        conn,
2653        sent_tx.tx(),
2654        Some(sent_tx.fee_amount()),
2655        Some(sent_tx.created()),
2656        Some(sent_tx.target_height()),
2657    )?;
2658
2659    let mut detectable_via_scanning = false;
2660
2661    // Mark notes as spent.
2662    //
2663    // This locks the notes so they aren't selected again by a subsequent call to
2664    // create_spend_to_address() before this transaction has been mined (at which point the notes
2665    // get re-marked as spent).
2666    //
2667    // Assumes that create_spend_to_address() will never be called in parallel, which is a
2668    // reasonable assumption for a light client such as a mobile phone.
2669    if let Some(bundle) = sent_tx.tx().sapling_bundle() {
2670        detectable_via_scanning = true;
2671        for spend in bundle.shielded_spends() {
2672            sapling::mark_sapling_note_spent(conn, tx_ref, spend.nullifier())?;
2673        }
2674    }
2675    if let Some(_bundle) = sent_tx.tx().orchard_bundle() {
2676        #[cfg(feature = "orchard")]
2677        {
2678            detectable_via_scanning = true;
2679            for action in _bundle.actions() {
2680                orchard::mark_orchard_note_spent(conn, tx_ref, action.nullifier())?;
2681            }
2682        }
2683
2684        #[cfg(not(feature = "orchard"))]
2685        panic!("Sent a transaction with Orchard Actions without `orchard` enabled?");
2686    }
2687
2688    #[cfg(feature = "transparent-inputs")]
2689    for utxo_outpoint in sent_tx.utxos_spent() {
2690        transparent::mark_transparent_utxo_spent(conn, tx_ref, utxo_outpoint)?;
2691    }
2692
2693    for output in sent_tx.outputs() {
2694        insert_sent_output(conn, params, tx_ref, *sent_tx.account_id(), output)?;
2695
2696        match output.recipient() {
2697            Recipient::External {
2698                recipient_address: _addr,
2699                output_pool: _pool,
2700                ..
2701            } => {
2702                // Nothing to do for external recipients.
2703            }
2704            Recipient::InternalAccount {
2705                receiving_account,
2706                note,
2707                ..
2708            } => match note.as_ref() {
2709                Note::Sapling(note) => {
2710                    sapling::put_received_note(
2711                        conn,
2712                        params,
2713                        &DecryptedOutput::new(
2714                            output.output_index(),
2715                            note.clone(),
2716                            *receiving_account,
2717                            output
2718                                .memo()
2719                                .map_or_else(MemoBytes::empty, |memo| memo.clone()),
2720                            TransferType::WalletInternal,
2721                        ),
2722                        tx_ref,
2723                        Some(sent_tx.target_height()),
2724                        None,
2725                    )?;
2726                }
2727                #[cfg(feature = "orchard")]
2728                Note::Orchard(note) => {
2729                    orchard::put_received_note(
2730                        conn,
2731                        params,
2732                        &DecryptedOutput::new(
2733                            output.output_index(),
2734                            *note,
2735                            *receiving_account,
2736                            output
2737                                .memo()
2738                                .map_or_else(MemoBytes::empty, |memo| memo.clone()),
2739                            TransferType::WalletInternal,
2740                        ),
2741                        tx_ref,
2742                        Some(sent_tx.target_height()),
2743                        None,
2744                    )?;
2745                }
2746            },
2747            #[cfg(feature = "transparent-inputs")]
2748            Recipient::EphemeralTransparent {
2749                ephemeral_address,
2750                outpoint,
2751                ..
2752            } => {
2753                // First check to verify that creation of this output does not result in reuse of
2754                // an ephemeral address.
2755                transparent::check_ephemeral_address_reuse(conn, params, ephemeral_address)?;
2756
2757                transparent::put_transparent_output(
2758                    conn,
2759                    &params,
2760                    outpoint,
2761                    &TxOut {
2762                        value: output.value(),
2763                        script_pubkey: ephemeral_address.script(),
2764                    },
2765                    None,
2766                    ephemeral_address,
2767                    true,
2768                )?;
2769            }
2770        }
2771    }
2772
2773    // Add the transaction to the set to be queried for transaction status. This is only necessary
2774    // at present for fully transparent transactions, because any transaction with a shielded
2775    // component will be detected via ordinary chain scanning and/or nullifier checking.
2776    if !detectable_via_scanning {
2777        queue_tx_retrieval(conn, std::iter::once(sent_tx.tx().txid()), None)?;
2778    }
2779
2780    Ok(())
2781}

2783pub(crate) fn set_transaction_status(
2784    conn: &rusqlite::Transaction,
2785    txid: TxId,
2786    status: TransactionStatus,
2787) -> Result<(), SqliteClientError> {
2788    // It is safe to unconditionally delete the request from `tx_retrieval_queue` below (both in
2789    // the expired case and the case where it has been mined), because we already have all the data
2790    // we need about this transaction:
2791    // * if the status is being set in response to a `GetStatus` request, we know that we already
2792    //   have the transaction data (`GetStatus` requests are only generated if we already have that
2793    //   data)
2794    // * if it is being set in response to an `Enhancement` request, we know that the status must
2795    //   be `TxidNotRecognized` because otherwise the transaction data should have been provided to
2796    //   the backend directly instead of calling `set_transaction_status`
2797    //
2798    // In general `Enhancement` requests are only generated in response to situations where a
2799    // transaction has already been mined - either the transaction was detected by scanning the
2800    // chain of `CompactBlock` values, or was discovered by walking backward from the inputs of a
2801    // transparent transaction; in the case that a transaction was read from the mempool, complete
2802    // transaction data will have been available and the only question that we are concerned with
2803    // is whether that transaction ends up being mined or expires.
2804    match status {
2805        TransactionStatus::TxidNotRecognized | TransactionStatus::NotInMainChain => {
2806            // If the transaction is now expired, remove it from the retrieval queue.
2807            if let Some(chain_tip) = chain_tip_height(conn)? {
2808                conn.execute(
2809                    "DELETE FROM tx_retrieval_queue WHERE txid IN (
2810                        SELECT txid FROM transactions
2811                        WHERE txid = :txid AND expiry_height < :chain_tip_minus_lookahead
2812                    )",
2813                    named_params![
2814                        ":txid": txid.as_ref(),
2815                        ":chain_tip_minus_lookahead": u32::from(chain_tip).saturating_sub(VERIFY_LOOKAHEAD)
2816                    ],
2817                )?;
2818            }
2819        }
2820        TransactionStatus::Mined(height) => {
2821            // The transaction has been mined, so we can set its mined height, associate it with
2822            // the appropriate block, and remove it from the retrieval queue.
2823            let sql_args = named_params![
2824                ":txid": txid.as_ref(),
2825                ":height": u32::from(height)
2826            ];
2827
2828            conn.execute(
2829                "UPDATE transactions
2830                 SET mined_height = :height
2831                 WHERE txid = :txid",
2832                sql_args,
2833            )?;
2834
2835            conn.execute(
2836                "UPDATE transactions
2837                 SET block = blocks.height
2838                 FROM blocks
2839                 WHERE txid = :txid
2840                 AND blocks.height = :height",
2841                sql_args,
2842            )?;
2843
2844            notify_tx_retrieved(conn, txid)?;
2845        }
2846    }
2847
2848    Ok(())
2849}

2851/// Truncates the database to at most the given height.
2852///
2853/// If the requested height is greater than or equal to the height of the last scanned
2854/// block, this function does nothing.
2855///
2856/// This should only be executed inside a transactional context.
2857///
2858/// Returns the block height to which the database was truncated.
2859pub(crate) fn truncate_to_height<P: consensus::Parameters>(
2860    conn: &rusqlite::Transaction,
2861    params: &P,
2862    #[cfg(feature = "transparent-inputs")] gap_limits: &GapLimits,
2863    max_height: BlockHeight,
2864) -> Result<BlockHeight, SqliteClientError> {
2865    // Determine a checkpoint to which we can rewind, if any.
2866    #[cfg(not(feature = "orchard"))]
2867    let truncation_height_query = r#"
2868        SELECT MAX(height) FROM blocks
2869        JOIN sapling_tree_checkpoints ON checkpoint_id = blocks.height
2870        WHERE blocks.height <= :block_height
2871    "#;
2872
2873    #[cfg(feature = "orchard")]
2874    let truncation_height_query = r#"
2875        SELECT MAX(height) FROM blocks
2876        JOIN sapling_tree_checkpoints sc ON sc.checkpoint_id = blocks.height
2877        JOIN orchard_tree_checkpoints oc ON oc.checkpoint_id = blocks.height
2878        WHERE blocks.height <= :block_height
2879    "#;
2880
2881    let truncation_height = conn
2882        .query_row(
2883            truncation_height_query,
2884            named_params! {":block_height": u32::from(max_height)},
2885            |row| row.get::<_, Option<u32>>(0),
2886        )
2887        .optional()?
2888        .flatten()
2889        .map_or_else(
2890            || {
2891                // If we don't have a checkpoint at a height less than or equal to the requested
2892                // truncation height, query for the minimum height to which it's possible for us to
2893                // truncate so that we can report it to the caller.
2894                #[cfg(not(feature = "orchard"))]
2895                let min_checkpoint_height_query =
2896                    "SELECT MIN(checkpoint_id) FROM sapling_tree_checkpoints";
2897                #[cfg(feature = "orchard")]
2898                let min_checkpoint_height_query = "SELECT MIN(sc.checkpoint_id)
2899                     FROM sapling_tree_checkpoints sc
2900                     JOIN orchard_tree_checkpoints oc
2901                     ON oc.checkpoint_id = sc.checkpoint_id";
2902
2903                let min_truncation_height = conn
2904                    .query_row(min_checkpoint_height_query, [], |row| {
2905                        row.get::<_, Option<u32>>(0)
2906                    })
2907                    .optional()?
2908                    .flatten()
2909                    .map(BlockHeight::from);
2910
2911                Err(SqliteClientError::RequestedRewindInvalid {
2912                    safe_rewind_height: min_truncation_height,
2913                    requested_height: max_height,
2914                })
2915            },
2916            |h| Ok(BlockHeight::from(h)),
2917        )?;
2918
2919    let last_scanned_height = conn.query_row("SELECT MAX(height) FROM blocks", [], |row| {
2920        let h = row.get::<_, Option<u32>>(0)?;
2921
2922        Ok(h.map_or_else(
2923            || {
2924                params
2925                    .activation_height(NetworkUpgrade::Sapling)
2926                    .expect("Sapling activation height must be available.")
2927                    - 1
2928            },
2929            BlockHeight::from,
2930        ))
2931    })?;
2932
2933    // Delete from the scanning queue any range with a start height greater than the
2934    // truncation height, and then truncate any remaining range by setting the end
2935    // equal to the truncation height + 1. This sets our view of the chain tip back
2936    // to the retained height.
2937    conn.execute(
2938        "DELETE FROM scan_queue
2939        WHERE block_range_start >= :new_end_height",
2940        named_params![":new_end_height": u32::from(truncation_height + 1)],
2941    )?;
2942    conn.execute(
2943        "UPDATE scan_queue
2944        SET block_range_end = :new_end_height
2945        WHERE block_range_end > :new_end_height",
2946        named_params![":new_end_height": u32::from(truncation_height + 1)],
2947    )?;
2948
2949    // Mark transparent utxos as un-mined. Since the TXO is now not mined, it would ideally be
2950    // considered to have been returned to the mempool; it _might_ be spendable in this state, but
2951    // we must also set its max_observed_unspent_height field to NULL because the transaction may
2952    // be rendered entirely invalid by a reorg that alters anchor(s) used in constructing shielded
2953    // spends in the transaction.
2954    conn.execute(
2955        "UPDATE transparent_received_outputs
2956         SET max_observed_unspent_height = CASE WHEN tx.mined_height <= :height THEN :height ELSE NULL END
2957         FROM transactions tx
2958         WHERE tx.id_tx = transaction_id
2959         AND max_observed_unspent_height > :height",
2960        named_params![":height": u32::from(truncation_height)],
2961    )?;
2962
2963    // Un-mine transactions. This must be done outside of the last_scanned_height check because
2964    // transaction entries may be created as a consequence of receiving transparent TXOs.
2965    conn.execute(
2966        "UPDATE transactions
2967         SET block = NULL, mined_height = NULL, tx_index = NULL
2968         WHERE mined_height > :height",
2969        named_params![":height": u32::from(truncation_height)],
2970    )?;
2971
2972    // If we're removing scanned blocks, we need to truncate the note commitment tree and remove
2973    // affected block records from the database.
2974    if truncation_height < last_scanned_height {
2975        // Truncate the note commitment trees
2976        let mut wdb = WalletDb {
2977            conn: SqlTransaction(conn),
2978            params: params.clone(),
2979            clock: (),
2980            rng: (),
2981            #[cfg(feature = "transparent-inputs")]
2982            gap_limits: *gap_limits,
2983        };
2984        wdb.with_sapling_tree_mut(|tree| {
2985            tree.truncate_to_checkpoint(&truncation_height)?;
2986            Ok::<_, SqliteClientError>(())
2987        })?;
2988        #[cfg(feature = "orchard")]
2989        wdb.with_orchard_tree_mut(|tree| {
2990            tree.truncate_to_checkpoint(&truncation_height)?;
2991            Ok::<_, SqliteClientError>(())
2992        })?;
2993
2994        // Do not delete sent notes; this can contain data that is not recoverable
2995        // from the chain. Wallets must continue to operate correctly in the
2996        // presence of stale sent notes that link to unmined transactions.
2997        // Also, do not delete received notes; they may contain memo data that is
2998        // not recoverable; balance APIs must ensure that un-mined received notes
2999        // do not count towards spendability or transaction balalnce.
3000
3001        // Now that they aren't depended on, delete un-mined blocks.
3002        conn.execute(
3003            "DELETE FROM blocks WHERE height > ?",
3004            [u32::from(truncation_height)],
3005        )?;
3006
3007        // Delete from the nullifier map any entries with a locator referencing a block
3008        // height greater than the truncation height.
3009        conn.execute(
3010            "DELETE FROM tx_locator_map
3011            WHERE block_height > :block_height",
3012            named_params![":block_height": u32::from(truncation_height)],
3013        )?;
3014    }
3015
3016    Ok(truncation_height)
3017}

3959/// Inserts the given entries into the nullifier map.
3960///
3961/// Returns an error if the new entries conflict with existing ones. This indicates either
3962/// corrupted data, or that a reorg has occurred and the caller needs to repair the wallet
3963/// state with [`truncate_to_height`].
3964pub(crate) fn insert_nullifier_map<N: AsRef<[u8]>>(
3965    conn: &rusqlite::Transaction<'_>,
3966    block_height: BlockHeight,
3967    spend_pool: ShieldedProtocol,
3968    new_entries: &[(TxId, u16, Vec<N>)],
3969) -> Result<(), SqliteClientError> {
3970    let mut stmt_select_tx_locators = conn.prepare_cached(
3971        "SELECT block_height, tx_index, txid
3972        FROM tx_locator_map
3973        WHERE (block_height = :block_height AND tx_index = :tx_index) OR txid = :txid",
3974    )?;
3975    let mut stmt_insert_tx_locator = conn.prepare_cached(
3976        "INSERT INTO tx_locator_map
3977        (block_height, tx_index, txid)
3978        VALUES (:block_height, :tx_index, :txid)",
3979    )?;
3980    let mut stmt_insert_nullifier_mapping = conn.prepare_cached(
3981        "INSERT INTO nullifier_map
3982        (spend_pool, nf, block_height, tx_index)
3983        VALUES (:spend_pool, :nf, :block_height, :tx_index)
3984        ON CONFLICT (spend_pool, nf) DO UPDATE
3985        SET block_height = :block_height,
3986            tx_index = :tx_index",
3987    )?;
3988
3989    for (txid, tx_index, nullifiers) in new_entries {
3990        let tx_args = named_params![
3991            ":block_height": u32::from(block_height),
3992            ":tx_index": tx_index,
3993            ":txid": txid.as_ref(),
3994        ];
3995
3996        // We cannot use an upsert here, because we use the tx locator as the foreign key
3997        // in `nullifier_map` instead of `txid` for database size efficiency. If an insert
3998        // into `tx_locator_map` were to conflict, we would need the resulting update to
3999        // cascade into `nullifier_map` as either:
4000        // - an update (if a transaction moved within a block), or
4001        // - a deletion (if the locator now points to a different transaction).
4002        //
4003        // `ON UPDATE` has `CASCADE` to always update, but has no deletion option. So we
4004        // instead set `ON UPDATE RESTRICT` on the foreign key relation, and require the
4005        // caller to manually rewind the database in this situation.
4006        let locator = stmt_select_tx_locators
4007            .query_map(tx_args, |row| {
4008                Ok((
4009                    BlockHeight::from_u32(row.get(0)?),
4010                    row.get::<_, u16>(1)?,
4011                    TxId::from_bytes(row.get(2)?),
4012                ))
4013            })?
4014            .try_fold(None, |acc, row| -> Result<_, SqliteClientError> {
4015                match (acc, row?) {
4016                    (None, rhs) => Ok(Some(Some(rhs))),
4017                    // If there was more than one row, then due to the uniqueness
4018                    // constraints on the `tx_locator_map` table, all of the rows conflict
4019                    // with the locator being inserted.
4020                    (Some(_), _) => Ok(Some(None)),
4021                }
4022            })?;
4023
4024        match locator {
4025            // If the locator in the table matches the one being inserted, do nothing.
4026            Some(Some(loc)) if loc == (block_height, *tx_index, *txid) => (),
4027            // If the locator being inserted would conflict, report it.
4028            Some(_) => Err(SqliteClientError::DbError(rusqlite::Error::SqliteFailure(
4029                rusqlite::ffi::Error::new(rusqlite::ffi::SQLITE_CONSTRAINT),
4030                Some("UNIQUE constraint failed: tx_locator_map.block_height, tx_locator_map.tx_index".into()),
4031            )))?,
4032            // If the locator doesn't exist, insert it.
4033            None => stmt_insert_tx_locator.execute(tx_args).map(|_| ())?,
4034        }
4035
4036        for nf in nullifiers {
4037            // Here it is okay to use an upsert, because per above we've confirmed that
4038            // the locator points to the same transaction.
4039            let nf_args = named_params![
4040                ":spend_pool": pool_code(PoolType::Shielded(spend_pool)),
4041                ":nf": nf.as_ref(),
4042                ":block_height": u32::from(block_height),
4043                ":tx_index": tx_index,
4044            ];
4045            stmt_insert_nullifier_mapping.execute(nf_args)?;
4046        }
4047    }
4048
4049    Ok(())
4050}

4123pub(crate) fn get_block_range(
4124    conn: &rusqlite::Connection,
4125    protocol: ShieldedProtocol,
4126    commitment_tree_address: incrementalmerkletree::Address,
4127) -> Result<Option<Range<BlockHeight>>, SqliteClientError> {
4128    let prefix = match protocol {
4129        ShieldedProtocol::Sapling => "sapling",
4130        ShieldedProtocol::Orchard => "orchard",
4131    };
4132    let mut stmt = conn.prepare_cached(&format!(
4133        "SELECT MIN(height), MAX(height), MAX({prefix}_commitment_tree_size)
4134         FROM blocks
4135         WHERE {prefix}_commitment_tree_size BETWEEN :min_tree_size AND :max_tree_size"
4136    ))?;
4137
4138    stmt.query_row(
4139        // BETWEEN is inclusive on both ends. However, we are comparing commitment tree sizes
4140        // to commitment tree positions, so we must add one to the start, and we do not subtract
4141        // one from the end.
4142        named_params! {
4143            ":min_tree_size": u64::from(commitment_tree_address.position_range_start()) + 1,
4144            ":max_tree_size": u64::from(commitment_tree_address.position_range_end()),
4145        },
4146        |row| {
4147            // The first block to be scanned is known to contain the start of the address range in
4148            // question because the tree size we compared against is measured as of the end of the
4149            // block.
4150            let min_height = row.get::<_, Option<u32>>(0)?.map(BlockHeight::from_u32);
4151            let max_height_inclusive = row.get::<_, Option<u32>>(1)?.map(BlockHeight::from_u32);
4152            let end_offset = row.get::<_, Option<u64>>(2)?.map(|max_height_tree_size| {
4153                // If the tree size at the end of the max-height block is less than the
4154                // end-exclusive maximum position of the address range, this means that the end of
4155                // the subtree referred to by that address is somewhere in the next block, so we
4156                // need to rescan an extra block to ensure that we have observed all of the note
4157                // commitments that aggregate up to that address.
4158                if max_height_tree_size < u64::from(commitment_tree_address.position_range_end()) {
4159                    1
4160                } else {
4161                    0
4162                }
4163            });
4164
4165            Ok(min_height
4166                .zip(max_height_inclusive)
4167                .zip(end_offset)
4168                .map(|((min, max_inclusive), offset)| min..(max_inclusive + offset + 1)))
4169        },
4170    )
4171    .map_err(SqliteClientError::from)
4172}

4367    fn check_block_fully_scanned<DsF: DataStoreFactory>(dsf: DsF) {
4368        let mut st = TestBuilder::new()
4369            .with_data_store_factory(dsf)
4370            .with_block_cache(BlockCache::new())
4371            .with_account_from_sapling_activation(BlockHash([0; 32]))
4372            .build();
4373
4374        let block_fully_scanned = |st: &TestState<_, DsF::DataStore, _>| {
4375            st.wallet()
4376                .block_fully_scanned()
4377                .unwrap()
4378                .map(|meta| meta.block_height())
4379        };
4380
4381        // A fresh wallet should have no fully-scanned block.
4382        assert_eq!(block_fully_scanned(&st), None);
4383
4384        // Scan a block above the wallet's birthday height.
4385        let not_our_key = ExtendedSpendingKey::master(&[]).to_diversifiable_full_viewing_key();
4386        let not_our_value = Zatoshis::const_from_u64(10000);
4387        let start_height = st.sapling_activation_height();
4388        let _ = st.generate_block_at(
4389            start_height,
4390            BlockHash([0; 32]),
4391            &[FakeCompactOutput::new(
4392                &not_our_key,
4393                AddressType::DefaultExternal,
4394                not_our_value,
4395            )],
4396            0,
4397            0,
4398            false,
4399        );
4400        let (mid_height, _, _) =
4401            st.generate_next_block(&not_our_key, AddressType::DefaultExternal, not_our_value);
4402        let (end_height, _, _) =
4403            st.generate_next_block(&not_our_key, AddressType::DefaultExternal, not_our_value);
4404
4405        // Scan the last block first
4406        st.scan_cached_blocks(end_height, 1);
4407
4408        // The wallet should still have no fully-scanned block, as no scanned block range
4409        // overlaps the wallet's birthday.
4410        assert_eq!(block_fully_scanned(&st), None);
4411
4412        // Scan the block at the wallet's birthday height.
4413        st.scan_cached_blocks(start_height, 1);
4414
4415        // The fully-scanned height should now be that of the scanned block.
4416        assert_eq!(block_fully_scanned(&st), Some(start_height));
4417
4418        // Scan the block in between the two previous blocks.
4419        st.scan_cached_blocks(mid_height, 1);
4420
4421        // The fully-scanned height should now be the latest block, as the two disjoint
4422        // ranges have been connected.
4423        assert_eq!(block_fully_scanned(&st), Some(end_height));
4424    }
