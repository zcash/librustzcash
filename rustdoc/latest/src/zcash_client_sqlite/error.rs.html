20/// The primary error type for the SQLite wallet backend.
21#[derive(Debug)]
22pub enum SqliteClientError {
23    /// Decoding of a stored value from its serialized form has failed.
24    CorruptedData(String),
25
26    /// An error occurred decoding a protobuf message.
27    Protobuf(prost::DecodeError),
28
29    /// The rcm value for a note cannot be decoded to a valid JubJub point.
30    InvalidNote,
31
32    /// Illegal attempt to reinitialize an already-initialized wallet database.
33    TableNotEmpty,
34
35    /// A Zcash key or address decoding error
36    DecodingError(ParseError),
37
38    /// An error produced in legacy transparent address derivation
39    #[cfg(feature = "transparent-inputs")]
40    TransparentDerivation(bip32::Error),
41
42    /// An error encountered in decoding a transparent address from its
43    /// serialized form.
44    #[cfg(feature = "transparent-inputs")]
45    TransparentAddress(TransparentCodecError),
46
47    /// Wrapper for rusqlite errors.
48    DbError(rusqlite::Error),
49
50    /// Wrapper for errors from the IO subsystem
51    Io(std::io::Error),
52
53    /// A received memo cannot be interpreted as a UTF-8 string.
54    InvalidMemo(zcash_protocol::memo::Error),
55
56    /// An attempt to update block data would overwrite the current hash for a block with a
57    /// different hash. This indicates that a required rewind was not performed.
58    BlockConflict(BlockHeight),
59
60    /// A range of blocks provided to the database as a unit was non-sequential
61    NonSequentialBlocks,
62
63    /// A requested rewind would violate invariants of the storage layer. The payload returned with
64    /// this error is (safe rewind height, requested height). If no safe rewind height can be
65    /// determined, the safe rewind height member will be `None`.
66    RequestedRewindInvalid {
67        safe_rewind_height: Option<BlockHeight>,
68        requested_height: BlockHeight,
69    },
70
71    /// An error occurred in generating a Zcash address.
72    AddressGeneration(AddressGenerationError),
73
74    /// The account for which information was requested does not belong to the wallet.
75    AccountUnknown,
76
77    /// The account being added collides with an existing account in the wallet with the given ID.
78    /// The collision can be on the seed and ZIP-32 account index, or a shared FVK component.
79    AccountCollision(AccountUuid),
80
81    /// The account was imported, and ZIP-32 derivation information is not known for it.
82    UnknownZip32Derivation,
83
84    /// An error occurred deriving a spending key from a seed and a ZIP-32 account index.
85    KeyDerivationError(zip32::AccountId),
86
87    /// An error occurred while processing an account due to a failure in deriving the account's keys.
88    BadAccountData(String),
89
90    /// A caller attempted to construct a new account with an invalid ZIP 32 account identifier.
91    Zip32AccountIndexOutOfRange,
92
93    /// The address associated with a record being inserted was not recognized as
94    /// belonging to the wallet.
95    #[cfg(feature = "transparent-inputs")]
96    AddressNotRecognized(TransparentAddress),
97
98    /// An error occurred in inserting data into or accessing data from one of the wallet's note
99    /// commitment trees.
100    CommitmentTree(ShardTreeError<commitment_tree::Error>),
101
102    /// The block at the specified height was not available from the block cache.
103    CacheMiss(BlockHeight),
104
105    /// The height of the chain was not available; a call to [`WalletWrite::update_chain_tip`] is
106    /// required before the requested operation can succeed.
107    ///
108    /// [`WalletWrite::update_chain_tip`]:
109    /// zcash_client_backend::data_api::WalletWrite::update_chain_tip
110    ChainHeightUnknown,
111
112    /// Unsupported pool type
113    UnsupportedPoolType(PoolType),
114
115    /// An error occurred in computing wallet balance
116    BalanceError(BalanceError),
117
118    /// A note selection query contained an invalid constant or was otherwise not supported.
119    NoteFilterInvalid(NoteFilter),
120
121    /// The proposal cannot be constructed until transactions with previously reserved
122    /// ephemeral address outputs have been mined. The parameters are the account UUID and
123    /// the index that could not safely be reserved.
124    #[cfg(feature = "transparent-inputs")]
125    ReachedGapLimit(AccountUuid, u32),
126
127    /// An ephemeral address would be reused. The parameters are the address in string
128    /// form, and the txid of the earliest transaction in which it is known to have been
129    /// used.
130    #[cfg(feature = "transparent-inputs")]
131    EphemeralAddressReuse(String, TxId),
132}
133
134impl error::Error for SqliteClientError {
135    fn source(&self) -> Option<&(dyn error::Error + 'static)> {
136        match &self {
137            SqliteClientError::InvalidMemo(e) => Some(e),
138            SqliteClientError::DbError(e) => Some(e),
139            SqliteClientError::Io(e) => Some(e),
140            SqliteClientError::BalanceError(e) => Some(e),
141            SqliteClientError::AddressGeneration(e) => Some(e),
142            _ => None,
143        }
144    }
145}
146
147impl fmt::Display for SqliteClientError {
148    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
149        match &self {
150            SqliteClientError::CorruptedData(reason) => {
151                write!(f, "Data DB is corrupted: {}", reason)
152            }
153            SqliteClientError::Protobuf(e) => write!(f, "Failed to parse protobuf-encoded record: {}", e),
154            SqliteClientError::InvalidNote => write!(f, "Invalid note"),
155            SqliteClientError::RequestedRewindInvalid { safe_rewind_height,  requested_height } => write!(
156                f,
157                "A rewind for your wallet may only target height {} or greater; the requested height was {}.",
158                safe_rewind_height.map_or("<unavailable>".to_owned(), |h0| format!("{}", h0)),
159               requested_height
160            ),
161            SqliteClientError::DecodingError(e) => write!(f, "{}", e),
162            #[cfg(feature = "transparent-inputs")]
163            SqliteClientError::TransparentDerivation(e) => write!(f, "{:?}", e),
164            #[cfg(feature = "transparent-inputs")]
165            SqliteClientError::TransparentAddress(e) => write!(f, "{}", e),
166            SqliteClientError::TableNotEmpty => write!(f, "Table is not empty"),
167            SqliteClientError::DbError(e) => write!(f, "{}", e),
168            SqliteClientError::Io(e) => write!(f, "{}", e),
169            SqliteClientError::InvalidMemo(e) => write!(f, "{}", e),
170            SqliteClientError::BlockConflict(h) => write!(f, "A block hash conflict occurred at height {}; rewind required.", u32::from(*h)),
171            SqliteClientError::NonSequentialBlocks => write!(f, "`put_blocks` requires that the provided block range be sequential"),
172            SqliteClientError::AddressGeneration(e) => write!(f, "{}", e),
173            SqliteClientError::AccountUnknown => write!(f, "The account with the given ID does not belong to this wallet."),
174            SqliteClientError::UnknownZip32Derivation => write!(f, "ZIP-32 derivation information is not known for this account."),
175            SqliteClientError::KeyDerivationError(zip32_index) => write!(f, "Key derivation failed for ZIP 32 account index {}", u32::from(*zip32_index)),
176            SqliteClientError::BadAccountData(e) => write!(f, "Failed to add account: {}", e),
177            SqliteClientError::Zip32AccountIndexOutOfRange => write!(f, "ZIP 32 account identifiers must be less than 0x7FFFFFFF."),
178            SqliteClientError::AccountCollision(account_uuid) => write!(f, "An account corresponding to the data provided already exists in the wallet with UUID {account_uuid:?}."),
179            #[cfg(feature = "transparent-inputs")]
180            SqliteClientError::AddressNotRecognized(_) => write!(f, "The address associated with a received txo is not identifiable as belonging to the wallet."),
181            SqliteClientError::CommitmentTree(err) => write!(f, "An error occurred accessing or updating note commitment tree data: {}.", err),
182            SqliteClientError::CacheMiss(height) => write!(f, "Requested height {} does not exist in the block cache.", height),
183            SqliteClientError::ChainHeightUnknown => write!(f, "Chain height unknown; please call `update_chain_tip`"),
184            SqliteClientError::UnsupportedPoolType(t) => write!(f, "Pool type is not currently supported: {}", t),
185            SqliteClientError::BalanceError(e) => write!(f, "Balance error: {}", e),
186            SqliteClientError::NoteFilterInvalid(s) => write!(f, "Could not evaluate filter query: {:?}", s),
187            #[cfg(feature = "transparent-inputs")]
188            SqliteClientError::ReachedGapLimit(account_id, bad_index) => write!(f,
189                "The proposal cannot be constructed until transactions with previously reserved ephemeral address outputs have been mined. \
190                 The ephemeral address in account {account_id:?} at index {bad_index} could not be safely reserved.",
191            ),
192            #[cfg(feature = "transparent-inputs")]
193            SqliteClientError::EphemeralAddressReuse(address_str, txid) => write!(f, "The ephemeral address {address_str} previously used in txid {txid} would be reused."),
194        }
195    }
196}
