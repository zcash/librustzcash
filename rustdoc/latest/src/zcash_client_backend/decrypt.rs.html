21/// An enumeration of the possible relationships a TXO can have to the wallet.
22#[derive(Debug, Copy, Clone, PartialEq, Eq)]
23pub enum TransferType {
24    /// The output was received on one of the wallet's external addresses via decryption using the
25    /// associated incoming viewing key, or at one of the wallet's transparent addresses.
26    Incoming,
27    /// The output was received on one of the wallet's internal-only shielded addresses via trial
28    /// decryption using one of the wallet's internal incoming viewing keys.
29    WalletInternal,
30    /// The output was decrypted using one of the wallet's outgoing viewing keys, or was created
31    /// in a transaction constructed by this wallet.
32    Outgoing,
33}

35/// A decrypted shielded output.
36pub struct DecryptedOutput<Note, AccountId> {
37    index: usize,
38    note: Note,
39    account: AccountId,
40    memo: MemoBytes,
41    transfer_type: TransferType,
42}

44impl<Note, AccountId> DecryptedOutput<Note, AccountId> {
45    pub fn new(
46        index: usize,
47        note: Note,
48        account: AccountId,
49        memo: MemoBytes,
50        transfer_type: TransferType,
51    ) -> Self {
52        Self {
53            index,
54            note,
55            account,
56            memo,
57            transfer_type,
58        }
59    }
60
61    /// The index of the output within the shielded outputs of the Sapling bundle or the actions of
62    /// the Orchard bundle, depending upon the type of [`Self::note`].
63    pub fn index(&self) -> usize {
64        self.index
65    }
66
67    /// The note within the output.
68    pub fn note(&self) -> &Note {
69        &self.note
70    }
71
72    /// The account that decrypted the note.
73    pub fn account(&self) -> &AccountId {
74        &self.account
75    }
76
77    /// The memo bytes included with the note.
78    pub fn memo(&self) -> &MemoBytes {
79        &self.memo
80    }
81
82    /// Returns a [`TransferType`] value that is determined based upon what type of key was used to
83    /// decrypt the transaction.
84    pub fn transfer_type(&self) -> TransferType {
85        self.transfer_type
86    }
87}

104/// Scans a [`Transaction`] for any information that can be decrypted by the set of
105/// [`UnifiedFullViewingKey`]s.
106pub fn decrypt_transaction<'a, P: consensus::Parameters, AccountId: Copy>(
107    params: &P,
108    height: BlockHeight,
109    tx: &'a Transaction,
110    ufvks: &HashMap<AccountId, UnifiedFullViewingKey>,
111) -> DecryptedTransaction<'a, AccountId> {
112    let zip212_enforcement = zip212_enforcement(params, height);
113    let sapling_bundle = tx.sapling_bundle();
114    let sapling_outputs = sapling_bundle
115        .iter()
116        .flat_map(|bundle| {
117            ufvks
118                .iter()
119                .flat_map(|(account, ufvk)| ufvk.sapling().into_iter().map(|dfvk| (*account, dfvk)))
120                .flat_map(|(account, dfvk)| {
121                    let sapling_domain = SaplingDomain::new(zip212_enforcement);
122                    let ivk_external =
123                        PreparedIncomingViewingKey::new(&dfvk.to_ivk(Scope::External));
124                    let ivk_internal =
125                        PreparedIncomingViewingKey::new(&dfvk.to_ivk(Scope::Internal));
126                    let ovk = dfvk.fvk().ovk;
127
128                    bundle
129                        .shielded_outputs()
130                        .iter()
131                        .enumerate()
132                        .flat_map(move |(index, output)| {
133                            try_note_decryption(&sapling_domain, &ivk_external, output)
134                                .map(|ret| (ret, TransferType::Incoming))
135                                .or_else(|| {
136                                    try_note_decryption(&sapling_domain, &ivk_internal, output)
137                                        .map(|ret| (ret, TransferType::WalletInternal))
138                                })
139                                .or_else(|| {
140                                    try_output_recovery_with_ovk(
141                                        &sapling_domain,
142                                        &ovk,
143                                        output,
144                                        output.cv(),
145                                        output.out_ciphertext(),
146                                    )
147                                    .map(|ret| (ret, TransferType::Outgoing))
148                                })
149                                .into_iter()
150                                .map(move |((note, _, memo), transfer_type)| {
151                                    DecryptedOutput::new(
152                                        index,
153                                        note,
154                                        account,
155                                        MemoBytes::from_bytes(&memo).expect("correct length"),
156                                        transfer_type,
157                                    )
158                                })
159                        })
160                })
161        })
162        .collect();
163
164    #[cfg(feature = "orchard")]
165    let orchard_bundle = tx.orchard_bundle();
166    #[cfg(feature = "orchard")]
167    let orchard_outputs = orchard_bundle
168        .iter()
169        .flat_map(|bundle| {
170            ufvks
171                .iter()
172                .flat_map(|(account, ufvk)| ufvk.orchard().into_iter().map(|fvk| (*account, fvk)))
173                .flat_map(|(account, fvk)| {
174                    let ivk_external = orchard::keys::PreparedIncomingViewingKey::new(
175                        &fvk.to_ivk(Scope::External),
176                    );
177                    let ivk_internal = orchard::keys::PreparedIncomingViewingKey::new(
178                        &fvk.to_ivk(Scope::Internal),
179                    );
180                    let ovk = fvk.to_ovk(Scope::External);
181
182                    bundle
183                        .actions()
184                        .iter()
185                        .enumerate()
186                        .flat_map(move |(index, action)| {
187                            let domain = OrchardDomain::for_action(action);
188                            try_note_decryption(&domain, &ivk_external, action)
189                                .map(|ret| (ret, TransferType::Incoming))
190                                .or_else(|| {
191                                    try_note_decryption(&domain, &ivk_internal, action)
192                                        .map(|ret| (ret, TransferType::WalletInternal))
193                                })
194                                .or_else(|| {
195                                    try_output_recovery_with_ovk(
196                                        &domain,
197                                        &ovk,
198                                        action,
199                                        action.cv_net(),
200                                        &action.encrypted_note().out_ciphertext,
201                                    )
202                                    .map(|ret| (ret, TransferType::Outgoing))
203                                })
204                                .into_iter()
205                                .map(move |((note, _, memo), transfer_type)| {
206                                    DecryptedOutput::new(
207                                        index,
208                                        note,
209                                        account,
210                                        MemoBytes::from_bytes(&memo).expect("correct length"),
211                                        transfer_type,
212                                    )
213                                })
214                        })
215                })
216        })
217        .collect();
218
219    DecryptedTransaction::new(
220        Some(height),
221        tx,
222        sapling_outputs,
223        #[cfg(feature = "orchard")]
224        orchard_outputs,
225    )
226}
