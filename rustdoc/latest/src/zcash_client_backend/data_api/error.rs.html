23/// Errors that can occur as a consequence of wallet operations.
24#[derive(Debug)]
25pub enum Error<DataSourceError, CommitmentTreeError, SelectionError, FeeError, ChangeErrT, NoteRefT>
26{
27    /// An error occurred retrieving data from the underlying data source
28    DataSource(DataSourceError),
29
30    /// An error in computations involving the note commitment trees.
31    CommitmentTree(ShardTreeError<CommitmentTreeError>),
32
33    /// An error in note selection
34    NoteSelection(SelectionError),
35
36    /// An error in change selection during transaction proposal construction
37    Change(ChangeError<ChangeErrT, NoteRefT>),
38
39    /// An error in transaction proposal construction
40    Proposal(ProposalError),
41
42    /// The proposal was structurally valid, but tried to do one of these unsupported things:
43    /// * spend a prior shielded output;
44    /// * pay to an output pool for which the corresponding feature is not enabled;
45    /// * pay to a TEX address if the "transparent-inputs" feature is not enabled.
46    ProposalNotSupported,
47
48    /// No account could be found corresponding to a provided ID.
49    AccountIdNotRecognized,
50
51    /// No account could be found corresponding to a provided spending key.
52    KeyNotRecognized,
53
54    /// The given account cannot be used for spending, because it is unable to maintain an
55    /// accurate balance.
56    AccountCannotSpend,
57
58    /// Zcash amount computation encountered an overflow or underflow.
59    BalanceError(BalanceError),
60
61    /// Unable to create a new spend because the wallet balance is not sufficient.
62    InsufficientFunds {
63        available: Zatoshis,
64        required: Zatoshis,
65    },
66
67    /// The wallet must first perform a scan of the blockchain before other
68    /// operations can be performed.
69    ScanRequired,
70
71    /// An error occurred building a new transaction.
72    Builder(builder::Error<FeeError>),
73
74    /// It is forbidden to provide a memo when constructing a transparent output.
75    MemoForbidden,
76
77    /// Attempted to send change to an unsupported pool.
78    ///
79    /// This is indicative of a programming error; execution of a transaction proposal that
80    /// presumes support for the specified pool was performed using an application that does not
81    /// provide such support.
82    UnsupportedChangeType(PoolType),
83
84    /// Attempted to create a spend to an unsupported Unified Address receiver
85    NoSupportedReceivers(Box<UnifiedAddress>),
86
87    /// A proposed transaction cannot be built because it requires spending an input of
88    /// a type for which a key required to construct the transaction is not available.
89    KeyNotAvailable(PoolType),
90
91    /// A note being spent does not correspond to either the internal or external
92    /// full viewing key for an account.
93    NoteMismatch(NoteId),
94
95    /// An error occurred parsing the address from a payment request.
96    Address(ConversionError<&'static str>),
97
98    /// The address associated with a record being inserted was not recognized as
99    /// belonging to the wallet.
100    #[cfg(feature = "transparent-inputs")]
101    AddressNotRecognized(TransparentAddress),
102
103    /// An error occurred while working with PCZTs.
104    #[cfg(feature = "pczt")]
105    Pczt(PcztError),
106}

108/// Errors that can occur while working with PCZTs.
109#[cfg(feature = "pczt")]
110#[derive(Debug)]
111pub enum PcztError {
112    /// An error occurred while building a PCZT.
113    Build,
114
115    /// An error occurred while finalizing the IO of a PCZT.
116    IoFinalization(pczt::roles::io_finalizer::Error),
117
118    /// An error occurred while updating the Orchard bundle of a PCZT.
119    UpdateOrchard(pczt::roles::updater::OrchardError),
120
121    /// An error occurred while updating the Sapling bundle of a PCZT.
122    UpdateSapling(pczt::roles::updater::SaplingError),
123
124    /// An error occurred while updating the transparent bundle of a PCZT.
125    UpdateTransparent(pczt::roles::updater::TransparentError),
126
127    /// An error occurred while finalizing the spends of a PCZT.
128    SpendFinalization(pczt::roles::spend_finalizer::Error),
129
130    /// An error occurred while extracting a transaction from a PCZT.
131    Extraction(pczt::roles::tx_extractor::Error),
132
133    /// PCZT parsing resulted in an invalid condition.
134    Invalid(String),
135}

137impl<DE, TE, SE, FE, CE, N> fmt::Display for Error<DE, TE, SE, FE, CE, N>
138where
139    DE: fmt::Display,
140    TE: fmt::Display,
141    SE: fmt::Display,
142    FE: fmt::Display,
143    CE: fmt::Display,
144    N: fmt::Display,
145{
146    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
147        use fmt::Write;
148
149        match self {
150            Error::DataSource(e) => {
151                write!(
152                    f,
153                    "The underlying datasource produced the following error: {e}"
154                )
155            }
156            Error::CommitmentTree(e) => {
157                write!(
158                    f,
159                    "An error occurred in querying or updating a note commitment tree: {e}"
160                )
161            }
162            Error::NoteSelection(e) => {
163                write!(f, "Note selection encountered the following error: {e}")
164            }
165            Error::Change(e) => {
166                write!(f, "Change output generation failed: {e}")
167            }
168            Error::Proposal(e) => {
169                write!(
170                    f,
171                    "Input selection attempted to construct an invalid proposal: {e}"
172                )
173            }
174            Error::ProposalNotSupported => write!(
175                f,
176                "The proposal was valid but tried to do something that is not supported \
177                 (spend shielded outputs of prior transaction steps or use a feature that \
178                 is not enabled).",
179            ),
180            Error::KeyNotRecognized => {
181                write!(
182                    f,
183                    "Wallet does not contain an account corresponding to the provided spending key"
184                )
185            }
186            Error::AccountCannotSpend => {
187                write!(
188                    f,
189                    "The given account cannot be used for spending, because it is unable to maintain an accurate balance.",
190                )
191            }
192            Error::AccountIdNotRecognized => {
193                write!(
194                    f,
195                    "Wallet does not contain an account corresponding to the provided ID"
196                )
197            }
198            Error::BalanceError(e) => write!(
199                f,
200                "The value lies outside the valid range of Zcash amounts: {e:?}."
201            ),
202            Error::InsufficientFunds {
203                available,
204                required,
205            } => write!(
206                f,
207                "Insufficient balance (have {}, need {} including fee)",
208                u64::from(*available),
209                u64::from(*required)
210            ),
211            Error::ScanRequired => write!(f, "Must scan blocks first"),
212            Error::Builder(e) => write!(f, "An error occurred building the transaction: {e}"),
213            Error::MemoForbidden => write!(
214                f,
215                "It is not possible to send a memo to a transparent address."
216            ),
217            Error::UnsupportedChangeType(t) => write!(
218                f,
219                "Attempted to send change to an unsupported pool type: {t}"
220            ),
221            Error::NoSupportedReceivers(ua) => write!(
222                f,
223                "A recipient's unified address does not contain any receivers to which the wallet can send funds; required one of {}",
224                ua.receiver_types()
225                    .iter()
226                    .enumerate()
227                    .fold(String::new(), |mut acc, (i, tc)| {
228                        let _ = write!(acc, "{}{:?}", if i > 0 { ", " } else { "" }, tc);
229                        acc
230                    })
231            ),
232            Error::KeyNotAvailable(pool) => write!(
233                f,
234                "A key required for transaction construction was not available for pool type {pool}"
235            ),
236            Error::NoteMismatch(n) => write!(
237                f,
238                "A note being spent ({n:?}) does not correspond to either the internal or external full viewing key for the provided spending key."
239            ),
240
241            Error::Address(e) => {
242                write!(
243                    f,
244                    "An error occurred decoding the address from a payment request: {e}."
245                )
246            }
247            #[cfg(feature = "transparent-inputs")]
248            Error::AddressNotRecognized(_) => {
249                write!(
250                    f,
251                    "The specified transparent address was not recognized as belonging to the wallet."
252                )
253            }
254            #[cfg(feature = "pczt")]
255            Error::Pczt(e) => write!(f, "PCZT error: {e}"),
256        }
257    }
258}

260#[cfg(feature = "pczt")]
261impl fmt::Display for PcztError {
262    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
263        match self {
264            PcztError::Build => {
265                write!(
266                    f,
267                    "Failed to generate the PCZT prior to proving or signing."
268                )
269            }
270            PcztError::IoFinalization(e) => {
271                write!(f, "Failed to finalize IO: {e:?}.")
272            }
273            PcztError::UpdateOrchard(e) => {
274                write!(f, "Failed to updating Orchard PCZT data: {e:?}.")
275            }
276            PcztError::UpdateSapling(e) => {
277                write!(f, "Failed to updating Sapling PCZT data: {e:?}.")
278            }
279            PcztError::UpdateTransparent(e) => {
280                write!(f, "Failed to updating transparent PCZT data: {e:?}.")
281            }
282            PcztError::SpendFinalization(e) => {
283                write!(f, "Failed to finalize the PCZT spends: {e:?}.")
284            }
285            PcztError::Extraction(e) => {
286                write!(f, "Failed to extract the final transaction: {e:?}.")
287            }
288            PcztError::Invalid(e) => {
289                write!(f, "PCZT parsing resulted in an invalid condition: {e}.")
290            }
291        }
292    }
293}

295impl<DE, TE, SE, FE, CE, N> error::Error for Error<DE, TE, SE, FE, CE, N>
296where
297    DE: Debug + Display + error::Error + 'static,
298    TE: Debug + Display + error::Error + 'static,
299    SE: Debug + Display + error::Error + 'static,
300    FE: Debug + Display + 'static,
301    CE: Debug + Display + error::Error + 'static,
302    N: Debug + Display + 'static,
303{
304    fn source(&self) -> Option<&(dyn error::Error + 'static)> {
305        match &self {
306            Error::DataSource(e) => Some(e),
307            Error::CommitmentTree(e) => Some(e),
308            Error::NoteSelection(e) => Some(e),
309            Error::Proposal(e) => Some(e),
310            Error::Builder(e) => Some(e),
311            #[cfg(feature = "pczt")]
312            Error::Pczt(e) => Some(e),
313            _ => None,
314        }
315    }
316}
