122    fn try_output_recovery<P: consensus::Parameters>(
123        params: &P,
124        height: BlockHeight,
125        tx: &Transaction,
126        fvk: &Self::Fvk,
127    ) -> Option<(Note, Address, MemoBytes)> {
128        for output in tx.sapling_bundle().unwrap().shielded_outputs() {
129            // Find the output that decrypts with the external OVK
130            let result = try_sapling_output_recovery(
131                &fvk.to_ovk(Scope::External),
132                output,
133                zip212_enforcement(params, height),
134            );
135
136            if result.is_some() {
137                return result.map(|(note, addr, memo)| {
138                    (
139                        Note::Sapling(note),
140                        addr.into(),
141                        MemoBytes::from_bytes(&memo).expect("correct length"),
142                    )
143                });
144            }
145        }
146
147        None
148    }

154    #[cfg(feature = "pczt")]
155    fn add_proof_generation_keys(
156        pczt: pczt::Pczt,
157        usk: &UnifiedSpendingKey,
158    ) -> Result<pczt::Pczt, pczt::roles::updater::SaplingError> {
159        let extsk = Self::usk_to_sk(usk);
160
161        Ok(pczt::roles::updater::Updater::new(pczt)
162            .update_sapling_with(|mut updater| {
163                let non_dummy_spends = updater
164                    .bundle()
165                    .spends()
166                    .iter()
167                    .enumerate()
168                    .filter_map(|(index, spend)| {
169                        // Dummy spends will already have a proof generation key.
170                        spend.proof_generation_key().is_none().then_some(index)
171                    })
172                    .collect::<Vec<_>>();
173
174                // Assume all non-dummy spent notes are from the same account.
175                for index in non_dummy_spends {
176                    updater.update_spend_with(index, |mut spend_updater| {
177                        spend_updater.set_proof_generation_key(extsk.expsk.proof_generation_key())
178                    })?;
179                }
180
181                Ok(())
182            })?
183            .finish())
184    }

186    #[cfg(feature = "pczt")]
187    fn apply_signatures_to_pczt(
188        signer: &mut pczt::roles::signer::Signer,
189        usk: &UnifiedSpendingKey,
190    ) -> Result<(), pczt::roles::signer::Error> {
191        let extsk = Self::usk_to_sk(usk);
192
193        // Figuring out which one is for us is hard. Let's just try signing all of them!
194        for index in 0.. {
195            match signer.sign_sapling(index, &extsk.expsk.ask) {
196                // Loop termination.
197                Err(pczt::roles::signer::Error::InvalidIndex) => break,
198                // Ignore any errors due to using the wrong key.
199                Ok(())
200                | Err(pczt::roles::signer::Error::SaplingSign(
201                    sapling::pczt::SignerError::WrongSpendAuthorizingKey,
202                )) => Ok(()),
203                // Raise any unexpected errors.
204                Err(e) => Err(e),
205            }?;
206        }
207
208        Ok(())
209    }
