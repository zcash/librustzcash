27/// Checks whether the transparent balance of the given test `account` is as `expected`
28/// considering the `min_confirmations`. It is assumed that zero or one `min_confirmations`
29/// are treated the same, and so this function also checks the other case when
30/// `min_confirmations` is 0 or 1.
31fn check_balance<DSF>(
32    st: &TestState<impl TestCache, <DSF as DataStoreFactory>::DataStore, LocalNetwork>,
33    account: &TestAccount<<DSF as DataStoreFactory>::Account>,
34    taddr: &TransparentAddress,
35    min_confirmations: u32,
36    expected: &Balance,
37) where
38    DSF: DataStoreFactory,
39{
40    // Check the wallet summary returns the expected transparent balance.
41    let summary = st
42        .wallet()
43        .get_wallet_summary(min_confirmations)
44        .unwrap()
45        .unwrap();
46    let balance = summary.account_balances().get(&account.id()).unwrap();
47
48    #[allow(deprecated)]
49    let old_unshielded_value = balance.unshielded();
50    assert_eq!(old_unshielded_value, expected.total());
51    assert_eq!(balance.unshielded_balance(), expected);
52
53    // Check the older APIs for consistency.
54    let mempool_height = st.wallet().chain_height().unwrap().unwrap() + 1;
55    assert_eq!(
56        st.wallet()
57            .get_transparent_balances(account.id(), mempool_height)
58            .unwrap()
59            .get(taddr)
60            .cloned()
61            .unwrap_or(Zatoshis::ZERO),
62        expected.total(),
63    );
64    assert_eq!(
65        st.wallet()
66            .get_spendable_transparent_outputs(taddr, mempool_height, min_confirmations)
67            .unwrap()
68            .into_iter()
69            .map(|utxo| utxo.value())
70            .sum::<Option<Zatoshis>>(),
71        Some(expected.spendable_value()),
72    );
73
74    // we currently treat min_confirmations the same regardless they are 0 (zero confirmations)
75    // or 1 (one block confirmation). We will check if this assumption holds until it's no
76    // longer made. If zero and one [`min_confirmations`] are treated differently in the future,
77    // this check should then be removed.
78    if min_confirmations == 0 || min_confirmations == 1 {
79        assert_eq!(
80            st.wallet()
81                .get_spendable_transparent_outputs(taddr, mempool_height, 1 - min_confirmations)
82                .unwrap()
83                .into_iter()
84                .map(|utxo| utxo.value())
85                .sum::<Option<Zatoshis>>(),
86            Some(expected.spendable_value()),
87        );
88    }
89}

91pub fn put_received_transparent_utxo<DSF>(dsf: DSF)
92where
93    DSF: DataStoreFactory,
94    <<DSF as DataStoreFactory>::DataStore as WalletWrite>::UtxoRef: std::fmt::Debug + PartialEq,
95{
96    let mut st = TestBuilder::new()
97        .with_data_store_factory(dsf)
98        .with_account_from_sapling_activation(BlockHash([0; 32]))
99        .build();
100
101    let birthday = st.test_account().unwrap().birthday().height();
102    let account_id = st.test_account().unwrap().id();
103    let uaddr = st
104        .wallet()
105        .get_last_generated_address_matching(account_id, UnifiedAddressRequest::AllAvailableKeys)
106        .unwrap()
107        .unwrap();
108    let taddr = uaddr.transparent().unwrap();
109
110    let height_1 = birthday + 12345;
111    st.wallet_mut().update_chain_tip(height_1).unwrap();
112
113    let bal_absent = st
114        .wallet()
115        .get_transparent_balances(account_id, height_1)
116        .unwrap();
117    assert!(bal_absent.is_empty());
118
119    // Create a fake transparent output.
120    let value = Zatoshis::const_from_u64(100000);
121    let outpoint = OutPoint::fake();
122    let txout = TxOut {
123        value,
124        script_pubkey: taddr.script(),
125    };
126
127    // Pretend the output's transaction was mined at `height_1`.
128    let utxo = WalletTransparentOutput::from_parts(outpoint.clone(), txout.clone(), Some(height_1))
129        .unwrap();
130    let res0 = st.wallet_mut().put_received_transparent_utxo(&utxo);
131    assert_matches!(res0, Ok(_));
132
133    // Confirm that we see the output unspent as of `height_1`.
134    assert_matches!(
135        st.wallet().get_spendable_transparent_outputs(
136            taddr,
137            height_1,
138            0
139        ).as_deref(),
140        Ok([ret]) if (ret.outpoint(), ret.txout(), ret.mined_height()) == (utxo.outpoint(), utxo.txout(), Some(height_1))
141    );
142    assert_matches!(
143        st.wallet().get_unspent_transparent_output(utxo.outpoint()),
144        Ok(Some(ret)) if (ret.outpoint(), ret.txout(), ret.mined_height()) == (utxo.outpoint(), utxo.txout(), Some(height_1))
145    );
146
147    // Change the mined height of the UTXO and upsert; we should get back
148    // the same `UtxoId`.
149    let height_2 = birthday + 34567;
150    st.wallet_mut().update_chain_tip(height_2).unwrap();
151    let utxo2 = WalletTransparentOutput::from_parts(outpoint, txout, Some(height_2)).unwrap();
152    let res1 = st.wallet_mut().put_received_transparent_utxo(&utxo2);
153    assert_matches!(res1, Ok(id) if id == res0.unwrap());
154
155    // Confirm that we no longer see any unspent outputs as of `height_1`.
156    assert_matches!(
157        st.wallet()
158            .get_spendable_transparent_outputs(taddr, height_1, 0)
159            .as_deref(),
160        Ok(&[])
161    );
162
163    // We can still look up the specific output, and it has the expected height.
164    assert_matches!(
165        st.wallet().get_unspent_transparent_output(utxo2.outpoint()),
166        Ok(Some(ret)) if (ret.outpoint(), ret.txout(), ret.mined_height()) == (utxo2.outpoint(), utxo2.txout(), Some(height_2))
167    );
168
169    // If we include `height_2` then the output is returned.
170    assert_matches!(
171        st.wallet()
172            .get_spendable_transparent_outputs(taddr, height_2, 0)
173            .as_deref(),
174        Ok([ret]) if (ret.outpoint(), ret.txout(), ret.mined_height()) == (utxo.outpoint(), utxo.txout(), Some(height_2))
175    );
176
177    assert_matches!(
178        st.wallet().get_transparent_balances(account_id, height_2),
179        Ok(h) if h.get(taddr) == Some(&value)
180    );
181}

183pub fn transparent_balance_across_shielding<DSF>(dsf: DSF, cache: impl TestCache)
184where
185    DSF: DataStoreFactory,
186{
187    let mut st = TestBuilder::new()
188        .with_data_store_factory(dsf)
189        .with_block_cache(cache)
190        .with_account_from_sapling_activation(BlockHash([0; 32]))
191        .build();
192
193    let account = st.test_account().cloned().unwrap();
194    let uaddr = st
195        .wallet()
196        .get_last_generated_address_matching(account.id(), UnifiedAddressRequest::AllAvailableKeys)
197        .unwrap()
198        .unwrap();
199    let taddr = uaddr.transparent().unwrap();
200
201    // Initialize the wallet with chain data that has no shielded notes for us.
202    let not_our_key = ExtendedSpendingKey::master(&[]).to_diversifiable_full_viewing_key();
203    let not_our_value = Zatoshis::const_from_u64(10000);
204    let (start_height, _, _) =
205        st.generate_next_block(&not_our_key, AddressType::DefaultExternal, not_our_value);
206    for _ in 1..10 {
207        st.generate_next_block(&not_our_key, AddressType::DefaultExternal, not_our_value);
208    }
209    st.scan_cached_blocks(start_height, 10);
210
211    // The wallet starts out with zero balance.
212    check_balance::<DSF>(&st, &account, taddr, 0, &Balance::ZERO);
213
214    // Create a fake transparent output.
215    let value = Zatoshis::from_u64(100000).unwrap();
216    let txout = TxOut {
217        value,
218        script_pubkey: taddr.script(),
219    };
220
221    // Pretend the output was received in the chain tip.
222    let height = st.wallet().chain_height().unwrap().unwrap();
223    let utxo = WalletTransparentOutput::from_parts(OutPoint::fake(), txout, Some(height)).unwrap();
224    st.wallet_mut()
225        .put_received_transparent_utxo(&utxo)
226        .unwrap();
227
228    // The wallet should detect the balance as available
229    let mut zero_or_one_conf_value = Balance::ZERO;
230
231    // add the spendable value to the expected balance
232    zero_or_one_conf_value.add_spendable_value(value).unwrap();
233
234    check_balance::<DSF>(&st, &account, taddr, 0, &zero_or_one_conf_value);
235
236    // Shield the output.
237    let input_selector = GreedyInputSelector::new();
238    let change_strategy = standard::SingleOutputChangeStrategy::new(
239        StandardFeeRule::Zip317,
240        None,
241        ShieldedProtocol::Sapling,
242        DustOutputPolicy::default(),
243    );
244    let txid = st
245        .shield_transparent_funds(
246            &input_selector,
247            &change_strategy,
248            value,
249            account.usk(),
250            &[*taddr],
251            account.id(),
252            1,
253        )
254        .unwrap()[0];
255
256    // The wallet should have zero transparent balance, because the shielding
257    // transaction can be mined.
258    check_balance::<DSF>(&st, &account, taddr, 0, &Balance::ZERO);
259
260    // Mine the shielding transaction.
261    let (mined_height, _) = st.generate_next_block_including(txid);
262    st.scan_cached_blocks(mined_height, 1);
263
264    // The wallet should still have zero transparent balance.
265    check_balance::<DSF>(&st, &account, taddr, 0, &Balance::ZERO);
266
267    // Unmine the shielding transaction via a reorg.
268    st.wallet_mut()
269        .truncate_to_height(mined_height - 1)
270        .unwrap();
271    assert_eq!(st.wallet().chain_height().unwrap(), Some(mined_height - 1));
272
273    // The wallet should still have zero transparent balance.
274    check_balance::<DSF>(&st, &account, taddr, 0, &Balance::ZERO);
275
276    // Expire the shielding transaction.
277    let expiry_height = st
278        .wallet()
279        .get_transaction(txid)
280        .unwrap()
281        .expect("Transaction exists in the wallet.")
282        .expiry_height();
283    st.wallet_mut().update_chain_tip(expiry_height).unwrap();
284
285    check_balance::<DSF>(&st, &account, taddr, 0, &zero_or_one_conf_value);
286}

288/// This test attempts to verify that transparent funds spendability is
289/// accounted for properly given the different minimum confirmations values
290/// that can be set when querying for balances.
291pub fn transparent_balance_spendability<DSF>(dsf: DSF, cache: impl TestCache)
292where
293    DSF: DataStoreFactory,
294{
295    let mut st = TestBuilder::new()
296        .with_data_store_factory(dsf)
297        .with_block_cache(cache)
298        .with_account_from_sapling_activation(BlockHash([0; 32]))
299        .build();
300
301    let account = st.test_account().cloned().unwrap();
302    let uaddr = st
303        .wallet()
304        .get_last_generated_address_matching(account.id(), UnifiedAddressRequest::AllAvailableKeys)
305        .unwrap()
306        .unwrap();
307    let taddr = uaddr.transparent().unwrap();
308
309    // Initialize the wallet with chain data that has no shielded notes for us.
310    let not_our_key = ExtendedSpendingKey::master(&[]).to_diversifiable_full_viewing_key();
311    let not_our_value = Zatoshis::const_from_u64(10000);
312    let (start_height, _, _) =
313        st.generate_next_block(&not_our_key, AddressType::DefaultExternal, not_our_value);
314    for _ in 1..10 {
315        st.generate_next_block(&not_our_key, AddressType::DefaultExternal, not_our_value);
316    }
317    st.scan_cached_blocks(start_height, 10);
318
319    // The wallet starts out with zero balance.
320    check_balance::<DSF>(
321        &st as &TestState<_, DSF::DataStore, _>,
322        &account,
323        taddr,
324        0,
325        &Balance::ZERO,
326    );
327
328    // Create a fake transparent output.
329    let value = Zatoshis::from_u64(100000).unwrap();
330    let txout = TxOut {
331        value,
332        script_pubkey: taddr.script(),
333    };
334
335    // Pretend the output was received in the chain tip.
336    let height = st.wallet().chain_height().unwrap().unwrap();
337    let utxo = WalletTransparentOutput::from_parts(OutPoint::fake(), txout, Some(height)).unwrap();
338    st.wallet_mut()
339        .put_received_transparent_utxo(&utxo)
340        .unwrap();
341
342    // The wallet should detect the balance as available
343    let mut zero_or_one_conf_value = Balance::ZERO;
344
345    // add the spendable value to the expected balance
346    zero_or_one_conf_value.add_spendable_value(value).unwrap();
347
348    check_balance::<DSF>(&st, &account, taddr, 0, &zero_or_one_conf_value);
349
350    // now if we increase the number of confirmations our spendable balance should
351    // be zero and the total balance equal to `value`
352    let mut not_confirmed_yet_value = Balance::ZERO;
353
354    not_confirmed_yet_value
355        .add_pending_spendable_value(value)
356        .unwrap();
357
358    check_balance::<DSF>(&st, &account, taddr, 2, &not_confirmed_yet_value);
359
360    // Add one extra block
361    st.generate_empty_block();
362
363    // Scan that block
364    st.scan_cached_blocks(height, 1);
365
366    // now we generate one more block and the balance should be the same as when the
367    // check_balance function was called with zero or one confirmation.
368    st.generate_empty_block();
369    st.scan_cached_blocks(height + 1, 1);
370
371    check_balance::<DSF>(&st, &account, taddr, 2, &zero_or_one_conf_value);
372}

403pub fn gap_limits<DSF>(ds_factory: DSF, cache: impl TestCache, gap_limits: GapLimits)
404where
405    DSF: DataStoreFactory,
406    <DSF as DataStoreFactory>::AccountId: std::fmt::Debug,
407{
408    let mut st = TestBuilder::new()
409        .with_data_store_factory(ds_factory)
410        .with_block_cache(cache)
411        .with_account_from_sapling_activation(BlockHash([0; 32]))
412        .build();
413
414    let test_account = st.test_account().cloned().unwrap();
415    let account_uuid = test_account.account().id();
416    let ufvk = test_account.account().ufvk().unwrap().clone();
417
418    let external_taddrs = st
419        .wallet()
420        .get_transparent_receivers(account_uuid, false)
421        .unwrap();
422    assert_eq!(
423        u32::try_from(external_taddrs.len()).unwrap(),
424        gap_limits.external()
425    );
426    let internal_taddrs = st
427        .wallet()
428        .get_transparent_receivers(account_uuid, true)
429        .unwrap();
430    assert_eq!(
431        u32::try_from(internal_taddrs.len()).unwrap(),
432        gap_limits.external() + gap_limits.internal()
433    );
434    let ephemeral_taddrs = st
435        .wallet()
436        .get_known_ephemeral_addresses(account_uuid, None)
437        .unwrap();
438    assert_eq!(
439        u32::try_from(ephemeral_taddrs.len()).unwrap(),
440        gap_limits.ephemeral()
441    );
442
443    // Add some funds to the wallet
444    let (h0, _, _) = st.generate_next_block(
445        &ufvk.sapling().unwrap(),
446        AddressType::DefaultExternal,
447        Zatoshis::const_from_u64(1000000),
448    );
449    st.scan_cached_blocks(h0, 1);
450
451    // The previous operation was shielded-only, but unified address usage within the
452    // valid transparent child index range still count towards the gap limit, so this
453    // updates the gap limit by the index of the default Sapling receiver
454    let external_taddrs = st
455        .wallet()
456        .get_transparent_receivers(account_uuid, false)
457        .unwrap();
458    assert_eq!(
459        u32::try_from(external_taddrs.len()).unwrap(),
460        gap_limits.external()
461            + (u32::try_from(ufvk.sapling().unwrap().default_address().0).unwrap() + 1)
462    );
463
464    // Pick an address half way through the set of external taddrs
465    let external_taddrs_sorted = external_taddrs
466        .into_iter()
467        .filter_map(|(addr, meta)| meta.map(|m| (m.address_index(), addr)))
468        .collect::<BTreeMap<_, _>>();
469    let to = Address::from(
470        *external_taddrs_sorted
471            .get(&transparent::keys::NonHardenedChildIndex::from_index(4).unwrap())
472            .expect("An address exists at index 4."),
473    )
474    .to_zcash_address(st.network());
475
476    // Create a transaction & scan the block. Since the txid corresponds to one our wallet
477    // generated, this should cause the gap limit to be bumped (generating addresses with index
478    // 10..15)
479    let txids = st
480        .create_standard_transaction(&test_account, to, Zatoshis::const_from_u64(20000))
481        .unwrap();
482    let (h1, _) = st.generate_next_block_including(txids.head);
483
484    // At this point, the transaction has been created, but since it has not been mined it does
485    // not cause an update to the gap limit; we have to wait for the transaction to actually be
486    // mined or we could bump the gap limit too soon and start generating addresses that will
487    // never be inspected on wallet recovery.
488    let external_taddrs = st
489        .wallet()
490        .get_transparent_receivers(account_uuid, false)
491        .unwrap();
492    assert_eq!(
493        u32::try_from(external_taddrs.len()).unwrap(),
494        gap_limits.external()
495            + (u32::try_from(ufvk.sapling().unwrap().default_address().0).unwrap() + 1)
496    );
497
498    // Mine the block, then use `decrypt_and_store_transaction` to ensure that the wallet sees
499    // the transaction as mined (since transparent handling doesn't get this from
500    // `scan_cached_blocks`)
501    st.scan_cached_blocks(h1, 1);
502    let tx = st.wallet().get_transaction(txids.head).unwrap().unwrap();
503    decrypt_and_store_transaction(&st.network().clone(), st.wallet_mut(), &tx, Some(h1)).unwrap();
504
505    // Now that the transaction has been mined, the gap limit should have increased.
506    let external_taddrs = st
507        .wallet()
508        .get_transparent_receivers(account_uuid, false)
509        .unwrap();
510    assert_eq!(
511        u32::try_from(external_taddrs.len()).unwrap(),
512        gap_limits.external() + 5
513    );
514
515    // The utxo query height should be equal to the minimum mined height among transactions
516    // sent to any of the set of {addresses in the gap limit range | address prior to the gap}.
517    let query_height = st.wallet().utxo_query_height(account_uuid).unwrap();
518    assert_eq!(query_height, h0);
519}
