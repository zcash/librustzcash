30/// Checks whether the transparent balance of the given test `account` is as `expected`
31/// considering the `confirmations_policy`.
32fn check_balance<DSF>(
33    st: &TestState<impl TestCache, <DSF as DataStoreFactory>::DataStore, LocalNetwork>,
34    account: &TestAccount<<DSF as DataStoreFactory>::Account>,
35    taddr: &TransparentAddress,
36    confirmations_policy: ConfirmationsPolicy,
37    expected: &Balance,
38) where
39    DSF: DataStoreFactory,
40{
41    // Check the wallet summary returns the expected transparent balance.
42    let summary = st
43        .wallet()
44        .get_wallet_summary(confirmations_policy)
45        .unwrap()
46        .unwrap();
47    let balance = summary.account_balances().get(&account.id()).unwrap();
48
49    #[allow(deprecated)]
50    let old_unshielded_value = balance.unshielded();
51    assert_eq!(old_unshielded_value, expected.total());
52    assert_eq!(balance.unshielded_balance(), expected);
53
54    // Check the older APIs for consistency.
55    let target_height = TargetHeight::from(st.wallet().chain_height().unwrap().unwrap() + 1);
56    assert_eq!(
57        st.wallet()
58            .get_transparent_balances(account.id(), target_height, confirmations_policy)
59            .unwrap()
60            .get(taddr)
61            .cloned()
62            .map_or(Zatoshis::ZERO, |(_, b)| b.spendable_value()),
63        expected.total(),
64    );
65    assert_eq!(
66        st.wallet()
67            .get_spendable_transparent_outputs(taddr, target_height, confirmations_policy)
68            .unwrap()
69            .into_iter()
70            .map(|utxo| utxo.value())
71            .sum::<Option<Zatoshis>>(),
72        Some(expected.spendable_value()),
73    );
74}

76pub fn put_received_transparent_utxo<DSF>(dsf: DSF)
77where
78    DSF: DataStoreFactory,
79    <<DSF as DataStoreFactory>::DataStore as WalletWrite>::UtxoRef: std::fmt::Debug + PartialEq,
80{
81    let mut st = TestBuilder::new()
82        .with_data_store_factory(dsf)
83        .with_account_from_sapling_activation(BlockHash([0; 32]))
84        .build();
85
86    let birthday = st.test_account().unwrap().birthday().height();
87    let account_id = st.test_account().unwrap().id();
88    let uaddr = st
89        .wallet()
90        .get_last_generated_address_matching(account_id, UnifiedAddressRequest::AllAvailableKeys)
91        .unwrap()
92        .unwrap();
93    let taddr = uaddr.transparent().unwrap();
94
95    let height_1 = birthday + 12345;
96    st.wallet_mut().update_chain_tip(height_1).unwrap();
97
98    let bal_absent = st
99        .wallet()
100        .get_transparent_balances(
101            account_id,
102            TargetHeight::from(height_1 + 1),
103            ConfirmationsPolicy::MIN,
104        )
105        .unwrap();
106    assert!(bal_absent.is_empty());
107
108    // Create a fake transparent output.
109    let value = Zatoshis::const_from_u64(100000);
110    let outpoint = OutPoint::fake();
111    let txout = TxOut::new(value, taddr.script().into());
112
113    // Pretend the output's transaction was mined at `height_1`.
114    let utxo = WalletTransparentOutput::from_parts(outpoint.clone(), txout.clone(), Some(height_1))
115        .unwrap();
116    let res0 = st.wallet_mut().put_received_transparent_utxo(&utxo);
117    assert_matches!(res0, Ok(_));
118
119    let target_height = TargetHeight::from(height_1 + 1);
120    // Confirm that we see the output unspent as of `height_1`.
121    assert_matches!(
122        st.wallet().get_spendable_transparent_outputs(
123            taddr,
124            target_height,
125            ConfirmationsPolicy::MIN
126        ).as_deref(),
127        Ok([ret])
128        if (ret.outpoint(), ret.txout(), ret.mined_height()) == (utxo.outpoint(), utxo.txout(), Some(height_1))
129    );
130    assert_matches!(
131        st.wallet().get_unspent_transparent_output(utxo.outpoint(), target_height),
132        Ok(Some(ret))
133        if (ret.outpoint(), ret.txout(), ret.mined_height()) == (utxo.outpoint(), utxo.txout(), Some(height_1))
134    );
135
136    // Change the mined height of the UTXO and upsert; we should get back
137    // the same `UtxoId`.
138    let height_2 = birthday + 34567;
139    st.wallet_mut().update_chain_tip(height_2).unwrap();
140    let utxo2 = WalletTransparentOutput::from_parts(outpoint, txout, Some(height_2)).unwrap();
141    let res1 = st.wallet_mut().put_received_transparent_utxo(&utxo2);
142    assert_matches!(res1, Ok(id) if id == res0.unwrap());
143
144    // Confirm that we no longer see any unspent outputs as of `height_1`.
145    assert_matches!(
146        st.wallet()
147            .get_spendable_transparent_outputs(taddr, target_height, ConfirmationsPolicy::MIN)
148            .as_deref(),
149        Ok(&[])
150    );
151
152    // We can still look up the specific output, and it has the expected height.
153    assert_matches!(
154        st.wallet().get_unspent_transparent_output(utxo2.outpoint(), target_height),
155        Ok(Some(ret))
156        if (ret.outpoint(), ret.txout(), ret.mined_height()) == (utxo2.outpoint(), utxo2.txout(), Some(height_2))
157    );
158
159    // If we include `height_2` then the output is returned.
160    assert_matches!(
161        st.wallet()
162            .get_spendable_transparent_outputs(taddr, TargetHeight::from(height_2 + 1), ConfirmationsPolicy::MIN)
163            .as_deref(),
164        Ok([ret]) if (ret.outpoint(), ret.txout(), ret.mined_height()) == (utxo.outpoint(), utxo.txout(), Some(height_2))
165    );
166
167    assert_matches!(
168        st.wallet().get_transparent_balances(
169            account_id,
170            TargetHeight::from(height_2 + 1),
171            ConfirmationsPolicy::MIN
172        ),
173        Ok(h) if h.get(taddr).map(|(_, b)| b.spendable_value()) == Some(value)
174    );
175}

177pub fn transparent_balance_across_shielding<DSF>(dsf: DSF, cache: impl TestCache)
178where
179    DSF: DataStoreFactory,
180{
181    let mut st = TestBuilder::new()
182        .with_data_store_factory(dsf)
183        .with_block_cache(cache)
184        .with_account_from_sapling_activation(BlockHash([0; 32]))
185        .build();
186
187    let account = st.test_account().cloned().unwrap();
188    let uaddr = st
189        .wallet()
190        .get_last_generated_address_matching(account.id(), UnifiedAddressRequest::AllAvailableKeys)
191        .unwrap()
192        .unwrap();
193    let taddr = uaddr.transparent().unwrap();
194
195    // Initialize the wallet with chain data that has no shielded notes for us.
196    let not_our_key = ExtendedSpendingKey::master(&[]).to_diversifiable_full_viewing_key();
197    let not_our_value = Zatoshis::const_from_u64(10000);
198    let (start_height, _, _) =
199        st.generate_next_block(&not_our_key, AddressType::DefaultExternal, not_our_value);
200    for _ in 1..10 {
201        st.generate_next_block(&not_our_key, AddressType::DefaultExternal, not_our_value);
202    }
203    st.scan_cached_blocks(start_height, 10);
204
205    // The wallet starts out with zero balance.
206    check_balance::<DSF>(
207        &st,
208        &account,
209        taddr,
210        ConfirmationsPolicy::MIN,
211        &Balance::ZERO,
212    );
213
214    // Create a fake transparent output.
215    let value = Zatoshis::from_u64(100000).unwrap();
216    let txout = TxOut::new(value, taddr.script().into());
217
218    // Pretend the output was received in the chain tip.
219    let height = st.wallet().chain_height().unwrap().unwrap();
220    let utxo = WalletTransparentOutput::from_parts(OutPoint::fake(), txout, Some(height)).unwrap();
221    st.wallet_mut()
222        .put_received_transparent_utxo(&utxo)
223        .unwrap();
224
225    // The wallet should detect the balance as available
226    let mut zero_or_one_conf_value = Balance::ZERO;
227
228    // add the spendable value to the expected balance
229    zero_or_one_conf_value.add_spendable_value(value).unwrap();
230
231    check_balance::<DSF>(
232        &st,
233        &account,
234        taddr,
235        ConfirmationsPolicy::MIN,
236        &zero_or_one_conf_value,
237    );
238
239    // Shield the output.
240    let input_selector = GreedyInputSelector::new();
241    let change_strategy = standard::SingleOutputChangeStrategy::new(
242        StandardFeeRule::Zip317,
243        None,
244        ShieldedProtocol::Sapling,
245        DustOutputPolicy::default(),
246    );
247    let txid = st
248        .shield_transparent_funds(
249            &input_selector,
250            &change_strategy,
251            value,
252            account.usk(),
253            &[*taddr],
254            account.id(),
255            ConfirmationsPolicy::MIN,
256        )
257        .unwrap()[0];
258
259    // The wallet should have zero transparent balance, because the shielding
260    // transaction can be mined.
261    check_balance::<DSF>(
262        &st,
263        &account,
264        taddr,
265        ConfirmationsPolicy::MIN,
266        &Balance::ZERO,
267    );
268
269    // Mine the shielding transaction.
270    let (mined_height, _) = st.generate_next_block_including(txid);
271    st.scan_cached_blocks(mined_height, 1);
272
273    // The wallet should still have zero transparent balance.
274    check_balance::<DSF>(
275        &st,
276        &account,
277        taddr,
278        ConfirmationsPolicy::MIN,
279        &Balance::ZERO,
280    );
281
282    // Unmine the shielding transaction via a reorg.
283    st.wallet_mut()
284        .truncate_to_height(mined_height - 1)
285        .unwrap();
286    assert_eq!(st.wallet().chain_height().unwrap(), Some(mined_height - 1));
287
288    // The wallet should still have zero transparent balance.
289    check_balance::<DSF>(
290        &st,
291        &account,
292        taddr,
293        ConfirmationsPolicy::MIN,
294        &Balance::ZERO,
295    );
296
297    // Expire the shielding transaction.
298    let expiry_height = st
299        .wallet()
300        .get_transaction(txid)
301        .unwrap()
302        .expect("Transaction exists in the wallet.")
303        .expiry_height();
304    st.wallet_mut().update_chain_tip(expiry_height).unwrap();
305
306    check_balance::<DSF>(
307        &st,
308        &account,
309        taddr,
310        ConfirmationsPolicy::MIN,
311        &zero_or_one_conf_value,
312    );
313}

315/// This test attempts to verify that transparent funds spendability is
316/// accounted for properly given the different minimum confirmations values
317/// that can be set when querying for balances.
318pub fn transparent_balance_spendability<DSF>(dsf: DSF, cache: impl TestCache)
319where
320    DSF: DataStoreFactory,
321{
322    let mut st = TestBuilder::new()
323        .with_data_store_factory(dsf)
324        .with_block_cache(cache)
325        .with_account_from_sapling_activation(BlockHash([0; 32]))
326        .build();
327
328    let account = st.test_account().cloned().unwrap();
329    let uaddr = st
330        .wallet()
331        .get_last_generated_address_matching(account.id(), UnifiedAddressRequest::AllAvailableKeys)
332        .unwrap()
333        .unwrap();
334    let taddr = uaddr.transparent().unwrap();
335
336    // Initialize the wallet with chain data that has no shielded notes for us.
337    let not_our_key = ExtendedSpendingKey::master(&[]).to_diversifiable_full_viewing_key();
338    let not_our_value = Zatoshis::const_from_u64(10000);
339    let (start_height, _, _) =
340        st.generate_next_block(&not_our_key, AddressType::DefaultExternal, not_our_value);
341    for _ in 1..10 {
342        st.generate_next_block(&not_our_key, AddressType::DefaultExternal, not_our_value);
343    }
344    st.scan_cached_blocks(start_height, 10);
345
346    // The wallet starts out with zero balance.
347    check_balance::<DSF>(
348        &st as &TestState<_, DSF::DataStore, _>,
349        &account,
350        taddr,
351        ConfirmationsPolicy::MIN,
352        &Balance::ZERO,
353    );
354
355    // Create a fake transparent output.
356    let value = Zatoshis::from_u64(100000).unwrap();
357    let txout = TxOut::new(value, taddr.script().into());
358
359    // Pretend the output was received in the chain tip.
360    let height = st.wallet().chain_height().unwrap().unwrap();
361    let utxo = WalletTransparentOutput::from_parts(OutPoint::fake(), txout, Some(height)).unwrap();
362    st.wallet_mut()
363        .put_received_transparent_utxo(&utxo)
364        .unwrap();
365
366    // The wallet should detect the balance as available
367    let mut zero_or_one_conf_value = Balance::ZERO;
368
369    // add the spendable value to the expected balance
370    zero_or_one_conf_value.add_spendable_value(value).unwrap();
371
372    check_balance::<DSF>(
373        &st,
374        &account,
375        taddr,
376        ConfirmationsPolicy::MIN,
377        &zero_or_one_conf_value,
378    );
379
380    // now if we increase the number of confirmations our spendable balance should
381    // be zero and the total balance equal to `value`
382    let mut not_confirmed_yet_value = Balance::ZERO;
383
384    not_confirmed_yet_value
385        .add_pending_spendable_value(value)
386        .unwrap();
387
388    check_balance::<DSF>(
389        &st,
390        &account,
391        taddr,
392        ConfirmationsPolicy::new_symmetrical_unchecked(2, false),
393        &not_confirmed_yet_value,
394    );
395
396    // Add one extra block
397    st.generate_empty_block();
398
399    // Scan that block
400    st.scan_cached_blocks(height, 1);
401
402    // now we generate one more block and the balance should be the same as when the
403    // check_balance function was called with zero or one confirmation.
404    st.generate_empty_block();
405    st.scan_cached_blocks(height + 1, 1);
406
407    check_balance::<DSF>(
408        &st,
409        &account,
410        taddr,
411        ConfirmationsPolicy::new_symmetrical_unchecked(2, true),
412        &zero_or_one_conf_value,
413    );
414}

445pub fn gap_limits<DSF>(ds_factory: DSF, cache: impl TestCache, gap_limits: GapLimits)
446where
447    DSF: DataStoreFactory,
448    <DSF as DataStoreFactory>::AccountId: std::fmt::Debug,
449{
450    let mut st = TestBuilder::new()
451        .with_data_store_factory(ds_factory)
452        .with_block_cache(cache)
453        .with_account_from_sapling_activation(BlockHash([0; 32]))
454        .build();
455
456    let test_account = st.test_account().cloned().unwrap();
457    let account_uuid = test_account.account().id();
458    let ufvk = test_account.account().ufvk().unwrap().clone();
459
460    let external_taddrs = st
461        .wallet()
462        .get_transparent_receivers(account_uuid, false, true)
463        .unwrap();
464    assert_eq!(
465        u32::try_from(external_taddrs.len()).unwrap(),
466        gap_limits.external()
467    );
468    let internal_taddrs = st
469        .wallet()
470        .get_transparent_receivers(account_uuid, true, false)
471        .unwrap();
472    assert_eq!(
473        u32::try_from(internal_taddrs.len()).unwrap(),
474        gap_limits.external() + gap_limits.internal()
475    );
476    let ephemeral_taddrs = st
477        .wallet()
478        .get_known_ephemeral_addresses(account_uuid, None)
479        .unwrap();
480    assert_eq!(
481        u32::try_from(ephemeral_taddrs.len()).unwrap(),
482        gap_limits.ephemeral()
483    );
484
485    // Add some funds to the wallet
486    let (h0, _, _) = st.generate_next_block(
487        &ufvk.sapling().unwrap(),
488        AddressType::DefaultExternal,
489        Zatoshis::const_from_u64(1000000),
490    );
491    st.scan_cached_blocks(h0, 1);
492
493    // The previous operation was shielded-only, but unified address usage within the
494    // valid transparent child index range still count towards the gap limit, so this
495    // updates the gap limit by the index of the default Sapling receiver
496    let external_taddrs = st
497        .wallet()
498        .get_transparent_receivers(account_uuid, false, true)
499        .unwrap();
500    assert_eq!(
501        u32::try_from(external_taddrs.len()).unwrap(),
502        gap_limits.external()
503            + (u32::try_from(ufvk.sapling().unwrap().default_address().0).unwrap() + 1)
504    );
505
506    // Pick an address half way through the set of external taddrs
507    let external_taddrs_sorted = external_taddrs
508        .into_iter()
509        .filter_map(|(addr, meta)| meta.address_index().map(|i| (i, addr)))
510        .collect::<BTreeMap<_, _>>();
511    let to = Address::from(
512        *external_taddrs_sorted
513            .get(&transparent::keys::NonHardenedChildIndex::from_index(4).unwrap())
514            .expect("An address exists at index 4."),
515    )
516    .to_zcash_address(st.network());
517
518    // Create a transaction & scan the block. Since the txid corresponds to one our wallet
519    // generated, this should cause the gap limit to be bumped (generating addresses with index
520    // 10..15)
521    let txids = st
522        .create_standard_transaction(&test_account, to, Zatoshis::const_from_u64(20000))
523        .unwrap();
524    let (h1, _) = st.generate_next_block_including(txids.head);
525
526    // At this point, the transaction has been created, but since it has not been mined it does
527    // not cause an update to the gap limit; we have to wait for the transaction to actually be
528    // mined or we could bump the gap limit too soon and start generating addresses that will
529    // never be inspected on wallet recovery.
530    let external_taddrs = st
531        .wallet()
532        .get_transparent_receivers(account_uuid, false, true)
533        .unwrap();
534    assert_eq!(
535        u32::try_from(external_taddrs.len()).unwrap(),
536        gap_limits.external()
537            + (u32::try_from(ufvk.sapling().unwrap().default_address().0).unwrap() + 1)
538    );
539
540    // Mine the block, then use `decrypt_and_store_transaction` to ensure that the wallet sees
541    // the transaction as mined (since transparent handling doesn't get this from
542    // `scan_cached_blocks`)
543    st.scan_cached_blocks(h1, 1);
544    let tx = st.wallet().get_transaction(txids.head).unwrap().unwrap();
545    decrypt_and_store_transaction(&st.network().clone(), st.wallet_mut(), &tx, Some(h1)).unwrap();
546
547    // Now that the transaction has been mined, the gap limit should have increased.
548    let external_taddrs = st
549        .wallet()
550        .get_transparent_receivers(account_uuid, false, true)
551        .unwrap();
552    assert_eq!(
553        u32::try_from(external_taddrs.len()).unwrap(),
554        gap_limits.external() + 5
555    );
556
557    // The utxo query height should be equal to the minimum mined height among transactions
558    // sent to any of the set of {addresses in the gap limit range | address prior to the gap}.
559    let query_height = st.wallet().utxo_query_height(account_uuid).unwrap();
560    assert_eq!(query_height, h0);
561}
