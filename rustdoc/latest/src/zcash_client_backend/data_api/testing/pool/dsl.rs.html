23/// A type-state wrapper struct that provides convenience methods.
24pub struct TestDsl<T> {
25    /// Either a `TestBuilder<Cache, DataStoreFactory>` or
26    /// `TestState<Cache, DataStore, LocalNetwork>`.
27    inner: T,
28}
29
30impl<T> Deref for TestDsl<T> {
31    type Target = T;
32
33    fn deref(&self) -> &Self::Target {
34        &self.inner
35    }
36}
37
38impl<T> DerefMut for TestDsl<T> {
39    fn deref_mut(&mut self) -> &mut Self::Target {
40        &mut self.inner
41    }
42}
43
44impl<T> From<T> for TestDsl<T> {
45    fn from(inner: T) -> Self {
46        Self { inner }
47    }
48}
49
50impl<T> TestDsl<T> {
51    /// Perform a state transition that may change the inner type.
52    pub fn map<X>(self, f: impl FnOnce(T) -> X) -> TestDsl<X> {
53        f(self.inner).into()
54    }
55}

57/// [`TestDsl`] provides convenience methods for common [`TestBuilder`] scenarios.
58impl<Cache, Dsf> TestDsl<TestBuilder<Cache, Dsf>>
59where
60    Dsf: DataStoreFactory,
61{
62    /// Equip the inner [`TestBuilder`] with the provided [`DataStoreFactory`]
63    /// and [`TestCache`], as well as an account that has a birthday at Sapling
64    /// activation.
65    ///
66    /// Shorthand for the following:
67    /// ```rust,ignore
68    /// let dsl: TestDsl<TestBuilder<_, _>> = TestBuilder::new()
69    ///     .with_data_store_factory(dsf)
70    ///     .with_block_cache(tc)
71    ///     .with_account_from_sapling_activation(BlockHash([0; 32]))
72    ///     .into();
73    /// ```
74    pub fn with_sapling_birthday_account(dsf: Dsf, tc: Cache) -> Self {
75        TestBuilder::new()
76            .with_data_store_factory(dsf)
77            .with_block_cache(tc)
78            .with_account_from_sapling_activation(BlockHash([0; 32]))
79            .into()
80    }
81
82    /// Build the builder, wrapping the resulting [`TestState`] in a [`TestDsl`] and [`TestScenario`].
83    pub fn build<T: ShieldedPoolTester>(self) -> TestDsl<TestScenario<T, Cache, Dsf>> {
84        let state = self.inner.build();
85        TestScenario {
86            state,
87            _phantom: PhantomData,
88        }
89        .into()
90    }
91}

93/// A proxy for `FakeCompactOutput` that allows test code to omit the `fvk` and
94/// `address_type` fields, which can be derived from the `TestState` in most cases.
95pub struct TestNoteConfig<T: ShieldedPoolTester> {
96    /// The amount of the note.
97    pub value: Zatoshis,
98    /// Diversifiable full viewing key of the recipient.
99    pub fvk: Option<T::Fvk>,

102}

108            fvk: None,
109            address_type: None,

111    }

126pub struct AddFundsStepResult<T: ShieldedPoolTester, C: TestCache> {
127    pub block_height: BlockHeight,
128    pub insert_result: C::InsertResult,
129    /// Empty when the step was to generate an empty block
130    pub nullifiers: Vec<<T::Fvk as TestFvk>::Nullifier>,
131}

133/// The input and output of one "add funds" step.
134pub struct AddFundsStep<T: ShieldedPoolTester, C: TestCache> {
135    pub notes: Vec<TestNoteConfig<T>>,
136    pub results: AddFundsStepResult<T, C>,
137}

139/// A collection of results from adding funds to a `TestState`.
140pub struct AddFundsSummary<T: ShieldedPoolTester, C: TestCache> {
141    pub steps: Vec<AddFundsStep<T, C>>,
142    pub scan_summary: Option<ScanSummary>,
143}

145impl<T: ShieldedPoolTester, C: TestCache> Default for AddFundsSummary<T, C> {
146    fn default() -> Self {
147        Self {
148            steps: Default::default(),
149            scan_summary: None,
150        }
151    }

154impl<T: ShieldedPoolTester, C: TestCache> AddFundsSummary<T, C> {
155    /// Return the first block height.
156    pub fn first_block_height(&self) -> Option<BlockHeight> {
157        self.steps.first().map(|step| step.results.block_height)
158    }
159
160    /// Return the latest block height after generating the blocks
161    /// that added funds.
162    pub fn block_height(&self) -> Option<BlockHeight> {
163        self.steps.last().map(|step| step.results.block_height)
164    }

196/// Add funds scenarios.
197impl<Cache, Dsf, T> TestScenario<T, Cache, Dsf>
198where
199    T: ShieldedPoolTester,
200    Cache: TestCache,
201    Dsf: DataStoreFactory,
202{
203    /// Return the current test account balance, if possible.
204    ///
205    /// Returns `None` when no account summary data is available, which is
206    /// the case before the wallet has scanned any blocks.
207    pub fn get_account_balance(
208        &self,
209        confirmations_policy: ConfirmationsPolicy,
210    ) -> Option<AccountBalance> {
211        let account = self.get_account();
212        let binding = self
213            .wallet()
214            .get_wallet_summary(confirmations_policy)
215            .unwrap()?;
216        let balance = binding.account_balances().get(&account.id())?;
217        Some(*balance)
218    }
219
220    /// Adds funds from a single note from an address of the given type.
221    ///
222    /// Returns the current block height, cache insert result and test viewing key nullifier.
223    ///
224    /// This is shorthand for:
225    /// ```rust,ignore
226    /// {
227    ///     let dfvk = T::test_account_fvk(&st);
228    ///     let output@(h, _, _) = st.generate_next_block(&dfvk, address_type, zatoshis);
229    ///     st.scan_cached_blocks(h, 1);
230    ///     output
231    /// }
232    /// ```
233    ///
234    /// This also verifies that the test account contains the expected funds as
235    /// part of the _total_, and that the funds are spendable with the minimum number
236    /// of confirmations.
237    pub fn add_a_single_note_checking_balance(
238        &mut self,
239        note: impl Into<TestNoteConfig<T>>,
240    ) -> (
241        BlockHeight,
242        Cache::InsertResult,
243        <T::Fvk as TestFvk>::Nullifier,
244    ) {
245        let mut summary = self.add_notes_checking_balance([[note]]);
246        let res = summary.steps.pop().unwrap().results;
247        (res.block_height, res.insert_result, res.nullifiers[0])
248    }
249
250    fn scanned_block_height(&self) -> BlockHeight {
251        self.wallet()
252            .block_max_scanned()
253            .unwrap()
254            .map(|meta| meta.block_height())
255            .unwrap_or_else(|| BlockHeight::from_u32(0))
256    }
257
258    /// Generates `N` empty blocks.
259    ///
260    /// Returns the current block height.
261    pub fn add_empty_blocks(&mut self, n: usize) -> BlockHeight {
262        let mut out_height = self.scanned_block_height();
263        for _ in 0..n {
264            let (h, _) = self.generate_empty_block();
265            out_height = h;
266        }
267        out_height
268    }

270    /// Returns the test account.
271    pub fn get_account(&self) -> TestAccount<Dsf::Account> {
272        self.test_account().expect("not configured").clone()
273    }

275    /// Creates a `FakeCompactOutput` from the given `TestNoteConfig`.
276    fn make_fake_output(&self, note_config: &TestNoteConfig<T>) -> FakeCompactOutput<T::Fvk> {
277        let TestNoteConfig {
278            value,
279            fvk,
280            address_type,
281        } = note_config;
282        FakeCompactOutput::new(
283            fvk.clone().unwrap_or_else(|| T::test_account_fvk(self)),
284            address_type.unwrap_or(AddressType::DefaultExternal),
285            *value,
286        )
287    }
288
289    /// Add funds from multiple notes in one or more blocks, or generate empty blocks.
290    ///
291    /// This step also verifies that the test account contains the expected
292    /// funds as part of the _total_. Keep in mind that these funds may not yet
293    /// be _spendable_ due to the number of confirmations required.
294    ///
295    /// Returns a summary of steps.
296    ///
297    /// ## Parameters
298    ///
299    /// * `blocks` - A collection of "blocks", where each "block" is a collection of
300    ///   "notes". More specifically, "notes" can be anything that can be converted
301    ///   into a [`TestNoteConfig`]. This allows you to add multiple blocks that each
302    ///   containing zero or more notes with one call to `add_notes`.
303    ///
304    /// ## Note
305    /// Keep in mind:
306    /// * Each block coalesces these notes into a single transaction.
307    /// * Funds are added to the default test account.
308    pub fn add_notes_checking_balance(
309        &mut self,
310        blocks: impl IntoIterator<Item = impl IntoIterator<Item = impl Into<TestNoteConfig<T>>>>,
311    ) -> AddFundsSummary<T, Cache> {
312        let mut from_height = None;
313        let mut current_height = self.scanned_block_height();
314        let mut limit = 0;
315        let account = self.get_account();
316        let starting_balance = self
317            .get_account_balance(ConfirmationsPolicy::MIN)
318            .map(|b| b.total())
319            .unwrap_or(Zatoshis::ZERO);
320        let mut expected_total = starting_balance;
321        let mut summary = AddFundsSummary::default();
322        for notes in blocks.into_iter() {
323            // Collect the notes while also counting their value.
324            let (fake_outputs, note_configs): (Vec<_>, Vec<_>) = notes
325                .into_iter()
326                .map(|into_note_config| {
327                    let note_config = into_note_config.into();
328                    if note_config.value > zip317::MARGINAL_FEE {
329                        // Don't include uneconomic (dust) notes in the expected
330                        // total, as the balance won't include them.
331                        expected_total = (expected_total + note_config.value).unwrap();
332                    }
333                    (self.make_fake_output(&note_config), note_config)
334                })
335                .unzip();
336            let step_result = if fake_outputs.is_empty() {
337                let (h, r) = self.generate_empty_block();
338                AddFundsStepResult {
339                    block_height: h,
340                    insert_result: r,
341                    nullifiers: vec![],
342                }
343            } else {
344                let (h, r, n) = self.generate_next_block_multi(&fake_outputs);
345                AddFundsStepResult {
346                    block_height: h,
347                    insert_result: r,
348                    nullifiers: n,
349                }
350            };
351            current_height = step_result.block_height;
352            if from_height.is_none() {
353                from_height = Some(current_height);
354            }
355            summary.steps.push(AddFundsStep {
356                notes: note_configs,
357                results: step_result,
358            });
359            limit += 1;
360        }
361        if let Some(from_height) = from_height {
362            summary.scan_summary = Some(self.scan_cached_blocks(from_height, limit));
363        }
364
365        // Do most of the assertions that we care about at the "add funds" callsites
366        assert_eq!(
367            self.get_total_balance(account.id()),
368            expected_total,
369            "Unexpected total balance"
370        );
371        assert_eq!(
372            self.wallet()
373                .block_max_scanned()
374                .unwrap()
375                .unwrap()
376                .block_height(),
377            current_height
378        );
379        assert_eq!(
380            self.get_spendable_balance(account.id(), ConfirmationsPolicy::MIN),
381            expected_total
382        );
383
384        summary
385    }
