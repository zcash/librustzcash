138    fn try_output_recovery<P: consensus::Parameters>(
139        _params: &P,
140        _: BlockHeight,
141        tx: &Transaction,
142        fvk: &Self::Fvk,
143    ) -> Option<(Note, Address, MemoBytes)> {
144        for action in tx.orchard_bundle().unwrap().actions() {
145            // Find the output that decrypts with the external OVK
146            let result = try_output_recovery_with_ovk(
147                &OrchardDomain::for_action(action),
148                &fvk.to_ovk(zip32::Scope::External),
149                action,
150                action.cv_net(),
151                &action.encrypted_note().out_ciphertext,
152            );
153
154            if result.is_some() {
155                return result.map(|(note, addr, memo)| {
156                    (
157                        Note::Orchard(note),
158                        UnifiedAddress::from_receivers(Some(addr), None, None)
159                            .unwrap()
160                            .into(),
161                        MemoBytes::from_bytes(&memo).expect("correct length"),
162                    )
163                });
164            }
165        }
166
167        None
168    }

183    #[cfg(feature = "pczt")]
184    fn apply_signatures_to_pczt(
185        signer: &mut pczt::roles::signer::Signer,
186        usk: &UnifiedSpendingKey,
187    ) -> Result<(), pczt::roles::signer::Error> {
188        let sk = Self::usk_to_sk(usk);
189        let ask = orchard::keys::SpendAuthorizingKey::from(sk);
190
191        // Figuring out which one is for us is hard. Let's just try signing all of them!
192        for index in 0.. {
193            match signer.sign_orchard(index, &ask) {
194                // Loop termination.
195                Err(pczt::roles::signer::Error::InvalidIndex) => break,
196                // Ignore any errors due to using the wrong key.
197                Ok(())
198                | Err(pczt::roles::signer::Error::OrchardSign(
199                    orchard::pczt::SignerError::WrongSpendAuthorizingKey,
200                )) => Ok(()),
201                // Raise any unexpected errors.
202                Err(e) => Err(e),
203            }?;
204        }
205
206        Ok(())
207    }
