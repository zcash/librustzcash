162    fn try_output_recovery<P: consensus::Parameters>(
163        _params: &P,
164        _: BlockHeight,
165        tx: &Transaction,
166        fvk: &Self::Fvk,
167    ) -> Option<(Note, Address, MemoBytes)> {
168        for action in tx.orchard_bundle().unwrap().actions() {
169            // Find the output that decrypts with the external OVK
170            let result = try_output_recovery_with_ovk(
171                &OrchardDomain::for_action(action),
172                &fvk.to_ovk(zip32::Scope::External),
173                action,
174                action.cv_net(),
175                &action.encrypted_note().out_ciphertext,
176            );
177
178            if result.is_some() {
179                return result.map(|(note, addr, memo)| {
180                    (
181                        Note::Orchard(note),
182                        UnifiedAddress::from_receivers(Some(addr), None, None)
183                            .unwrap()
184                            .into(),
185                        MemoBytes::from_bytes(&memo).expect("correct length"),
186                    )
187                });
188            }
189        }
190
191        None
192    }

207    #[cfg(feature = "pczt")]
208    fn apply_signatures_to_pczt(
209        signer: &mut pczt::roles::signer::Signer,
210        usk: &UnifiedSpendingKey,
211    ) -> Result<(), pczt::roles::signer::Error> {
212        let sk = Self::usk_to_sk(usk);
213        let ask = orchard::keys::SpendAuthorizingKey::from(sk);
214
215        // Figuring out which one is for us is hard. Let's just try signing all of them!
216        for index in 0.. {
217            match signer.sign_orchard(index, &ask) {
218                // Loop termination.
219                Err(pczt::roles::signer::Error::InvalidIndex) => break,
220                // Ignore any errors due to using the wrong key.
221                Ok(())
222                | Err(pczt::roles::signer::Error::OrchardSign(
223                    orchard::pczt::SignerError::WrongSpendAuthorizingKey,
224                )) => Ok(()),
225                // Raise any unexpected errors.
226                Err(e) => Err(e),
227            }?;
228        }
229
230        Ok(())
231    }
