181/// Tests sending funds within the given shielded pool in a single transaction.
182///
183/// The test:
184/// - Adds funds to the wallet in a single note.
185/// - Checks that the wallet balances are correct.
186/// - Constructs a request to spend part of that balance to an external address in the
187///   same pool.
188/// - Builds the transaction.
189/// - Checks that the transaction was stored, and that the outputs are decryptable and
190///   have the expected details.
191pub fn send_single_step_proposed_transfer<T: ShieldedPoolTester>(
192    dsf: impl DataStoreFactory,
193    cache: impl TestCache,
194) {
195    let mut st = TestBuilder::new()
196        .with_data_store_factory(dsf)
197        .with_block_cache(cache)
198        .with_account_from_sapling_activation(BlockHash([0; 32]))
199        .build();
200
201    let account = st.test_account().cloned().unwrap();
202    let dfvk = T::test_account_fvk(&st);
203
204    // Add funds to the wallet in a single note
205    let value = Zatoshis::const_from_u64(60000);
206    let (h, _, _) = st.generate_next_block(&dfvk, AddressType::DefaultExternal, value);
207    st.scan_cached_blocks(h, 1);
208
209    // Spendable balance matches total balance
210    assert_eq!(st.get_total_balance(account.id()), value);
211    assert_eq!(st.get_spendable_balance(account.id(), 1), value);
212
213    assert_eq!(
214        st.wallet()
215            .block_max_scanned()
216            .unwrap()
217            .unwrap()
218            .block_height(),
219        h
220    );
221
222    let to_extsk = T::sk(&[0xf5; 32]);
223    let to: Address = T::sk_default_address(&to_extsk);
224    let request = zip321::TransactionRequest::new(vec![Payment::without_memo(
225        to.to_zcash_address(st.network()),
226        Zatoshis::const_from_u64(10000),
227    )])
228    .unwrap();
229
230    let fee_rule = StandardFeeRule::Zip317;
231
232    let change_memo = "Test change memo".parse::<Memo>().unwrap();
233    let change_strategy = standard::SingleOutputChangeStrategy::new(
234        fee_rule,
235        Some(change_memo.clone().into()),
236        T::SHIELDED_PROTOCOL,
237        DustOutputPolicy::default(),
238    );
239    let input_selector = GreedyInputSelector::new();
240
241    let proposal = st
242        .propose_transfer(
243            account.id(),
244            &input_selector,
245            &change_strategy,
246            request,
247            NonZeroU32::new(1).unwrap(),
248        )
249        .unwrap();
250
251    let create_proposed_result = st.create_proposed_transactions::<Infallible, _, Infallible>(
252        account.usk(),
253        OvkPolicy::Sender,
254        &proposal,
255    );
256    assert_matches!(&create_proposed_result, Ok(txids) if txids.len() == 1);
257
258    let sent_tx_id = create_proposed_result.unwrap()[0];
259
260    // Verify that the sent transaction was stored and that we can decrypt the memos
261    let tx = st
262        .wallet()
263        .get_transaction(sent_tx_id)
264        .unwrap()
265        .expect("Created transaction was stored.");
266    let ufvks = [(account.id(), account.usk().to_unified_full_viewing_key())]
267        .into_iter()
268        .collect();
269    let d_tx = decrypt_transaction(st.network(), None, Some(h), &tx, &ufvks);
270    assert_eq!(T::decrypted_pool_outputs_count(&d_tx), 2);
271
272    let mut found_tx_change_memo = false;
273    let mut found_tx_empty_memo = false;
274    T::with_decrypted_pool_memos(&d_tx, |memo| {
275        if Memo::try_from(memo).unwrap() == change_memo {
276            found_tx_change_memo = true
277        }
278        if Memo::try_from(memo).unwrap() == Memo::Empty {
279            found_tx_empty_memo = true
280        }
281    });
282    assert!(found_tx_change_memo);
283    assert!(found_tx_empty_memo);
284
285    // Verify that the stored sent notes match what we're expecting
286    let sent_note_ids = st
287        .wallet()
288        .get_sent_note_ids(&sent_tx_id, T::SHIELDED_PROTOCOL)
289        .unwrap();
290    assert_eq!(sent_note_ids.len(), 2);
291
292    // The sent memo should be the empty memo for the sent output, and the
293    // change output's memo should be as specified.
294    let mut found_sent_change_memo = false;
295    let mut found_sent_empty_memo = false;
296    for sent_note_id in sent_note_ids {
297        match st
298            .wallet()
299            .get_memo(sent_note_id)
300            .expect("Note id is valid")
301            .as_ref()
302        {
303            Some(m) if m == &change_memo => {
304                found_sent_change_memo = true;
305            }
306            Some(m) if m == &Memo::Empty => {
307                found_sent_empty_memo = true;
308            }
309            Some(other) => panic!("Unexpected memo value: {:?}", other),
310            None => panic!("Memo should not be stored as NULL"),
311        }
312    }
313    assert!(found_sent_change_memo);
314    assert!(found_sent_empty_memo);
315
316    // Check that querying for a nonexistent sent note returns None
317    assert_matches!(
318        st.wallet()
319            .get_memo(NoteId::new(sent_tx_id, T::SHIELDED_PROTOCOL, 12345)),
320        Ok(None)
321    );
322
323    let tx_history = st.wallet().get_tx_history().unwrap();
324    assert_eq!(tx_history.len(), 2);
325    {
326        let tx_0 = &tx_history[0];
327        assert_eq!(tx_0.total_spent(), Zatoshis::const_from_u64(0));
328        assert_eq!(tx_0.total_received(), Zatoshis::const_from_u64(60000));
329    }
330
331    {
332        let tx_1 = &tx_history[1];
333        assert_eq!(tx_1.total_spent(), Zatoshis::const_from_u64(60000));
334        assert_eq!(tx_1.total_received(), Zatoshis::const_from_u64(40000));
335    }
336
337    let network = *st.network();
338    assert_matches!(
339        decrypt_and_store_transaction(&network, st.wallet_mut(), &tx, None),
340        Ok(_)
341    );
342}

882    // Now we've reserved everything we can, we can't reserve one more
883    reservation_should_fail(&mut st, 1, gap_limits.ephemeral() * 2);

2653pub fn reorg_to_checkpoint<T: ShieldedPoolTester, DSF, C>(ds_factory: DSF, cache: C)
2654where
2655    DSF: DataStoreFactory,
2656    <DSF as DataStoreFactory>::AccountId: std::fmt::Debug,
2657    C: TestCache,
2658{

2660        .with_data_store_factory(ds_factory)
2661        .with_block_cache(cache)
2662        .with_account_from_sapling_activation(BlockHash([0; 32]))
2663        .build();
2664
2665    let account = st.test_account().cloned().unwrap();
2666
2667    // Create a sequence of blocks to serve as the foundation of our chain state.
2668    let p0_fvk = T::random_fvk(st.rng_mut());
2669    let gen_random_block = |st: &mut TestState<C, DSF::DataStore, LocalNetwork>,
2670                            output_count: usize| {
2671        let fake_outputs =
2672            std::iter::repeat_with(|| FakeCompactOutput::random(st.rng_mut(), p0_fvk.clone()))
2673                .take(output_count)
2674                .collect::<Vec<_>>();
2675        st.generate_next_block_multi(&fake_outputs[..]);
2676        output_count
2677    };
2678
2679    // The stable portion of the tree will contain 20 notes.
2680    for _ in 0..10 {
2681        gen_random_block(&mut st, 4);
2682    }
2683
2684    // We will reorg to this height.
2685    let reorg_height = account.birthday().height() + 4;
2686    let reorg_position = Position::from(19);
2687
2688    // Scan the first 5 blocks. The last block in this sequence will be where we simulate a
2689    // reorg.
2690    st.scan_cached_blocks(account.birthday().height(), 5);
2691    assert_eq!(
2692        st.wallet()
2693            .block_max_scanned()
2694            .unwrap()
2695            .unwrap()
2696            .block_height(),
2697        reorg_height
2698    );
2699
2700    // There will be 6 checkpoints: one for the prior block frontier, and then one for each scanned
2701    // block.
2702    let checkpoints = st
2703        .wallet()
2704        .get_checkpoint_history(&T::SHIELDED_PROTOCOL)
2705        .unwrap();
2706    assert_eq!(checkpoints.len(), 6);
2707    assert_eq!(
2708        checkpoints.last(),
2709        Some(&(reorg_height, Some(reorg_position)))
2710    );
2711
2712    // Scan another block, then simulate a reorg.
2713    st.scan_cached_blocks(reorg_height + 1, 1);
2714    assert_eq!(
2715        st.wallet()
2716            .block_max_scanned()
2717            .unwrap()
2718            .unwrap()
2719            .block_height(),
2720        reorg_height + 1
2721    );
2722    let checkpoints = st
2723        .wallet()
2724        .get_checkpoint_history(&T::SHIELDED_PROTOCOL)
2725        .unwrap();
2726    assert_eq!(checkpoints.len(), 7);
2727    assert_eq!(
2728        checkpoints.last(),
2729        Some(&(reorg_height + 1, Some(reorg_position + 4)))
2730    );
2731
2732    //        /\  /\  /\
2733    //  .... /\/\/\/\/\/\
2734    //          c   d   e
2735
2736    // Truncate back to the reorg height, but retain the block cache.
2737    st.truncate_to_height_retaining_cache(reorg_height);
2738
2739    // The following error-prone tree state is generated by the a previous (buggy) truncate
2740    // implementation:
2741    //        /\  /\
2742    //  .... /\/\/\/\
2743    //          c
2744
2745    // We have pruned back to the original checkpoints & tree state.
2746    let checkpoints = st
2747        .wallet()
2748        .get_checkpoint_history(&T::SHIELDED_PROTOCOL)
2749        .unwrap();
2750    assert_eq!(checkpoints.len(), 6);
2751    assert_eq!(
2752        checkpoints.last(),
2753        Some(&(reorg_height, Some(reorg_position)))
2754    );
2755
2756    // Skip two blocks, then (re) scan the same block.
2757    st.scan_cached_blocks(reorg_height + 2, 1);
2758
2759    // Given the buggy truncation, this would result in this the following tree state:
2760    //        /\  /\   \  /\
2761    //  .... /\/\/\/\   \/\/\
2762    //          c       e   f
2763
2764    let checkpoints = st
2765        .wallet()
2766        .get_checkpoint_history(&T::SHIELDED_PROTOCOL)
2767        .unwrap();
2768    // Even though we only scanned one block, we get a checkpoint at both the start and the end of
2769    // the block due to the insertion of the prior block frontier.
2770    assert_eq!(checkpoints.len(), 8);
2771    assert_eq!(
2772        checkpoints.last(),
2773        Some(&(reorg_height + 2, Some(reorg_position + 8)))
2774    );
2775
2776    // Now, fully truncate back to the reorg height. This should leave the tree in a state
2777    // where it can be added to with arbitrary notes.
2778    st.truncate_to_height(reorg_height);
2779
2780    // Generate some new random blocks
2781    for _ in 0..10 {
2782        let output_count = st.rng_mut().gen_range(2..10);
2783        gen_random_block(&mut st, output_count);
2784    }
2785
2786    // The previous truncation retained the cache, so re-scanning the same blocks would have
2787    // resulted in the same note commitment tree state, and hence no conflicts; could occur. Now
2788    // that we have cleared the cache and generated a different sequence blocks, if truncation did
2789    // not completely clear the tree state this would generates a note commitment tree conflict.
2790    st.scan_cached_blocks(reorg_height + 1, 1);
2791}
