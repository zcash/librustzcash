67/// A type that represents the recipient of a transaction output:
68///
69/// * a recipient address;
70/// * for external unified addresses, the pool to which the payment is sent;
71/// * for wallet-internal outputs, the internal account ID and metadata about the note.
72/// * if the `transparent-inputs` feature is enabled, for ephemeral transparent outputs, the
73///   internal account ID and metadata about the outpoint;

76    External {
77        recipient_address: ZcashAddress,

80    #[cfg(feature = "transparent-inputs")]

82        receiving_account: AccountId,
83        ephemeral_address: TransparentAddress,
84        outpoint: OutPoint,
85    },
86    InternalAccount {
87        receiving_account: AccountId,
88        external_address: Option<ZcashAddress>,
89        note: Box<Note>,
90    },

362/// An enumeration of supported shielded note types for use in [`ReceivedNote`]
363#[derive(Debug, Clone, PartialEq, Eq)]
364pub enum Note {
365    Sapling(sapling::Note),
366    #[cfg(feature = "orchard")]
367    Orchard(orchard::Note),
368}
369
370impl Note {
371    pub fn value(&self) -> Zatoshis {
372        match self {
373            Note::Sapling(n) => n.value().inner().try_into().expect(
374                "Sapling notes must have values in the range of valid non-negative ZEC values.",
375            ),
376            #[cfg(feature = "orchard")]
377            Note::Orchard(n) => Zatoshis::from_u64(n.value().inner()).expect(
378                "Orchard notes must have values in the range of valid non-negative ZEC values.",
379            ),
380        }
381    }
382
383    /// Returns the shielded protocol used by this note.
384    pub fn protocol(&self) -> ShieldedProtocol {
385        match self {
386            Note::Sapling(_) => ShieldedProtocol::Sapling,
387            #[cfg(feature = "orchard")]
388            Note::Orchard(_) => ShieldedProtocol::Orchard,
389        }
390    }
391}

594/// Describes a policy for which outgoing viewing key should be able to decrypt
595/// transaction outputs.
596///
597/// For details on what transaction information is visible to the holder of an outgoing
598/// viewing key, refer to [ZIP 310].
599///
600/// [ZIP 310]: https://zips.z.cash/zip-0310
601#[derive(Debug, Clone)]
602pub enum OvkPolicy {
603    /// Use the outgoing viewing key from the sender's [`UnifiedFullViewingKey`].
604    ///
605    /// Transaction outputs will be decryptable by the sender, in addition to the
606    /// recipients.
607    ///
608    /// [`UnifiedFullViewingKey`]: zcash_keys::keys::UnifiedFullViewingKey
609    Sender,
610
611    /// Use custom outgoing viewing keys. These might for instance be derived from a
612    /// different seed than the wallet's spending keys.
613    ///
614    /// Transaction outputs will be decryptable by the recipients, and whoever controls
615    /// the provided outgoing viewing keys.
616    Custom {
617        sapling: sapling::keys::OutgoingViewingKey,
618        #[cfg(feature = "orchard")]
619        orchard: orchard::keys::OutgoingViewingKey,
620    },
621    /// Use no outgoing viewing keys. Transaction outputs will be decryptable by their
622    /// recipients, but not by the sender.
623    Discard,
624}
625
626impl OvkPolicy {
627    /// Constructs an [`OvkPolicy::Custom`] value from a single arbitrary 32-byte key.
628    ///
629    /// Outputs of transactions created with this OVK policy will be recoverable using
630    /// this key irrespective of the output pool.
631    pub fn custom_from_common_bytes(key: &[u8; 32]) -> Self {
632        OvkPolicy::Custom {
633            sapling: sapling::keys::OutgoingViewingKey(*key),
634            #[cfg(feature = "orchard")]
635            orchard: orchard::keys::OutgoingViewingKey::from(*key),
636        }
637    }
638}

640/// Metadata describing the gap limit position of a transparent address.
641#[derive(Debug, Clone, Copy, PartialEq, Eq)]
642pub enum GapMetadata {
643    /// The address, or an address at a greater child index, has received transparent funds and
644    /// will be discovered by wallet recovery by exploration over the space of
645    /// [`NonHardenedChildIndex`]es using the provided gap limit.
646    GapRecoverable { gap_limit: u32 },
647    /// The address exists within an address gap of the given limit size, and will be discovered by
648    /// wallet recovery by exploration using the provided gap limit. In the view of the wallet, no
649    /// addresses at the given position or greater (up to the gap limit) have received funds. The
650    /// number of addresses remaining within the gap limit before no additional addresses can be

652    InGap {

656        /// without any of those addresses having received funds.
657        gap_limit: u32,
658    },
659    /// The wallet does not contain derivation information for the associated address, and so its
660    /// relationship to other addresses in the wallet cannot be determined.
661    DerivationUnknown,

664/// Metadata describing whether and when a transparent address was exposed by the wallet.
665#[derive(Debug, Clone, Copy, PartialEq, Eq)]
666#[cfg(feature = "transparent-inputs")]
667pub enum Exposure {
668    /// The address has been exposed by the wallet.
669    Exposed {
670        /// The address was first exposed to the wider ecosystem at this height, to the best
671        /// of our knowledge.
672        ///
673        /// - For user-generated addresses, this is the chain tip height at the time that the
674        ///   address was generated by an explicit request by the user or reserved for use in
675        ///   a ZIP 320 transaction. These heights are not recoverable from chain.
676        /// - In the case of an address with its first use discovered in a transaction
677        ///   obtained by scanning the chain, this will be set to the mined height of that
678        ///   transaction. In recover from seed cases, this is what user-generated addresses
679        ///   will be assigned.
680        at_height: BlockHeight,
681        /// Transparent address gap metadata, as of the time the query that produced this exposure
682        /// metadata was executed.
683        gap_metadata: GapMetadata,
684    },
685    /// The address is not known to have been exposed to an external caller by the wallet.
686    ///
687    /// The wallet makes its determination based on observed chain data and inference from
688    /// standard wallet address generation patterns. In particular, this is the state that
689    /// an address is in when it has been generated by the advancement of the transparent
690    /// address gap. This judgement may be incorrect for restored wallets.
691    Unknown,
692    /// It is not possible for the wallet to determine whether the address has been exposed,
693    /// given the information the wallet has access to.
694    CannotKnow,
695}

697/// Information about a transparent address controlled by the wallet.
698#[derive(Debug, Clone, PartialEq, Eq)]
699#[cfg(feature = "transparent-inputs")]
700pub struct TransparentAddressMetadata {

702    exposure: Exposure,
703    next_check_time: Option<SystemTime>,
704}

707impl TransparentAddressMetadata {

721    /// Returns a [`TransparentAddressMetadata`] with [`TransparentAddressSource::Derived`] source

734            exposure,
735            next_check_time,

739    /// Returns a [`TransparentAddressMetadata`] with [`TransparentAddressSource::Standalone`] source

742    pub fn standalone(
743        pubkey: secp256k1::PublicKey,

749            exposure,

751        }
752    }

754    /// Returns the source metadata for the address.

757    }
758
759    /// Returns the exposure metadata for this transparent address.

762    }

764    /// Returns a copy of this metadata, with its exposure metadata updated

784    /// only one such query should be performed at a time, to avoid linking multiple transparent

800    }

803/// Source information for a transparent address.
804#[derive(Debug, Clone, PartialEq, Eq)]
805#[cfg(feature = "transparent-inputs")]
806pub enum TransparentAddressSource {
807    /// BIP 44 path derivation information for the address below account pubkey level, i.e. the
808    /// `change` and `index` elements of the path.
809    Derived {
810        scope: TransparentKeyScope,
811        address_index: NonHardenedChildIndex,
812    },
813    /// The address was derived from a secp256k1 public key for which derivation information is
814    /// unknown or for which the associated spending key was produced from system randomness.
815    /// This variant provides the public key directly.
816    #[cfg(feature = "transparent-key-import")]
817    Standalone(secp256k1::PublicKey),
818}
819
820#[cfg(feature = "transparent-inputs")]
821impl TransparentAddressSource {
822    /// Returns the [`TransparentKeyScope`] of the private key from which the address was derived,
823    /// if known. Returns `None` for standalone addresses in the wallet.
824    pub fn scope(&self) -> Option<TransparentKeyScope> {
825        match self {
826            TransparentAddressSource::Derived { scope, .. } => Some(*scope),
827            #[cfg(feature = "transparent-key-import")]
828            TransparentAddressSource::Standalone(_) => None,
829        }
830    }
831
832    /// Returns the BIP 44 [`NonHardenedChildIndex`] at which the address was derived, if known.
833    /// Returns `None` for standalone addresses in the wallet.
834    pub fn address_index(&self) -> Option<NonHardenedChildIndex> {
835        match self {
836            TransparentAddressSource::Derived { address_index, .. } => Some(*address_index),
837            #[cfg(feature = "transparent-key-import")]
838            TransparentAddressSource::Standalone(_) => None,
839        }
840    }
841}
