375    /// Parses this tree state into a [`ChainState`] for use with [`scan_cached_blocks`].
376    ///
377    /// [`scan_cached_blocks`]: crate::data_api::chain::scan_cached_blocks
378    pub fn to_chain_state(&self) -> io::Result<ChainState> {
379        let mut hash_bytes = hex::decode(&self.hash).map_err(|e| {
380            io::Error::new(
381                io::ErrorKind::InvalidData,
382                format!("Block hash is not valid hex: {e:?}"),
383            )
384        })?;
385        // Zcashd hex strings for block hashes are byte-reversed.
386        hash_bytes.reverse();
387
388        Ok(ChainState::new(
389            self.height
390                .try_into()
391                .map_err(|_| io::Error::new(io::ErrorKind::InvalidData, "Invalid block height"))?,
392            BlockHash::try_from_slice(&hash_bytes).ok_or_else(|| {
393                io::Error::new(io::ErrorKind::InvalidData, "Invalid block hash length.")
394            })?,
395            self.sapling_tree()?.to_frontier(),
396            #[cfg(feature = "orchard")]
397            self.orchard_tree()?.to_frontier(),
398        ))
399    }

405/// Errors that can occur in the process of decoding a [`Proposal`] from its protobuf
406/// representation.
407#[derive(Debug, Clone)]
408pub enum ProposalDecodingError<DbError> {
409    /// The encoded proposal contained no steps.
410    NoSteps,
411    /// The ZIP 321 transaction request URI was invalid.
412    Zip321(Zip321Error),
413    /// A proposed input was null.
414    NullInput(usize),
415    /// A transaction identifier string did not decode to a valid transaction ID.
416    TxIdInvalid(TryFromSliceError),
417    /// An invalid value pool identifier was encountered.
418    ValuePoolNotSupported(i32),
419    /// A failure occurred trying to retrieve an unspent note or UTXO from the wallet database.
420    InputRetrieval(DbError),
421    /// The unspent note or UTXO corresponding to a proposal input was not found in the wallet
422    /// database.
423    InputNotFound(TxId, PoolType, u32),
424    /// The transaction balance, or a component thereof, failed to decode correctly.
425    BalanceInvalid,
426    /// Failed to decode a ZIP-302-compliant memo from the provided memo bytes.
427    MemoInvalid(memo::Error),
428    /// The serialization version returned by the protobuf was not recognized.
429    VersionInvalid(u32),
430    /// The fee rule specified by the proposal is not supported by the wallet.
431    FeeRuleNotSupported(proposal::FeeRule),
432    /// The proposal violated balance or structural constraints.
433    ProposalInvalid(ProposalError),
434    /// An inputs field for the given protocol was present, but contained no input note references.
435    EmptyShieldedInputs(ShieldedProtocol),
436    /// A memo field was provided for a transparent output.
437    TransparentMemo,
438    /// Change outputs to the specified pool are not supported.
439    InvalidChangeRecipient(PoolType),
440    /// Ephemeral outputs to the specified pool are not supported.
441    InvalidEphemeralRecipient(PoolType),
442}

450impl<E: Display> Display for ProposalDecodingError<E> {
451    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
452        match self {
453            ProposalDecodingError::NoSteps => write!(f, "The proposal had no steps."),
454            ProposalDecodingError::Zip321(err) => write!(f, "Transaction request invalid: {err}"),
455            ProposalDecodingError::NullInput(i) => {
456                write!(f, "Proposed input was null at index {i}")
457            }
458            ProposalDecodingError::TxIdInvalid(err) => {
459                write!(f, "Invalid transaction id: {err:?}")
460            }
461            ProposalDecodingError::ValuePoolNotSupported(id) => {
462                write!(f, "Invalid value pool identifier: {id:?}")
463            }
464            ProposalDecodingError::InputRetrieval(err) => {
465                write!(f, "An error occurred retrieving a transaction input: {err}")
466            }
467            ProposalDecodingError::InputNotFound(txid, pool, idx) => {
468                write!(f, "No {pool} input found for txid {txid}, index {idx}")
469            }
470            ProposalDecodingError::BalanceInvalid => {
471                write!(f, "An error occurred decoding the proposal balance.")
472            }
473            ProposalDecodingError::MemoInvalid(err) => {
474                write!(f, "An error occurred decoding a proposed memo: {err}")
475            }
476            ProposalDecodingError::VersionInvalid(v) => {
477                write!(f, "Unrecognized proposal version {v}")
478            }
479            ProposalDecodingError::FeeRuleNotSupported(r) => {
480                write!(
481                    f,
482                    "Fee calculation using the {r:?} fee rule is not supported."
483                )
484            }
485            ProposalDecodingError::ProposalInvalid(err) => write!(f, "{err}"),
486            ProposalDecodingError::EmptyShieldedInputs(protocol) => write!(
487                f,
488                "An inputs field was present for {protocol:?}, but contained no note references."
489            ),
490            ProposalDecodingError::TransparentMemo => {
491                write!(f, "Transparent outputs cannot have memos.")
492            }
493            ProposalDecodingError::InvalidChangeRecipient(pool_type) => write!(
494                f,
495                "Change outputs to the {pool_type} pool are not supported."
496            ),
497            ProposalDecodingError::InvalidEphemeralRecipient(pool_type) => write!(
498                f,
499                "Ephemeral outputs to the {pool_type} pool are not supported."

501        }
502    }

559impl proposal::Proposal {
560    /// Serializes a [`Proposal`] based upon a supported [`StandardFeeRule`] to its protobuf
561    /// representation.
562    pub fn from_standard_proposal<NoteRef>(value: &Proposal<StandardFeeRule, NoteRef>) -> Self {
563        use proposal::proposed_input;
564        use proposal::{PriorStepChange, PriorStepOutput, ReceivedOutput};
565        let steps = value
566            .steps()
567            .iter()
568            .map(|step| {
569                let transaction_request = step.transaction_request().to_uri();
570
571                let anchor_height = step
572                    .shielded_inputs()
573                    .map_or_else(|| 0, |i| u32::from(i.anchor_height()));
574
575                let inputs = step
576                    .transparent_inputs()
577                    .iter()
578                    .map(|utxo| proposal::ProposedInput {
579                        value: Some(proposed_input::Value::ReceivedOutput(ReceivedOutput {
580                            txid: utxo.outpoint().hash().to_vec(),
581                            value_pool: proposal::ValuePool::Transparent.into(),
582                            index: utxo.outpoint().n(),
583                            value: utxo.txout().value().into(),
584                        })),
585                    })
586                    .chain(step.shielded_inputs().iter().flat_map(|s_in| {
587                        s_in.notes().iter().map(|rec_note| proposal::ProposedInput {
588                            value: Some(proposed_input::Value::ReceivedOutput(ReceivedOutput {
589                                txid: rec_note.txid().as_ref().to_vec(),
590                                value_pool: proposal::ValuePool::from(rec_note.note().protocol())
591                                    .into(),
592                                index: rec_note.output_index().into(),
593                                value: rec_note.note().value().into(),
594                            })),
595                        })
596                    }))
597                    .chain(step.prior_step_inputs().iter().map(|p_in| {
598                        match p_in.output_index() {
599                            StepOutputIndex::Payment(i) => proposal::ProposedInput {
600                                value: Some(proposed_input::Value::PriorStepOutput(
601                                    PriorStepOutput {
602                                        step_index: p_in
603                                            .step_index()
604                                            .try_into()
605                                            .expect("Step index fits into a u32"),
606                                        payment_index: i
607                                            .try_into()
608                                            .expect("Payment index fits into a u32"),
609                                    },
610                                )),
611                            },
612                            StepOutputIndex::Change(i) => proposal::ProposedInput {
613                                value: Some(proposed_input::Value::PriorStepChange(
614                                    PriorStepChange {
615                                        step_index: p_in
616                                            .step_index()
617                                            .try_into()
618                                            .expect("Step index fits into a u32"),
619                                        change_index: i
620                                            .try_into()
621                                            .expect("Payment index fits into a u32"),
622                                    },
623                                )),
624                            },
625                        }
626                    }))
627                    .collect();
628
629                let payment_output_pools = step
630                    .payment_pools()
631                    .iter()
632                    .map(|(idx, pool_type)| proposal::PaymentOutputPool {
633                        payment_index: u32::try_from(*idx).expect("Payment index fits into a u32"),
634                        value_pool: proposal::ValuePool::from(*pool_type).into(),
635                    })
636                    .collect();
637
638                let balance = Some(proposal::TransactionBalance {
639                    proposed_change: step
640                        .balance()
641                        .proposed_change()
642                        .iter()
643                        .map(|change| proposal::ChangeValue {
644                            value: change.value().into(),
645                            value_pool: proposal::ValuePool::from(change.output_pool()).into(),
646                            memo: change.memo().map(|memo_bytes| proposal::MemoBytes {
647                                value: memo_bytes.as_slice().to_vec(),
648                            }),
649                            is_ephemeral: change.is_ephemeral(),
650                        })
651                        .collect(),
652                    fee_required: step.balance().fee_required().into(),
653                });
654
655                proposal::ProposalStep {
656                    transaction_request,
657                    payment_output_pools,
658                    anchor_height,
659                    inputs,
660                    balance,
661                    is_shielding: step.is_shielding(),
662                }
663            })
664            .collect();
665
666        proposal::Proposal {
667            proto_version: PROPOSAL_SER_V1,
668            fee_rule: match value.fee_rule() {
669                StandardFeeRule::Zip317 => proposal::FeeRule::Zip317,
670            }
671            .into(),
672            min_target_height: value.min_target_height().into(),
673            steps,
674        }
675    }
676
677    /// Attempts to parse a [`Proposal`] based upon a supported [`StandardFeeRule`] from its
678    /// protobuf representation.
679    pub fn try_into_standard_proposal<DbT, DbError>(
680        &self,
681        wallet_db: &DbT,
682    ) -> Result<Proposal<StandardFeeRule, DbT::NoteRef>, ProposalDecodingError<DbError>>
683    where
684        DbT: InputSource<Error = DbError>,
685    {
686        use self::proposal::proposed_input::Value::*;
687        match self.proto_version {
688            PROPOSAL_SER_V1 => {
689                let fee_rule = match self.fee_rule() {
690                    proposal::FeeRule::Zip317 => StandardFeeRule::Zip317,
691                    other => {
692                        return Err(ProposalDecodingError::FeeRuleNotSupported(other));
693                    }
694                };
695
696                let target_height = TargetHeight::from(self.min_target_height);
697
698                let mut steps = Vec::with_capacity(self.steps.len());
699                for step in &self.steps {
700                    let transaction_request =
701                        TransactionRequest::from_uri(&step.transaction_request)?;
702
703                    let payment_pools = step
704                        .payment_output_pools
705                        .iter()
706                        .map(|pop| {
707                            Ok((
708                                usize::try_from(pop.payment_index)
709                                    .expect("Payment index fits into a usize"),
710                                pool_type(pop.value_pool)?,
711                            ))
712                        })
713                        .collect::<Result<BTreeMap<usize, PoolType>, ProposalDecodingError<DbError>>>()?;
714
715                    #[allow(unused_mut)]
716                    let mut transparent_inputs = vec![];
717                    let mut received_notes = vec![];
718                    let mut prior_step_inputs = vec![];
719                    for (i, input) in step.inputs.iter().enumerate() {
720                        match input
721                            .value
722                            .as_ref()
723                            .ok_or(ProposalDecodingError::NullInput(i))?
724                        {
725                            ReceivedOutput(out) => {
726                                let txid = out
727                                    .parse_txid()
728                                    .map_err(ProposalDecodingError::TxIdInvalid)?;
729
730                                match out.pool_type()? {
731                                    PoolType::Transparent => {
732                                        #[cfg(not(feature = "transparent-inputs"))]
733                                        return Err(ProposalDecodingError::ValuePoolNotSupported(
734                                            out.value_pool,
735                                        ));
736
737                                        #[cfg(feature = "transparent-inputs")]
738                                        {
739                                            let outpoint = OutPoint::new(txid.into(), out.index);
740                                            transparent_inputs.push(
741                                                wallet_db
742                                                    .get_unspent_transparent_output(
743                                                        &outpoint,
744                                                        target_height,
745                                                    )
746                                                    .map_err(ProposalDecodingError::InputRetrieval)?
747                                                    .ok_or({
748                                                        ProposalDecodingError::InputNotFound(
749                                                            txid,
750                                                            PoolType::TRANSPARENT,
751                                                            out.index,
752                                                        )
753                                                    })?
754                                                    .into_wallet_output(),
755                                            );
756                                        }
757                                    }
758                                    PoolType::Shielded(protocol) => received_notes.push(
759                                        wallet_db
760                                            .get_spendable_note(
761                                                &txid,
762                                                protocol,
763                                                out.index,
764                                                target_height,
765                                            )
766                                            .map_err(ProposalDecodingError::InputRetrieval)
767                                            .and_then(|opt| {
768                                                opt.ok_or({
769                                                    ProposalDecodingError::InputNotFound(
770                                                        txid,
771                                                        PoolType::Shielded(protocol),
772                                                        out.index,
773                                                    )
774                                                })
775                                            })?,
776                                    ),
777                                }
778                            }
779                            PriorStepOutput(s_ref) => {
780                                prior_step_inputs.push(StepOutput::new(
781                                    s_ref
782                                        .step_index
783                                        .try_into()
784                                        .expect("Step index fits into a usize"),
785                                    StepOutputIndex::Payment(
786                                        s_ref
787                                            .payment_index
788                                            .try_into()
789                                            .expect("Payment index fits into a usize"),
790                                    ),
791                                ));
792                            }
793                            PriorStepChange(s_ref) => {
794                                prior_step_inputs.push(StepOutput::new(
795                                    s_ref
796                                        .step_index
797                                        .try_into()
798                                        .expect("Step index fits into a usize"),
799                                    StepOutputIndex::Change(
800                                        s_ref
801                                            .change_index
802                                            .try_into()
803                                            .expect("Payment index fits into a usize"),
804                                    ),
805                                ));
806                            }
807                        }
808                    }
809
810                    let shielded_inputs = NonEmpty::from_vec(received_notes)
811                        .map(|notes| ShieldedInputs::from_parts(step.anchor_height.into(), notes));
812
813                    let proto_balance = step
814                        .balance
815                        .as_ref()
816                        .ok_or(ProposalDecodingError::BalanceInvalid)?;
817                    let balance = TransactionBalance::new(
818                        proto_balance
819                            .proposed_change
820                            .iter()
821                            .map(|cv| -> Result<ChangeValue, ProposalDecodingError<_>> {
822                                let value = Zatoshis::from_u64(cv.value)
823                                    .map_err(|_| ProposalDecodingError::BalanceInvalid)?;
824                                let memo = cv
825                                    .memo
826                                    .as_ref()
827                                    .map(|bytes| {
828                                        MemoBytes::from_bytes(&bytes.value)
829                                            .map_err(ProposalDecodingError::MemoInvalid)
830                                    })
831                                    .transpose()?;
832                                match (cv.pool_type()?, cv.is_ephemeral) {
833                                    (PoolType::Shielded(ShieldedProtocol::Sapling), false) => {
834                                        Ok(ChangeValue::sapling(value, memo))
835                                    }
836                                    #[cfg(feature = "orchard")]
837                                    (PoolType::Shielded(ShieldedProtocol::Orchard), false) => {
838                                        Ok(ChangeValue::orchard(value, memo))
839                                    }
840                                    (PoolType::Transparent, _) if memo.is_some() => {
841                                        Err(ProposalDecodingError::TransparentMemo)
842                                    }
843                                    #[cfg(feature = "transparent-inputs")]
844                                    (PoolType::Transparent, true) => {
845                                        Ok(ChangeValue::ephemeral_transparent(value))
846                                    }
847                                    (pool, false) => {
848                                        Err(ProposalDecodingError::InvalidChangeRecipient(pool))
849                                    }
850                                    (pool, true) => {
851                                        Err(ProposalDecodingError::InvalidEphemeralRecipient(pool))
852                                    }
853                                }
854                            })
855                            .collect::<Result<Vec<_>, _>>()?,
856                        Zatoshis::from_u64(proto_balance.fee_required)
857                            .map_err(|_| ProposalDecodingError::BalanceInvalid)?,
858                    )
859                    .map_err(|_| ProposalDecodingError::BalanceInvalid)?;
860
861                    let step = Step::from_parts(
862                        &steps,
863                        transaction_request,
864                        payment_pools,
865                        transparent_inputs,
866                        shielded_inputs,
867                        prior_step_inputs,
868                        balance,
869                        step.is_shielding,
870                    )
871                    .map_err(ProposalDecodingError::ProposalInvalid)?;
872
873                    steps.push(step);
874                }
875
876                Proposal::multi_step(
877                    fee_rule,
878                    target_height,
879                    NonEmpty::from_vec(steps).ok_or(ProposalDecodingError::NoSteps)?,
880                )
881                .map_err(ProposalDecodingError::ProposalInvalid)
882            }
883            other => Err(ProposalDecodingError::VersionInvalid(other)),
884        }
885    }
886}
