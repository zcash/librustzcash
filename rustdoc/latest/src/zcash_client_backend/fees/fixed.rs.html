24/// A change strategy that proposes change as a single output. The output pool is chosen
25/// as the most current pool that avoids unnecessary pool-crossing (with a specified
26/// fallback when the transaction has no shielded inputs). Fee calculation is delegated
27/// to the provided fee rule.
28pub struct SingleOutputChangeStrategy<I> {
29    fee_rule: FixedFeeRule,
30    change_memo: Option<MemoBytes>,
31    fallback_change_pool: ShieldedProtocol,
32    dust_output_policy: DustOutputPolicy,
33    meta_source: PhantomData<I>,
34}
35
36impl<I> SingleOutputChangeStrategy<I> {
37    /// Constructs a new [`SingleOutputChangeStrategy`] with the specified fee rule
38    /// and change memo.
39    ///
40    /// `fallback_change_pool` is used when more than one shielded pool is enabled via
41    /// feature flags, and the transaction has no shielded inputs.
42    pub fn new(
43        fee_rule: FixedFeeRule,
44        change_memo: Option<MemoBytes>,
45        fallback_change_pool: ShieldedProtocol,
46        dust_output_policy: DustOutputPolicy,
47    ) -> Self {
48        Self {
49            fee_rule,
50            change_memo,
51            fallback_change_pool,
52            dust_output_policy,
53            meta_source: PhantomData,
54        }
55    }
56}
57
58impl<I: InputSource> ChangeStrategy for SingleOutputChangeStrategy<I> {
59    type FeeRule = FixedFeeRule;
60    type Error = BalanceError;
61    type MetaSource = I;
62    type AccountMetaT = ();
63
64    fn fee_rule(&self) -> &Self::FeeRule {
65        &self.fee_rule
66    }
67
68    fn fetch_wallet_meta(
69        &self,
70        _meta_source: &Self::MetaSource,
71        _account: <Self::MetaSource as InputSource>::AccountId,
72        _target_height: TargetHeight,
73        _exclude: &[<Self::MetaSource as crate::data_api::InputSource>::NoteRef],
74    ) -> Result<Self::AccountMetaT, <Self::MetaSource as crate::data_api::InputSource>::Error> {
75        Ok(())
76    }
77
78    fn compute_balance<P: consensus::Parameters, NoteRefT: Clone>(
79        &self,
80        params: &P,
81        target_height: TargetHeight,
82        transparent_inputs: &[impl transparent::InputView],
83        transparent_outputs: &[impl transparent::OutputView],
84        sapling: &impl sapling_fees::BundleView<NoteRefT>,
85        #[cfg(feature = "orchard")] orchard: &impl orchard_fees::BundleView<NoteRefT>,
86        ephemeral_balance: Option<EphemeralBalance>,
87        _wallet_meta: &Self::AccountMetaT,
88    ) -> Result<TransactionBalance, ChangeError<Self::Error, NoteRefT>> {
89        let split_policy = SplitPolicy::single_output();
90        let cfg = SinglePoolBalanceConfig::new(
91            params,
92            &self.fee_rule,
93            &self.dust_output_policy,
94            self.fee_rule.fixed_fee(),
95            &split_policy,
96            self.fallback_change_pool,
97            Zatoshis::ZERO,
98            0,
99        );
100
101        single_pool_output_balance(
102            cfg,
103            None,
104            target_height,
105            transparent_inputs,
106            transparent_outputs,
107            sapling,
108            #[cfg(feature = "orchard")]
109            orchard,
110            self.change_memo.as_ref(),
111            ephemeral_balance,
112        )
113    }
114}

140    #[test]
141    fn change_without_dust() {
142        let fee_rule = FixedFeeRule::non_standard(MINIMUM_FEE);
143        let change_strategy = SingleOutputChangeStrategy::<MockWalletDb>::new(
144            fee_rule,
145            None,
146            ShieldedProtocol::Sapling,
147            DustOutputPolicy::default(),
148        );
149
150        // spend a single Sapling note that is sufficient to pay the fee
151        let result = change_strategy.compute_balance(
152            &Network::TestNetwork,
153            Network::TestNetwork
154                .activation_height(NetworkUpgrade::Nu5)
155                .unwrap()
156                .into(),
157            &[] as &[TestTransparentInput],
158            &[] as &[TxOut],
159            &(
160                sapling::builder::BundleType::DEFAULT,
161                &[TestSaplingInput {
162                    note_id: 0,
163                    value: Zatoshis::const_from_u64(60000),
164                }][..],
165                &[SaplingPayment::new(Zatoshis::const_from_u64(40000))][..],
166            ),
167            #[cfg(feature = "orchard")]
168            &orchard_fees::EmptyBundleView,
169            None,
170            &(),
171        );
172
173        assert_matches!(
174            result,
175            Ok(balance) if
176                balance.proposed_change() == [ChangeValue::sapling(Zatoshis::const_from_u64(10000), None)] &&
177                balance.fee_required() == MINIMUM_FEE
178        );
179    }

184        let change_strategy = SingleOutputChangeStrategy::<MockWalletDb>::new(
185            fee_rule,
186            None,

197                .into(),
198            &[] as &[TestTransparentInput],
199            &[] as &[TxOut],

201                sapling::builder::BundleType::DEFAULT,
202                &[
203                    TestSaplingInput {

205                        value: Zatoshis::const_from_u64(40000),

207                    // enough to pay a fee, plus dust
208                    TestSaplingInput {
209                        note_id: 0,
210                        value: Zatoshis::const_from_u64(10100),
211                    },
212                ][..],
213                &[SaplingPayment::new(Zatoshis::const_from_u64(40000))][..],

215            #[cfg(feature = "orchard")]
216            &orchard_fees::EmptyBundleView,
217            None,
218            &(),
219        );
220
221        assert_matches!(
222            result,
223            Err(ChangeError::InsufficientFunds { available, required })
224            if available == Zatoshis::const_from_u64(50100) && required == Zatoshis::const_from_u64(60000)
225        );
226    }
