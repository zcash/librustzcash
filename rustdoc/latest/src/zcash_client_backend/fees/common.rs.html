24pub(crate) struct NetFlows {
25    t_in: Zatoshis,
26    t_out: Zatoshis,
27    sapling_in: Zatoshis,
28    sapling_out: Zatoshis,
29    orchard_in: Zatoshis,
30    orchard_out: Zatoshis,
31}
32
33impl NetFlows {
34    fn total_in(&self) -> Result<Zatoshis, BalanceError> {
35        (self.t_in + self.sapling_in + self.orchard_in).ok_or(BalanceError::Overflow)
36    }
37    fn total_out(&self) -> Result<Zatoshis, BalanceError> {
38        (self.t_out + self.sapling_out + self.orchard_out).ok_or(BalanceError::Overflow)
39    }
40    /// Returns true iff the flows excluding change are fully transparent.
41    fn is_transparent(&self) -> bool {
42        !(self.sapling_in.is_positive()
43            || self.sapling_out.is_positive()
44            || self.orchard_in.is_positive()
45            || self.orchard_out.is_positive())
46    }
47}

49#[allow(clippy::too_many_arguments)]
50pub(crate) fn calculate_net_flows<NoteRefT: Clone, F: FeeRule, E>(
51    transparent_inputs: &[impl transparent::InputView],
52    transparent_outputs: &[impl transparent::OutputView],
53    sapling: &impl sapling_fees::BundleView<NoteRefT>,
54    #[cfg(feature = "orchard")] orchard: &impl orchard_fees::BundleView<NoteRefT>,
55    ephemeral_balance: Option<&EphemeralBalance>,
56) -> Result<NetFlows, ChangeError<E, NoteRefT>>
57where
58    E: From<F::Error> + From<BalanceError>,
59{
60    let overflow = || ChangeError::StrategyError(E::from(BalanceError::Overflow));
61
62    let t_in = transparent_inputs
63        .iter()
64        .map(|t_in| t_in.coin().value)
65        .chain(ephemeral_balance.and_then(|b| b.ephemeral_input_amount()))
66        .sum::<Option<_>>()
67        .ok_or_else(overflow)?;
68    let t_out = transparent_outputs
69        .iter()
70        .map(|t_out| t_out.value())
71        .chain(ephemeral_balance.and_then(|b| b.ephemeral_output_amount()))
72        .sum::<Option<_>>()
73        .ok_or_else(overflow)?;
74    let sapling_in = sapling
75        .inputs()
76        .iter()
77        .map(sapling_fees::InputView::<NoteRefT>::value)
78        .sum::<Option<_>>()
79        .ok_or_else(overflow)?;
80    let sapling_out = sapling
81        .outputs()
82        .iter()
83        .map(sapling_fees::OutputView::value)
84        .sum::<Option<_>>()
85        .ok_or_else(overflow)?;
86
87    #[cfg(feature = "orchard")]
88    let orchard_in = orchard
89        .inputs()
90        .iter()
91        .map(orchard_fees::InputView::<NoteRefT>::value)
92        .sum::<Option<_>>()
93        .ok_or_else(overflow)?;
94    #[cfg(not(feature = "orchard"))]
95    let orchard_in = Zatoshis::ZERO;
96
97    #[cfg(feature = "orchard")]
98    let orchard_out = orchard
99        .outputs()
100        .iter()
101        .map(orchard_fees::OutputView::value)
102        .sum::<Option<_>>()
103        .ok_or_else(overflow)?;
104    #[cfg(not(feature = "orchard"))]
105    let orchard_out = Zatoshis::ZERO;
106
107    Ok(NetFlows {
108        t_in,
109        t_out,
110        sapling_in,
111        sapling_out,
112        orchard_in,
113        orchard_out,
114    })
115}

117/// Decide which shielded pool change should go to if there is any.
118pub(crate) fn select_change_pool(
119    _net_flows: &NetFlows,
120    _fallback_change_pool: ShieldedProtocol,
121) -> ShieldedProtocol {
122    // TODO: implement a less naive strategy for selecting the pool to which change will be sent.
123    #[cfg(feature = "orchard")]
124    if _net_flows.orchard_in.is_positive() || _net_flows.orchard_out.is_positive() {
125        // Send change to Orchard if we're spending any Orchard inputs or creating any Orchard outputs.
126        ShieldedProtocol::Orchard
127    } else if _net_flows.sapling_in.is_positive() || _net_flows.sapling_out.is_positive() {
128        // Otherwise, send change to Sapling if we're spending any Sapling inputs or creating any
129        // Sapling outputs, so that we avoid pool-crossing.
130        ShieldedProtocol::Sapling
131    } else {
132        // The flows are transparent, so there may not be change. If there is, the caller
133        // gets to decide where to shield it.
134        _fallback_change_pool
135    }
136    #[cfg(not(feature = "orchard"))]
137    ShieldedProtocol::Sapling
138}

203#[allow(clippy::too_many_arguments)]
204pub(crate) fn single_pool_output_balance<P: consensus::Parameters, NoteRefT: Clone, F: FeeRule, E>(
205    cfg: SinglePoolBalanceConfig<P, F>,
206    wallet_meta: Option<&AccountMeta>,
207    target_height: BlockHeight,
208    transparent_inputs: &[impl transparent::InputView],
209    transparent_outputs: &[impl transparent::OutputView],
210    sapling: &impl sapling_fees::BundleView<NoteRefT>,
211    #[cfg(feature = "orchard")] orchard: &impl orchard_fees::BundleView<NoteRefT>,
212    change_memo: Option<&MemoBytes>,
213    ephemeral_balance: Option<&EphemeralBalance>,
214) -> Result<TransactionBalance, ChangeError<E, NoteRefT>>
215where
216    E: From<F::Error> + From<BalanceError>,
217{
218    // The change memo, if any, must be attached to the change in the intermediate step that
219    // produces the ephemeral output, and so it should be discarded in the ultimate step; this is
220    // distinguished by identifying that this transaction has ephemeral inputs.
221    let change_memo = change_memo.filter(|_| ephemeral_balance.map_or(true, |b| !b.is_input()));
222
223    let overflow = || ChangeError::StrategyError(E::from(BalanceError::Overflow));
224    let underflow = || ChangeError::StrategyError(E::from(BalanceError::Underflow));
225
226    let net_flows = calculate_net_flows::<NoteRefT, F, E>(
227        transparent_inputs,
228        transparent_outputs,
229        sapling,
230        #[cfg(feature = "orchard")]
231        orchard,
232        ephemeral_balance,
233    )?;
234
235    let change_pool = select_change_pool(&net_flows, cfg.fallback_change_pool);
236    let target_change_count = wallet_meta.map_or(1, |m| {
237        usize::from(cfg.split_policy.target_output_count)
238            // If we cannot determine a total note count, fall back to a single output
239            .saturating_sub(m.total_note_count().unwrap_or(usize::MAX))
240            .max(1)
241    });
242    let target_change_counts = OutputManifest {
243        transparent: 0,
244        sapling: if change_pool == ShieldedProtocol::Sapling {
245            target_change_count
246        } else {
247            0
248        },
249        orchard: if change_pool == ShieldedProtocol::Orchard {
250            target_change_count
251        } else {
252            0
253        },
254    };
255
256    // We don't create a fully-transparent transaction if a change memo is used.
257    let transparent = net_flows.is_transparent() && change_memo.is_none();
258
259    // If we have a non-zero marginal fee, we need to check for uneconomic inputs.
260    // This is basically assuming that fee rules with non-zero marginal fee are
261    // "ZIP 317-like", but we can generalize later if needed.
262    if cfg.marginal_fee.is_positive() {
263        // Is it certain that there will be a change output? If it is not certain,
264        // we should call `check_for_uneconomic_inputs` with `possible_change`
265        // including both possibilities.
266        let possible_change = {
267            // These are the situations where we might not have a change output.
268            if transparent
269                || (cfg.dust_output_policy.action() == DustAction::AddDustToFee
270                    && change_memo.is_none())
271            {
272                vec![OutputManifest::ZERO, target_change_counts]
273            } else {
274                vec![target_change_counts]
275            }
276        };
277
278        check_for_uneconomic_inputs(
279            transparent_inputs,
280            transparent_outputs,
281            sapling,
282            #[cfg(feature = "orchard")]
283            orchard,
284            cfg.marginal_fee,
285            cfg.grace_actions,
286            &possible_change[..],
287            ephemeral_balance,
288        )?;
289    }
290
291    let total_in = net_flows
292        .total_in()
293        .map_err(|e| ChangeError::StrategyError(E::from(e)))?;
294    let total_out = net_flows
295        .total_out()
296        .map_err(|e| ChangeError::StrategyError(E::from(e)))?;
297
298    let sapling_input_count = sapling
299        .bundle_type()
300        .num_spends(sapling.inputs().len())
301        .map_err(ChangeError::BundleError)?;
302    let sapling_output_count = |change_count| {
303        sapling
304            .bundle_type()
305            .num_outputs(
306                sapling.inputs().len(),
307                sapling.outputs().len() + change_count,
308            )
309            .map_err(ChangeError::BundleError)
310    };
311
312    #[cfg(feature = "orchard")]
313    let orchard_action_count = |change_count| {
314        orchard
315            .bundle_type()
316            .num_actions(
317                orchard.inputs().len(),
318                orchard.outputs().len() + change_count,
319            )
320            .map_err(ChangeError::BundleError)
321    };
322    #[cfg(not(feature = "orchard"))]
323    let orchard_action_count = |change_count: usize| -> Result<usize, ChangeError<E, NoteRefT>> {
324        if change_count != 0 {
325            Err(ChangeError::BundleError(
326                "Nonzero Orchard change requested but the `orchard` feature is not enabled.",
327            ))
328        } else {
329            Ok(0)
330        }
331    };
332
333    // Once we calculate the balance with and without change, there are five cases:
334    //
335    // 1. Insufficient funds even without change.
336    // 2. The fee amount without change exactly cancels out the net flow balance.
337    // 3. The fee amount without change is smaller than the change.
338    //    3a. Insufficient funds once the change output is added.
339    //    3b. The fee amount with change exactly cancels out the net flow balance.
340    //    3c. The fee amount with change leaves a non-zero change value.
341    //
342    // Case 2 happens for the second transaction of a ZIP 320 pair. In that case
343    // the transaction will be fully transparent, and there must be no change.
344    //
345    // If cases 2 or 3b happen for a transaction with any shielded flows, we
346    // want there to be a zero-value shielded change output anyway (i.e. treat
347    // case 2 as case 3, and case 3b as case 3c), because:
348    // * being able to distinguish these cases potentially leaks too much
349    //   information (an adversary that knows the number of external recipients
350    //   and the sum of their outputs learns the sum of the inputs if no change
351    //   output is present); and
352    // * we will then always have an shielded output in which to put change_memo,
353    //   if one is used.
354    //
355    // Note that using the `DustAction::AddDustToFee` policy inherently leaks
356    // more information.
357
358    let transparent_input_sizes = transparent_inputs
359        .iter()
360        .map(|i| i.serialized_size())
361        .chain(
362            ephemeral_balance
363                .and_then(|b| b.ephemeral_input_amount())
364                .map(|_| transparent::InputSize::STANDARD_P2PKH),
365        );
366    let transparent_output_sizes = transparent_outputs
367        .iter()
368        .map(|i| i.serialized_size())
369        .chain(
370            ephemeral_balance
371                .and_then(|b| b.ephemeral_output_amount())
372                .map(|_| P2PKH_STANDARD_OUTPUT_SIZE),
373        );
374
375    let fee_without_change = cfg
376        .fee_rule
377        .fee_required(
378            cfg.params,
379            target_height,
380            transparent_input_sizes.clone(),
381            transparent_output_sizes.clone(),
382            sapling_input_count,
383            sapling_output_count(0)?,
384            orchard_action_count(0)?,
385        )
386        .map_err(|fee_error| ChangeError::StrategyError(E::from(fee_error)))?;
387
388    let fee_with_change = max(
389        fee_without_change,
390        cfg.fee_rule
391            .fee_required(
392                cfg.params,
393                target_height,
394                transparent_input_sizes.clone(),
395                transparent_output_sizes.clone(),
396                sapling_input_count,
397                sapling_output_count(target_change_counts.sapling())?,
398                orchard_action_count(target_change_counts.orchard())?,
399            )
400            .map_err(|fee_error| ChangeError::StrategyError(E::from(fee_error)))?,
401    );
402
403    let total_out_plus_fee_without_change =
404        (total_out + fee_without_change).ok_or_else(overflow)?;
405    let total_out_plus_fee_with_change = (total_out + fee_with_change).ok_or_else(overflow)?;
406
407    #[allow(unused_mut)]
408    let (mut change, fee) = {
409        if transparent && total_in < total_out_plus_fee_without_change {
410            // Case 1 for a tx with all transparent flows.
411            return Err(ChangeError::InsufficientFunds {
412                available: total_in,
413                required: total_out_plus_fee_without_change,
414            });
415        } else if transparent && total_in == total_out_plus_fee_without_change {
416            // Case 2 for a tx with all transparent flows.
417            (vec![], fee_without_change)
418        } else if total_in < total_out_plus_fee_with_change {
419            // Case 3a, or case 1 or 2 with non-transparent flows.
420            return Err(ChangeError::InsufficientFunds {
421                available: total_in,
422                required: total_out_plus_fee_with_change,
423            });
424        } else {
425            // Case 3b or 3c.
426            let proposed_change =
427                (total_in - total_out_plus_fee_with_change).expect("checked above");
428
429            // We obtain a split count based on the total number of notes of sufficient size
430            // available in the wallet, irrespective of pool. If we don't have any wallet metadata
431            // available, we fall back to generating a single change output.
432            let split_count = wallet_meta.map_or(NonZeroUsize::MIN, |wm| {
433                cfg.split_policy.split_count(
434                    wm.total_note_count(),
435                    wm.total_value(),
436                    proposed_change,
437                )
438            });
439            let per_output_change = proposed_change.div_with_remainder(
440                NonZeroU64::new(
441                    u64::try_from(usize::from(split_count)).expect("usize fits into u64"),
442                )
443                .unwrap(),
444            );
445
446            // If we don't have as many change outputs as we expected, recompute the fee.
447            let (fee_with_change, excess_fee) =
448                if usize::from(split_count) < target_change_counts.total_shielded() {
449                    let new_fee_with_change = cfg
450                        .fee_rule
451                        .fee_required(
452                            cfg.params,
453                            target_height,
454                            transparent_input_sizes,
455                            transparent_output_sizes,
456                            sapling_input_count,
457                            sapling_output_count(if change_pool == ShieldedProtocol::Sapling {
458                                usize::from(split_count)
459                            } else {
460                                0
461                            })?,
462                            orchard_action_count(if change_pool == ShieldedProtocol::Orchard {
463                                usize::from(split_count)
464                            } else {
465                                0
466                            })?,
467                        )
468                        .map_err(|fee_error| ChangeError::StrategyError(E::from(fee_error)))?;
469                    (
470                        new_fee_with_change,
471                        (fee_with_change - new_fee_with_change).unwrap_or(Zatoshis::ZERO),
472                    )
473                } else {
474                    (fee_with_change, Zatoshis::ZERO)
475                };
476
477            let simple_case = || {
478                (
479                    (0usize..split_count.into())
480                        .map(|i| {
481                            ChangeValue::shielded(
482                                change_pool,
483                                if i == 0 {
484                                    // Add any remainder to the first output only
485                                    (*per_output_change.quotient()
486                                        + *per_output_change.remainder()
487                                        + excess_fee)
488                                        .unwrap()
489                                } else {
490                                    // For any other output, the change value will just be the
491                                    // quotient.
492                                    *per_output_change.quotient()
493                                },
494                                change_memo.cloned(),
495                            )
496                        })
497                        .collect(),
498                    fee_with_change,
499                )
500            };
501
502            let change_dust_threshold = cfg
503                .dust_output_policy
504                .dust_threshold()
505                .unwrap_or(cfg.default_dust_threshold);
506
507            if proposed_change < change_dust_threshold {
508                match cfg.dust_output_policy.action() {
509                    DustAction::Reject => {
510                        // Always allow zero-valued change even for the `Reject` policy:
511                        // * it should be allowed in order to record change memos and to improve
512                        //   indistinguishability;
513                        // * this case occurs in practice when sending all funds from an account;
514                        // * zero-valued notes do not require witness tracking;
515                        // * the effect on trial decryption overhead is small.
516                        if proposed_change.is_zero() && excess_fee.is_zero() {
517                            simple_case()
518                        } else {
519                            let shortfall =
520                                (change_dust_threshold - proposed_change).ok_or_else(underflow)?;
521
522                            return Err(ChangeError::InsufficientFunds {
523                                available: total_in,
524                                required: (total_in + shortfall).ok_or_else(overflow)?,
525                            });
526                        }
527                    }
528                    DustAction::AllowDustChange => simple_case(),
529                    DustAction::AddDustToFee => {
530                        // Zero-valued change is also always allowed for this policy, but when
531                        // no change memo is given, we might omit the change output instead.
532
533                        let fee_with_dust = (total_in - total_out)
534                            .expect("we already checked for sufficient funds");
535                        // We can add a change output if necessary.
536                        assert!(fee_with_change <= fee_with_dust);
537
538                        let reasonable_fee = (fee_with_change + (MINIMUM_FEE * 10u64).unwrap())
539                            .ok_or_else(overflow)?;
540
541                        if fee_with_dust > reasonable_fee {
542                            // Defend against losing money by using AddDustToFee with a too-high
543                            // dust threshold.
544                            simple_case()
545                        } else if change_memo.is_some() {
546                            (
547                                vec![ChangeValue::shielded(
548                                    change_pool,
549                                    Zatoshis::ZERO,
550                                    change_memo.cloned(),
551                                )],
552                                fee_with_dust,
553                            )
554                        } else {
555                            (vec![], fee_with_dust)
556                        }
557                    }
558                }
559            } else {
560                simple_case()
561            }
562        }
563    };
564    #[cfg(feature = "transparent-inputs")]
565    change.extend(
566        ephemeral_balance
567            .and_then(|b| b.ephemeral_output_amount())
568            .map(ChangeValue::ephemeral_transparent),
569    );
570
571    TransactionBalance::new(change, fee).map_err(|_| overflow())
572}

574/// Returns a `[ChangeStrategy::DustInputs]` error if some of the inputs provided
575/// to the transaction have value less than the marginal fee, and could not be
576/// determined to have any economic value in the context of this input selection.
577///
578/// This determination is potentially conservative in the sense that outputs
579/// with value less than the marginal fee might be excluded, even though in
580/// practice they would not cause the fee to increase. Outputs with value
581/// greater than the marginal fee will never be excluded.
582///
583/// `possible_change` is a slice of `(transparent_change, sapling_change, orchard_change)`
584/// tuples indicating possible combinations of how many change outputs (0 or 1)
585/// might be included in the transaction for each pool. The shape of the tuple
586/// does not depend on which protocol features are enabled.
587#[allow(clippy::too_many_arguments)]
588pub(crate) fn check_for_uneconomic_inputs<NoteRefT: Clone, E>(
589    transparent_inputs: &[impl transparent::InputView],
590    transparent_outputs: &[impl transparent::OutputView],
591    sapling: &impl sapling_fees::BundleView<NoteRefT>,
592    #[cfg(feature = "orchard")] orchard: &impl orchard_fees::BundleView<NoteRefT>,
593    marginal_fee: Zatoshis,
594    grace_actions: usize,
595    possible_change: &[OutputManifest],
596    ephemeral_balance: Option<&EphemeralBalance>,
597) -> Result<(), ChangeError<E, NoteRefT>> {
598    let mut t_dust: Vec<_> = transparent_inputs
599        .iter()
600        .filter_map(|i| {
601            // For now, we're just assuming P2PKH inputs, so we don't check the
602            // size of the input script.
603            if i.coin().value <= marginal_fee {
604                Some(i.outpoint().clone())
605            } else {
606                None
607            }
608        })
609        .collect();
610
611    let mut s_dust: Vec<_> = sapling
612        .inputs()
613        .iter()
614        .filter_map(|i| {
615            if sapling_fees::InputView::<NoteRefT>::value(i) <= marginal_fee {
616                Some(sapling_fees::InputView::<NoteRefT>::note_id(i).clone())
617            } else {
618                None
619            }
620        })
621        .collect();
622
623    #[cfg(feature = "orchard")]
624    let mut o_dust: Vec<NoteRefT> = orchard
625        .inputs()
626        .iter()
627        .filter_map(|i| {
628            if orchard_fees::InputView::<NoteRefT>::value(i) <= marginal_fee {
629                Some(orchard_fees::InputView::<NoteRefT>::note_id(i).clone())
630            } else {
631                None
632            }
633        })
634        .collect();
635    #[cfg(not(feature = "orchard"))]
636    let mut o_dust: Vec<NoteRefT> = vec![];
637
638    // If we don't have any dust inputs, there is nothing to check.
639    if t_dust.is_empty() && s_dust.is_empty() && o_dust.is_empty() {
640        return Ok(());
641    }
642
643    let (t_inputs_len, t_outputs_len) = (
644        transparent_inputs.len() + usize::from(ephemeral_balance.is_some_and(|b| b.is_input())),
645        transparent_outputs.len() + usize::from(ephemeral_balance.is_some_and(|b| b.is_output())),
646    );
647    let (s_inputs_len, s_outputs_len) = (sapling.inputs().len(), sapling.outputs().len());
648    #[cfg(feature = "orchard")]
649    let (o_inputs_len, o_outputs_len) = (orchard.inputs().len(), orchard.outputs().len());
650    #[cfg(not(feature = "orchard"))]
651    let (o_inputs_len, o_outputs_len) = (0usize, 0usize);
652
653    let t_non_dust = t_inputs_len.checked_sub(t_dust.len()).unwrap();
654    let s_non_dust = s_inputs_len.checked_sub(s_dust.len()).unwrap();
655    let o_non_dust = o_inputs_len.checked_sub(o_dust.len()).unwrap();
656
657    // Return the number of allowed dust inputs from each pool.
658    let allowed_dust = |change: &OutputManifest| {
659        // Here we assume a "ZIP 317-like" fee model in which the existence of an output
660        // to a given pool implies that a corresponding input from that pool can be
661        // provided without increasing the fee. (This is also likely to be true for
662        // future fee models if we do not want to penalize use of Orchard relative to
663        // other pools.)
664        //
665        // Under that assumption, we want to calculate the maximum number of dust inputs
666        // from each pool, out of the ones we actually have, that can be economically
667        // spent along with the non-dust inputs. Get an initial estimate by calculating
668        // the number of dust inputs in each pool that will be allowed regardless of
669        // padding or grace.
670
671        let t_allowed = min(
672            t_dust.len(),
673            (t_outputs_len + change.transparent).saturating_sub(t_non_dust),
674        );
675        let s_allowed = min(
676            s_dust.len(),
677            (s_outputs_len + change.sapling).saturating_sub(s_non_dust),
678        );
679        let o_allowed = min(
680            o_dust.len(),
681            (o_outputs_len + change.orchard).saturating_sub(o_non_dust),
682        );
683
684        // We'll be spending the non-dust and allowed dust in each pool.
685        let t_req_inputs = t_non_dust + t_allowed;
686        let s_req_inputs = s_non_dust + s_allowed;
687        #[cfg(feature = "orchard")]
688        let o_req_inputs = o_non_dust + o_allowed;
689
690        // This calculates the hypothetical number of actions with given extra inputs,
691        // for ZIP 317 and the padding rules in effect. The padding rules for each
692        // pool are subtle (they also depend on `bundle_required` for example), so we
693        // must actually call them rather than try to predict their effect. To tell
694        // whether we can freely add an extra input from a given pool, we need to call
695        // them both with and without that input; if the number of actions does not
696        // increase, then the input is free to add.
697        let hypothetical_actions = |t_extra, s_extra, _o_extra| {
698            let s_spend_count = sapling
699                .bundle_type()
700                .num_spends(s_req_inputs + s_extra)
701                .map_err(ChangeError::BundleError)?;
702
703            let s_output_count = sapling
704                .bundle_type()
705                .num_outputs(s_req_inputs + s_extra, s_outputs_len + change.sapling)
706                .map_err(ChangeError::BundleError)?;
707
708            #[cfg(feature = "orchard")]
709            let o_action_count = orchard
710                .bundle_type()
711                .num_actions(o_req_inputs + _o_extra, o_outputs_len + change.orchard)
712                .map_err(ChangeError::BundleError)?;
713            #[cfg(not(feature = "orchard"))]
714            let o_action_count = 0;
715
716            // To calculate the number of unused actions, we assume that transparent inputs
717            // and outputs are P2PKH.
718            Ok(
719                max(t_req_inputs + t_extra, t_outputs_len + change.transparent)
720                    + max(s_spend_count, s_output_count)
721                    + o_action_count,
722            )
723        };
724
725        // First calculate the baseline number of logical actions with only the definitely
726        // allowed inputs estimated above. If it is less than `grace_actions`, try to allocate
727        // a grace input first to transparent dust, then to Sapling dust, then to Orchard dust.
728        // If the number of actions increases, it was not possible to allocate that input for
729        // free. This approach is sufficient because at most one such input can be allocated,
730        // since `grace_actions` is at most 2 for ZIP 317 and there must be at least one
731        // logical action. (If `grace_actions` were greater than 2 then the code would still
732        // be correct, it would just not find all potential extra inputs.)
733
734        let baseline = hypothetical_actions(0, 0, 0)?;
735
736        let (t_extra, s_extra, o_extra) = if baseline >= grace_actions {
737            (0, 0, 0)
738        } else if t_dust.len() > t_allowed && hypothetical_actions(1, 0, 0)? <= baseline {
739            (1, 0, 0)
740        } else if s_dust.len() > s_allowed && hypothetical_actions(0, 1, 0)? <= baseline {
741            (0, 1, 0)
742        } else if o_dust.len() > o_allowed && hypothetical_actions(0, 0, 1)? <= baseline {
743            (0, 0, 1)
744        } else {
745            (0, 0, 0)
746        };
747        Ok(OutputManifest {
748            transparent: t_allowed + t_extra,
749            sapling: s_allowed + s_extra,
750            orchard: o_allowed + o_extra,
751        })
752    };
753
754    // Find the least number of allowed dust inputs for each pool for any `possible_change`.
755    let allowed = possible_change
756        .iter()
757        .map(allowed_dust)
758        .collect::<Result<Vec<_>, _>>()?
759        .into_iter()
760        .reduce(|l, r| OutputManifest {
761            transparent: min(l.transparent, r.transparent),
762            sapling: min(l.sapling, r.sapling),
763            orchard: min(l.orchard, r.orchard),
764        })
765        .expect("possible_change is nonempty");
766
767    // The inputs in the tail of each list after the first `*_allowed` are returned as uneconomic.
768    // The caller should order the inputs from most to least preferred to spend.
769    let t_dust = t_dust.split_off(allowed.transparent);
770    let s_dust = s_dust.split_off(allowed.sapling);
771    let o_dust = o_dust.split_off(allowed.orchard);
772
773    if t_dust.is_empty() && s_dust.is_empty() && o_dust.is_empty() {
774        Ok(())
775    } else {
776        Err(ChangeError::DustInputs {
777            transparent: t_dust,
778            sapling: s_dust,
779            #[cfg(feature = "orchard")]
780            orchard: o_dust,
781        })
782    }
783}
