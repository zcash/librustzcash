28/// How a particular connection failure should be retried.
29pub enum Retry {
30    /// Retry using the same Tor circuits that resulted in this error.
31    Same,
32    /// Retry using separate Tor circuits isolated from any other Tor usage.
33    Isolated,
34}

36pub(super) fn url_is_https(url: &Uri) -> Result<bool, HttpError> {
37    Ok(url.scheme().ok_or_else(|| HttpError::NonHttpUrl)? == &Scheme::HTTPS)
38}

40pub(super) fn parse_url(url: &Uri) -> Result<(bool, String, u16), Error> {
41    let is_https = url_is_https(url)?;
42
43    let host = url.host().ok_or_else(|| HttpError::NonHttpUrl)?.to_string();
44
45    let port = match url.port_u16() {
46        Some(port) => port,
47        None if is_https => 443,
48        None => 80,
49    };
50
51    Ok((is_https, host, port))
52}

54impl Client {
55    /// Makes an HTTP GET request over Tor.
56    ///
57    /// The `request` closure can be used to modify or append HTTP request headers. You
58    /// must not call the following [`Builder`] methods within it:
59    /// - [`Builder::method`] (this is internally set to `GET`).
60    /// - [`Builder::uri`] (this is internally set to `url`).
61    /// - [`Builder::header`] with header name `"Host"` (this is internally set based on
62    ///   `url`).
63    ///
64    /// Returns `Ok(response)` if an HTTP response is received, even if the HTTP status
65    /// code is not in the 200-299 success range (i.e. [`HttpError::Unsuccessful`] is
66    /// never returned).
67    ///
68    /// There are two arguments for controlling retry behaviour:
69    /// - `retry_limit` is the maximum number of times that a failed request should be
70    ///   retried. You can disable retries by setting this to 0.
71    /// - `retry_filter` can be used to only retry requests that fail in specific ways,
72    ///   and control how the retry is performed. You can disable retries by setting this
73    ///   to `|_| None`, and you can ensure the same circuit is reused by setting this to
74    ///   `|res| res.is_err().then_some(Retry::Same)` (e.g. if you require a persistent
75    ///   Tor client identity across queries).
76    #[tracing::instrument(skip(self, request, parse_response, retry_filter))]
77    pub async fn http_get<T, F: Future<Output = Result<T, Error>>>(
78        &self,
79        url: Uri,
80        request: impl Fn(Builder) -> Builder,
81        parse_response: impl FnOnce(Incoming) -> F,
82        retry_limit: u8,
83        retry_filter: impl Fn(Result<StatusCode, &Error>) -> Option<Retry>,
84    ) -> Result<Response<T>, Error> {
85        self.http_request(
86            url,
87            |builder| request(builder).method("GET"),
88            Empty::<Bytes>::new(),
89            parse_response,
90            retry_limit,
91            retry_filter,
92        )
93        .await
94    }

144    /// Makes an HTTP request over Tor.
145    ///
146    /// There are two arguments for controlling retry behaviour:
147    /// - `retry_limit` is the maximum number of times that a failed request should be
148    ///   retried. You can disable retries by setting this to 0.
149    /// - `retry_filter` can be used to only retry requests that fail in specific ways,
150    ///   and control how the retry is performed. You can disable retries by setting this
151    ///   to `|_| None`, and you can ensure the same circuit is reused by setting this to
152    ///   `|res| res.is_err().then_some(Retry::Same)` (e.g. if you require a persistent
153    ///   Tor client identity across queries).
154    async fn http_request<B, T, F>(
155        &self,
156        url: Uri,
157        request: impl Fn(Builder) -> Builder,
158        body: B,
159        parse_response: impl FnOnce(Incoming) -> F,
160        retry_limit: u8,
161        retry_filter: impl Fn(Result<StatusCode, &Error>) -> Option<Retry>,
162    ) -> Result<Response<T>, Error>
163    where
164        B: Body + Clone + Send + 'static,
165        B::Data: Send,
166        B::Error: Into<Box<dyn std::error::Error + Send + Sync>>,
167        F: Future<Output = Result<T, Error>>,
168    {
169        let mut retries_remaining = retry_limit;
170        let mut client = None;
171
172        let (parts, body) = loop {
173            let response = one_http_request(
174                &client.as_ref().unwrap_or(self).inner,
175                url.clone(),
176                &request,
177                body.clone(),
178            )
179            .await;
180
181            match (
182                retries_remaining.checked_sub(1),
183                retry_filter(response.as_ref().map(|response| response.status())),
184            ) {
185                (Some(retries), Some(retry)) => {
186                    debug!("Retrying due to filter match");
187                    retries_remaining = retries;
188
189                    match retry {
190                        Retry::Same => (),
191                        Retry::Isolated => {
192                            debug!("Switching to isolated Tor circuit for retry");
193                            client = Some(self.isolated_client());
194                        }
195                    }
196                }
197                (None, _) | (_, None) => break response,
198            }
199        }?
200        .into_parts();
201
202        Ok(Response::from_parts(parts, parse_response(body).await?))
203    }

205    /// Makes an HTTP GET request over Tor, parsing the response as JSON.
206    ///
207    /// This is a simple wapper around [`Self::http_get`]. Use that method if you need
208    /// more control over the request headers or response parsing.
209    ///
210    /// Returns `Ok(response)` if an HTTP response is received, even if the HTTP status
211    /// code is not in the 200-299 success range (i.e. [`HttpError::Unsuccessful`] is
212    /// never returned).
213    ///
214    /// There are two arguments for controlling retry behaviour:
215    /// - `retry_limit` is the maximum number of times that a failed request should be
216    ///   retried. You can disable retries by setting this to 0.
217    /// - `retry_filter` can be used to only retry requests that fail in specific ways,
218    ///   and control how the retry is performed. You can disable retries by setting this
219    ///   to `|_| None`, and you can ensure the same circuit is reused by setting this to
220    ///   `|res| res.is_err().then_some(Retry::Same)` (e.g. if you require a persistent
221    ///   Tor client identity across queries).
222    pub async fn http_get_json<T: DeserializeOwned>(
223        &self,
224        url: Uri,
225        retry_limit: u8,
226        retry_filter: impl Fn(Result<StatusCode, &Error>) -> Option<Retry>,
227    ) -> Result<Response<T>, Error> {
228        self.http_get(
229            url,
230            |builder| builder.header(hyper::header::ACCEPT, "application/json"),
231            |body| async {
232                Ok(serde_json::from_reader(
233                    body.collect()
234                        .await
235                        .map_err(HttpError::from)?
236                        .aggregate()
237                        .reader(),
238                )
239                .map_err(HttpError::from)?)
240            },
241            retry_limit,
242            retry_filter,
243        )
244        .await
245    }

248async fn one_http_request<B>(
249    tor_client: &TorClient<PreferredRuntime>,
250    url: Uri,
251    request: impl FnOnce(Builder) -> Builder,
252    body: B,
253) -> Result<Response<Incoming>, Error>
254where
255    B: Body + Send + 'static,
256    B::Data: Send,
257    B::Error: Into<Box<dyn std::error::Error + Send + Sync>>,
258{
259    let (is_https, host, port) = parse_url(&url)?;
260
261    // Connect to the server.
262    debug!("Connecting through Tor to {}:{}", host, port);
263    let stream = tor_client.connect((host.as_str(), port)).await?;
264
265    if is_https {
266        // On apple-darwin targets there's an issue with the native TLS implementation
267        // when used over Tor circuits. We use Rustls instead.
268        //
269        // https://gitlab.torproject.org/tpo/core/arti/-/issues/715
270        let root_store = RootCertStore {
271            roots: webpki_roots::TLS_SERVER_ROOTS.to_vec(),
272        };
273        let config = ClientConfig::builder()
274            .with_root_certificates(root_store)
275            .with_no_client_auth();
276        let connector = TlsConnector::from(Arc::new(config));
277        let dnsname = ServerName::try_from(host).expect("Already checked");
278        let stream = connector
279            .connect(dnsname, stream)
280            .await
281            .map_err(HttpError::Tls)?;
282        make_http_request(stream, url, request, body).await
283    } else {
284        make_http_request(stream, url, request, body).await
285    }
286}

288async fn make_http_request<B>(
289    stream: impl AsyncRead + AsyncWrite + Unpin + Send + 'static,
290    url: Uri,
291    request: impl FnOnce(Builder) -> Builder,
292    body: B,
293) -> Result<Response<Incoming>, Error>
294where
295    B: Body + Send + 'static,
296    B::Data: Send,
297    B::Error: Into<Box<dyn std::error::Error + Send + Sync>>,
298{
299    debug!("Making request");
300    let (mut sender, connection) = conn::http1::handshake(TokioIo::new(stream))
301        .await
302        .map_err(HttpError::from)?;
303
304    // Spawn a task to poll the connection and drive the HTTP state.
305    PreferredRuntime::current()?
306        .spawn(async move {
307            if let Err(e) = connection.await {
308                error!("Connection failed: {}", e);
309            }
310        })
311        .map_err(HttpError::from)?;
312
313    // Build the request. We let the caller make whatever request modifications they need,
314    // and then set the Host and URI afterwards so that they are guaranteed to match the
315    // circuit and TLS connection.
316    let req = request(Request::builder())
317        .header(
318            hyper::header::HOST,
319            url.authority().expect("Already checked").as_str(),
320        )
321        .uri(url)
322        .body(body)
323        .map_err(HttpError::from)?;
324    let response = sender.send_request(req).await.map_err(HttpError::from)?;
325    debug!("Response status code: {}", response.status());
326
327    Ok(response)
328}

330/// Errors that can occurr while using HTTP-over-Tor.
331#[derive(Debug)]
332pub enum HttpError {
333    /// A non-HTTP URL was encountered.
334    NonHttpUrl,
335    /// An HTTP error.
336    Http(hyper::http::Error),
337    /// A [`hyper`] error.
338    Hyper(hyper::Error),
339    /// A JSON parsing error.
340    Json(serde_json::Error),
341    /// An error occurred while spawning a background worker task for driving the HTTP
342    /// connection.
343    Spawn(futures_util::task::SpawnError),
344    /// A TLS-specific IO error.
345    Tls(io::Error),
346    /// The status code indicated that the request was unsuccessful.
347    ///
348    /// This is only returned by APIs that make specific queries, such as
349    /// [`Client::get_latest_zec_to_usd_rate`]. Generic APIs like [`Client::http_get`]
350    /// will not return this error variant.
351    Unsuccessful(hyper::http::StatusCode),
352}
353
354impl fmt::Display for HttpError {
355    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
356        match self {
357            HttpError::NonHttpUrl => write!(f, "Only HTTP or HTTPS URLs are supported"),
358            HttpError::Http(e) => write!(f, "HTTP error: {e}"),
359            HttpError::Hyper(e) => write!(f, "Hyper error: {e}"),
360            HttpError::Json(e) => write!(f, "Failed to parse JSON: {e}"),
361            HttpError::Spawn(e) => write!(f, "Failed to spawn task: {e}"),
362            HttpError::Tls(e) => write!(f, "TLS error: {e}"),
363            HttpError::Unsuccessful(status) => write!(f, "Request was unsuccessful ({status:?})"),
364        }
365    }
366}

368impl std::error::Error for HttpError {
369    fn source(&self) -> Option<&(dyn std::error::Error + 'static)> {
370        match self {
371            HttpError::NonHttpUrl => None,
372            HttpError::Http(e) => Some(e),
373            HttpError::Hyper(e) => Some(e),
374            HttpError::Json(e) => Some(e),
375            HttpError::Spawn(e) => Some(e),
376            HttpError::Tls(e) => Some(e),
377            HttpError::Unsuccessful(_) => None,
378        }
379    }
380}

407mod live_network_tests {
408    use http_body_util::BodyExt;
409    use hyper::body::Buf;
410
411    use crate::tor::{
412        Client,
413        http::{HttpError, Retry},

415
416    #[test]
417    fn httpbin() {
418        let tor_dir = tempfile::tempdir().unwrap();
419
420        tokio::runtime::Runtime::new().unwrap().block_on(async {
421            // Start a new Tor client.
422            let client = Client::create(tor_dir.path(), |_| ()).await.unwrap();
423
424            // Test HTTP GET
425            let get_response = client
426                .http_get_json::<serde_json::Value>(
427                    "https://httpbin.org/get".parse().unwrap(),
428                    3,
429                    |res| res.is_err().then_some(Retry::Same),
430                )
431                .await
432                .unwrap();
433            assert_eq!(
434                get_response.body().get("url").and_then(|v| v.as_str()),
435                Some("https://httpbin.org/get"),
436            );
437            assert_eq!(
438                get_response
439                    .body()
440                    .get("headers")
441                    .and_then(|v| v.as_object())
442                    .and_then(|h| h.get("Host"))
443                    .and_then(|v| v.as_str()),
444                Some("httpbin.org"),
445            );
446            assert!(
447                get_response
448                    .body()
449                    .get("args")
450                    .unwrap()
451                    .as_object()
452                    .unwrap()
453                    .is_empty()
454            );
455
456            // Test HTTP POST
457            let post_body = "Some body";
458            let post_response = client
459                .http_post(
460                    "https://httpbin.org/post".parse().unwrap(),
461                    |builder| builder.header(hyper::header::ACCEPT, "application/json"),
462                    http_body_util::Full::new(post_body.as_bytes()),
463                    |body| async {
464                        Ok(serde_json::from_reader::<_, serde_json::Value>(
465                            body.collect()
466                                .await
467                                .map_err(HttpError::from)?
468                                .aggregate()
469                                .reader(),
470                        )
471                        .map_err(HttpError::from)?)
472                    },
473                    3,
474                    |res| res.is_err().then_some(Retry::Same),
475                )
476                .await
477                .unwrap();
478            assert!(
479                post_response
480                    .body()
481                    .get("args")
482                    .unwrap()
483                    .as_object()
484                    .unwrap()
485                    .is_empty()
486            );
487            assert_eq!(
488                post_response.body().get("data").and_then(|v| v.as_str()),
489                Some(post_body),
490            );
491        })
