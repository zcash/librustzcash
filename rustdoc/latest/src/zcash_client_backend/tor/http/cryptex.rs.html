26/// A retry filter that attempts to avoid Tor exit node connection failures.
27///
28/// A common failure with HTTP requests over Tor is a particular exit node being blocked
29/// by the server. This may be due to other exit node activity causing some IP rate limit
30/// to be exceeded, for example.
31///
32/// If the HTTP request doesn't require a persistent Tor client identity across queries,
33/// we can retry with an isolated client in order to use new circuits that have a decent
34/// chance of using a different exit node. The isolation is not for privacy; the server
35/// can trivially link the two requests together via timing.
36///
37/// This filter attempts retries as follows:
38/// - A successful request that resulted in a client error (HTTP 400-499) will cause a
39///   retry with an isolated client.
40/// - All other errors will cause a retry with the same client.
41fn retry_filter(res: Result<StatusCode, &Error>) -> Option<Retry> {
42    match res {
43        Ok(status) => {
44            if status.is_client_error() {
45                Some(Retry::Isolated)
46            } else {
47                (!status.is_success()).then_some(Retry::Same)
48            }
49        }
50        Err(_) => Some(Retry::Same),
51    }
52}

69/// An exchange that can be queried for ZEC data.
70#[trait_variant::make(Exchange: Send)]
71#[dynosaur::dynosaur(DynExchange = dyn(box) Exchange)]
72pub trait LocalExchange {
73    /// Queries data about the USD/ZEC pair.
74    ///
75    /// The returned bid and ask data must be denominated in USD, i.e. the latest bid and
76    /// ask for 1 ZEC.
77    async fn query_zec_to_usd(&self, client: &Client) -> Result<ExchangeData, Error>;
78}

80/// Data queried from an [`Exchange`].
81#[derive(Debug)]
82pub struct ExchangeData {
83    /// The highest current bid.
84    pub bid: Decimal,
85
86    /// The lowest current ask.
87    pub ask: Decimal,
88}
89
90impl ExchangeData {
91    /// Returns the mid-point between current best bid and current best ask, to avoid
92    /// manipulation by targeted trade fulfilment.
93    fn exchange_rate(&self) -> Decimal {
94        (self.bid + self.ask) / Decimal::TWO
95    }
96}

98/// A set of [`Exchange`]s that can be queried for ZEC data.
99pub struct Exchanges {

104impl Exchanges {

116            .with(exchanges::KuCoin::unauthenticated())
117            .with(exchanges::Mexc::unauthenticated())
118            .with(exchanges::Xt::unauthenticated())
119            .build()
120    }

122    /// Returns an `Exchanges` builder.
123    ///
124    /// The `trusted` exchange will always have its data used, _if_ data is successfully
125    /// obtained via Tor (i.e. no transient failures).
126    pub fn builder(trusted: impl Exchange + 'static) -> ExchangesBuilder {
127        ExchangesBuilder::new(trusted)

131/// Builder type for [`Exchanges`].
132///
133/// Every [`Exchanges`] is configured with a "trusted" [`Exchange`] that will always have
134/// its data used, if data is successfully obtained via Tor (i.e. no transient failures).
135/// Additional data sources can be provided to [`ExchangesBuilder::with`] for resiliency
136/// against transient network failures or adversarial market manipulation on individual
137/// sources.
138///
139/// The number of times [`ExchangesBuilder::with`] is called will affect the behaviour of
140/// the final [`Exchanges`]:
141/// - With no additional sources, the trusted [`Exchange`] is used on its own.
142/// - With one additional source, the trusted [`Exchange`] is used preferentially,
143///   with the additional source as a backup if the trusted source cannot be queried.
144/// - With two or more additional sources, a minimum of three successful responses are
145///   required from any of the sources.
146pub struct ExchangesBuilder(Exchanges);
147
148impl ExchangesBuilder {
149    /// Constructs a new [`Exchanges`] builder.
150    ///
151    /// The `trusted` exchange will always have its data used, _if_ data is successfully
152    /// obtained via Tor (i.e. no transient failures).
153    pub fn new(trusted: impl Exchange + 'static) -> Self {
154        Self(Exchanges {
155            trusted: DynExchange::new_box(trusted),
156            others: vec![],
157        })
158    }
159
160    /// Adds another [`Exchange`] as a data source.
161    pub fn with(mut self, other: impl Exchange + 'static) -> Self {
162        self.0.others.push(DynExchange::new_box(other));
163        self
164    }
165
166    /// Builds the [`Exchanges`].
167    pub fn build(self) -> Exchanges {
168        self.0
169    }
170}

172impl Client {
173    /// Fetches the latest USD/ZEC exchange rate, derived from the given exchanges.
174    ///
175    /// Returns:
176    /// - `Ok(rate)` if at least one exchange request succeeds.
177    /// - `Err(_)` if none of the exchange queries succeed.
178    pub async fn get_latest_zec_to_usd_rate(
179        &self,
180        exchanges: &Exchanges,
181    ) -> Result<Decimal, Error> {
182        self.ensure_bootstrapped().await?;
183
184        // Fetch the data in parallel.
185        let res = join!(
186            DynExchange::query_zec_to_usd(&exchanges.trusted, self),
187            join_all(
188                exchanges
189                    .others
190                    .iter()
191                    .map(|e| DynExchange::query_zec_to_usd(e, self))
192            )
193        );
194        trace!(?res, "Data results");
195        let (trusted_res, other_res) = res;
196
197        // Split into successful queries and errors.
198        let mut rates: Vec<Decimal> = vec![];
199        let mut errors = vec![];
200        for res in other_res {
201            match res {
202                Ok(d) => rates.push(d.exchange_rate()),
203                Err(e) => errors.push(e),
204            }
205        }
206
207        // "Never go to sea with two chronometers; take one or three."
208        // Randomly drop one rate if necessary to have an odd number of rates, as long as
209        // we have either at least three rates, or fewer than three sources.
210        if exchanges.others.len() >= 2 && rates.len() + usize::from(trusted_res.is_ok()) < 3 {
211            error!("Too many exchange requests failed");
212            return Err(errors
213                .into_iter()
214                .next()
215                .expect("At least one request failed"));
216        }
217        let evict_random = |s: &mut Vec<Decimal>| {
218            if let Some(index) = (0..s.len()).choose(&mut thread_rng()) {
219                s.remove(index);
220            }
221        };
222        match trusted_res {
223            Ok(trusted) => {
224                if rates.len() % 2 != 0 {
225                    evict_random(&mut rates);
226                }
227                rates.push(trusted.exchange_rate());
228            }
229            Err(e) => {
230                if rates.len() % 2 == 0 {
231                    evict_random(&mut rates);
232                }
233                errors.push(e);
234            }
235        }
236
237        // If all of the requests failed, log all errors and return one of them.
238        if rates.is_empty() {
239            error!("All exchange requests failed");
240            Err(errors.into_iter().next().expect("All requests failed"))
241        } else {
242            // We have an odd number of rates; take the median.
243            assert!(rates.len() % 2 != 0);
244            rates.sort();
245            let median = rates.len() / 2;
246            Ok(rates[median])
247        }
248    }
