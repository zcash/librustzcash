19#[derive(Debug, Deserialize)]

23    change: Decimal,
24    base_vol: Decimal,

33#[derive(Debug, Deserialize)]
34#[allow(dead_code)]
35struct DigiFinexResponse {
36    date: u64,
37    code: u32,
38    ticker: (DigiFinexData,),
39}

41impl Exchange for DigiFinex {
42    async fn query_zec_to_usd(&self, client: &Client) -> Result<ExchangeData, Error> {
43        // API documentation:
44        // https://docs.digifinex.com/en-ww/spot/v3/rest.html#ticker-price
45        let res = client
46            .http_get_json::<DigiFinexResponse>(
47                "https://openapi.digifinex.com/v3/ticker?symbol=zec_usdt"
48                    .parse()
49                    .unwrap(),
50                RETRY_LIMIT,
51                retry_filter,
52            )
53            .await?;
54        let data = res.into_body().ticker.0;
55        Ok(ExchangeData {
56            bid: data.buy,
57            ask: data.sell,
58        })
59    }
