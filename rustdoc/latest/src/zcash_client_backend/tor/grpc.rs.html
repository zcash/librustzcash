18impl Client {
19    /// Connects to the `lightwalletd` server at the given endpoint.
20    pub async fn connect_to_lightwalletd(
21        &self,
22        endpoint: Uri,
23    ) -> Result<CompactTxStreamerClient<Channel>, Error> {
24        self.ensure_bootstrapped().await?;
25
26        let is_https = http::url_is_https(&endpoint)?;
27
28        let channel = Endpoint::from(endpoint);
29        let channel = if is_https {
30            channel
31                .tls_config(ClientTlsConfig::new().with_webpki_roots())
32                .map_err(GrpcError::Tonic)?
33        } else {
34            channel
35        };
36
37        let conn = channel
38            .connect_with_connector(self.http_tcp_connector())
39            .await
40            .map_err(GrpcError::Tonic)?;
41
42        Ok(CompactTxStreamerClient::new(conn))
43    }
44
45    fn http_tcp_connector(&self) -> HttpTcpConnector {
46        HttpTcpConnector {
47            client: self.clone(),
48        }
49    }
50}

52struct HttpTcpConnector {
53    client: Client,
54}
55
56impl Service<Uri> for HttpTcpConnector {
57    type Response = TokioIo<DataStream>;
58    type Error = Error;
59    type Future = Pin<Box<dyn Future<Output = Result<Self::Response, Self::Error>> + Send>>;
60
61    fn poll_ready(&mut self, _: &mut Context<'_>) -> Poll<Result<(), Self::Error>> {
62        Poll::Ready(Ok(()))
63    }
64
65    fn call(&mut self, endpoint: Uri) -> Self::Future {
66        let parsed = http::parse_url(&endpoint);
67        let client = self.client.clone();
68
69        let fut = async move {
70            let (_, host, port) = parsed?;
71
72            debug!("Connecting through Tor to {}:{}", host, port);
73            let stream = client.inner.connect((host.as_str(), port)).await?;
74
75            Ok(TokioIo::new(stream))
76        };
77
78        Box::pin(fut)
79    }
80}

82/// Errors that can occurr while using HTTP-over-Tor.
83#[derive(Debug)]
84pub enum GrpcError {
85    /// A [`tonic`] error.
86    Tonic(tonic::transport::Error),
87}
88
89impl fmt::Display for GrpcError {
90    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
91        match self {
92            GrpcError::Tonic(e) => {
93                if let Some(source) = e.source() {
94                    // Tonic doesn't include the source error in its `Display` impl;
95                    // add it manually for the benefit of our downstreams.
96                    write!(f, "Tonic error: {e}: {source}")
97                } else {
98                    write!(f, "Tonic error: {e}")
99                }
100            }
101        }
102    }
103}
104
105impl std::error::Error for GrpcError {
106    fn source(&self) -> Option<&(dyn std::error::Error + 'static)> {
107        match self {
108            GrpcError::Tonic(e) => Some(e),
109        }
110    }
111}
112
113impl From<tonic::transport::Error> for GrpcError {
114    fn from(e: tonic::transport::Error) -> Self {
115        GrpcError::Tonic(e)
116    }
