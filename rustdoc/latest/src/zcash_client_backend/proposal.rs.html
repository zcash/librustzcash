19/// Errors that can occur in construction of a [`Step`].
20#[derive(Debug, Clone)]
21pub enum ProposalError {
22    /// The total output value of the transaction request is not a valid Zcash amount.
23    RequestTotalInvalid,
24    /// The total of transaction inputs overflows the valid range of Zcash values.
25    Overflow,
26    /// The input total and output total of the payment request are not equal to one another. The
27    /// sum of transaction outputs, change, and fees is required to be exactly equal to the value
28    /// of provided inputs.
29    BalanceError {
30        input_total: Zatoshis,
31        output_total: Zatoshis,
32    },
33    /// The `is_shielding` flag may only be set to `true` under the following conditions:
34    /// * The total of transparent inputs is nonzero
35    /// * There exist no Sapling inputs
36    /// * There provided transaction request is empty; i.e. the only output values specified
37    ///   are change and fee amounts.
38    ShieldingInvalid,
39    /// No anchor information could be obtained for the specified block height.
40    AnchorNotFound(BlockHeight),
41    /// A reference to the output of a prior step is invalid.
42    ReferenceError(StepOutput),
43    /// An attempted double-spend of a prior step output was detected.
44    StepDoubleSpend(StepOutput),
45    /// An attempted double-spend of an output belonging to the wallet was detected.
46    ChainDoubleSpend(PoolType, TxId, u32),
47    /// There was a mismatch between the payments in the proposal's transaction request
48    /// and the payment pool selection values.
49    PaymentPoolsMismatch,
50    /// The proposal tried to spend a change output. Mark the `ChangeValue` as ephemeral if this is intended.
51    SpendsChange(StepOutput),
52    /// The proposal results in an invalid payment request according to ZIP-321.
53    Zip321(Zip321Error),
54    /// A proposal step created an ephemeral output that was not spent in any later step.
55    #[cfg(feature = "transparent-inputs")]
56    EphemeralOutputLeftUnspent(StepOutput),
57    /// The proposal included a payment to a TEX address and a spend from a shielded input in the same step.
58    #[cfg(feature = "transparent-inputs")]
59    PaysTexFromShielded,
60    /// The change strategy provided to input selection failed to correctly generate an ephemeral
61    /// change output when needed for sending to a TEX address.
62    #[cfg(feature = "transparent-inputs")]
63    EphemeralOutputsInvalid,
64    /// The requested proposal would link activity on an ephemeral address to other wallet
65    /// activity.
66    #[cfg(feature = "transparent-inputs")]
67    EphemeralAddressLinkability,
68}
69
70impl Display for ProposalError {
71    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
72        match self {
73            ProposalError::RequestTotalInvalid => write!(
74                f,
75                "The total requested output value is not a valid Zcash amount."
76            ),
77            ProposalError::Overflow => write!(
78                f,
79                "The total of transaction inputs overflows the valid range of Zcash values."
80            ),
81            ProposalError::BalanceError {
82                input_total,
83                output_total,
84            } => write!(
85                f,
86                "Balance error: the output total {} was not equal to the input total {}",
87                u64::from(*output_total),
88                u64::from(*input_total)
89            ),
90            ProposalError::ShieldingInvalid => write!(
91                f,
92                "The proposal violates the rules for a shielding transaction."
93            ),
94            ProposalError::AnchorNotFound(h) => {
95                write!(f, "Unable to compute anchor for block height {h:?}")
96            }
97            ProposalError::ReferenceError(r) => {
98                write!(f, "No prior step output found for reference {r:?}")
99            }
100            ProposalError::StepDoubleSpend(r) => write!(
101                f,
102                "The proposal uses the output of step {r:?} in more than one place."
103            ),
104            ProposalError::ChainDoubleSpend(pool, txid, index) => write!(
105                f,
106                "The proposal attempts to spend the same output twice: {pool}, {txid}, {index}"
107            ),
108            ProposalError::PaymentPoolsMismatch => write!(
109                f,
110                "The chosen payment pools did not match the payments of the transaction request."
111            ),
112            ProposalError::SpendsChange(r) => write!(
113                f,
114                "The proposal attempts to spends the change output created at step {r:?}.",
115            ),
116            ProposalError::Zip321(r) => {
117                write!(f, "The proposal results in an invalid payment {r:?}.",)
118            }
119            #[cfg(feature = "transparent-inputs")]
120            ProposalError::EphemeralOutputLeftUnspent(r) => write!(
121                f,
122                "The proposal created an ephemeral output at step {r:?} that was not spent in any later step.",
123            ),
124            #[cfg(feature = "transparent-inputs")]
125            ProposalError::PaysTexFromShielded => write!(
126                f,
127                "The proposal included a payment to a TEX address and a spend from a shielded input in the same step.",
128            ),
129            #[cfg(feature = "transparent-inputs")]
130            ProposalError::EphemeralOutputsInvalid => write!(
131                f,
132                "The proposal generator failed to correctly generate an ephemeral change output when needed for sending to a TEX address."
133            ),
134            #[cfg(feature = "transparent-inputs")]
135            ProposalError::EphemeralAddressLinkability => write!(
136                f,
137                "The proposal requested spending funds in a way that would link activity on an ephemeral address to other wallet activity."
138            ),
139        }
140    }
141}

176/// A proposal for a series of transactions to be created.
177///
178/// Each step of the proposal represents a separate transaction to be created. At present, only
179/// transparent outputs of earlier steps may be spent in later steps; the ability to chain shielded
180/// transaction steps may be added in a future update.
181#[derive(Clone, PartialEq, Eq)]
182pub struct Proposal<FeeRuleT, NoteRef> {
183    fee_rule: FeeRuleT,
184    min_target_height: TargetHeight,
185    steps: NonEmpty<Step<NoteRef>>,
186}
187
188impl<FeeRuleT, NoteRef> Proposal<FeeRuleT, NoteRef> {
189    /// Constructs a validated multi-step [`Proposal`].
190    ///
191    /// This operation validates the proposal for agreement between outputs and inputs
192    /// in the case of multi-step proposals, and ensures that no double-spends are being
193    /// proposed.
194    ///
195    /// Parameters:
196    /// * `fee_rule`: The fee rule observed by the proposed transaction.
197    /// * `min_target_height`: The minimum block height at which the transaction may be created.
198    /// * `steps`: A vector of steps that make up the proposal.
199    pub fn multi_step(
200        fee_rule: FeeRuleT,
201        min_target_height: TargetHeight,
202        steps: NonEmpty<Step<NoteRef>>,
203    ) -> Result<Self, ProposalError> {
204        let mut consumed_chain_inputs: BTreeSet<(PoolType, TxId, u32)> = BTreeSet::new();
205        let mut consumed_prior_inputs: BTreeSet<StepOutput> = BTreeSet::new();
206
207        for (i, step) in steps.iter().enumerate() {
208            for prior_ref in step.prior_step_inputs() {
209                // check that there are no forward references
210                if prior_ref.step_index() >= i {
211                    return Err(ProposalError::ReferenceError(*prior_ref));
212                }
213                // check that the reference is valid
214                let prior_step = &steps[prior_ref.step_index()];
215                match prior_ref.output_index() {
216                    StepOutputIndex::Payment(idx) => {
217                        if prior_step.transaction_request().payments().len() <= idx {
218                            return Err(ProposalError::ReferenceError(*prior_ref));
219                        }
220                    }
221                    StepOutputIndex::Change(idx) => {
222                        if prior_step.balance().proposed_change().len() <= idx {
223                            return Err(ProposalError::ReferenceError(*prior_ref));
224                        }
225                    }
226                }
227                // check that there are no double-spends
228                if !consumed_prior_inputs.insert(*prior_ref) {
229                    return Err(ProposalError::StepDoubleSpend(*prior_ref));
230                }
231            }
232
233            for t_out in step.transparent_inputs() {
234                let key = (
235                    PoolType::TRANSPARENT,
236                    TxId::from_bytes(*t_out.outpoint().hash()),
237                    t_out.outpoint().n(),
238                );
239                if !consumed_chain_inputs.insert(key) {
240                    return Err(ProposalError::ChainDoubleSpend(key.0, key.1, key.2));
241                }
242            }
243
244            for s_out in step.shielded_inputs().iter().flat_map(|i| i.notes().iter()) {
245                let key = (
246                    match &s_out.note() {
247                        Note::Sapling(_) => PoolType::SAPLING,
248                        #[cfg(feature = "orchard")]
249                        Note::Orchard(_) => PoolType::ORCHARD,
250                    },
251                    *s_out.txid(),
252                    s_out.output_index().into(),
253                );
254                if !consumed_chain_inputs.insert(key) {
255                    return Err(ProposalError::ChainDoubleSpend(key.0, key.1, key.2));
256                }
257            }
258        }
259
260        Ok(Self {
261            fee_rule,
262            min_target_height,
263            steps,
264        })
265    }
266
267    /// Constructs a validated [`Proposal`] having only a single step from its constituent parts.
268    ///
269    /// This operation validates the proposal for balance consistency and agreement between
270    /// the `is_shielding` flag and the structure of the proposal.
271    ///
272    /// Parameters:
273    /// * `transaction_request`: The ZIP 321 transaction request describing the payments to be
274    ///   made.
275    /// * `payment_pools`: A map from payment index to pool type.
276    /// * `transparent_inputs`: The set of previous transparent outputs to be spent.
277    /// * `shielded_inputs`: The sets of previous shielded outputs to be spent.
278    /// * `balance`: The change outputs to be added the transaction and the fee to be paid.
279    /// * `fee_rule`: The fee rule observed by the proposed transaction.
280    /// * `min_target_height`: The minimum block height at which the transaction may be created.
281    /// * `is_shielding`: A flag that identifies whether this is a wallet-internal shielding
282    ///   transaction.
283    #[allow(clippy::too_many_arguments)]
284    pub fn single_step(
285        transaction_request: TransactionRequest,
286        payment_pools: BTreeMap<usize, PoolType>,
287        transparent_inputs: Vec<WalletTransparentOutput>,
288        shielded_inputs: Option<ShieldedInputs<NoteRef>>,
289        balance: TransactionBalance,
290        fee_rule: FeeRuleT,
291        min_target_height: TargetHeight,
292        is_shielding: bool,
293    ) -> Result<Self, ProposalError> {
294        Ok(Self {
295            fee_rule,
296            min_target_height,
297            steps: NonEmpty::singleton(Step::from_parts(
298                &[],
299                transaction_request,
300                payment_pools,
301                transparent_inputs,
302                shielded_inputs,
303                vec![],
304                balance,
305                is_shielding,
306            )?),
307        })
308    }
309
310    /// Returns the fee rule to be used by the transaction builder.
311    pub fn fee_rule(&self) -> &FeeRuleT {
312        &self.fee_rule
313    }
314
315    /// Returns the target height for which the proposal was prepared.
316    ///
317    /// The chain must contain at least this many blocks in order for the proposal to
318    /// be executed.
319    pub fn min_target_height(&self) -> TargetHeight {
320        self.min_target_height
321    }
322
323    /// Returns the steps of the proposal. Each step corresponds to an independent transaction to
324    /// be generated as a result of this proposal.
325    pub fn steps(&self) -> &NonEmpty<Step<NoteRef>> {
326        &self.steps
327    }
328}

375/// The inputs to be consumed and outputs to be produced in a proposed transaction.
376#[derive(Clone, PartialEq, Eq)]
377pub struct Step<NoteRef> {
378    transaction_request: TransactionRequest,
379    payment_pools: BTreeMap<usize, PoolType>,
380    transparent_inputs: Vec<WalletTransparentOutput>,
381    shielded_inputs: Option<ShieldedInputs<NoteRef>>,
382    prior_step_inputs: Vec<StepOutput>,
383    balance: TransactionBalance,
384    is_shielding: bool,
385}
386
387impl<NoteRef> Step<NoteRef> {
388    /// Constructs a validated [`Step`] from its constituent parts.
389    ///
390    /// This operation validates the proposal for balance consistency and agreement between
391    /// the `is_shielding` flag and the structure of the proposal.
392    ///
393    /// Parameters:
394    /// * `transaction_request`: The ZIP 321 transaction request describing the payments
395    ///   to be made.
396    /// * `payment_pools`: A map from payment index to pool type. The set of payment indices
397    ///   provided here must exactly match the set of payment indices in the [`TransactionRequest`],
398    ///   and the selected pool for an index must correspond to a valid receiver of the
399    ///   address at that index (or the address itself in the case of bare transparent or Sapling
400    ///   addresses).
401    /// * `transparent_inputs`: The set of previous transparent outputs to be spent.
402    /// * `shielded_inputs`: The sets of previous shielded outputs to be spent.
403    /// * `balance`: The change outputs to be added the transaction and the fee to be paid.
404    /// * `is_shielding`: A flag that identifies whether this is a wallet-internal shielding
405    ///   transaction.
406    #[allow(clippy::too_many_arguments)]
407    pub fn from_parts(
408        prior_steps: &[Step<NoteRef>],
409        transaction_request: TransactionRequest,
410        payment_pools: BTreeMap<usize, PoolType>,
411        transparent_inputs: Vec<WalletTransparentOutput>,
412        shielded_inputs: Option<ShieldedInputs<NoteRef>>,
413        prior_step_inputs: Vec<StepOutput>,
414        balance: TransactionBalance,
415        is_shielding: bool,
416    ) -> Result<Self, ProposalError> {
417        // Verify that the set of payment pools matches exactly a set of valid payment recipients
418        if transaction_request.payments().len() != payment_pools.len() {
419            return Err(ProposalError::PaymentPoolsMismatch);
420        }
421        for (idx, pool) in &payment_pools {
422            if !transaction_request
423                .payments()
424                .get(idx)
425                .iter()
426                .any(|payment| payment.recipient_address().can_receive_as(*pool))
427            {
428                return Err(ProposalError::PaymentPoolsMismatch);
429            }
430        }
431
432        let transparent_input_total = transparent_inputs
433            .iter()
434            .map(|out| out.txout().value())
435            .try_fold(Zatoshis::ZERO, |acc, a| {
436                (acc + a).ok_or(ProposalError::Overflow)
437            })?;
438
439        let shielded_input_total = shielded_inputs
440            .iter()
441            .flat_map(|s_in| s_in.notes().iter())
442            .map(|out| out.note().value())
443            .try_fold(Zatoshis::ZERO, |acc, a| acc + a)
444            .ok_or(ProposalError::Overflow)?;
445
446        let prior_step_input_total = prior_step_inputs
447            .iter()
448            .map(|s_ref| {
449                let step = prior_steps
450                    .get(s_ref.step_index)
451                    .ok_or(ProposalError::ReferenceError(*s_ref))?;
452                Ok(match s_ref.output_index {
453                    StepOutputIndex::Payment(i) => step
454                        .transaction_request
455                        .payments()
456                        .get(&i)
457                        .ok_or(ProposalError::ReferenceError(*s_ref))?
458                        .amount(),
459                    StepOutputIndex::Change(i) => step
460                        .balance
461                        .proposed_change()
462                        .get(i)
463                        .ok_or(ProposalError::ReferenceError(*s_ref))?
464                        .value(),
465                })
466            })
467            .collect::<Result<Vec<_>, _>>()?
468            .into_iter()
469            .try_fold(Zatoshis::ZERO, |acc, a| acc + a)
470            .ok_or(ProposalError::Overflow)?;
471
472        let input_total = (transparent_input_total + shielded_input_total + prior_step_input_total)
473            .ok_or(ProposalError::Overflow)?;
474
475        let request_total = transaction_request
476            .total()
477            .map_err(|_| ProposalError::RequestTotalInvalid)?;
478        let output_total = (request_total + balance.total()).ok_or(ProposalError::Overflow)?;
479
480        if is_shielding
481            && (transparent_input_total == Zatoshis::ZERO
482                || shielded_input_total > Zatoshis::ZERO
483                || request_total > Zatoshis::ZERO)
484        {
485            return Err(ProposalError::ShieldingInvalid);
486        }
487
488        if input_total == output_total {
489            Ok(Self {
490                transaction_request,
491                payment_pools,
492                transparent_inputs,
493                shielded_inputs,
494                prior_step_inputs,
495                balance,
496                is_shielding,
497            })
498        } else {
499            Err(ProposalError::BalanceError {
500                input_total,
501                output_total,
502            })
503        }
504    }
505
506    /// Returns the transaction request that describes the payments to be made.
507    pub fn transaction_request(&self) -> &TransactionRequest {
508        &self.transaction_request
509    }
510    /// Returns the map from payment index to the pool that has been selected
511    /// for the output that will fulfill that payment.
512    pub fn payment_pools(&self) -> &BTreeMap<usize, PoolType> {
513        &self.payment_pools
514    }
515    /// Returns the transparent inputs that have been selected to fund the transaction.
516    pub fn transparent_inputs(&self) -> &[WalletTransparentOutput] {
517        &self.transparent_inputs
518    }
519    /// Returns the shielded inputs that have been selected to fund the transaction.
520    pub fn shielded_inputs(&self) -> Option<&ShieldedInputs<NoteRef>> {
521        self.shielded_inputs.as_ref()
522    }
523    /// Returns the inputs that should be obtained from the outputs of the transaction
524    /// created to satisfy a previous step of the proposal.
525    pub fn prior_step_inputs(&self) -> &[StepOutput] {
526        self.prior_step_inputs.as_ref()
527    }
528    /// Returns the change outputs to be added to the transaction and the fee to be paid.
529    pub fn balance(&self) -> &TransactionBalance {
530        &self.balance
531    }
532    /// Returns a flag indicating whether or not the proposed transaction
533    /// is exclusively wallet-internal (if it does not involve any external
534    /// recipients).
535    pub fn is_shielding(&self) -> bool {
536        self.is_shielding
537    }
538
539    /// Returns whether or not this proposal requires interaction with the specified pool.
540    pub fn involves(&self, pool_type: PoolType) -> bool {
541        let input_in_this_pool = || match pool_type {
542            PoolType::Transparent => self.is_shielding || !self.transparent_inputs.is_empty(),
543            PoolType::Shielded(ShieldedProtocol::Sapling) => {
544                self.shielded_inputs.iter().any(|s_in| {
545                    s_in.notes()
546                        .iter()
547                        .any(|note| matches!(note.note(), Note::Sapling(_)))
548                })
549            }
550            #[cfg(feature = "orchard")]
551            PoolType::Shielded(ShieldedProtocol::Orchard) => {
552                self.shielded_inputs.iter().any(|s_in| {
553                    s_in.notes()
554                        .iter()
555                        .any(|note| matches!(note.note(), Note::Orchard(_)))
556                })
557            }
558            #[cfg(not(feature = "orchard"))]
559            PoolType::Shielded(ShieldedProtocol::Orchard) => false,
560        };
561        let output_in_this_pool = || self.payment_pools().values().any(|pool| *pool == pool_type);
562        let change_in_this_pool = || {
563            self.balance
564                .proposed_change()
565                .iter()
566                .any(|c| c.output_pool() == pool_type)
567        };
568
569        input_in_this_pool() || output_in_this_pool() || change_in_this_pool()
570    }
571}

576            .field("transaction_request", &self.transaction_request)
