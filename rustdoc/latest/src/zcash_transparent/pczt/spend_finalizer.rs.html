1use ripemd::Ripemd160;
2use sha2::{Digest, Sha256};
3use zcash_script::{pattern::push_script, pv, script, solver};
4
5use crate::address::TransparentAddress;
6
7impl super::Bundle {
8    /// Finalizes the spends for this bundle.
9    ///
10    /// Returns an error if any spend uses an unsupported script format. The supported
11    /// script formats are:
12    /// - P2PKH
13    /// - P2SH with one of these redeem script formats:
14    ///   - P2MS
15    pub fn finalize_spends(&mut self) -> Result<(), SpendFinalizerError> {
16        // For each input, the Spend Finalizer determines if the input has enough data to
17        // pass validation. If it does, it must construct the `script_sig` and place it
18        // into the PCZT input. If `script_sig` is empty for an input, the field should
19        // remain unset rather than assigned an empty array.
20        for input in &mut self.inputs {
21            TransparentAddress::from_script_from_chain(&input.script_pubkey)
22                .ok_or(SpendFinalizerError::UnsupportedScriptPubkey)
23                .and_then(|address| match address {
24                    TransparentAddress::PublicKeyHash(hash) => {
25                        let mut iter = input.partial_signatures.iter();
26                        match (iter.next(), iter.next()) {
27                            (Some(entry), None) => Ok(entry),
28                            (None, _) => Err(SpendFinalizerError::MissingSignature),
29                            (Some(_), Some(_)) => Err(SpendFinalizerError::UnexpectedSignatures),
30                        }
31                        .and_then(|(pubkey, sig_bytes)| {
32                            // Check that the signature is for this input.
33                            if hash[..] != Ripemd160::digest(Sha256::digest(pubkey))[..] {
34                                Err(SpendFinalizerError::UnexpectedSignatures)
35                            } else {
36                                // P2PKH scriptSig
37                                input.script_sig = Some(script::Component(vec![
38                                    pv::push_value(sig_bytes).expect("short enough"),
39                                    pv::push_value(pubkey).expect("short enough"),
40                                ]));
41                                Ok(())
42                            }
43                        })
44                    }
45                    TransparentAddress::ScriptHash(_) => {
46                        let redeem_script = input
47                            .redeem_script
48                            .as_ref()
49                            .ok_or(SpendFinalizerError::MissingRedeemScript)?
50                            .refine()
51                            .map_err(|_| SpendFinalizerError::UnsupportedRedeemScript)?;
52
53                        match solver::standard(&redeem_script) {
54                            Some(solver::ScriptKind::MultiSig { required, pubkeys }) => {
55                                // P2MS-in-P2SH `script_sig` format is:
56                                // - Dummy OP_0 to bypass OP_CHECKMULTISIG bug.
57                                let mut script_sig = vec![pv::_0];
58
59                                // - PushData(signature) * required
60                                //
61                                // The OP_CHECKMULTISIG logic matches pubkeys and
62                                // signatures together sequentially, so we look for
63                                // signatures in the order that the pubkeys exist in
64                                // `redeem_script`.
65                                let mut signatures_found = 0;
66                                for pubkey in pubkeys {
67                                    // Once we reach the threshold of required signatures,
68                                    // any additional signatures present in the PCZT are
69                                    // discarded.
70                                    if signatures_found == required {
71                                        break;
72                                    }
73
74                                    // PCZT requires compressed pubkeys.
75                                    let pubkey =
76                                        <[u8; 33]>::try_from(pubkey.as_slice()).map_err(|_| {
77                                            SpendFinalizerError::UncompressedPubkeyInScript
78                                        })?;
79
80                                    // If we have a signature from this pubkey, use it.
81                                    if let Some(sig) = input.partial_signatures.get(&pubkey) {
82                                        // Valid signatures always fit into `PushData`s.
83                                        script_sig.push(
84                                            pv::push_value(sig)
85                                                .ok_or(SpendFinalizerError::InvalidSignature)?,
86                                        );
87                                        signatures_found += 1;
88                                    }
89                                }
90                                if signatures_found < required {
91                                    return Err(SpendFinalizerError::MissingSignature);
92                                }
93
94                                // - PushData(redeem_script)
95                                script_sig.push(
96                                    push_script(&redeem_script)
97                                        .ok_or(SpendFinalizerError::RedeemScriptTooLong)?,
98                                );
99
100                                // P2SH scriptSig
101                                input.script_sig = Some(script::Component(script_sig));
102
103                                Ok(())
104                            }
105                            _ => Err(SpendFinalizerError::UnsupportedScriptPubkey),
106                        }
107                    }
108                })?
109        }
110
111        // All other data except the UTXO and proprietary fields in the input should be
112        // cleared from the PSBT. The UTXO should be kept to allow Transaction Extractors
113        // to verify the final network serialized transaction.
114        for input in &mut self.inputs {
115            input.required_time_lock_time = None;
116            input.required_height_lock_time = None;
117            input.redeem_script = None;
118            input.partial_signatures.clear();
119            input.bip32_derivation.clear();
120            input.ripemd160_preimages.clear();
121            input.sha256_preimages.clear();
122            input.hash160_preimages.clear();
123            input.hash256_preimages.clear();
124        }
125
126        Ok(())
127    }
128}

130/// Errors that can occur while finalizing the transparent inputs of a PCZT bundle.
131#[derive(Debug)]
132pub enum SpendFinalizerError {
133    /// `script_pubkey` is a P2SH script, but `redeem_script` is not set.
134    MissingRedeemScript,
135    /// `script_pubkey` is a P2SH script, but `redeem_script` is too long for a `PushData`.
136    RedeemScriptTooLong,
137    /// `partial_signatures` contained too few signatures.
138    MissingSignature,
139    /// `partial_signatures` contained an invalid signature.
140    InvalidSignature,
141    /// `redeem_script` contained an uncompressed pubkey, which PCZT does not support.
142    UncompressedPubkeyInScript,
143    /// `partial_signatures` contained unexpected signatures.
144    UnexpectedSignatures,
145    /// The `script_pubkey` kind is unsupported.
146    UnsupportedScriptPubkey,
147    /// The `redeem_script` kind is unsupported.
148    UnsupportedRedeemScript,
149}
