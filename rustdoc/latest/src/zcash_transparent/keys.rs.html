19/// The scope of a transparent key.
20///
21/// This type can represent [`zip32`] internal and external scopes, as well as custom scopes that
22/// may be used in non-hardened derivation at the `change` level of the BIP 44 key path.
23#[derive(Debug, Clone, Copy, PartialEq, Eq)]
24pub struct TransparentKeyScope(u32);
25
26impl TransparentKeyScope {
27    /// Returns an arbitrary custom `TransparentKeyScope`.
28    ///
29    /// This should be used with care: funds associated with keys derived under a custom
30    /// scope may not be recoverable if the wallet seed is restored in another wallet. It
31    /// is usually preferable to use standardized key scopes.
32    pub const fn custom(i: u32) -> Option<Self> {
33        if i < (1 << 31) {
34            Some(TransparentKeyScope(i))
35        } else {
36            None
37        }
38    }
39
40    /// The scope used to derive keys for external transparent addresses,
41    /// intended to be used to send funds to this wallet.
42    pub const EXTERNAL: Self = TransparentKeyScope(0);
43
44    /// The scope used to derive keys for internal wallet operations, e.g.
45    /// change or UTXO management.
46    pub const INTERNAL: Self = TransparentKeyScope(1);
47
48    /// The scope used to derive keys for ephemeral transparent addresses.
49    pub const EPHEMERAL: Self = TransparentKeyScope(2);
50}
51
52impl From<zip32::Scope> for TransparentKeyScope {
53    fn from(value: zip32::Scope) -> Self {
54        match value {
55            zip32::Scope::External => TransparentKeyScope::EXTERNAL,
56            zip32::Scope::Internal => TransparentKeyScope::INTERNAL,
57        }
58    }
59}
60
61impl From<TransparentKeyScope> for ChildNumber {
62    fn from(value: TransparentKeyScope) -> Self {
63        ChildNumber::new(value.0, false).expect("TransparentKeyScope is correct by construction")
64    }
65}

67/// A child index for a derived transparent address.
68///
69/// Only NON-hardened derivation is supported.
70#[derive(Clone, Copy, Debug, PartialEq, Eq)]
71pub struct NonHardenedChildIndex(u32);
72
73impl ConstantTimeEq for NonHardenedChildIndex {
74    fn ct_eq(&self, other: &Self) -> Choice {
75        self.0.ct_eq(&other.0)
76    }
77}
78
79impl NonHardenedChildIndex {
80    pub const ZERO: NonHardenedChildIndex = NonHardenedChildIndex(0);
81
82    /// Parses the given ZIP 32 child index.
83    ///
84    /// Returns `None` if the hardened bit is set.
85    pub fn from_index(i: u32) -> Option<Self> {
86        if i < (1 << 31) {
87            Some(NonHardenedChildIndex(i))
88        } else {
89            None
90        }
91    }
92
93    /// Returns the index as a 32-bit integer.
94    pub fn index(&self) -> u32 {
95        self.0
96    }
97
98    pub fn next(&self) -> Option<Self> {
99        // overflow cannot happen because self.0 is 31 bits, and the next index is at most 32 bits
100        // which in that case would lead from_index to return None.
101        Self::from_index(self.0 + 1)
102    }
103}
104
105impl TryFrom<ChildNumber> for NonHardenedChildIndex {
106    type Error = ();
107
108    fn try_from(value: ChildNumber) -> Result<Self, Self::Error> {
109        if value.is_hardened() {
110            Err(())
111        } else {
112            NonHardenedChildIndex::from_index(value.index()).ok_or(())
113        }
114    }
115}
116
117impl From<NonHardenedChildIndex> for ChildNumber {
118    fn from(value: NonHardenedChildIndex) -> Self {
119        Self::new(value.index(), false).expect("NonHardenedChildIndex is correct by construction")
120    }
121}

123/// A [BIP44] private key at the account path level `m/44'/<coin_type>'/<account>'`.
124///
125/// [BIP44]: https://github.com/bitcoin/bips/blob/master/bip-0044.mediawiki
126#[derive(Clone, Debug)]
127#[cfg(feature = "transparent-inputs")]
128pub struct AccountPrivKey(ExtendedPrivateKey<secp256k1::SecretKey>);
129
130#[cfg(feature = "transparent-inputs")]
131impl AccountPrivKey {
132    /// Performs derivation of the extended private key for the BIP44 path:
133    /// `m/44'/<coin_type>'/<account>'`.
134    ///
135    /// This produces the root of the derivation tree for transparent
136    /// viewing keys and addresses for the provided account.
137    pub fn from_seed<P: consensus::Parameters>(
138        params: &P,
139        seed: &[u8],
140        account: AccountId,
141    ) -> Result<AccountPrivKey, bip32::Error> {
142        ExtendedPrivateKey::new(seed)?
143            .derive_child(ChildNumber::new(44, true)?)?
144            .derive_child(ChildNumber::new(params.coin_type(), true)?)?
145            .derive_child(ChildNumber::new(account.into(), true)?)
146            .map(AccountPrivKey)
147    }
148
149    pub fn from_extended_privkey(extprivkey: ExtendedPrivateKey<secp256k1::SecretKey>) -> Self {
150        AccountPrivKey(extprivkey)
151    }
152
153    pub fn to_account_pubkey(&self) -> AccountPubKey {
154        AccountPubKey(ExtendedPublicKey::from(&self.0))
155    }
156
157    /// Derives the BIP44 private spending key for the child path
158    /// `m/44'/<coin_type>'/<account>'/<scope>/<address_index>`.
159    pub fn derive_secret_key(
160        &self,
161        scope: TransparentKeyScope,
162        address_index: NonHardenedChildIndex,
163    ) -> Result<secp256k1::SecretKey, bip32::Error> {
164        self.0
165            .derive_child(scope.into())?
166            .derive_child(address_index.into())
167            .map(|k| *k.private_key())
168    }
169
170    /// Derives the BIP44 private spending key for the external (incoming payment) child path
171    /// `m/44'/<coin_type>'/<account>'/0/<address_index>`.
172    pub fn derive_external_secret_key(
173        &self,
174        address_index: NonHardenedChildIndex,
175    ) -> Result<secp256k1::SecretKey, bip32::Error> {
176        self.derive_secret_key(zip32::Scope::External.into(), address_index)
177    }
178
179    /// Derives the BIP44 private spending key for the internal (change) child path
180    /// `m/44'/<coin_type>'/<account>'/1/<address_index>`.
181    pub fn derive_internal_secret_key(
182        &self,
183        address_index: NonHardenedChildIndex,
184    ) -> Result<secp256k1::SecretKey, bip32::Error> {
185        self.derive_secret_key(zip32::Scope::Internal.into(), address_index)
186    }
187
188    /// Returns the `AccountPrivKey` serialized using the encoding for a
189    /// [BIP 32](https://en.bitcoin.it/wiki/BIP_0032) ExtendedPrivateKey, excluding the
190    /// 4 prefix bytes.
191    pub fn to_bytes(&self) -> Vec<u8> {
192        // Convert to `xprv` encoding.
193        let xprv_encoded = self.0.to_extended_key(Prefix::XPRV).to_string();
194
195        // Now decode it and return the bytes we want.
196        bs58::decode(xprv_encoded)
197            .with_check(None)
198            .into_vec()
199            .expect("correct")
200            .split_off(Prefix::LENGTH)
201    }
202
203    /// Decodes the `AccountPrivKey` from the encoding specified for a
204    /// [BIP 32](https://en.bitcoin.it/wiki/BIP_0032) ExtendedPrivateKey, excluding the
205    /// 4 prefix bytes.
206    pub fn from_bytes(b: &[u8]) -> Option<Self> {
207        // Convert to `xprv` encoding.
208        let mut bytes = Prefix::XPRV.to_bytes().to_vec();
209        bytes.extend_from_slice(b);
210        let xprv_encoded = bs58::encode(bytes).with_check().into_string();
211
212        // Now we can parse it.
213        xprv_encoded
214            .parse::<ExtendedKey>()
215            .ok()
216            .and_then(|k| ExtendedPrivateKey::try_from(k).ok())
217            .map(AccountPrivKey::from_extended_privkey)
218    }
219}

257    /// Derives the BIP44 public key at the "address level" path corresponding to the given scope
258    /// and address index.
259    pub fn derive_address_pubkey(
260        &self,
261        scope: TransparentKeyScope,
262        address_index: NonHardenedChildIndex,
263    ) -> Result<secp256k1::PublicKey, bip32::Error> {
264        Ok(*self
265            .0
266            .derive_child(scope.into())?
267            .derive_child(address_index.into())?
268            .public_key())
269    }

271    /// Derives the public key corresponding to the given full BIP 32 path.
272    ///
273    /// This enforces that the path has a prefix that could have been used to derive this
274    /// `AccountPubKey`.
275    pub fn derive_pubkey_at_bip32_path<P: consensus::Parameters>(
276        &self,
277        params: &P,
278        expected_account_index: AccountId,
279        path: &[ChildNumber],
280    ) -> Result<secp256k1::PublicKey, bip32::Error> {
281        if path.len() < 3 {
282            Err(bip32::Error::ChildNumber)
283        } else {
284            match path.split_at(3) {
285                ([purpose, coin_type, account_index], sub_path)
286                    if purpose.is_hardened()
287                        && purpose.index() == 44
288                        && coin_type.is_hardened()
289                        && coin_type.index() == params.network_type().coin_type()
290                        && account_index.is_hardened()
291                        && account_index.index() == expected_account_index.into() =>
292                {
293                    sub_path
294                        .iter()
295                        .try_fold(self.0.clone(), |acc, child_index| {
296                            acc.derive_child(*child_index)
297                        })
298                        .map(|k| *k.public_key())
299                }
300                _ => Err(bip32::Error::ChildNumber),
301            }
302        }
303    }

305    /// Derives the internal ovk and external ovk corresponding to this
306    /// transparent fvk. As specified in [ZIP 316][transparent-ovk].
307    ///
308    /// [transparent-ovk]: https://zips.z.cash/zip-0316#deriving-internal-keys
309    pub fn ovks_for_shielding(&self) -> (InternalOvk, ExternalOvk) {
310        let i_ovk = PrfExpand::TRANSPARENT_ZIP316_OVK
311            .with(&self.0.attrs().chain_code, &self.0.public_key().serialize());
312        let ovk_external = ExternalOvk(i_ovk[..32].try_into().unwrap());
313        let ovk_internal = InternalOvk(i_ovk[32..].try_into().unwrap());
314
315        (ovk_internal, ovk_external)
316    }
317
318    /// Derives the internal ovk corresponding to this transparent fvk.
319    pub fn internal_ovk(&self) -> InternalOvk {
320        self.ovks_for_shielding().0
321    }
322
323    /// Derives the external ovk corresponding to this transparent fvk.
324    pub fn external_ovk(&self) -> ExternalOvk {
325        self.ovks_for_shielding().1
326    }
327
328    pub fn serialize(&self) -> Vec<u8> {
329        let mut buf = self.0.attrs().chain_code.to_vec();
330        buf.extend_from_slice(&self.0.public_key().serialize());
331        buf
332    }
333
334    pub fn deserialize(data: &[u8; 65]) -> Result<Self, bip32::Error> {
335        let chain_code = data[..32].try_into().expect("correct length");
336        let public_key = PublicKey::from_slice(&data[32..])?;
337        Ok(AccountPubKey(ExtendedPublicKey::new(
338            public_key,
339            ExtendedKeyAttrs {
340                depth: 3,
341                // We do not expose the inner `ExtendedPublicKey`, so we can use dummy
342                // values for the fields that are not encoded in an `AccountPubKey`.
343                parent_fingerprint: [0xff, 0xff, 0xff, 0xff],
344                child_number: ChildNumber::new(0, true).expect("correct"),
345                chain_code,
346            },
347        )))
348    }
349}

372/// Trait representing a transparent "incoming viewing key".
373///
374/// Unlike the Sapling and Orchard shielded protocols (which have viewing keys built into
375/// their key trees and bound to specific spending keys), the transparent protocol has no
376/// "viewing key" concept. Transparent viewing keys are instead emulated by making two
377/// observations:
378///
379/// - [BIP32] hierarchical derivation is structured as a tree.
380/// - The [BIP44] key paths use non-hardened derivation below the account level.
381///
382/// A transparent viewing key for an account is thus defined as the root of a specific
383/// non-hardened subtree underneath the account's path.
384///
385/// [BIP32]: https://github.com/bitcoin/bips/blob/master/bip-0032.mediawiki
386/// [BIP44]: https://github.com/bitcoin/bips/blob/master/bip-0044.mediawiki
387#[cfg(feature = "transparent-inputs")]
388pub trait IncomingViewingKey: private::SealedChangeLevelKey + core::marker::Sized {
389    /// Derives a transparent address at the provided child index.
390    #[allow(deprecated)]
391    fn derive_address(
392        &self,
393        address_index: NonHardenedChildIndex,
394    ) -> Result<TransparentAddress, bip32::Error> {
395        let child_key = self.extended_pubkey().derive_child(address_index.into())?;
396        Ok(pubkey_to_address(child_key.public_key()))
397    }
398
399    /// Searches the space of child indexes for an index that will
400    /// generate a valid transparent address, and returns the resulting
401    /// address and the index at which it was generated.
402    fn default_address(&self) -> (TransparentAddress, NonHardenedChildIndex) {
403        let mut address_index = NonHardenedChildIndex::ZERO;
404        loop {
405            match self.derive_address(address_index) {
406                Ok(addr) => {
407                    return (addr, address_index);
408                }
409                Err(_) => {
410                    address_index = address_index.next().unwrap_or_else(|| {
411                        panic!("Exhausted child index space attempting to find a default address.");
412                    });
413                }
414            }
415        }
416    }
417
418    fn serialize(&self) -> Vec<u8> {
419        let extpubkey = self.extended_pubkey();
420        let mut buf = extpubkey.attrs().chain_code.to_vec();
421        buf.extend_from_slice(&extpubkey.public_key().serialize());
422        buf
423    }
424
425    fn deserialize(data: &[u8; 65]) -> Result<Self, bip32::Error> {
426        let chain_code = data[..32].try_into().expect("correct length");
427        let public_key = PublicKey::from_slice(&data[32..])?;
428        Ok(Self::from_extended_pubkey(ExtendedPublicKey::new(
429            public_key,
430            ExtendedKeyAttrs {
431                depth: 4,
432                // We do not expose the inner `ExtendedPublicKey`, so we can use a dummy
433                // value for the `parent_fingerprint` that is not encoded in an
434                // `IncomingViewingKey`.
435                parent_fingerprint: [0xff, 0xff, 0xff, 0xff],
436                child_number: Self::SCOPE.into(),
437                chain_code,
438            },
439        )))
440    }
441}
