13/// Errors that can occur in the parsing of Bitcoin-style base58-encoded secret key material
14#[derive(Debug)]
15pub enum ParseError {
16    /// The data being decoded had an incorrect length, an incorrect prefix, or has the correct
17    /// length for a compressed encoding but has its final byte not equal to `1`.
18    InvalidEncoding,
19    /// Errors that occur in base58 decoding.
20    Base58(bs58::decode::Error),
21    /// Errors that occur when a decoded binary value does not correspond to a valid secp256k1 secret key.
22    Bip32(bip32::Error),
23}
24
25impl From<bs58::decode::Error> for ParseError {
26    fn from(value: bs58::decode::Error) -> Self {
27        ParseError::Base58(value)
28    }
29}
30
31impl From<TryFromSliceError> for ParseError {
32    fn from(_: TryFromSliceError) -> Self {
33        ParseError::InvalidEncoding
34    }
35}
36
37impl From<bip32::Error> for ParseError {
38    fn from(value: bip32::Error) -> Self {
39        ParseError::Bip32(value)
40    }
41}

43/// A secp256k1 secret key, along with a flag indicating whether a compressed encoding should be
44/// used when performing DER serialization.
45pub struct Key {

47    compressed: bool,
48}

50impl Key {
51    /// Constructs a new key value from a secret key and a flag indicating whether
52    /// the compressed encoding should be used when performing DER serialization.
53    pub fn new(secret: SecretKey, compressed: bool) -> Self {
54        Self { secret, compressed }
55    }

57    /// Decodes a base58-encoded secret key.
58    ///
59    /// This corresponds to <https://github.com/zcash/zcash/blob/1f1f7a385adc048154e7f25a3a0de76f3658ca09/src/key_io.cpp#L282>
60    pub fn decode_base58<N: NetworkConstants>(
61        network: &N,
62        encoded: &SecretString,
63    ) -> Result<Self, ParseError> {
64        let decoded = SecretVec::new(
65            bs58::decode(encoded.expose_secret())
66                .with_check(None)
67                .into_vec()?,
68        );
69        let prefix = network.b58_secret_key_prefix();
70        let decoded_len = decoded.expose_secret().len();
71        let compressed =
72            decoded_len == (33 + prefix.len()) && decoded.expose_secret().last() == Some(&1);
73        if (decoded_len == 32 + prefix.len() || compressed)
74            && decoded.expose_secret()[0..prefix.len()] == prefix
75        {
76            let key_end = decoded_len - if compressed { 1 } else { 0 };
77            let bytes = PrivateKeyBytes::try_from(&decoded.expose_secret()[prefix.len()..key_end])?;
78            Ok(Self {
79                secret: SecretKey::from_bytes(&bytes)?,
80                compressed,
81            })
82        } else {
83            Err(ParseError::InvalidEncoding)
84        }
85    }
86
87    /// Encodes a base58-encoded secret key.
88    ///
89    /// This corresponds to <https://github.com/zcash/zcash/blob/1f1f7a385adc048154e7f25a3a0de76f3658ca09/src/key_io.cpp#L298>
90    pub fn encode_base58<N: NetworkConstants>(&self, network: &N) -> SecretString {
91        let input = SecretVec::new(
92            network
93                .b58_secret_key_prefix()
94                .iter()
95                .chain(self.secret.secret_bytes().iter())
96                .chain(self.compressed.then_some(&1))
97                .copied()
98                .collect(),
99        );
100
101        SecretString::new(
102            bs58::encode(input.expose_secret())
103                .with_check()
104                .into_string(),
105        )
106    }

221    // Decodes a secret key from the "openssl-inspired" DER encoding used by zcashd.
222    pub fn der_decode(encoded: &SecretVec<u8>, compressed: bool) -> Result<Self, ()> {
223        // Ported from https://github.com/zcash/zcash/blob/1f1f7a385adc048154e7f25a3a0de76f3658ca09/src/key.cpp#L36
224        // The original c++ code is retained as comments.
225
226        let seckey = encoded.expose_secret();
227
228        //    /* sequence header */
229        //    if (end - seckey < 1 || *seckey != 0x30u) {
230        //        return 0;
231        //    }
232        //    seckey++;
233        let seckey = match seckey.split_first() {
234            Some((&0x30, rest)) => Ok(rest),
235            _ => Err(()),
236        }?;
237
238        //    /* sequence length constructor */
239        //    if (end - seckey < 1 || !(*seckey & 0x80u)) {
240        //        return 0;
241        //    }
242        //    size_t lenb = *seckey & ~0x80u; seckey++;
243        //    if (lenb < 1 || lenb > 2) {
244        //        return 0;
245        //    }
246        //    if (end - seckey < lenb) {
247        //        return 0;
248        //    }
249        let (lenb, seckey) = match seckey.split_first() {
250            Some((lenb, seckey)) if lenb & 0x80 != 0 => Ok((usize::from(lenb & !0x80), seckey)),
251            _ => Err(()),
252        }?;
253        if !(1..=2).contains(&lenb) {
254            return Err(());
255        }
256        if seckey.len() < lenb {
257            return Err(());
258        }
259
260        //    /* sequence length */
261        //    size_t len = seckey[lenb-1] | (lenb > 1 ? seckey[lenb-2] << 8 : 0u);
262        //    seckey += lenb;
263        //    if (end - seckey < len) {
264        //        return 0;
265        //    }
266        let len_low_bits = usize::from(seckey[lenb - 1]);
267        let len_high_bits = if lenb > 1 {
268            usize::from(seckey[lenb - 2]) << 8
269        } else {
270            0
271        };
272        let len = len_low_bits | len_high_bits;
273        let seckey = &seckey[lenb..];
274        if seckey.len() < len {
275            return Err(());
276        }
277
278        //    /* sequence element 0: version number (=1) */
279        //    if (end - seckey < 3 || seckey[0] != 0x02u || seckey[1] != 0x01u || seckey[2] != 0x01u) {
280        //        return 0;
281        //    }
282        //    seckey += 3;
283        let seckey = match seckey.split_at(3) {
284            (&[0x02, 0x01, 0x01], rest) => Ok(rest),
285            _ => Err(()),
286        }?;
287
288        //    /* sequence element 1: octet string, up to 32 bytes */
289        //    if (end - seckey < 2 || seckey[0] != 0x04u) {
290        //        return 0;
291        //    }
292        //    size_t oslen = seckey[1];
293        //    seckey += 2;
294        //    if (oslen > 32 || end - seckey < oslen) {
295        //        return 0;
296        //    }
297        //    memcpy(out32 + (32 - oslen), seckey, oslen);
298        //    if (!secp256k1_ec_seckey_verify(ctx, out32)) {
299        //        memset(out32, 0, 32);
300        //        return 0;
301        //    }
302        //    return 1;
303        //}
304        if seckey.len() < 2 || seckey[0] != 0x04 {
305            return Err(());
306        }
307        let oslen = usize::from(seckey[1]);
308        let seckey = &seckey[2..];
309        if oslen > 32 || seckey.len() < oslen {
310            return Err(());
311        }
312        let mut secret_buf = [0u8; 32];
313        secret_buf[(32 - oslen)..].copy_from_slice(&seckey[..oslen]);
314        let secret_key = SecretKey::from_bytes(&secret_buf);
315        secret_buf.zeroize();
316
317        secret_key
318            .map(|secret| Self { secret, compressed })
319            .map_err(|_| ())
320    }

349    #[test]
350    fn der_encoding_roundtrip() {
351        let mut rng = ChaChaRng::from_seed([0u8; 32]);
352        let secp = Secp256k1::new();
353        for _ in 0..100 {
354            let secret = SecretKey::new(&mut rng);
355            let compressed = rng.gen_bool(0.5);
356            let key = Key { secret, compressed };
357
358            let encoded = key.der_encode_with_context(&secp);
359            let decoded = Key::der_decode(&encoded, compressed).unwrap();
360
361            assert_eq!(key.secret(), decoded.secret());
362        }
363    }

365    /// Checks that parsed keys match test payload.
366    #[test]
367    fn base58_keys_valid_parse() {
368        for v in VALID {
369            match v.kind {
370                VectorKind::Privkey { is_compressed } => {
371                    // Must be valid private key
372                    let secret = &SecretString::new(v.base58_encoding.into());
373                    let privkey = Key::decode_base58(&v.network, secret).unwrap();
374                    assert_eq!(privkey.compressed, is_compressed);
375                    assert_eq!(hex::encode(privkey.secret.as_ref()), v.raw_bytes_hex);
376
377                    // Private key must be invalid public key
378                    assert_eq!(
379                        zcash_address::ZcashAddress::try_from_encoded(v.base58_encoding),
380                        Err(zcash_address::ParseError::NotZcash),
381                    );
382                }
383                VectorKind::Pubkey => {
384                    // Must be valid public key
385                    let destination: TransparentAddress =
386                        zcash_address::ZcashAddress::try_from_encoded(v.base58_encoding)
387                            .unwrap()
388                            .convert_if_network(v.network)
389                            .unwrap();
390                    let script = destination.script();
391                    assert_eq!(hex::encode(script.to_bytes()), v.raw_bytes_hex);
392
393                    // Public key must be invalid private key
394                    assert!(
395                        Key::decode_base58(
396                            &v.network,
397                            &SecretString::new(v.base58_encoding.into())
398                        )
399                        .is_err()
400                    );
401                }
402            }
403        }
404    }
