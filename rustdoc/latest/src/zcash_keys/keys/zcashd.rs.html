10// Derives a mnemonic phrase, using the pre-BIP-39 seed as entropy.
11//
12// zcashd produced a mnemonic from the pre-BIP-39 seed by incrementing the first byte of the
13// pre-BIP-39 seed until we found a value that was usable as valid entropy for seed phrase
14// generation.
15pub fn derive_mnemonic(legacy_seed: &SecretVec<u8>) -> Option<Mnemonic> {
16    if legacy_seed.expose_secret().len() != 32 {
17        return None;
18    }
19
20    let mut offset = 0u8;
21    loop {
22        let mut entropy = legacy_seed.expose_secret().clone();
23        entropy[0] += offset;
24        match Mnemonic::<English>::from_entropy(entropy) {
25            Ok(m) => {
26                return Some(m);
27            }
28            Err(_) => {
29                if offset == 0xFF {
30                    return None;
31                } else {
32                    offset += 1;
33                }
34            }
35        }
36    }
37}

39/// A type-safe wrapper for account identifiers.
40///
41/// Accounts are 31-bit unsigned integers, and are always treated as hardened in
42/// derivation paths.
43#[derive(Debug, Default, Copy, Clone, PartialEq, Eq, PartialOrd, Ord, Hash)]
44pub struct LegacyAddressIndex(u32);
45
46impl TryFrom<u32> for LegacyAddressIndex {
47    type Error = ();
48
49    fn try_from(id: u32) -> Result<Self, Self::Error> {
50        // Account IDs are always hardened in derivation paths, so they are effectively at
51        // most 31 bits.
52        if id < (1 << 31) {
53            Ok(Self(id))
54        } else {
55            Err(())
56        }
57    }
58}
59
60impl From<LegacyAddressIndex> for u32 {
61    fn from(id: LegacyAddressIndex) -> Self {
62        id.0
63    }
64}
65
66impl From<LegacyAddressIndex> for ChildIndex {
67    fn from(id: LegacyAddressIndex) -> Self {
68        // Account IDs are always hardened in derivation paths.
69        ChildIndex::hardened(id.0)
70    }
71}

73/// An enumeration of the address derivation algorithm variants used by `zcashd`.
74#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
75pub enum ZcashdHdDerivation {
76    /// A unified account derived according to ZIP 32.
77    Zip32 { account_id: AccountId },
78    /// The address was derived under a nonstandard path of the form:
79    ///
80    /// `m/32'/<coin_type>'/0x7FFFFFFF'/<address_index>'
81    Post470LegacySapling { address_index: LegacyAddressIndex },
82}
83
84/// Errors that can occur in parsing the string representation of an HD key path.
85#[derive(Debug)]
86pub enum PathParseError {
87    /// The string did not match the format of the subset of HD derivation paths that are produced
88    /// by zcashd.
89    PathInvalid,
90    /// The coin type component of the HD derivation path did match a supported zcashd coin type.
91    CoinTypeInvalid(String),
92    /// The coin type component of the HD derivation did not match the coin type for the specified
93    /// network.
94    CoinTypeMismatch { expected: u32, actual: u32 },
95    /// The account index component of the HD derivation path could not be parsed as a value in the
96    /// range of a `u32`.
97    AccountIndexInvalid(String),
98    /// The account index component of the HD derivation path is not in the range of valid ZIP 32
99    /// account indices.
100    AccountIdInvalid(u32),
101    /// The derivation path did not contain an address index component.
102    AddressIndexMissing,
103    /// The address index component of the HD derivation path could not be parsed as a value in the
104    /// range of valid ZIP 32 address indices.
105    AddressIndexInvalid(String),
106}
107
108impl ZcashdHdDerivation {
109    const ZCASHD_LEGACY_ACCOUNT_INDEX: u32 = 0x7FFFFFFF;
110
111    // Parses a HD key path to obtain the coin type and account ID or legacy Sapling address index.
112    //
113    // Historically, each key generated via the `z_getnewaddress` RPC method was treated as a separate
114    // pool of funds, much like a ZIP 32 account. After the introduction of BIP 39 mnemonic seed
115    // phrases in Zcashd v4.7.0, we wanted to retain these semantics but did not want `z_getnewaddress`
116    // to implicitly create a new ZIP 32 account. Instead, we used a fixed keypath scheme of
117    //
118    // `m/32'/coin_type'/0x7FFFFFFF'/addressIndex'`
119    //
120    // for each new address. This is not a "standard" path, but instead is a predictable location for
121    // legacy zcashd-derived keys that is minimally different from the UA account path, while unlikely
122    // to collide with normal UA account usage.
123    //
124    // For post v4.7.0 Zcashd accounts, we use standard ZIP 32 HD derivation from the master key to
125    // obtain the unified spending key.
126    pub fn parse_hd_path<C: NetworkConstants>(
127        network: &C,
128        path: &str,
129    ) -> Result<Self, PathParseError> {
130        let re = Regex::new(r"^m/32'/(\d+)'/(\d+)'(?:/(\d+)')?$").expect("checked to be valid");
131        let parts = re.captures(path).ok_or(PathParseError::PathInvalid)?;
132        assert!(parts.len() <= 4);
133
134        let coin_type = parts
135            .get(1)
136            .map_or(Err(PathParseError::PathInvalid), |part| {
137                let part_str = part.as_str();
138                part_str
139                    .parse::<u32>()
140                    .map_err(|_| PathParseError::CoinTypeInvalid(part_str.to_string()))
141            })?;
142        if coin_type != network.coin_type() {
143            return Err(PathParseError::CoinTypeMismatch {
144                expected: network.coin_type(),
145                actual: coin_type,
146            });
147        }
148
149        let account_index = parts
150            .get(2)
151            .map_or(Err(PathParseError::PathInvalid), |part| {
152                let part_str = part.as_str();
153                part_str
154                    .parse::<u32>()
155                    .map_err(|_| PathParseError::AccountIndexInvalid(part_str.to_string()))
156            })?;
157
158        if account_index == Self::ZCASHD_LEGACY_ACCOUNT_INDEX {
159            let address_index =
160                parts
161                    .get(3)
162                    .map_or(Err(PathParseError::AddressIndexMissing), |part| {
163                        let part_str = part.as_str();
164
165                        match part_str.parse::<u32>() {
166                            Ok(v) if v < (1 << 31) => Ok(LegacyAddressIndex(v)),
167                            _ => Err(PathParseError::AddressIndexInvalid(part_str.to_string())),
168                        }
169                    })?;
170
171            Ok(ZcashdHdDerivation::Post470LegacySapling { address_index })
172        } else {
173            if parts.get(3).is_some() {
174                return Err(PathParseError::PathInvalid);
175            }
176            let account_id = AccountId::try_from(account_index)
177                .map_err(|_| PathParseError::AccountIdInvalid(account_index))?;
178            Ok(ZcashdHdDerivation::Zip32 { account_id })
179        }
180    }
181}

183#[cfg(test)]
184mod tests {
185    use assert_matches::assert_matches;
186
187    use zcash_protocol::consensus::{NetworkConstants, NetworkType};
188    use zip32::AccountId;
189
190    use super::{PathParseError, ZcashdHdDerivation};
191
192    #[test]
193    fn parse_zcashd_hd_path() {
194        let zip32_bip39_path = format!("m/32'/{}'/0'", NetworkType::Main.coin_type());
195        assert_matches!(
196            ZcashdHdDerivation::parse_hd_path(&NetworkType::Main, &zip32_bip39_path),
197            Ok(ZcashdHdDerivation::Zip32 {
198                account_id: id
199            })
200            if id == AccountId::ZERO
201        );
202
203        let zip32_bip39_path_coin_invalid = format!("m/32'/{}'/0'", u64::MAX);
204        assert_matches!(
205            ZcashdHdDerivation::parse_hd_path(&NetworkType::Main, &zip32_bip39_path_coin_invalid),
206            Err(PathParseError::CoinTypeInvalid(_))
207        );
208
209        let zip32_bip39_path_coin_mismatch = format!("m/32'/{}'/0'", NetworkType::Test.coin_type());
210        assert_matches!(
211            ZcashdHdDerivation::parse_hd_path(&NetworkType::Main, &zip32_bip39_path_coin_mismatch),
212            Err(PathParseError::CoinTypeMismatch { .. })
213        );
214
215        let zip32_bip39_account_idx_invalid =
216            format!("m/32'/{}'/{}'/0'", NetworkType::Main.coin_type(), u64::MAX);
217        assert_matches!(
218            ZcashdHdDerivation::parse_hd_path(&NetworkType::Main, &zip32_bip39_account_idx_invalid),
219            Err(PathParseError::AccountIndexInvalid(_))
220        );
221
222        let zip32_bip39_account_id_invalid = format!(
223            "m/32'/{}'/{}'",
224            NetworkType::Main.coin_type(),
225            ZcashdHdDerivation::ZCASHD_LEGACY_ACCOUNT_INDEX + 1
226        );
227        assert_matches!(
228            ZcashdHdDerivation::parse_hd_path(&NetworkType::Main, &zip32_bip39_account_id_invalid),
229            Err(PathParseError::AccountIdInvalid(_))
230        );
231
232        let zcashd_legacy_path = format!(
233            "m/32'/{}'/{}'/0'",
234            NetworkType::Main.coin_type(),
235            ZcashdHdDerivation::ZCASHD_LEGACY_ACCOUNT_INDEX
236        );
237        assert_matches!(
238            ZcashdHdDerivation::parse_hd_path(&NetworkType::Main, &zcashd_legacy_path),
239            Ok(ZcashdHdDerivation::Post470LegacySapling {
240                address_index
241            })
242            if address_index.0 == 0
243        );
244
245        let too_long_path = format!("m/32'/{}'/1'/0'", NetworkType::Main.coin_type(),);
246        assert_matches!(
247            ZcashdHdDerivation::parse_hd_path(&NetworkType::Main, &too_long_path),
248            Err(PathParseError::PathInvalid)
249        );
250    }
