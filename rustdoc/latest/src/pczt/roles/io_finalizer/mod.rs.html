1//! The IO Finalizer role (anyone can execute).
2//!
3//! - Sets the appropriate bits in `Global.tx_modifiable` to 0.
4//! - Updates the various bsk values using the rcv information from spends and outputs.
5
6use rand_core::OsRng;
7use zcash_primitives::transaction::{
8    sighash::SignableInput, sighash_v5::v5_signature_hash, txid::TxIdDigester,
9};
10
11use crate::{
12    Pczt,
13    common::{
14        FLAG_SHIELDED_MODIFIABLE, FLAG_TRANSPARENT_INPUTS_MODIFIABLE,
15        FLAG_TRANSPARENT_OUTPUTS_MODIFIABLE,
16    },
17};
18use zcash_protocol::constants::{V5_TX_VERSION, V5_VERSION_GROUP_ID};
19
20use super::signer::pczt_to_tx_data;
21
22pub struct IoFinalizer {
23    pczt: Pczt,
24}
25
26impl IoFinalizer {
27    /// Instantiates the IO Finalizer role with the given PCZT.
28    pub fn new(pczt: Pczt) -> Self {
29        Self { pczt }
30    }
31
32    /// Finalizes the IO of the PCZT.
33    pub fn finalize_io(self) -> Result<Pczt, Error> {
34        let Self { pczt } = self;
35
36        let has_shielded_spends =
37            !(pczt.sapling.spends.is_empty() && pczt.orchard.actions.is_empty());
38        let has_shielded_outputs =
39            !(pczt.sapling.outputs.is_empty() && pczt.orchard.actions.is_empty());
40
41        // We can't build a transaction that has no spends or outputs.
42        // However, we don't attempt to reject an entirely dummy transaction.
43        if pczt.transparent.inputs.is_empty() && !has_shielded_spends {
44            return Err(Error::NoSpends);
45        }
46        if pczt.transparent.outputs.is_empty() && !has_shielded_outputs {
47            return Err(Error::NoOutputs);
48        }
49
50        let Pczt {
51            mut global,
52            transparent,
53            sapling,
54            orchard,
55        } = pczt;
56
57        // After shielded IO finalization, the transaction effects cannot be modified
58        // because dummy spends will have been signed.
59        if has_shielded_spends || has_shielded_outputs {
60            global.tx_modifiable &= !(FLAG_TRANSPARENT_INPUTS_MODIFIABLE
61                | FLAG_TRANSPARENT_OUTPUTS_MODIFIABLE
62                | FLAG_SHIELDED_MODIFIABLE);
63        }
64
65        let transparent = transparent.into_parsed().map_err(Error::TransparentParse)?;
66        let mut sapling = sapling.into_parsed().map_err(Error::SaplingParse)?;
67        let mut orchard = orchard.into_parsed().map_err(Error::OrchardParse)?;
68
69        let tx_data = pczt_to_tx_data(&global, &transparent, &sapling, &orchard)?;
70        let txid_parts = tx_data.digest(TxIdDigester);
71
72        // TODO: Pick sighash based on tx version.
73        match (global.tx_version, global.version_group_id) {
74            (V5_TX_VERSION, V5_VERSION_GROUP_ID) => Ok(()),
75            (version, version_group_id) => Err(Error::UnsupportedTxVersion {
76                version,
77                version_group_id,
78            }),
79        }?;
80        let shielded_sighash = v5_signature_hash(&tx_data, &SignableInput::Shielded, &txid_parts)
81            .as_ref()
82            .try_into()
83            .expect("correct length");
84
85        sapling
86            .finalize_io(shielded_sighash, OsRng)
87            .map_err(Error::SaplingFinalize)?;
88        orchard
89            .finalize_io(shielded_sighash, OsRng)
90            .map_err(Error::OrchardFinalize)?;
91
92        Ok(Pczt {
93            global,
94            transparent: crate::transparent::Bundle::serialize_from(transparent),
95            sapling: crate::sapling::Bundle::serialize_from(sapling),
96            orchard: crate::orchard::Bundle::serialize_from(orchard),
97        })
98    }
99}

101/// Errors that can occur while finalizing the IO of a PCZT.
102#[derive(Debug)]
103pub enum Error {
104    NoOutputs,
105    NoSpends,
106    OrchardFinalize(orchard::pczt::IoFinalizerError),
107    OrchardParse(orchard::pczt::ParseError),
108    SaplingFinalize(sapling::pczt::IoFinalizerError),
109    SaplingParse(sapling::pczt::ParseError),
110    Sign(super::signer::Error),
111    TransparentParse(transparent::pczt::ParseError),
112    UnsupportedTxVersion { version: u32, version_group_id: u32 },
113}
114
115impl From<super::signer::Error> for Error {
116    fn from(e: super::signer::Error) -> Self {
117        match e {
118            super::signer::Error::OrchardParse(parse_error) => Error::OrchardParse(parse_error),
119            super::signer::Error::SaplingParse(parse_error) => Error::SaplingParse(parse_error),
120            super::signer::Error::TransparentParse(parse_error) => {
121                Error::TransparentParse(parse_error)
122            }
123            _ => Error::Sign(e),
124        }
125    }
126}
