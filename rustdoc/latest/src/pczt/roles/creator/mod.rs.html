97    /// Builds a PCZT from the output of a [`Builder`].
98    ///
99    /// Returns `None` if the `TxVersion` is incompatible with PCZTs.
100    ///
101    /// [`Builder`]: zcash_primitives::transaction::builder::Builder
102    #[cfg(feature = "zcp-builder")]
103    pub fn build_from_parts<P: zcash_protocol::consensus::Parameters>(
104        parts: zcash_primitives::transaction::builder::PcztParts<P>,
105    ) -> Option<Pczt> {
106        use ::transparent::sighash::{SIGHASH_ANYONECANPAY, SIGHASH_SINGLE};
107        use zcash_protocol::consensus::NetworkConstants;
108
109        use crate::{common::FLAG_HAS_SIGHASH_SINGLE, SAPLING_TX_VERSION};
110
111        let tx_version = match parts.version {
112            zcash_primitives::transaction::TxVersion::Sprout(_)
113            | zcash_primitives::transaction::TxVersion::Overwinter => None,
114            zcash_primitives::transaction::TxVersion::Sapling => Some(SAPLING_TX_VERSION),
115            zcash_primitives::transaction::TxVersion::Zip225 => Some(V5_TX_VERSION),
116            #[cfg(zcash_unstable = "zfuture")]
117            zcash_primitives::transaction::TxVersion::ZFuture => None,
118        }?;
119
120        // Spends and outputs not modifiable.
121        let mut tx_modifiable = 0b0000_0000;
122        // Check if any input is using `SIGHASH_SINGLE` (with or without `ANYONECANPAY`).
123        if parts.transparent.as_ref().is_some_and(|bundle| {
124            bundle.inputs().iter().any(|input| {
125                (input.sighash_type().encode() & !SIGHASH_ANYONECANPAY) == SIGHASH_SINGLE
126            })
127        }) {
128            tx_modifiable |= FLAG_HAS_SIGHASH_SINGLE;
129        }
130
131        Some(Pczt {
132            global: crate::common::Global {
133                tx_version,
134                version_group_id: parts.version.version_group_id(),
135                consensus_branch_id: parts.consensus_branch_id.into(),
136                fallback_lock_time: Some(parts.lock_time),
137                expiry_height: parts.expiry_height.into(),
138                coin_type: parts.params.network_type().coin_type(),
139                tx_modifiable,
140                proprietary: BTreeMap::new(),
141            },
142            transparent: parts
143                .transparent
144                .map(crate::transparent::Bundle::serialize_from)
145                .unwrap_or_else(|| crate::transparent::Bundle {
146                    inputs: vec![],
147                    outputs: vec![],
148                }),
149            sapling: parts
150                .sapling
151                .map(crate::sapling::Bundle::serialize_from)
152                .unwrap_or_else(|| crate::sapling::Bundle {
153                    spends: vec![],
154                    outputs: vec![],
155                    value_sum: 0,
156                    anchor: sapling::Anchor::empty_tree().to_bytes(),
157                    bsk: None,
158                }),
159            orchard: parts
160                .orchard
161                .map(crate::orchard::Bundle::serialize_from)
162                .unwrap_or_else(|| crate::orchard::Bundle {
163                    actions: vec![],
164                    flags: ORCHARD_SPENDS_AND_OUTPUTS_ENABLED,
165                    value_sum: (0, true),
166                    anchor: orchard::Anchor::empty_tree().to_bytes(),
167                    zkproof: None,
168                    bsk: None,
169                }),
170        })
171    }
