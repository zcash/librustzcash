1use rand_core::OsRng;
2use sapling::{
3    BatchValidator, Bundle,
4    bundle::Authorized,
5    circuit::{OutputVerifyingKey, SpendVerifyingKey},
6    pczt::Unbound,
7};
8use zcash_protocol::value::ZatBalance;
9
10pub(super) fn extract_bundle(
11    bundle: crate::sapling::Bundle,
12) -> Result<Option<Bundle<Unbound, ZatBalance>>, SaplingError> {
13    bundle
14        .into_parsed()
15        .map_err(SaplingError::Parse)?
16        .extract()
17        .map_err(SaplingError::Extract)
18}
19
20pub(super) fn verify_bundle(
21    bundle: &Bundle<Authorized, ZatBalance>,
22    spend_vk: &SpendVerifyingKey,
23    output_vk: &OutputVerifyingKey,
24    sighash: [u8; 32],
25) -> Result<(), SaplingError> {
26    let mut validator = BatchValidator::new();
27
28    if !validator.check_bundle(bundle.clone(), sighash) {
29        return Err(SaplingError::ConsensusRuleViolation);
30    }
31
32    if !validator.validate(spend_vk, output_vk, OsRng) {
33        return Err(SaplingError::InvalidProofsOrSignatures);
34    }
35
36    Ok(())
37}
38
39#[derive(Debug)]
40pub enum SaplingError {
41    ConsensusRuleViolation,
42    Extract(sapling::pczt::TxExtractorError),
43    InvalidProofsOrSignatures,
44    Parse(sapling::pczt::ParseError),
45}
