1//! The Signer role (capability holders can contribute).
2//!
3//! - Needs the spend authorization randomizers to create signatures.
4//! - Needs sufficient information to verify that the proof is over the correct data,
5//!   without needing to verify the proof itself.
6//! - A Signer should only need to implement:
7//!   - Pedersen commitments using Jubjub / Pallas arithmetic (for note and value
8//!     commitments)
9//!   - BLAKE2b and BLAKE2s (and the various PRFs / CRHs they are used in)
10//!   - Nullifier check (using Jubjub / Pallas arithmetic)
11//!   - KDF plus note decryption (AEAD_CHACHA20_POLY1305)
12//!   - SignatureHash algorithm
13//!   - Signatures (RedJubjub / RedPallas)
14//!   - A source of randomness.
15

43const V5_VERSION_GROUP_ID: u32 = 0x26A7270A;

45pub struct Signer {
46    global: Global,
47    transparent: transparent::pczt::Bundle,
48    sapling: sapling::pczt::Bundle,
49    orchard: orchard::pczt::Bundle,
50    /// Cached across multiple signatures.
51    tx_data: TransactionData<EffectsOnly>,
52    txid_parts: TxDigests<Blake2bHash>,
53    shielded_sighash: [u8; 32],
54    secp: secp256k1::Secp256k1<secp256k1::All>,
55}

57impl Signer {
58    /// Instantiates the Signer role with the given PCZT.
59    pub fn new(pczt: Pczt) -> Result<Self, Error> {
60        let Pczt {
61            global,
62            transparent,
63            sapling,
64            orchard,
65        } = pczt;
66
67        let transparent = transparent.into_parsed().map_err(Error::TransparentParse)?;
68        let sapling = sapling.into_parsed().map_err(Error::SaplingParse)?;
69        let orchard = orchard.into_parsed().map_err(Error::OrchardParse)?;
70
71        let tx_data = pczt_to_tx_data(&global, &transparent, &sapling, &orchard)?;
72        let txid_parts = tx_data.digest(TxIdDigester);
73
74        // TODO: Pick sighash based on tx version.
75        match (global.tx_version, global.version_group_id) {
76            (V5_TX_VERSION, V5_VERSION_GROUP_ID) => Ok(()),
77            (version, version_group_id) => Err(Error::Global(GlobalError::UnsupportedTxVersion {
78                version,
79                version_group_id,
80            })),
81        }?;
82        let shielded_sighash = sighash(&tx_data, &SignableInput::Shielded, &txid_parts);
83
84        Ok(Self {
85            global,
86            transparent,
87            sapling,
88            orchard,
89            tx_data,
90            txid_parts,
91            shielded_sighash,
92            secp: secp256k1::Secp256k1::new(),
93        })
94    }

96    /// Calculates the signature digest that must be signed to authorize shielded spends.
97    ///
98    /// This can be used to produce a signature externally suitable for passing to e.g.
99    /// [`Self::apply_orchard_signature`].}
100    pub fn shielded_sighash(&self) -> [u8; 32] {
101        self.shielded_sighash
102    }
103
104    /// Calculates the signature digest that must be signed to authorize the transparent
105    /// spend at the given index.
106    ///
107    /// This can be used to produce a signature externally suitable for passing to e.g.
108    /// [`Self::append_transparent_signature`].}
109    ///
110    /// Returns an error if `index` is invalid for this PCZT.
111    pub fn transparent_sighash(&self, index: usize) -> Result<[u8; 32], Error> {
112        let input = self
113            .transparent
114            .inputs()
115            .get(index)
116            .ok_or(Error::InvalidIndex)?;
117
118        input.with_signable_input(index, |signable_input| {
119            Ok(sighash(
120                &self.tx_data,
121                &SignableInput::Transparent(signable_input),
122                &self.txid_parts,
123            ))
124        })
125    }
126
127    /// Signs the transparent spend at the given index with the given spending key.
128    ///
129    /// It is the caller's responsibility to perform any semantic validity checks on the
130    /// PCZT (for example, comfirming that the change amounts are correct) before calling
131    /// this method.
132    pub fn sign_transparent(
133        &mut self,
134        index: usize,
135        sk: &secp256k1::SecretKey,
136    ) -> Result<(), Error> {
137        self.generate_or_append_transparent_signature(index, |input, tx_data, txid_parts, secp| {
138            input.sign(
139                index,
140                |input| sighash(tx_data, &SignableInput::Transparent(input), txid_parts),
141                sk,
142                secp,
143            )
144        })
145    }
146
147    /// Appends the given signature to the transparent spend.
148    ///
149    /// It is the caller's responsibility to perform any semantic validity checks on the
150    /// PCZT (for example, comfirming that the change amounts are correct) before calling
151    /// this method.
152    pub fn append_transparent_signature(
153        &mut self,
154        index: usize,
155        signature: secp256k1::ecdsa::Signature,
156    ) -> Result<(), Error> {
157        self.generate_or_append_transparent_signature(index, |input, tx_data, txid_parts, secp| {
158            input.append_signature(
159                index,
160                |input| sighash(tx_data, &SignableInput::Transparent(input), txid_parts),
161                signature,
162                secp,
163            )
164        })
165    }
166
167    fn generate_or_append_transparent_signature<F>(
168        &mut self,
169        index: usize,
170        f: F,
171    ) -> Result<(), Error>
172    where
173        F: FnOnce(
174            &mut transparent::pczt::Input,
175            &TransactionData<EffectsOnly>,
176            &TxDigests<Blake2bHash>,
177            &secp256k1::Secp256k1<secp256k1::All>,
178        ) -> Result<(), transparent::pczt::SignerError>,
179    {
180        let input = self
181            .transparent
182            .inputs_mut()
183            .get_mut(index)
184            .ok_or(Error::InvalidIndex)?;
185
186        // Check consistency of the input being signed.
187        // TODO
188
189        // Generate or apply the signature.
190        f(input, &self.tx_data, &self.txid_parts, &self.secp).map_err(Error::TransparentSign)?;
191
192        // Update transaction modifiability:
193        // - If the Signer added a signature that does not use `SIGHASH_ANYONECANPAY`, the
194        //   Transparent Inputs Modifiable Flag must be set to False (because the
195        //   signature commits to all inputs, not just the one at `index`).
196        if input.sighash_type().encode() & SIGHASH_ANYONECANPAY == 0 {
197            self.global.tx_modifiable &= !FLAG_TRANSPARENT_INPUTS_MODIFIABLE;
198        }
199        // - If the Signer added a signature that does not use `SIGHASH_NONE`, the
200        //   Transparent Outputs Modifiable Flag must be set to False. Note that this
201        //   applies to `SIGHASH_SINGLE` because we could otherwise remove the output at
202        //   `index`, which would not remove the signature.
203        if (input.sighash_type().encode() & !SIGHASH_ANYONECANPAY) != SIGHASH_NONE {
204            self.global.tx_modifiable &= !FLAG_TRANSPARENT_OUTPUTS_MODIFIABLE;
205        }
206        // - If the Signer added a signature that uses `SIGHASH_SINGLE`, the Has
207        //   `SIGHASH_SINGLE` flag must be set to True.
208        if (input.sighash_type().encode() & !SIGHASH_ANYONECANPAY) == SIGHASH_SINGLE {
209            self.global.tx_modifiable |= FLAG_HAS_SIGHASH_SINGLE;
210        }
211        // - Always set the Shielded Modifiable Flag to False.
212        self.global.tx_modifiable &= !FLAG_SHIELDED_MODIFIABLE;
213
214        Ok(())
215    }

217    /// Signs the Sapling spend at the given index with the given spend authorizing key.
218    ///
219    /// Requires the spend's `proof_generation_key` field to be set (because the API does
220    /// not take an FVK).
221    ///
222    /// It is the caller's responsibility to perform any semantic validity checks on the
223    /// PCZT (for example, comfirming that the change amounts are correct) before calling
224    /// this method.
225    pub fn sign_sapling(
226        &mut self,
227        index: usize,
228        ask: &sapling::keys::SpendAuthorizingKey,
229    ) -> Result<(), Error> {
230        self.generate_or_apply_sapling_signature(index, |spend, shielded_sighash| {
231            spend.sign(shielded_sighash, ask, OsRng)
232        })
233    }

235    /// Applies the given signature to the Sapling spend.
236    ///
237    /// It is the caller's responsibility to perform any semantic validity checks on the
238    /// PCZT (for example, comfirming that the change amounts are correct) before calling
239    /// this method.
240    pub fn apply_sapling_signature(
241        &mut self,
242        index: usize,
243        signature: redjubjub::Signature<redjubjub::SpendAuth>,
244    ) -> Result<(), Error> {
245        self.generate_or_apply_sapling_signature(index, |spend, shielded_sighash| {
246            spend.apply_signature(shielded_sighash, signature)
247        })
248    }
249
250    fn generate_or_apply_sapling_signature<F>(&mut self, index: usize, f: F) -> Result<(), Error>
251    where
252        F: FnOnce(&mut sapling::pczt::Spend, [u8; 32]) -> Result<(), sapling::pczt::SignerError>,
253    {
254        let spend = self
255            .sapling
256            .spends_mut()
257            .get_mut(index)
258            .ok_or(Error::InvalidIndex)?;
259
260        // Check consistency of the input being signed if we have its note components.
261        match spend.verify_nullifier(None) {
262            Err(
263                sapling::pczt::VerifyError::MissingRecipient
264                | sapling::pczt::VerifyError::MissingValue
265                | sapling::pczt::VerifyError::MissingRandomSeed,
266            ) => Ok(()),
267            r => r,
268        }
269        .map_err(Error::SaplingVerify)?;
270
271        // Generate or apply the signature.
272        f(spend, self.shielded_sighash).map_err(Error::SaplingSign)?;
273
274        // Update transaction modifiability: all transaction effects have been committed
275        // to by the signature.
276        self.global.tx_modifiable &= !(FLAG_TRANSPARENT_INPUTS_MODIFIABLE
277            | FLAG_TRANSPARENT_OUTPUTS_MODIFIABLE
278            | FLAG_SHIELDED_MODIFIABLE);
279
280        Ok(())
281    }

315    fn generate_or_apply_orchard_signature<F>(&mut self, index: usize, f: F) -> Result<(), Error>
316    where
317        F: FnOnce(&mut orchard::pczt::Action, [u8; 32]) -> Result<(), orchard::pczt::SignerError>,
318    {
319        let action = self
320            .orchard
321            .actions_mut()
322            .get_mut(index)
323            .ok_or(Error::InvalidIndex)?;
324
325        // Check consistency of the input being signed if we have its note components.
326        match action.spend().verify_nullifier(None) {
327            Err(
328                orchard::pczt::VerifyError::MissingRecipient
329                | orchard::pczt::VerifyError::MissingValue
330                | orchard::pczt::VerifyError::MissingRho
331                | orchard::pczt::VerifyError::MissingRandomSeed,
332            ) => Ok(()),
333            r => r,
334        }
335        .map_err(Error::OrchardVerify)?;
336
337        // Generate or apply the signature.
338        f(action, self.shielded_sighash).map_err(Error::OrchardSign)?;
339
340        // Update transaction modifiability: all transaction effects have been committed
341        // to by the signature.
342        self.global.tx_modifiable &= !(FLAG_TRANSPARENT_INPUTS_MODIFIABLE
343            | FLAG_TRANSPARENT_OUTPUTS_MODIFIABLE
344            | FLAG_SHIELDED_MODIFIABLE);
345
346        Ok(())
347    }

360/// Extracts an unauthorized `TransactionData` from the PCZT.
361///
362/// We don't care about existing proofs or signatures here, because they do not affect the
363/// sighash; we only want the effects of the transaction.
364pub(crate) fn pczt_to_tx_data(
365    global: &Global,
366    transparent: &transparent::pczt::Bundle,
367    sapling: &sapling::pczt::Bundle,
368    orchard: &orchard::pczt::Bundle,
369) -> Result<TransactionData<EffectsOnly>, Error> {
370    let version = match (global.tx_version, global.version_group_id) {
371        (V5_TX_VERSION, V5_VERSION_GROUP_ID) => Ok(TxVersion::V5),
372        (version, version_group_id) => Err(Error::Global(GlobalError::UnsupportedTxVersion {
373            version,
374            version_group_id,
375        })),
376    }?;
377
378    let consensus_branch_id = BranchId::try_from(global.consensus_branch_id)
379        .map_err(|_| Error::Global(GlobalError::UnknownConsensusBranchId))?;
380
381    let transparent_bundle = transparent
382        .extract_effects()
383        .map_err(Error::TransparentExtract)?;
384
385    let sapling_bundle = sapling.extract_effects().map_err(Error::SaplingExtract)?;
386
387    let orchard_bundle = orchard.extract_effects().map_err(Error::OrchardExtract)?;
388
389    Ok(TransactionData::from_parts(
390        version,
391        consensus_branch_id,
392        determine_lock_time(global, transparent.inputs()).ok_or(Error::IncompatibleLockTimes)?,
393        global.expiry_height.into(),
394        #[cfg(all(
395            any(zcash_unstable = "nu7", zcash_unstable = "zfuture"),
396            feature = "zip-233"
397        ))]
398        Zatoshis::ZERO,
399        transparent_bundle,
400        None,
401        sapling_bundle,
402        orchard_bundle,
403    ))
404}

416/// Helper to produce the correct sighash for a PCZT.
417fn sighash(
418    tx_data: &TransactionData<EffectsOnly>,
419    signable_input: &SignableInput,
420    txid_parts: &TxDigests<Blake2bHash>,
421) -> [u8; 32] {
422    v5_signature_hash(tx_data, signable_input, txid_parts)
423        .as_ref()
424        .try_into()
425        .expect("correct length")
426}

428/// Errors that can occur while creating signatures for a PCZT.

432    IncompatibleLockTimes,
433    InvalidIndex,
434    OrchardExtract(orchard::pczt::TxExtractorError),
435    OrchardParse(orchard::pczt::ParseError),
436    OrchardSign(orchard::pczt::SignerError),
437    OrchardVerify(orchard::pczt::VerifyError),
438    SaplingExtract(sapling::pczt::TxExtractorError),
439    SaplingParse(sapling::pczt::ParseError),
440    SaplingSign(sapling::pczt::SignerError),
441    SaplingVerify(sapling::pczt::VerifyError),
442    TransparentExtract(transparent::pczt::TxExtractorError),
443    TransparentParse(transparent::pczt::ParseError),
444    TransparentSign(transparent::pczt::SignerError),
445}
