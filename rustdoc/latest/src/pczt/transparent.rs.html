20/// PCZT fields that are specific to producing the transaction's transparent bundle (if
21/// any).
22#[derive(Clone, Debug, Serialize, Deserialize, Getters)]
23pub struct Bundle {
24    #[getset(get = "pub")]
25    pub(crate) inputs: Vec<Input>,
26    #[getset(get = "pub")]
27    pub(crate) outputs: Vec<Output>,
28}

30/// Information about a transparent input within a transaction.
31#[serde_as]
32#[derive(Clone, Debug, Serialize, Deserialize, Getters)]
33pub struct Input {
34    //
35    // Transparent effecting data.
36    //
37    // These are required fields that are part of the final transaction, and are filled in
38    // by the Constructor when adding an output.
39    //
40    #[getset(get = "pub")]
41    pub(crate) prevout_txid: [u8; 32],
42    #[getset(get = "pub")]
43    pub(crate) prevout_index: u32,
44
45    /// The sequence number of this input.
46    ///
47    /// - This is set by the Constructor.
48    /// - If omitted, the sequence number is assumed to be the final sequence number
49    ///   (`0xffffffff`).
50    #[getset(get = "pub")]
51    pub(crate) sequence: Option<u32>,
52
53    /// The minimum Unix timstamp that this input requires to be set as the transaction's
54    /// lock time.
55    ///
56    /// - This is set by the Constructor.
57    /// - This must be greater than or equal to 500000000.
58    pub(crate) required_time_lock_time: Option<u32>,
59
60    /// The minimum block height that this input requires to be set as the transaction's
61    /// lock time.
62    ///
63    /// - This is set by the Constructor.
64    /// - This must be greater than 0 and less than 500000000.
65    pub(crate) required_height_lock_time: Option<u32>,
66
67    /// A satisfying witness for the `script_pubkey` of the input being spent.
68    ///
69    /// This is set by the Spend Finalizer.
70    pub(crate) script_sig: Option<Vec<u8>>,
71
72    // These are required by the Transaction Extractor, to derive the shielded sighash
73    // needed for computing the binding signatures.
74    #[getset(get = "pub")]
75    pub(crate) value: u64,
76    #[getset(get = "pub")]
77    pub(crate) script_pubkey: Vec<u8>,
78
79    /// The script required to spend this output, if it is P2SH.
80    ///
81    /// Set to `None` if this is a P2PKH output.
82    pub(crate) redeem_script: Option<Vec<u8>>,
83
84    /// A map from a pubkey to a signature created by it.
85    ///
86    /// - Each pubkey should appear in `script_pubkey` or `redeem_script`.
87    /// - Each entry is set by a Signer, and should contain an ECDSA signature that is
88    ///   valid under the corresponding pubkey.
89    /// - These are required by the Spend Finalizer to assemble `script_sig`.
90    #[serde_as(as = "BTreeMap<[_; 33], _>")]
91    pub(crate) partial_signatures: BTreeMap<[u8; 33], Vec<u8>>,
92
93    /// The sighash type to be used for this input.
94    ///
95    /// - Signers must use this sighash type to produce their signatures. Signers that
96    ///   cannot produce signatures for this sighash type must not provide a signature.
97    /// - Spend Finalizers must fail to finalize inputs which have signatures not matching
98    ///   this sighash type.
99    pub(crate) sighash_type: u8,
100
101    /// A map from a pubkey to the BIP 32 derivation path at which its corresponding
102    /// spending key can be found.
103    ///
104    /// - The pubkeys should appear in `script_pubkey` or `redeem_script`.
105    /// - Each entry is set by an Updater.
106    /// - Individual entries may be required by a Signer.
107    /// - It is not required that the map include entries for all of the used pubkeys.
108    ///   In particular, it is not possible to include entries for non-BIP-32 pubkeys.
109    #[serde_as(as = "BTreeMap<[_; 33], _>")]
110    pub(crate) bip32_derivation: BTreeMap<[u8; 33], Zip32Derivation>,
111
112    /// Mappings of the form `key = RIPEMD160(value)`.
113    ///
114    /// - These may be used by the Signer to inspect parts of `script_pubkey` or
115    ///   `redeem_script`.
116    pub(crate) ripemd160_preimages: BTreeMap<[u8; 20], Vec<u8>>,
117
118    /// Mappings of the form `key = SHA256(value)`.
119    ///
120    /// - These may be used by the Signer to inspect parts of `script_pubkey` or
121    ///   `redeem_script`.
122    pub(crate) sha256_preimages: BTreeMap<[u8; 32], Vec<u8>>,
123
124    /// Mappings of the form `key = RIPEMD160(SHA256(value))`.
125    ///
126    /// - These may be used by the Signer to inspect parts of `script_pubkey` or
127    ///   `redeem_script`.
128    pub(crate) hash160_preimages: BTreeMap<[u8; 20], Vec<u8>>,
129
130    /// Mappings of the form `key = SHA256(SHA256(value))`.
131    ///
132    /// - These may be used by the Signer to inspect parts of `script_pubkey` or
133    ///   `redeem_script`.
134    pub(crate) hash256_preimages: BTreeMap<[u8; 32], Vec<u8>>,
135
136    /// Proprietary fields related to the note being spent.
137    #[getset(get = "pub")]
138    pub(crate) proprietary: BTreeMap<String, Vec<u8>>,
139}

141/// Information about a transparent output within a transaction.
142#[serde_as]
143#[derive(Clone, Debug, Serialize, Deserialize, Getters)]
144pub struct Output {
145    //
146    // Transparent effecting data.
147    //
148    // These are required fields that are part of the final transaction, and are filled in
149    // by the Constructor when adding an output.
150    //
151    #[getset(get = "pub")]
152    pub(crate) value: u64,
153    #[getset(get = "pub")]
154    pub(crate) script_pubkey: Vec<u8>,
155
156    /// The script required to spend this output, if it is P2SH.
157    ///
158    /// Set to `None` if this is a P2PKH output.
159    pub(crate) redeem_script: Option<Vec<u8>>,
160
161    /// A map from a pubkey to the BIP 32 derivation path at which its corresponding
162    /// spending key can be found.
163    ///
164    /// - The pubkeys should appear in `script_pubkey` or `redeem_script`.
165    /// - Each entry is set by an Updater.
166    /// - Individual entries may be required by a Signer.
167    /// - It is not required that the map include entries for all of the used pubkeys.
168    ///   In particular, it is not possible to include entries for non-BIP-32 pubkeys.
169    #[serde_as(as = "BTreeMap<[_; 33], _>")]
170    pub(crate) bip32_derivation: BTreeMap<[u8; 33], Zip32Derivation>,
171
172    /// The user-facing address to which this output is being sent, if any.
173    ///
174    /// - This is set by an Updater.
175    /// - Signers must parse this address (if present) and confirm that it contains
176    ///   `recipient` (either directly, or e.g. as a receiver within a Unified Address).
177    #[getset(get = "pub")]
178    pub(crate) user_address: Option<String>,
179
180    /// Proprietary fields related to the note being spent.
181    #[getset(get = "pub")]
182    pub(crate) proprietary: BTreeMap<String, Vec<u8>>,
183}

185impl Bundle {
186    /// Merges this bundle with another.
187    ///
188    /// Returns `None` if the bundles have conflicting data.
189    pub(crate) fn merge(
190        mut self,
191        other: Self,
192        self_global: &Global,
193        other_global: &Global,
194    ) -> Option<Self> {
195        // Destructure `other` to ensure we handle everything.
196        let Self {
197            mut inputs,
198            mut outputs,
199        } = other;
200
201        match (
202            self_global.inputs_modifiable(),
203            other_global.inputs_modifiable(),
204            self.inputs.len().cmp(&inputs.len()),
205        ) {
206            // Fail if the merge would add inputs to a non-modifiable bundle.
207            (false, _, Ordering::Less) | (_, false, Ordering::Greater) => return None,
208            // If the other bundle has more inputs than us, move them over; these cannot
209            // conflict by construction.
210            (true, _, Ordering::Less) => self.inputs.extend(inputs.drain(self.inputs.len()..)),
211            // Do nothing otherwise.
212            (_, _, Ordering::Equal) | (_, true, Ordering::Greater) => (),
213        }
214
215        match (
216            self_global.outputs_modifiable(),
217            other_global.outputs_modifiable(),
218            self.outputs.len().cmp(&outputs.len()),
219        ) {
220            // Fail if the merge would add outputs to a non-modifiable bundle.
221            (false, _, Ordering::Less) | (_, false, Ordering::Greater) => return None,
222            // If the other bundle has more outputs than us, move them over; these cannot
223            // conflict by construction.
224            (true, _, Ordering::Less) => self.outputs.extend(outputs.drain(self.outputs.len()..)),
225            // Do nothing otherwise.
226            (_, _, Ordering::Equal) | (_, true, Ordering::Greater) => (),
227        }
228
229        // Leverage the early-exit behaviour of zip to confirm that the remaining data in
230        // the other bundle matches this one.
231        for (lhs, rhs) in self.inputs.iter_mut().zip(inputs.into_iter()) {
232            // Destructure `rhs` to ensure we handle everything.
233            let Input {
234                prevout_txid,
235                prevout_index,
236                sequence,
237                required_time_lock_time,
238                required_height_lock_time,
239                script_sig,
240                value,
241                script_pubkey,
242                redeem_script,
243                partial_signatures,
244                sighash_type,
245                bip32_derivation,
246                ripemd160_preimages,
247                sha256_preimages,
248                hash160_preimages,
249                hash256_preimages,
250                proprietary,
251            } = rhs;
252
253            if lhs.prevout_txid != prevout_txid
254                || lhs.prevout_index != prevout_index
255                || lhs.value != value
256                || lhs.script_pubkey != script_pubkey
257                || lhs.sighash_type != sighash_type
258            {
259                return None;
260            }
261
262            if !(merge_optional(&mut lhs.sequence, sequence)
263                && merge_optional(&mut lhs.required_time_lock_time, required_time_lock_time)
264                && merge_optional(
265                    &mut lhs.required_height_lock_time,
266                    required_height_lock_time,
267                )
268                && merge_optional(&mut lhs.script_sig, script_sig)
269                && merge_optional(&mut lhs.redeem_script, redeem_script)
270                && merge_map(&mut lhs.partial_signatures, partial_signatures)
271                && merge_map(&mut lhs.bip32_derivation, bip32_derivation)
272                && merge_map(&mut lhs.ripemd160_preimages, ripemd160_preimages)
273                && merge_map(&mut lhs.sha256_preimages, sha256_preimages)
274                && merge_map(&mut lhs.hash160_preimages, hash160_preimages)
275                && merge_map(&mut lhs.hash256_preimages, hash256_preimages)
276                && merge_map(&mut lhs.proprietary, proprietary))
277            {
278                return None;
279            }
280        }
281
282        for (lhs, rhs) in self.outputs.iter_mut().zip(outputs.into_iter()) {
283            // Destructure `rhs` to ensure we handle everything.
284            let Output {
285                value,
286                script_pubkey,
287                redeem_script,
288                bip32_derivation,
289                user_address,
290                proprietary,
291            } = rhs;
292
293            if lhs.value != value || lhs.script_pubkey != script_pubkey {
294                return None;
295            }
296
297            if !(merge_optional(&mut lhs.redeem_script, redeem_script)
298                && merge_map(&mut lhs.bip32_derivation, bip32_derivation)
299                && merge_optional(&mut lhs.user_address, user_address)
300                && merge_map(&mut lhs.proprietary, proprietary))
301            {
302                return None;
303            }
304        }
305
306        Some(self)
307    }
308}

311impl Bundle {
312    pub(crate) fn into_parsed(
313        self,
314    ) -> Result<transparent::pczt::Bundle, transparent::pczt::ParseError> {
315        let inputs = self
316            .inputs
317            .into_iter()
318            .map(|input| {
319                transparent::pczt::Input::parse(
320                    input.prevout_txid,
321                    input.prevout_index,
322                    input.sequence,
323                    input.required_time_lock_time,
324                    input.required_height_lock_time,
325                    input.script_sig,
326                    input.value,
327                    input.script_pubkey,
328                    input.redeem_script,
329                    input.partial_signatures,
330                    input.sighash_type,
331                    input
332                        .bip32_derivation
333                        .into_iter()
334                        .map(|(k, v)| {
335                            transparent::pczt::Bip32Derivation::parse(
336                                v.seed_fingerprint,
337                                v.derivation_path,
338                            )
339                            .map(|v| (k, v))
340                        })
341                        .collect::<Result<_, _>>()?,
342                    input.ripemd160_preimages,
343                    input.sha256_preimages,
344                    input.hash160_preimages,
345                    input.hash256_preimages,
346                    input.proprietary,
347                )
348            })
349            .collect::<Result<_, _>>()?;
350
351        let outputs = self
352            .outputs
353            .into_iter()
354            .map(|output| {
355                transparent::pczt::Output::parse(
356                    output.value,
357                    output.script_pubkey,
358                    output.redeem_script,
359                    output
360                        .bip32_derivation
361                        .into_iter()
362                        .map(|(k, v)| {
363                            transparent::pczt::Bip32Derivation::parse(
364                                v.seed_fingerprint,
365                                v.derivation_path,
366                            )
367                            .map(|v| (k, v))
368                        })
369                        .collect::<Result<_, _>>()?,
370                    output.user_address,
371                    output.proprietary,
372                )
373            })
374            .collect::<Result<_, _>>()?;
375
376        transparent::pczt::Bundle::parse(inputs, outputs)
377    }

379    pub(crate) fn serialize_from(bundle: transparent::pczt::Bundle) -> Self {
380        let inputs = bundle
381            .inputs()
382            .iter()
383            .map(|input| Input {
384                prevout_txid: (*input.prevout_txid()).into(),
385                prevout_index: *input.prevout_index(),
386                sequence: *input.sequence(),
387                required_time_lock_time: *input.required_time_lock_time(),
388                required_height_lock_time: *input.required_height_lock_time(),
389                script_sig: input
390                    .script_sig()
391                    .as_ref()
392                    .map(|script_sig| script_sig.to_bytes()),
393                value: input.value().into_u64(),
394                script_pubkey: input.script_pubkey().to_bytes(),
395                redeem_script: input
396                    .redeem_script()
397                    .as_ref()
398                    .map(|redeem_script| redeem_script.to_bytes()),
399                partial_signatures: input.partial_signatures().clone(),
400                sighash_type: input.sighash_type().encode(),
401                bip32_derivation: input
402                    .bip32_derivation()
403                    .iter()
404                    .map(|(k, v)| {
405                        (
406                            *k,
407                            Zip32Derivation {
408                                seed_fingerprint: *v.seed_fingerprint(),
409                                derivation_path: v
410                                    .derivation_path()
411                                    .iter()
412                                    .copied()
413                                    .map(u32::from)
414                                    .collect(),
415                            },
416                        )
417                    })
418                    .collect(),
419                ripemd160_preimages: input.ripemd160_preimages().clone(),
420                sha256_preimages: input.sha256_preimages().clone(),
421                hash160_preimages: input.hash160_preimages().clone(),
422                hash256_preimages: input.hash256_preimages().clone(),
423                proprietary: input.proprietary().clone(),
424            })
425            .collect();
426
427        let outputs = bundle
428            .outputs()
429            .iter()
430            .map(|output| Output {
431                value: output.value().into_u64(),
432                script_pubkey: output.script_pubkey().to_bytes(),
433                redeem_script: output
434                    .redeem_script()
435                    .as_ref()
436                    .map(|redeem_script| redeem_script.to_bytes()),
437                bip32_derivation: output
438                    .bip32_derivation()
439                    .iter()
440                    .map(|(k, v)| {
441                        (
442                            *k,
443                            Zip32Derivation {
444                                seed_fingerprint: *v.seed_fingerprint(),
445                                derivation_path: v
446                                    .derivation_path()
447                                    .iter()
448                                    .copied()
449                                    .map(u32::from)
450                                    .collect(),
451                            },
452                        )
453                    })
454                    .collect(),
455                user_address: output.user_address().clone(),
456                proprietary: output.proprietary().clone(),
457            })
458            .collect();
459
460        Self { inputs, outputs }
461    }
