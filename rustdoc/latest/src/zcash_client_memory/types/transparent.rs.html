18/// Stores the transparent outputs received by the wallet.
19#[derive(Debug, Default, Clone, PartialEq)]
20pub struct TransparentReceivedOutputs(pub(crate) BTreeMap<OutPoint, ReceivedTransparentOutput>);
21
22impl TransparentReceivedOutputs {
23    pub fn new() -> Self {
24        Self(BTreeMap::new())
25    }
26
27    #[cfg(feature = "transparent-inputs")]
28    pub fn detect_spending_accounts<'a>(
29        &self,
30        spent: impl Iterator<Item = &'a OutPoint>,
31    ) -> Result<BTreeSet<AccountId>, Error> {
32        let mut acc = BTreeSet::new();
33        for prevout in spent {
34            if let Some(output) = self.0.get(prevout) {
35                acc.insert(output.account_id);
36            }
37        }
38        Ok(acc)
39    }
40}
41
42impl Deref for TransparentReceivedOutputs {
43    type Target = BTreeMap<OutPoint, ReceivedTransparentOutput>;
44
45    fn deref(&self) -> &Self::Target {
46        &self.0
47    }
48}
49
50impl DerefMut for TransparentReceivedOutputs {
51    fn deref_mut(&mut self) -> &mut Self::Target {
52        &mut self.0
53    }
54}

84#[derive(Debug, Clone, PartialEq, Eq)]
85pub struct ReceivedTransparentOutput {
86    // Reference to the transaction in which this TXO was created
87    pub(crate) transaction_id: TxId,
88    // The account that controls spend authority for this TXO
89    pub(crate) account_id: AccountId,
90    // The address to which this TXO was sent
91    pub(crate) address: TransparentAddress,
92    // The key scope at which the address was derived
93    pub(crate) key_scope: TransparentKeyScope,
94    // script, value_zat
95    pub(crate) txout: TxOut,
96    /// The maximum block height at which this TXO was either
97    /// observed to be a member of the UTXO set at the start of the block, or observed
98    /// to be an output of a transaction mined in the block. This is intended to be used to
99    /// determine when the TXO is no longer a part of the UTXO set, in the case that the
100    /// transaction that spends it is not detected by the wallet.
101    pub(crate) max_observed_unspent_height: Option<BlockHeight>,
102}
103
104impl ReceivedTransparentOutput {
105    pub fn new(
106        transaction_id: TxId,
107        account_id: AccountId,
108        address: TransparentAddress,
109        key_scope: TransparentKeyScope,
110        txout: TxOut,
111        max_observed_unspent_height: BlockHeight,
112    ) -> Self {
113        Self {
114            transaction_id,
115            account_id,
116            address,
117            key_scope,
118            txout,
119            max_observed_unspent_height: Some(max_observed_unspent_height),
120        }
121    }
122
123    pub fn to_wallet_transparent_output(
124        &self,
125        outpoint: &OutPoint,
126        mined_height: Option<BlockHeight>,
127    ) -> Option<WalletTransparentOutput> {
128        WalletTransparentOutput::from_parts(outpoint.clone(), self.txout.clone(), mined_height)
129    }
130}

132/// A cache of the relationship between a transaction and the prevout data of its
133/// transparent inputs.
134///
135/// Output may be attempted to be spent in multiple transactions, even though only one will ever be mined
136/// which is why can cannot just rely on TransparentReceivedOutputSpends or implement this as as map
137#[derive(Debug, Default, PartialEq)]
138pub struct TransparentSpendCache(pub(crate) BTreeSet<(TxId, OutPoint)>);
139
140impl TransparentSpendCache {
141    pub fn new() -> Self {
142        Self(BTreeSet::new())
143    }
144
145    /// Get all the outpoints for a given transaction ID.
146    #[cfg(feature = "transparent-inputs")]
147    pub fn contains(&self, txid: &TxId, outpoint: &OutPoint) -> bool {
148        self.0.contains(&(*txid, outpoint.clone()))
149    }
150}

167    impl From<ReceivedTransparentOutput> for proto::ReceivedTransparentOutput {
168        fn from(output: ReceivedTransparentOutput) -> Self {
169            Self {
170                transaction_id: output.transaction_id.as_ref().to_vec(),
171                account_id: *output.account_id,
172                address: output.address.encode(&EncodingParams),
173                txout: Some(output.txout.into()),
174                max_observed_unspent_height: output.max_observed_unspent_height.map(|h| h.into()),
175            }
176        }
177    }
