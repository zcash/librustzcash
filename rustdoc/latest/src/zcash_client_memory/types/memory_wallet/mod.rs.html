107impl<P: consensus::Parameters + PartialEq> PartialEq for MemoryWalletDb<P> {
108    /// Tests for equality between two `MemoryWalletDb` instances.
109    /// but does NOT compare the sapling_tree and orchard_tree fields.
110    fn eq(&self, other: &Self) -> bool {
111        #[cfg(feature = "orchard")]
112        let orchard_comparisons =
113            { self.orchard_tree_shard_end_heights == other.orchard_tree_shard_end_heights };
114        #[cfg(not(feature = "orchard"))]
115        let orchard_comparisons = true;
116
117        #[cfg(feature = "transparent-inputs")]
118        let transparent_comparisons = {
119            self.transparent_received_outputs == other.transparent_received_outputs
120                && self.transparent_received_output_spends
121                    == other.transparent_received_output_spends
122                && self.transparent_spend_map == other.transparent_spend_map
123        };
124        #[cfg(not(feature = "transparent-inputs"))]
125        let transparent_comparisons = true;
126
127        self.params == other.params
128            && self.accounts == other.accounts
129            && self.blocks == other.blocks
130            && self.tx_table == other.tx_table
131            && self.received_notes == other.received_notes
132            && self.received_note_spends == other.received_note_spends
133            && self.nullifiers == other.nullifiers
134            && self.sent_notes == other.sent_notes
135            && self.tx_locator == other.tx_locator
136            && self.scan_queue == other.scan_queue
137            && self.sapling_tree_shard_end_heights == other.sapling_tree_shard_end_heights
138            && orchard_comparisons
139            && transparent_comparisons
140            && self.transaction_data_request_queue == other.transaction_data_request_queue
141    }

174    pub(crate) fn add_account(
175        &mut self,
176        account_name: &str,
177        kind: AccountSource,
178        viewing_key: UnifiedFullViewingKey,
179        birthday: AccountBirthday,
180    ) -> Result<(AccountId, Account), Error> {
181        let (id, account) = self.accounts.new_account(
182            account_name,
183            kind,
184            viewing_key.to_owned(),
185            birthday.clone(),
186        )?;
187
188        // If a birthday frontier is available, insert it into the note commitment tree. If the
189        // birthday frontier is the empty frontier, we don't need to do anything.
190        if let Some(frontier) = birthday.sapling_frontier().value() {
191            tracing::debug!("Inserting Sapling frontier into ShardTree: {:?}", frontier);
192            self.sapling_tree.insert_frontier_nodes(
193                frontier.clone(),
194                Retention::Checkpoint {
195                    // This subtraction is safe, because all leaves in the tree appear in blocks, and
196                    // the invariant that birthday.height() always corresponds to the block for which
197                    // `frontier` is the tree state at the start of the block. Together, this means
198                    // there exists a prior block for which frontier is the tree state at the end of
199                    // the block.
200                    id: birthday.height() - 1,
201                    marking: Marking::Reference,
202                },
203            )?;
204        }
205
206        #[cfg(feature = "orchard")]
207        if let Some(frontier) = birthday.orchard_frontier().value() {
208            tracing::debug!("Inserting Orchard frontier into ShardTree: {:?}", frontier);
209            self.orchard_tree.insert_frontier_nodes(
210                frontier.clone(),
211                Retention::Checkpoint {
212                    // This subtraction is safe, because all leaves in the tree appear in blocks, and
213                    // the invariant that birthday.height() always corresponds to the block for which
214                    // `frontier` is the tree state at the start of the block. Together, this means
215                    // there exists a prior block for which frontier is the tree state at the end of
216                    // the block.
217                    id: birthday.height() - 1,
218                    marking: Marking::Reference,
219                },
220            )?;
221        }
222
223        // The ignored range always starts at Sapling activation
224        let sapling_activation_height = self
225            .params
226            .activation_height(NetworkUpgrade::Sapling)
227            .expect("Sapling activation height must be available.");
228
229        // Add the ignored range up to the birthday height.
230        if sapling_activation_height < birthday.height() {
231            let ignored_range = sapling_activation_height..birthday.height();
232            self.scan_queue.replace_queue_entries(
233                &ignored_range,
234                Some(ScanRange::from_parts(
235                    ignored_range.clone(),
236                    ScanPriority::Ignored,
237                ))
238                .into_iter(),
239                false,
240            )?;
241        };
242
243        // Rewrite the scan ranges from the birthday height up to the chain tip so that we'll ensure we
244        // re-scan to find any notes that might belong to the newly added account.
245        if let Some(t) = self.chain_height()? {
246            let rescan_range = birthday.height()..(t + 1);
247            self.scan_queue.replace_queue_entries(
248                &rescan_range,
249                Some(ScanRange::from_parts(
250                    rescan_range.clone(),
251                    ScanPriority::Historic,
252                ))
253                .into_iter(),
254                true, // force rescan
255            )?;
256        }
257
258        Ok((id, account))
259    }

262    pub(crate) fn first_unsafe_index(&self, account_id: AccountId) -> Result<u32, Error> {
263        let first_unmined_index = if let Some(account) = self.accounts.get(account_id) {
264            let mut idx = 0;
265            for (tidx, eph_addr) in account.ephemeral_addresses.iter().rev() {
266                if eph_addr
267                    .seen
268                    .and_then(|txid| self.tx_table.get(&txid))
269                    .and_then(|tx| tx.mined_height())
270                    .is_some()
271                {
272                    idx = tidx.checked_add(1).unwrap();
273                    break;
274                }
275            }
276            idx
277        } else {
278            0
279        };
280
281        Ok(core::cmp::min(
282            1 << 31,
283            first_unmined_index
284                .checked_add(account::EPHEMERAL_GAP_LIMIT)
285                .unwrap(),
286        ))
287    }

290        &self,
291        tx: &Transaction,
292    ) -> Result<BTreeSet<AccountId>, Error> {
293        let mut funding_accounts = BTreeSet::new();
294        #[cfg(feature = "transparent-inputs")]

312        #[cfg(feature = "orchard")]
313        funding_accounts.extend(
314            self.received_notes.detect_orchard_spending_accounts(
315                tx.orchard_bundle()
316                    .iter()
317                    .flat_map(|bundle| bundle.actions().iter().map(|action| action.nullifier())),
318            )?,
319        );

321        Ok(funding_accounts)
322    }

350    /// Returns true if the note is in the spent notes table and the transaction that spent it is
351    /// in the transaction table and has either been mined or can be mined in the future
352    /// (i.e. it hasn't or will not expire)
353    pub(crate) fn note_is_spent(
354        &self,
355        note: &ReceivedNote,
356        target_height: TargetHeight,
357    ) -> Result<bool, Error> {
358        let spend = self.received_note_spends.get(&note.note_id());
359
360        let spent = match spend {
361            Some(txid) => {
362                let spending_tx = self
363                    .tx_table
364                    .get(txid)
365                    .ok_or(Error::TransactionNotFound(*txid))?;
366                match spending_tx.status() {
367                    TransactionStatus::Mined(_height) => true,
368                    TransactionStatus::TxidNotRecognized => unreachable!(),
369                    TransactionStatus::NotInMainChain => {
370                        // transaction either never expires, or expires in the future.
371                        spending_tx
372                            .expiry_height()
373                            .iter()
374                            .all(|h| *h >= BlockHeight::from(target_height))
375                    }
376                }
377            }
378            None => false,
379        };
380        Ok(spent)
381    }

383    /// To be spendable a note must be:
384    /// - unspent (obviously)
385    /// - not dust (value > 5000 ZATs)
386    /// - be associated with an account with a ufvk
387    /// - have a recipient key scope
388    /// - We know the nullifier
389    /// - We know the commitment tree position
390    /// - be in a block less than or equal to the anchor height
391    /// - not be in the given exclude list
392    ///
393    /// Additionally the tree shard containing the node must not be in an unscanned range
394    /// excluding ranges that start above the anchor height or end below the wallet birthday.
395    /// This is determined by looking at the scan queue
396    pub(crate) fn note_is_spendable(
397        &self,
398        note: &ReceivedNote,
399        birthday_height: BlockHeight,
400        target_height: TargetHeight,
401        confirmations_policy: ConfirmationsPolicy,
402        exclude: &[<MemoryWalletDb<P> as InputSource>::NoteRef],
403    ) -> Result<bool, Error> {
404        let note_account = self
405            .get_account(note.account_id())?
406            .ok_or(Error::AccountUnknown(note.account_id))?;
407        let note_txn = self
408            .tx_table
409            .get(&note.txid())
410            .ok_or_else(|| Error::TransactionNotFound(note.txid()))?;
411
412        let unscanned_ranges = self.unscanned_ranges();
413        let anchor_height = target_height - u32::from(confirmations_policy.trusted());
414
415        let note_in_unscanned_range =
416            unscanned_ranges
417                .iter()
418                .any(|(start_height, end_height, start, end_exclusive)| {
419                    let in_range = note.commitment_tree_position.is_some_and(|pos| {
420                        if let (Some(start), Some(end_exclusive)) = (start, end_exclusive) {
421                            pos >= *start && pos < *end_exclusive
422                        } else {
423                            true
424                        }
425                    });
426                    in_range && *end_height > birthday_height && *start_height <= anchor_height
427                });
428
429        Ok(!self.note_is_spent(note, target_height)?
430            && !note_in_unscanned_range
431            && note.note.value().into_u64() > 5000
432            && note_account.ufvk().is_some()
433            && note.nullifier().is_some()
434            && note.commitment_tree_position.is_some()
435            && note_txn.mined_height().is_some()
436            && match note.recipient_key_scope {
437                Some(Scope::External) => {
438                    target_height.saturating_sub(u32::from(confirmations_policy.untrusted()))
439                        >= note_txn.mined_height().unwrap()
440                }
441                Some(Scope::Internal) => {
442                    target_height.saturating_sub(u32::from(confirmations_policy.trusted()))
443                        >= note_txn.mined_height().unwrap()
444                }
445                None => false,
446            }
447            && !exclude.contains(&note.note_id()))
448    }

450    /// To be pending a note must be:
451    /// - mined, but without sufficient confirmations to be spent; or
452    /// - unmined and unexpired
453    pub(crate) fn note_is_pending(
454        &self,
455        note: &ReceivedNote,
456        target_height: TargetHeight,
457        confirmations_policy: ConfirmationsPolicy,
458    ) -> Result<bool, Error> {
459        let tx = self
460            .tx_table
461            .get(&note.txid())
462            .ok_or_else(|| Error::TransactionNotFound(note.txid()))?;
463
464        Ok(tx.mined_height().map_or_else(
465            || {
466                tx.expiry_height()
467                    .iter()
468                    .all(|h| *h >= BlockHeight::from(target_height))
469            },
470            |h| match note.recipient_key_scope {
471                // If we don't know the recipient key scope, we treat the note as untrusted.
472                Some(Scope::External) | None => {
473                    BlockHeight::from(target_height) - h
474                        < u32::from(confirmations_policy.untrusted())
475                }
476                Some(Scope::Internal) => {
477                    BlockHeight::from(target_height) - h < u32::from(confirmations_policy.trusted())
478                }
479            },
480        ))
481    }

557    pub(crate) fn insert_sapling_nullifier_map(
558        &mut self,
559        block_height: BlockHeight,
560        new_entries: &[(TxId, u16, Vec<sapling::Nullifier>)],
561    ) -> Result<(), Error> {
562        for (txid, tx_index, nullifiers) in new_entries {
563            for nf in nullifiers.iter() {
564                self.nullifiers
565                    .insert(block_height, *tx_index as u32, Nullifier::Sapling(*nf));
566            }
567            match self.tx_locator.entry((block_height, *tx_index as u32)) {
568                Entry::Occupied(x) => {
569                    if txid == x.get() {
570                        // This is a duplicate entry
571                        continue;
572                    } else {
573                        return Err(Error::ConflictingTxLocator);
574                    }
575                }
576                Entry::Vacant(entry) => {
577                    entry.insert(*txid);
578                }
579            }
580        }
581        Ok(())
582    }

612    pub(crate) fn block_height_extrema(&self) -> Option<RangeInclusive<BlockHeight>> {
613        let (min, max) = self.blocks.keys().fold((None, None), |(min, max), height| {
614            (
615                Some(min.map_or(height, |min| std::cmp::min(min, height))),
616                Some(max.map_or(height, |max| std::cmp::max(max, height))),
617            )
618        });
619        if let (Some(min), Some(max)) = (min, max) {
620            Some(*min..=*max)
621        } else {
622            None
623        }
624    }

639    ) -> Result<Option<BlockHeight>, Error> {

641            u32::from(chain_tip_height).saturating_sub(u32::from(min_confirmations) - 1);

645            if self.sapling_tree.store().get_checkpoint(&height)?.is_some() {
646                return Ok(Some(height));

706    /// Makes the required changes to the scan queue to reflect the completion of a scan
707    pub(crate) fn scan_complete(
708        &mut self,
709        range: Range<BlockHeight>,
710        wallet_note_positions: &[(ShieldedProtocol, Position)],
711    ) -> Result<(), Error> {
712        let wallet_birthday = self.get_wallet_birthday()?;
713
714        // Determine the range of block heights for which we will be updating the scan queue.
715        let extended_range = {
716            // If notes have been detected in the scan, we need to extend any adjacent un-scanned
717            // ranges starting from the wallet birthday to include the blocks needed to complete
718            // the note commitment tree subtrees containing the positions of the discovered notes.
719            // We will query by subtree index to find these bounds.
720            let mut required_sapling_subtrees = BTreeSet::new();
721            #[cfg(feature = "orchard")]
722            let mut required_orchard_subtrees = BTreeSet::new();
723            for (protocol, position) in wallet_note_positions {
724                match protocol {
725                    ShieldedProtocol::Sapling => {
726                        required_sapling_subtrees.insert(
727                            Address::above_position(SAPLING_SHARD_HEIGHT.into(), *position).index(),
728                        );
729                    }
730                    ShieldedProtocol::Orchard => {
731                        #[cfg(feature = "orchard")]
732                        required_orchard_subtrees.insert(
733                            Address::above_position(ORCHARD_SHARD_HEIGHT.into(), *position).index(),
734                        );
735
736                        #[cfg(not(feature = "orchard"))]
737                        return Err(Error::OrchardNotEnabled);
738                    }
739                }
740            }
741
742            let extended_range = self.extend_range(
743                &ShieldedProtocol::Sapling,
744                &range,
745                required_sapling_subtrees,
746                self.params.activation_height(NetworkUpgrade::Sapling),
747                wallet_birthday,
748            )?;
749
750            #[cfg(feature = "orchard")]
751            let extended_range = self
752                .extend_range(
753                    &ShieldedProtocol::Orchard,
754                    extended_range.as_ref().unwrap_or(&range),
755                    required_orchard_subtrees,
756                    self.params.activation_height(NetworkUpgrade::Nu5),
757                    wallet_birthday,
758                )?
759                .or(extended_range);
760
761            #[allow(clippy::let_and_return)]
762            extended_range
763        };
764
765        let query_range = extended_range.clone().unwrap_or_else(|| range.clone());
766
767        let scanned = ScanRange::from_parts(range.clone(), ScanPriority::Scanned);
768
769        // If any of the extended range actually extends beyond the scanned range, we need to
770        // scan that extension in order to make the found note(s) spendable. We need to avoid
771        // creating empty ranges here, as that acts as an optimization barrier preventing
772        // `SpanningTree` from merging non-empty scanned ranges on either side.
773        let extended_before = extended_range
774            .as_ref()
775            .map(|extended| {
776                ScanRange::from_parts(extended.start..range.start, ScanPriority::FoundNote)
777            })
778            .filter(|range| !range.is_empty());
779        let extended_after = extended_range
780            .map(|extended| ScanRange::from_parts(range.end..extended.end, ScanPriority::FoundNote))
781            .filter(|range| !range.is_empty());
782
783        let replacement = Some(scanned)
784            .into_iter()
785            .chain(extended_before)
786            .chain(extended_after);
787
788        self.scan_queue
789            .replace_queue_entries(&query_range, replacement, false)
790    }

792    // Given a range of block heights, extend the range to include the subtrees containing the
793    // given subtree indices, bounded by the wallet birthday and the fallback start height.
794    fn extend_range(
795        &self,
796        pool: &ShieldedProtocol,
797        range: &Range<BlockHeight>,
798        required_subtree_indices: BTreeSet<u64>,
799        fallback_start_height: Option<BlockHeight>,
800        birthday_height: Option<BlockHeight>,
801    ) -> Result<Option<Range<BlockHeight>>, Error> {
802        // we'll either have both min and max bounds, or we'll have neither
803        let subtree_index_bounds = required_subtree_indices
804            .iter()
805            .min()
806            .zip(required_subtree_indices.iter().max());
807
808        let shard_end = |index| -> Result<_, Error> {
809            match pool {
810                ShieldedProtocol::Sapling => Ok(self
811                    .sapling_tree_shard_end_heights
812                    .get(&Address::from_parts(SAPLING_SHARD_HEIGHT.into(), index))
813                    .cloned()),
814                ShieldedProtocol::Orchard => {
815                    #[cfg(feature = "orchard")]
816                    {
817                        Ok(self
818                            .orchard_tree_shard_end_heights
819                            .get(&Address::from_parts(ORCHARD_SHARD_HEIGHT.into(), index))
820                            .cloned())
821                    }
822                    #[cfg(not(feature = "orchard"))]
823                    panic!("Unsupported pool")
824                }
825            }
826        };
827
828        // If no notes belonging to the wallet were found, we don't need to extend the scanning
829        // range suggestions to include the associated subtrees, and our bounds are just the
830        // scanned range. Otherwise, ensure that all shard ranges starting from the wallet
831        // birthday are included.
832        subtree_index_bounds
833            .map(|(min_idx, max_idx)| {
834                let range_min = if *min_idx > 0 {
835                    // get the block height of the end of the previous shard
836                    shard_end(*min_idx - 1)?
837                } else {
838                    // our lower bound is going to be the fallback height
839                    fallback_start_height
840                };
841
842                // bound the minimum to the wallet birthday
843                let range_min =
844                    range_min.map(|h| birthday_height.map_or(h, |b| std::cmp::max(b, h)));
845
846                // Get the block height for the end of the current shard, and make it an
847                // exclusive end bound.
848                let range_max = shard_end(*max_idx)?.map(|end| end + 1);
849
850                Ok(Range {
851                    start: range.start.min(range_min.unwrap_or(range.start)),
852                    end: range.end.max(range_max.unwrap_or(range.end)),
853                })
854            })
855            .transpose()
856    }

862    pub(crate) fn sapling_scan_progress(
863        &self,
864        birthday_height: &BlockHeight,
865        fully_scanned_height: &BlockHeight,
866        chain_tip_height: &BlockHeight,
867    ) -> Result<Option<Ratio<u64>>, Error> {
868        if fully_scanned_height == chain_tip_height {
869            let outputs_sum = self
870                .blocks
871                .iter()
872                .filter(|(height, _)| height >= &birthday_height)
873                .fold(0, |sum, (_, block)| {
874                    sum + block.sapling_output_count.unwrap_or(0)
875                });
876            Ok(Some(Ratio::new(outputs_sum as u64, outputs_sum as u64)))
877        } else {
878            // Get the starting note commitment tree size from the wallet birthday, or failing that
879            // from the blocks table.
880            let start_size = self
881                .accounts
882                .iter()
883                .filter_map(|(_, account)| {
884                    if account.birthday().height() == *birthday_height {
885                        Some(account.birthday().sapling_frontier().tree_size())
886                    } else {
887                        None
888                    }
889                })
890                .next()
891                .or_else(|| {
892                    self.blocks
893                        .iter()
894                        .filter(|(height, _)| height <= &birthday_height)
895                        .map(|(_, block)| {
896                            (block.sapling_commitment_tree_size.unwrap_or(0)
897                                - block.sapling_output_count.unwrap_or(0))
898                                as u64
899                        })
900                        .max()
901                });
902
903            // Compute the total blocks scanned so far above the starting height
904            let scanned_count = self
905                .blocks
906                .iter()
907                .filter(|(height, _)| height > &birthday_height)
908                .fold(0_u64, |acc, (_, block)| {
909                    acc + block.sapling_output_count.unwrap_or(0) as u64
910                });
911
912            // We don't have complete information on how many outputs will exist in the shard at
913            // the chain tip without having scanned the chain tip block, so we overestimate by
914            // computing the maximum possible number of notes directly from the shard indices.
915            //
916            // TODO: it would be nice to be able to reliably have the size of the commitment tree
917            // at the chain tip without having to have scanned that block.
918
919            let shard_index_iter = self
920                .sapling_tree_shard_end_heights
921                .iter()
922                .filter(|(_, height)| height > &birthday_height)
923                .map(|(address, _)| address.index());
924
925            let min_idx = shard_index_iter.clone().min().unwrap_or(0);
926            let max_idx = shard_index_iter.max().unwrap_or(0);
927
928            let max_tree_size = Some(min_idx << SAPLING_SHARD_HEIGHT);
929            let min_tree_size = Some((max_idx + 1) << SAPLING_SHARD_HEIGHT);
930
931            Ok(start_size.or(min_tree_size).zip(max_tree_size).map(
932                |(min_tree_size, max_tree_size)| {
933                    Ratio::new(scanned_count, max_tree_size - min_tree_size)
934                },
935            ))
936        }
937    }

1040    #[cfg(feature = "transparent-inputs")]
1041    #[allow(unreachable_code, unused_variables)] //FIXME: need address key scope detection
1042    pub(crate) fn put_transparent_output(
1043        &mut self,
1044        output: &WalletTransparentOutput,
1045        receiving_account: &AccountId,
1046        known_unspent: bool,
1047    ) -> Result<OutPoint, Error> {
1048        use crate::types::transparent::ReceivedTransparentOutput;
1049
1050        let address = output.recipient_address();
1051        // get the block height of the block that mined the output only if we have it in the block table
1052        // otherwise return None
1053        let block = output
1054            .mined_height()
1055            .and_then(|h| self.blocks.get(&h).map(|b| b.height));
1056        let txid = TxId::from_bytes(output.outpoint().hash().to_vec().try_into().unwrap());
1057
1058        // insert a new tx into the transactions table for the one that spent this output. If there is already one then do an update
1059        self.tx_table
1060            .put_tx_partial(&txid, &block, output.mined_height());
1061
1062        // look for a spent_height for this output by querying transparent_received_output_spends.
1063        // If there isn't one then return None (this is an unspent output)
1064        // otherwise return the height found by joining on the tx table
1065        let spent_height = self
1066            .transparent_received_output_spends
1067            .get(output.outpoint())
1068            .and_then(|txid| {
1069                self.tx_table
1070                    .tx_status(txid)
1071                    .and_then(|status| match status {
1072                        TransactionStatus::Mined(height) => Some(height),
1073                        _ => None,
1074                    })
1075            });
1076
1077        // The max observed unspent height is either the spending transaction's mined height - 1, or
1078        // the current chain tip height if the UTXO was received via a path that confirmed that it was
1079        // unspent, such as by querying the UTXO set of the network.
1080        let max_observed_unspent = match spent_height {
1081            Some(h) => Some(h - 1),
1082            None => {
1083                if known_unspent {
1084                    self.chain_height()?
1085                } else {
1086                    None
1087                }
1088            }
1089        };
1090
1091        // insert into transparent_received_outputs table. Update if it exists
1092        #[allow(clippy::diverging_sub_expression)] // FIXME
1093        match self
1094            .transparent_received_outputs
1095            .entry(output.outpoint().clone())
1096        {
1097            Entry::Occupied(mut entry) => {
1098                entry.get_mut().transaction_id = txid;
1099                entry.get_mut().address = *address;
1100                entry.get_mut().account_id = *receiving_account;
1101                entry.get_mut().txout = output.txout().clone();
1102            }
1103            Entry::Vacant(entry) => {
1104                entry.insert(ReceivedTransparentOutput::new(
1105                    txid,
1106                    *receiving_account,
1107                    *address,
1108                    todo!("look up the key scope for the address"),
1109                    output.txout().clone(),
1110                    max_observed_unspent.unwrap_or(BlockHeight::from(0)),
1111                ));
1112            }
1113        }
1114
1115        // look in transparent_spend_map for a record of the output already having been spent, then mark it as spent using the
1116        // stored reference to the spending transaction.
1117        if self
1118            .transparent_spend_map
1119            .contains(&txid, output.outpoint())
1120        {
1121            self.mark_transparent_output_spent(&txid, output.outpoint())?;
1122        }
1123
1124        Ok(output.outpoint().clone())
1125    }
