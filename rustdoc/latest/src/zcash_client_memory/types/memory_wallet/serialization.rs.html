11impl<P: Parameters> MemoryWalletDb<P> {
12    /// Encode a memory wallet db as a protobuf byte buffer
13    /// Always uses the latest version of the wire protocol
14    pub fn encode<B: BufMut>(&self, buf: &mut B) -> Result<()> {
15        let proto_wallet: proto::MemoryWallet = self.into();
16        proto_wallet.encode(buf)?;
17        Ok(())
18    }
19
20    /// Create a mew memory wallet db from a protobuf encoded byte buffer with version awareness
21    pub fn decode_new<B: Buf>(buf: B, params: P, max_checkpoints: usize) -> Result<Self> {
22        let proto_wallet = proto::MemoryWallet::decode(buf)?;
23        Self::new_from_proto(proto_wallet, params, max_checkpoints)
24    }
25
26    /// Build a memory wallet db from protobuf type with version awareness
27    pub fn new_from_proto(
28        proto_wallet: proto::MemoryWallet,
29        params: P,
30        max_checkpoints: usize,
31    ) -> Result<Self> {
32        match proto_wallet.version {
33            1 => Self::new_from_proto_v1(proto_wallet, params, max_checkpoints),
34            _ => Err(Error::UnsupportedProtoVersion(1, proto_wallet.version)),
35        }
36    }
37
38    fn new_from_proto_v1(
39        proto_wallet: proto::MemoryWallet,
40        params: P,
41        max_checkpoints: usize,
42    ) -> Result<Self> {
43        if proto_wallet.version != 1 {
44            return Err(Error::UnsupportedProtoVersion(1, proto_wallet.version));
45        }
46
47        let mut wallet = MemoryWalletDb::new(params, max_checkpoints);
48
49        wallet.accounts = {
50            let proto_accounts = read_optional!(proto_wallet, accounts)?;
51            let accounts = proto_accounts
52                .accounts
53                .into_iter()
54                .map(|proto_account| {
55                    let id = proto_account.account_id;
56                    let account = Account::try_from(proto_account)?;
57                    Ok((AccountId::from(id), account))
58                })
59                .collect::<Result<_>>()?;
60            Ok::<Accounts, Error>(Accounts {
61                accounts,
62                nonce: proto_accounts.account_nonce,
63            })
64        }?;
65
66        wallet.blocks = proto_wallet
67            .blocks
68            .into_iter()
69            .map(|proto_block| {
70                Ok((
71                    proto_block.height.into(),
72                    MemoryWalletBlock::try_from(proto_block)?,
73                ))
74            })
75            .collect::<Result<_>>()?;
76
77        wallet.tx_table = TransactionTable(
78            proto_wallet
79                .tx_table
80                .into_iter()
81                .map(|proto_tx| {
82                    let txid = read_optional!(proto_tx, tx_id)?;
83                    let tx = read_optional!(proto_tx, tx_entry)?;
84                    Ok((txid.try_into()?, tx.try_into()?))
85                })
86                .collect::<Result<_>>()?,
87        );
88
89        wallet.received_notes = ReceivedNoteTable(
90            proto_wallet
91                .received_note_table
92                .into_iter()
93                .map(ReceivedNote::try_from)
94                .collect::<Result<_>>()?,
95        );
96
97        wallet.received_note_spends = ReceievedNoteSpends(
98            proto_wallet
99                .received_note_spends
100                .into_iter()
101                .map(|proto_spend| {
102                    let note_id = read_optional!(proto_spend, note_id)?;
103                    let tx_id = read_optional!(proto_spend, tx_id)?;
104                    Ok((note_id.try_into()?, tx_id.try_into()?))
105                })
106                .collect::<Result<_>>()?,
107        );
108
109        wallet.nullifiers = NullifierMap(
110            proto_wallet
111                .nullifiers
112                .into_iter()
113                .map(|proto_nullifier| {
114                    let block_height = proto_nullifier.block_height.into();
115                    let tx_index = proto_nullifier.tx_index;
116                    let nullifier = read_optional!(proto_nullifier, nullifier)?.try_into()?;
117                    Ok((nullifier, (block_height, tx_index)))
118                })
119                .collect::<Result<_>>()?,
120        );
121
122        wallet.sent_notes = SentNoteTable(
123            proto_wallet
124                .sent_notes
125                .into_iter()
126                .map(|proto_sent_note| {
127                    let sent_note_id = read_optional!(proto_sent_note, sent_note_id)?;
128                    let sent_note = read_optional!(proto_sent_note, sent_note)?;
129                    Ok((sent_note_id.try_into()?, SentNote::try_from(sent_note)?))
130                })
131                .collect::<Result<_>>()?,
132        );
133
134        wallet.tx_locator = TxLocatorMap(
135            proto_wallet
136                .tx_locator
137                .into_iter()
138                .map(|proto_locator| {
139                    let block_height = proto_locator.block_height.into();
140                    let tx_index = proto_locator.tx_index;
141                    let tx_id = read_optional!(proto_locator, tx_id)?.try_into()?;
142                    Ok(((block_height, tx_index), tx_id))
143                })
144                .collect::<Result<_>>()?,
145        );
146
147        wallet.scan_queue = ScanQueue(
148            proto_wallet
149                .scan_queue
150                .into_iter()
151                .map(|item| item.into())
152                .collect(),
153        );
154
155        wallet.sapling_tree =
156            tree_from_protobuf(read_optional!(proto_wallet, sapling_tree)?, 100, 16.into())?;
157
158        wallet.sapling_tree_shard_end_heights = proto_wallet
159            .sapling_tree_shard_end_heights
160            .into_iter()
161            .map(|proto_end_height| {
162                let address = Address::from_parts(
163                    Level::from(u8::try_from(proto_end_height.level)?),
164                    proto_end_height.index,
165                );
166                let height = proto_end_height.block_height.into();
167                Ok((address, height))
168            })
169            .collect::<Result<_>>()?;
170
171        #[cfg(feature = "orchard")]
172        {
173            wallet.orchard_tree =
174                tree_from_protobuf(read_optional!(proto_wallet, orchard_tree)?, 100, 16.into())?;
175        };
176
177        #[cfg(feature = "orchard")]
178        {
179            wallet.orchard_tree_shard_end_heights = proto_wallet
180                .orchard_tree_shard_end_heights
181                .into_iter()
182                .map(|proto_end_height| {
183                    let address = Address::from_parts(
184                        Level::from(u8::try_from(proto_end_height.level)?),
185                        proto_end_height.index,
186                    );
187                    let height = proto_end_height.block_height.into();
188                    Ok((address, height))
189                })
190                .collect::<Result<_>>()?;
191        };
192
193        wallet.transparent_received_outputs = TransparentReceivedOutputs(
194            proto_wallet
195                .transparent_received_outputs
196                .into_iter()
197                .map(|proto_output| {
198                    let outpoint = read_optional!(proto_output, outpoint)?;
199                    let output = read_optional!(proto_output, output)?.try_into()?;
200                    Ok((OutPoint::try_from(outpoint)?, output))
201                })
202                .collect::<Result<_>>()?,
203        );
204
205        wallet.transparent_received_output_spends = TransparentReceivedOutputSpends(
206            proto_wallet
207                .transparent_received_output_spends
208                .into_iter()
209                .map(|proto_spend| {
210                    let outpoint = read_optional!(proto_spend, outpoint)?;
211                    let txid = read_optional!(proto_spend, tx_id)?.try_into()?;
212                    Ok((OutPoint::try_from(outpoint)?, txid))
213                })
214                .collect::<Result<_>>()?,
215        );
216
217        wallet.transparent_spend_map = TransparentSpendCache(
218            proto_wallet
219                .transparent_spend_map
220                .into_iter()
221                .map(|proto_spend| {
222                    let txid = read_optional!(proto_spend, tx_id)?.try_into()?;
223                    let outpoint = read_optional!(proto_spend, outpoint)?;
224                    Ok((txid, OutPoint::try_from(outpoint)?))
225                })
226                .collect::<Result<_>>()?,
227        );
228
229        wallet.transaction_data_request_queue = TransactionDataRequestQueue(
230            proto_wallet
231                .transaction_data_requests
232                .into_iter()
233                .map(TryInto::try_into)
234                .collect::<Result<_>>()?,
235        );
236
237        Ok(wallet)
238    }
