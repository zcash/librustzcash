6#[derive(Debug, Default, PartialEq)]
7pub struct TransactionDataRequestQueue(pub(crate) VecDeque<TransactionDataRequest>);

9impl TransactionDataRequestQueue {

11        Self(VecDeque::new())
12    }
13
14    pub fn queue_status_retrieval(&mut self, txid: &TxId) {
15        self.0.push_back(TransactionDataRequest::GetStatus(*txid));
16    }
17}

27mod serialization {
28    use super::*;
29    use crate::{error::Error, proto::memwallet as proto, read_optional};
30
31    #[cfg(feature = "transparent-inputs")]
32    use {
33        ::transparent::address::TransparentAddress, zcash_keys::encoding::AddressCodec as _,
34        zcash_protocol::consensus::Network::MainNetwork as EncodingParams,
35    };
36
37    impl From<TransactionDataRequest> for proto::TransactionDataRequest {
38        fn from(request: TransactionDataRequest) -> Self {
39            match request {
40                TransactionDataRequest::GetStatus(txid) => Self {
41                    request_type: proto::TransactionDataRequestType::GetStatus as i32,
42                    tx_id: Some(txid.into()),
43                    address: None,
44                    block_range_start: None,
45                    block_range_end: None,
46                },
47                TransactionDataRequest::Enhancement(txid) => Self {
48                    request_type: proto::TransactionDataRequestType::Enhancement as i32,
49                    tx_id: Some(txid.into()),
50                    address: None,
51                    block_range_start: None,
52                    block_range_end: None,
53                },
54                #[cfg(feature = "transparent-inputs")]
55                TransactionDataRequest::TransactionsInvolvingAddress(req) => Self {
56                    request_type: proto::TransactionDataRequestType::SpendsFromAddress as i32,
57                    tx_id: None,
58                    address: Some(req.address().encode(&EncodingParams).as_bytes().to_vec()),
59                    block_range_start: Some(u32::from(req.block_range_start())),
60                    block_range_end: req.block_range_end().map(u32::from),
61                },
62            }
63        }
64    }
65
66    impl TryFrom<proto::TransactionDataRequest> for TransactionDataRequest {
67        type Error = crate::Error;
68
69        fn try_from(request: proto::TransactionDataRequest) -> Result<Self, crate::Error> {
70            Ok(match request.request_type() {
71                proto::TransactionDataRequestType::GetStatus => {
72                    TransactionDataRequest::GetStatus(read_optional!(request, tx_id)?.try_into()?)
73                }
74                proto::TransactionDataRequestType::Enhancement => {
75                    TransactionDataRequest::Enhancement(read_optional!(request, tx_id)?.try_into()?)
76                }
77                #[cfg(feature = "transparent-inputs")]
78                proto::TransactionDataRequestType::SpendsFromAddress => {
79                    use zcash_client_backend::data_api::{
80                        OutputStatusFilter, TransactionStatusFilter,
81                    };
82
83                    TransactionDataRequest::transactions_involving_address(
84                        TransparentAddress::decode(
85                            &EncodingParams,
86                            &String::from_utf8(read_optional!(request, address)?)?,
87                        )?,
88                        read_optional!(request, block_range_start)?.into(),
89                        Some(read_optional!(request, block_range_end)?.into()),
90                        None,
91                        TransactionStatusFilter::Mined,
92                        OutputStatusFilter::All,
93                    )
94                }
95                #[cfg(not(feature = "transparent-inputs"))]
96                _ => panic!("invalid request type"),
97            })
98        }
99    }
