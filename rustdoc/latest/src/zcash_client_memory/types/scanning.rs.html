10/// A queue of scanning ranges. Contains the start and end heights of each range, along with the
11/// priority of scanning that range.
12#[derive(Debug, Clone, PartialEq)]
13pub(crate) struct ScanQueue(pub(crate) Vec<(BlockHeight, BlockHeight, ScanPriority)>);
14
15impl ScanQueue {
16    pub(crate) fn new() -> Self {
17        ScanQueue(Vec::new())
18    }
19
20    pub(crate) fn suggest_scan_ranges(&self, min_priority: ScanPriority) -> Vec<ScanRange> {
21        let mut priorities: Vec<_> = self
22            .0
23            .iter()
24            .filter(|(_, _, p)| *p >= min_priority)
25            .collect();
26        priorities.sort_by(|(_, _, a), (_, _, b)| b.cmp(a));
27
28        priorities
29            .into_iter()
30            .map(|(start, end, priority)| {
31                let range = Range {
32                    start: *start,
33                    end: *end,
34                };
35                ScanRange::from_parts(range, *priority)
36            })
37            .collect()
38    }
39    fn insert_queue_entries<'a>(
40        &mut self,
41        entries: impl Iterator<Item = &'a ScanRange>,
42    ) -> Result<(), Error> {
43        for entry in entries {
44            if entry.block_range().start >= entry.block_range().end {
45                return Err(Error::InvalidScanRange(
46                    entry.block_range().start,
47                    entry.block_range().end,
48                    "start must be less than end".to_string(),
49                ));
50            }
51
52            for (start, end, _) in &self.0 {
53                if *start == entry.block_range().start || *end == entry.block_range().end {
54                    return Err(Error::InvalidScanRange(
55                        entry.block_range().start,
56                        entry.block_range().end,
57                        "at least part of range is already covered by another range".to_string(),
58                    ));
59                }
60            }
61
62            self.0.push((
63                entry.block_range().start,
64                entry.block_range().end,
65                entry.priority(),
66            ));
67        }
68        Ok(())
69    }
70    pub(crate) fn replace_queue_entries(
71        &mut self,
72        query_range: &Range<BlockHeight>,
73        entries: impl Iterator<Item = ScanRange>,
74        force_rescans: bool,
75    ) -> Result<(), Error> {
76        let (to_create, to_delete_ends) = {
77            let mut q_ranges: Vec<_> = self
78                .0
79                .iter()
80                .filter(|(start, end, _)| {
81                    //  Ignore ranges that do not overlap and are not adjacent to the query range.
82                    !(start > &query_range.end || &query_range.start > end)
83                })
84                .collect();
85            q_ranges.sort_by(|(_, end_a, _), (_, end_b, _)| end_a.cmp(end_b));
86
87            // Iterate over the ranges in the scan queue that overlap the range that we have
88            // identified as needing to be fully scanned. For each such range add it to the
89            // spanning tree (these should all be nonoverlapping ranges, but we might coalesce
90            // some in the process).
91            let mut to_create: Option<SpanningTree> = None;
92            let mut to_delete_ends: Vec<BlockHeight> = vec![];
93
94            let q_ranges = q_ranges.into_iter();
95            for (start, end, priority) in q_ranges {
96                let entry = ScanRange::from_parts(
97                    Range {
98                        start: *start,
99                        end: *end,
100                    },
101                    *priority,
102                );
103                to_delete_ends.push(entry.block_range().end);
104                to_create = if let Some(cur) = to_create {
105                    Some(cur.insert(entry, force_rescans))
106                } else {
107                    Some(SpanningTree::Leaf(entry))
108                };
109            }
110
111            // Update the tree that we read from the database, or if we didn't find any ranges
112            // start with the scanned range.
113            for entry in entries {
114                to_create = if let Some(cur) = to_create {
115                    Some(cur.insert(entry, force_rescans))
116                } else {
117                    Some(SpanningTree::Leaf(entry))
118                };
119            }
120            (to_create, to_delete_ends)
121        };
122
123        if let Some(tree) = to_create {
124            self.0.retain(|(_, block_range_end, _)| {
125                // if the block_range_end is equal to any in to_delete_ends, remove it
126                !to_delete_ends.contains(block_range_end)
127            });
128            let scan_ranges = tree.into_vec();
129            self.insert_queue_entries(scan_ranges.iter())?;
130        }
131        Ok(())
132    }
133
134    pub fn delete_starts_greater_than_equal_to(&mut self, height: BlockHeight) {
135        self.0.retain(|(start, _, _)| *start < height);
136    }
137
138    pub fn truncate_ends_to(&mut self, height: BlockHeight) {
139        self.0.iter_mut().for_each(|(_, end, _)| {
140            if *end > height {
141                *end = height;
142            }
143        });
144    }

147// We deref to slice so that we can reuse the slice impls
148impl Deref for ScanQueue {
149    type Target = [(BlockHeight, BlockHeight, ScanPriority)];
150
151    fn deref(&self) -> &Self::Target {
152        &self.0[..]
153    }
154}
155impl DerefMut for ScanQueue {

165    impl From<(BlockHeight, BlockHeight, ScanPriority)> for proto::ScanQueueRecord {
166        fn from(
167            (start_height, end_height, priority): (BlockHeight, BlockHeight, ScanPriority),
168        ) -> Self {
169            Self {
170                start_height: start_height.into(),
171                end_height: end_height.into(),
172                priority: match priority {
173                    ScanPriority::Ignored => proto::ScanPriority::Ignored as i32,
174                    ScanPriority::Scanned => proto::ScanPriority::Scanned as i32,
175                    ScanPriority::Historic => proto::ScanPriority::Historic as i32,
176                    ScanPriority::OpenAdjacent => proto::ScanPriority::OpenAdjacent as i32,
177                    ScanPriority::FoundNote => proto::ScanPriority::FoundNote as i32,
178                    ScanPriority::ChainTip => proto::ScanPriority::ChainTip as i32,
179                    ScanPriority::Verify => proto::ScanPriority::Verify as i32,
180                },
181            }
182        }
183    }
184
185    impl From<proto::ScanQueueRecord> for (BlockHeight, BlockHeight, ScanPriority) {
186        fn from(record: proto::ScanQueueRecord) -> Self {
187            (
188                record.start_height.into(),
189                record.end_height.into(),
190                match record.priority() {
191                    proto::ScanPriority::Ignored => ScanPriority::Ignored,
192                    proto::ScanPriority::Scanned => ScanPriority::Scanned,
193                    proto::ScanPriority::Historic => ScanPriority::Historic,
194                    proto::ScanPriority::OpenAdjacent => ScanPriority::OpenAdjacent,
195                    proto::ScanPriority::FoundNote => ScanPriority::FoundNote,
196                    proto::ScanPriority::ChainTip => ScanPriority::ChainTip,
197                    proto::ScanPriority::Verify => ScanPriority::Verify,
198                },
199            )
200        }
201    }
