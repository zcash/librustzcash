10/// Internal wallet representation of a Block.
11#[derive(Clone, Debug, PartialEq)]
12pub(crate) struct MemoryWalletBlock {
13    pub(crate) height: BlockHeight,
14    pub(crate) hash: BlockHash,
15    pub(crate) block_time: u32,
16    // Just the transactions that involve an account in this wallet
17    pub(crate) _transactions: HashSet<TxId>,
18    pub(crate) _memos: HashMap<NoteId, MemoBytes>,
19    pub(crate) sapling_commitment_tree_size: Option<u32>,
20    pub(crate) sapling_output_count: Option<u32>,
21    #[cfg(feature = "orchard")]
22    pub(crate) orchard_commitment_tree_size: Option<u32>,
23    #[cfg(feature = "orchard")]
24    pub(crate) orchard_action_count: Option<u32>,
25}
26
27impl Eq for MemoryWalletBlock {}
28
29impl PartialOrd for MemoryWalletBlock {
30    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
31        Some(self.cmp(other))
32    }
33}
34
35impl Ord for MemoryWalletBlock {
36    fn cmp(&self, other: &Self) -> Ordering {
37        (self.height, self.block_time).cmp(&(other.height, other.block_time))
38    }
39}

47    impl From<MemoryWalletBlock> for proto::WalletBlock {
48        fn from(block: MemoryWalletBlock) -> Self {
49            Self {
50                height: block.height.into(),
51                hash: block.hash.0.to_vec(),
52                block_time: block.block_time,
53                transactions: block
54                    ._transactions
55                    .into_iter()
56                    .map(|txid| txid.as_ref().to_vec())
57                    .collect(),
58                memos: block
59                    ._memos
60                    .into_iter()
61                    .map(|(note_id, memo)| proto::Memo {
62                        note_id: Some(note_id.into()),
63                        memo: memo.as_array().to_vec(),
64                    })
65                    .collect(),
66                sapling_commitment_tree_size: block.sapling_commitment_tree_size,
67                sapling_output_count: block.sapling_output_count,
68                #[cfg(feature = "orchard")]
69                orchard_commitment_tree_size: block.orchard_commitment_tree_size,
70                #[cfg(not(feature = "orchard"))]
71                orchard_commitment_tree_size: None,
72                #[cfg(feature = "orchard")]
73                orchard_action_count: block.orchard_action_count,
74                #[cfg(not(feature = "orchard"))]
75                orchard_action_count: None,
76            }
77        }
78    }

80    impl TryFrom<proto::WalletBlock> for MemoryWalletBlock {
81        type Error = crate::Error;
82        fn try_from(block: proto::WalletBlock) -> Result<Self> {
83            Ok(Self {
84                height: block.height.into(),
85                hash: BlockHash(block.hash.try_into()?),
86                block_time: block.block_time,
87                _transactions: block
88                    .transactions
89                    .into_iter()
90                    .map(|txid| Ok(TxId::from_bytes(txid.try_into()?)))
91                    .collect::<Result<_>>()?,
92                _memos: block
93                    .memos
94                    .into_iter()
95                    .map(|memo| {
96                        let note_id = read_optional!(memo, note_id)?;
97                        Ok((
98                            NoteId::new(
99                                read_optional!(note_id.clone(), tx_id)?.try_into()?,
100                                match note_id.pool() {
101                                    proto::PoolType::ShieldedSapling => {
102                                        zcash_protocol::ShieldedProtocol::Sapling
103                                    }
104                                    #[cfg(feature = "orchard")]
105                                    proto::PoolType::ShieldedOrchard => {
106                                        zcash_protocol::ShieldedProtocol::Orchard
107                                    }
108                                    _ => unreachable!(),
109                                },
110                                note_id.output_index as u16,
111                            ),
112                            MemoBytes::from_bytes(&memo.memo)?,
113                        ))
114                    })
115                    .collect::<Result<_>>()?,
116                sapling_commitment_tree_size: block.sapling_commitment_tree_size,
117                sapling_output_count: block.sapling_output_count,
118                #[cfg(feature = "orchard")]
119                orchard_commitment_tree_size: block.orchard_commitment_tree_size,
120                #[cfg(feature = "orchard")]
121                orchard_action_count: block.orchard_action_count,
122            })
123        }
124    }

135                height: 1.into(),

137                block_time: 2,

144                #[cfg(feature = "orchard")]
145                orchard_action_count: Some(6),
146            };
147
148            let proto: proto::WalletBlock = block.clone().into();
149            let recovered: MemoryWalletBlock = proto.clone().try_into().unwrap();
