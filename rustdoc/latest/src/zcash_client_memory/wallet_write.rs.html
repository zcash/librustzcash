59    fn create_account(
60        &mut self,
61        account_name: &str,
62        seed: &SecretVec<u8>,
63        birthday: &AccountBirthday,
64        key_source: Option<&str>,
65    ) -> Result<(Self::AccountId, UnifiedSpendingKey), Self::Error> {
66        if cfg!(not(test)) {
67            unimplemented!(
68                "Memwallet does not support adding accounts from seed phrases. 
69    Instead derive the ufvk in the calling code and import it using `import_account_ufvk`"
70            )
71        } else {
72            let seed_fingerprint = SeedFingerprint::from_seed(seed.expose_secret())
73                .ok_or(Self::Error::InvalidSeedLength)?;
74            let account_index = self
75                .max_zip32_account_index(&seed_fingerprint)?
76                .map(|a| a.next().ok_or(Self::Error::AccountOutOfRange))
77                .transpose()?
78                .unwrap_or(zip32::AccountId::ZERO);
79
80            let usk =
81                UnifiedSpendingKey::from_seed(&self.params, seed.expose_secret(), account_index)?;
82            let ufvk = usk.to_unified_full_viewing_key();
83
84            let (id, _account) = self.add_account(
85                account_name,
86                AccountSource::Derived {
87                    derivation: Zip32Derivation::new(
88                        seed_fingerprint,
89                        account_index,
90                        #[cfg(feature = "zcashd-compat")]
91                        None,
92                    ),
93                    key_source: key_source.map(|s| s.to_string()),
94                },
95                ufvk,
96                birthday.clone(),
97            )?;
98
99            Ok((id, usk))
100        }
101    }

116    fn update_chain_tip(&mut self, tip_height: BlockHeight) -> Result<(), Self::Error> {
117        tracing::debug!("update_chain_tip");
118        // If the caller provided a chain tip that is before Sapling activation, do nothing.
119        let sapling_activation = match self.params.activation_height(NetworkUpgrade::Sapling) {
120            Some(h) if h <= tip_height => h,
121            _ => return Ok(()),
122        };
123
124        let max_scanned = self.block_height_extrema().map(|range| *range.end());
125        let wallet_birthday = self.get_wallet_birthday()?;
126
127        // If the chain tip is below the prior max scanned height, then the caller has caught
128        // the chain in the middle of a reorg. Do nothing; the caller will continue using the
129        // old scan ranges and either:
130        // - encounter an error trying to fetch the blocks (and thus trigger the same handling
131        //   logic as if this happened with the old linear scanning code); or
132        // - encounter a discontinuity error in `scan_cached_blocks`, at which point they will
133        //   call `WalletDb::truncate_to_height` as part of their reorg handling which will
134        //   resolve the problem.
135        //
136        // We don't check the shard height, as normal usage would have the caller update the
137        // shard state prior to this call, so it is possible and expected to be in a situation
138        // where we should update the tip-related scan ranges but not the shard-related ones.
139        match max_scanned {
140            Some(h) if tip_height < h => return Ok(()),
141            _ => (),
142        };
143
144        // `ScanRange` uses an exclusive upper bound.
145        let chain_end = tip_height + 1;
146
147        let sapling_shard_tip = self.sapling_tip_shard_end_height();
148        // TODO: Handle orchard case as well. See zcash_client_sqlite scanning.rs update_chain_tip
149        let min_shard_tip = sapling_shard_tip;
150
151        // Create a scanning range for the fragment of the last shard leading up to new tip.
152        // We set a lower bound at the wallet birthday (if known), because account creation
153        // requires specifying a tree frontier that ensures we don't need tree information
154        // prior to the birthday.
155        let tip_shard_entry = min_shard_tip.filter(|h| h < &chain_end).map(|h| {
156            let min_to_scan = wallet_birthday.filter(|b| b > &h).unwrap_or(h);
157            ScanRange::from_parts(min_to_scan..chain_end, ScanPriority::ChainTip)
158        });
159
160        // Create scan ranges to either validate potentially invalid blocks at the wallet's
161        // view of the chain tip, or connect the prior tip to the new tip.
162        let tip_entry = max_scanned.map_or_else(
163            || {
164                // No blocks have been scanned, so we need to anchor the start of the new scan
165                // range to something else.
166                wallet_birthday.map_or_else(
167                    // We don't have a wallet birthday, which means we have no accounts yet.
168                    // We can therefore ignore all blocks up to the chain tip.
169                    || ScanRange::from_parts(sapling_activation..chain_end, ScanPriority::Ignored),
170                    // We have a wallet birthday, so mark all blocks between that and the
171                    // chain tip as `Historic` (performing wallet recovery).
172                    |wallet_birthday| {
173                        ScanRange::from_parts(wallet_birthday..chain_end, ScanPriority::Historic)
174                    },
175                )
176            },
177            |max_scanned| {
178                // The scan range starts at the block after the max scanned height. Since
179                // `scan_cached_blocks` retrieves the metadata for the block being connected to
180                // (if it exists), the connectivity of the scan range to the max scanned block
181                // will always be checked if relevant.
182                let min_unscanned = max_scanned + 1;
183
184                // If we don't have shard metadata, this means we're doing linear scanning, so
185                // create a scan range from the prior tip to the current tip with `Historic`
186                // priority.
187                if tip_shard_entry.is_none() {
188                    ScanRange::from_parts(min_unscanned..chain_end, ScanPriority::Historic)
189                } else {
190                    // Determine the height to which we expect new blocks retrieved from the
191                    // block source to be stable and not subject to being reorg'ed.
192                    let stable_height = tip_height.saturating_sub(PRUNING_DEPTH);
193
194                    // If the wallet's max scanned height is above the stable height,
195                    // prioritize the range between it and the new tip as `ChainTip`.
196                    if max_scanned > stable_height {
197                        // We are in the steady-state case, where a wallet is close to the
198                        // chain tip and just needs to catch up.
199                        //
200                        // This overlaps the `tip_shard_entry` range and so will be coalesced
201                        // with it.
202                        ScanRange::from_parts(min_unscanned..chain_end, ScanPriority::ChainTip)
203                    } else {
204                        // In this case, the max scanned height is considered stable relative
205                        // to the chain tip. However, it may be stable or unstable relative to
206                        // the prior chain tip, which we could determine by looking up the
207                        // prior chain tip height from the scan queue. For simplicity we merge
208                        // these two cases together, and proceed as though the max scanned
209                        // block is unstable relative to the prior chain tip.
210                        //
211                        // To confirm its stability, prioritize the `VERIFY_LOOKAHEAD` blocks
212                        // above the max scanned height as `Verify`:
213                        //
214                        // - We use `Verify` to ensure that a connectivity check is performed,
215                        //   along with any required rewinds, before any `ChainTip` ranges
216                        //   (from this or any prior `update_chain_tip` call) are scanned.
217                        //
218                        // - We prioritize `VERIFY_LOOKAHEAD` blocks because this is expected
219                        //   to be 12.5 minutes, within which it is reasonable for a user to
220                        //   have potentially received a transaction (if they opened their
221                        //   wallet to provide an address to someone else, or spent their own
222                        //   funds creating a change output), without necessarily having left
223                        //   their wallet open long enough for the transaction to be mined and
224                        //   the corresponding block to be scanned.
225                        //
226                        // - We limit the range to at most the stable region, to prevent any
227                        //   `Verify` ranges from being susceptible to reorgs, and potentially
228                        //   interfering with subsequent `Verify` ranges defined by future
229                        //   calls to `update_chain_tip`. Any gap between `stable_height` and
230                        //   `shard_start_height` will be filled by the scan range merging
231                        //   logic with a `Historic` range.
232                        //
233                        // If `max_scanned == stable_height` then this is a zero-length range.
234                        // In this case, any non-empty `(stable_height+1)..shard_start_height`
235                        // will be marked `Historic`, minimising the prioritised blocks at the
236                        // chain tip and allowing for other ranges (for example, `FoundNote`)
237                        // to take priority.
238                        ScanRange::from_parts(
239                            min_unscanned
240                                ..std::cmp::min(
241                                    stable_height + 1,
242                                    min_unscanned + VERIFY_LOOKAHEAD,
243                                ),
244                            ScanPriority::Verify,
245                        )
246                    }
247                }
248            },
249        );
250        if let Some(entry) = &tip_shard_entry {
251            tracing::debug!("{} will update latest shard", entry);
252        }
253        tracing::debug!("{} will connect prior scanned state to new tip", tip_entry);
254
255        let query_range = match tip_shard_entry.as_ref() {
256            Some(se) => Range {
257                start: std::cmp::min(se.block_range().start, tip_entry.block_range().start),
258                end: std::cmp::max(se.block_range().end, tip_entry.block_range().end),
259            },
260            None => tip_entry.block_range().clone(),
261        };
262
263        self.scan_queue.replace_queue_entries(
264            &query_range,
265            tip_shard_entry.into_iter().chain(Some(tip_entry)),
266            false,
267        )?;
268        Ok(())
269    }

271    /// Adds a sequence of blocks to the data store.
272    ///
273    /// Assumes blocks will be here in order.
274    fn put_blocks(
275        &mut self,
276        from_state: &ChainState,
277        blocks: Vec<ScannedBlock<Self::AccountId>>,
278    ) -> Result<(), Self::Error> {
279        tracing::debug!("put_blocks");
280        let mut last_scanned_height = None;
281        struct BlockPositions {
282            height: BlockHeight,
283            sapling_start_position: Position,
284            #[cfg(feature = "orchard")]
285            orchard_start_position: Position,
286        }
287        let start_positions = blocks.first().map(|block| BlockPositions {
288            height: block.height(),
289            sapling_start_position: Position::from(
290                u64::from(block.sapling().final_tree_size())
291                    - u64::try_from(block.sapling().commitments().len()).unwrap(),
292            ),
293            #[cfg(feature = "orchard")]
294            orchard_start_position: Position::from(
295                u64::from(block.orchard().final_tree_size())
296                    - u64::try_from(block.orchard().commitments().len()).unwrap(),
297            ),
298        });
299
300        let mut sapling_commitments = vec![];
301        #[cfg(feature = "orchard")]
302        let mut orchard_commitments = vec![];
303        let mut note_positions = vec![];
304        for block in blocks.into_iter() {
305            let mut transactions = HashMap::new();
306            let mut memos = HashMap::new();
307            if last_scanned_height
308                .iter()
309                .any(|prev| block.height() != *prev + 1)
310            {
311                return Err(Error::NonSequentialBlocks);
312            }
313
314            for transaction in block.transactions().iter() {
315                let txid = transaction.txid();
316
317                // Mark the Sapling nullifiers of the spent notes as spent in the `sapling_spends` map.
318                for spend in transaction.sapling_spends() {
319                    println!(
320                        "marking note {:?} as spent in transaction {:?}",
321                        spend.nf(),
322                        txid
323                    );
324                    self.mark_sapling_note_spent(*spend.nf(), txid)?;
325                }
326
327                // Mark the Orchard nullifiers of the spent notes as spent in the `orchard_spends` map.
328                #[cfg(feature = "orchard")]
329                for spend in transaction.orchard_spends() {
330                    self.mark_orchard_note_spent(*spend.nf(), txid)?;
331                }
332
333                for output in transaction.sapling_outputs() {
334                    // Insert the memo into the `memos` map.
335                    let note_id = NoteId::new(
336                        txid,
337                        Sapling,
338                        u16::try_from(output.index())
339                            .expect("output indices are representable as u16"),
340                    );
341                    if let Ok(Some(memo)) = self.get_memo(note_id) {
342                        memos.insert(note_id, memo.encode());
343                    }
344                    // Check whether this note was spent in a later block range that
345                    // we previously scanned.
346                    let spent_in = output
347                        .nf()
348                        .and_then(|nf| self.nullifiers.get(&Nullifier::Sapling(*nf)))
349                        .and_then(|(height, tx_idx)| self.tx_locator.get(*height, *tx_idx))
350                        .copied();
351
352                    self.insert_received_sapling_note(note_id, output, spent_in);
353                }
354
355                #[cfg(feature = "orchard")]
356                for output in transaction.orchard_outputs().iter() {
357                    // Insert the memo into the `memos` map.
358                    let note_id = NoteId::new(
359                        txid,
360                        Orchard,
361                        u16::try_from(output.index())
362                            .expect("output indices are representable as u16"),
363                    );
364                    if let Ok(Some(memo)) = self.get_memo(note_id) {
365                        memos.insert(note_id, memo.encode());
366                    }
367                    // Check whether this note was spent in a later block range that
368                    // we previously scanned.
369                    let spent_in = output
370                        .nf()
371                        .and_then(|nf| self.nullifiers.get(&Nullifier::Orchard(*nf)))
372                        .and_then(|(height, tx_idx)| self.tx_locator.get(*height, *tx_idx))
373                        .copied();
374
375                    self.insert_received_orchard_note(note_id, output, spent_in)
376                }
377
378                transactions.insert(txid, transaction.clone());
379            }
380
381            // Insert the new nullifiers from this block into the nullifier map
382            self.insert_sapling_nullifier_map(block.height(), block.sapling().nullifier_map())?;
383            #[cfg(feature = "orchard")]
384            self.insert_orchard_nullifier_map(block.height(), block.orchard().nullifier_map())?;
385            note_positions.extend(block.transactions().iter().flat_map(|wtx| {
386                let iter = wtx.sapling_outputs().iter().map(|out| {
387                    (
388                        ShieldedProtocol::Sapling,
389                        out.note_commitment_tree_position(),
390                    )
391                });
392                #[cfg(feature = "orchard")]
393                let iter = iter.chain(wtx.orchard_outputs().iter().map(|out| {
394                    (
395                        ShieldedProtocol::Orchard,
396                        out.note_commitment_tree_position(),
397                    )
398                }));
399
400                iter
401            }));
402
403            let memory_block = MemoryWalletBlock {
404                height: block.height(),
405                hash: block.block_hash(),
406                block_time: block.block_time(),
407                _transactions: transactions.keys().cloned().collect(),
408                _memos: memos,
409                sapling_commitment_tree_size: Some(block.sapling().final_tree_size()),
410                sapling_output_count: Some(block.sapling().commitments().len().try_into().unwrap()),
411                #[cfg(feature = "orchard")]
412                orchard_commitment_tree_size: Some(block.orchard().final_tree_size()),
413                #[cfg(feature = "orchard")]
414                orchard_action_count: Some(block.orchard().commitments().len().try_into().unwrap()),
415            };
416
417            // Insert transaction metadata into the transaction table
418            transactions
419                .into_iter()
420                .for_each(|(_id, tx)| self.tx_table.put_tx_meta(tx, block.height()));
421
422            // Insert the block into the block map
423            self.blocks.insert(block.height(), memory_block);
424            last_scanned_height = Some(block.height());
425
426            let block_commitments = block.into_commitments();
427            sapling_commitments.extend(block_commitments.sapling.into_iter().map(Some));
428            #[cfg(feature = "orchard")]
429            orchard_commitments.extend(block_commitments.orchard.into_iter().map(Some));
430        }
431
432        if let Some((start_positions, last_scanned_height)) =
433            start_positions.zip(last_scanned_height)
434        {
435            // Create subtrees from the note commitments in parallel.
436            const CHUNK_SIZE: usize = 1024;
437            let sapling_subtrees = sapling_commitments
438                .par_chunks_mut(CHUNK_SIZE)
439                .enumerate()
440                .filter_map(|(i, chunk)| {
441                    let start = start_positions.sapling_start_position + (i * CHUNK_SIZE) as u64;
442                    let end = start + chunk.len() as u64;
443
444                    shardtree::LocatedTree::from_iter(
445                        start..end,
446                        SAPLING_SHARD_HEIGHT.into(),
447                        chunk.iter_mut().map(|n| n.take().expect("always Some")),
448                    )
449                })
450                .map(|res| (res.subtree, res.checkpoints))
451                .collect::<Vec<_>>();
452
453            #[cfg(feature = "orchard")]
454            let orchard_subtrees = orchard_commitments
455                .par_chunks_mut(CHUNK_SIZE)
456                .enumerate()
457                .filter_map(|(i, chunk)| {
458                    let start = start_positions.orchard_start_position + (i * CHUNK_SIZE) as u64;
459                    let end = start + chunk.len() as u64;
460
461                    shardtree::LocatedTree::from_iter(
462                        start..end,
463                        ORCHARD_SHARD_HEIGHT.into(),
464                        chunk.iter_mut().map(|n| n.take().expect("always Some")),
465                    )
466                })
467                .map(|res| (res.subtree, res.checkpoints))
468                .collect::<Vec<_>>();
469
470            // Collect the complete set of Sapling checkpoints
471            #[cfg(feature = "orchard")]
472            let sapling_checkpoint_positions: BTreeMap<BlockHeight, Position> = sapling_subtrees
473                .iter()
474                .flat_map(|(_, checkpoints)| checkpoints.iter())
475                .map(|(k, v)| (*k, *v))
476                .collect();
477
478            #[cfg(feature = "orchard")]
479            let orchard_checkpoint_positions: BTreeMap<BlockHeight, Position> = orchard_subtrees
480                .iter()
481                .flat_map(|(_, checkpoints)| checkpoints.iter())
482                .map(|(k, v)| (*k, *v))
483                .collect();
484
485            #[cfg(feature = "orchard")]
486            let (missing_sapling_checkpoints, missing_orchard_checkpoints) = (
487                ensure_checkpoints(
488                    orchard_checkpoint_positions.keys(),
489                    &sapling_checkpoint_positions,
490                    from_state.final_sapling_tree(),
491                ),
492                ensure_checkpoints(
493                    sapling_checkpoint_positions.keys(),
494                    &orchard_checkpoint_positions,
495                    from_state.final_orchard_tree(),
496                ),
497            );
498
499            // Update the Sapling note commitment tree with all newly read note commitments
500            {
501                let mut sapling_subtrees_iter = sapling_subtrees.into_iter();
502                self.with_sapling_tree_mut::<_, _, Self::Error>(|sapling_tree| {
503                    sapling_tree.insert_frontier(
504                        from_state.final_sapling_tree().clone(),
505                        Retention::Checkpoint {
506                            id: from_state.block_height(),
507                            marking: Marking::Reference,
508                        },
509                    )?;
510
511                    for (tree, checkpoints) in &mut sapling_subtrees_iter {
512                        sapling_tree.insert_tree(tree, checkpoints)?;
513                    }
514
515                    // Ensure we have a Sapling checkpoint for each checkpointed Orchard block height.
516                    // We skip all checkpoints below the minimum retained checkpoint in the
517                    // Sapling tree, because branches below this height may be pruned.
518                    #[cfg(feature = "orchard")]
519                    {
520                        let min_checkpoint_height = sapling_tree
521                            .store()
522                            .min_checkpoint_id()
523                            .map_err(ShardTreeError::Storage)?
524                            .expect("At least one checkpoint was inserted (by insert_frontier)");
525
526                        for (height, checkpoint) in &missing_sapling_checkpoints {
527                            if *height > min_checkpoint_height {
528                                sapling_tree
529                                    .store_mut()
530                                    .add_checkpoint(*height, checkpoint.clone())
531                                    .map_err(ShardTreeError::Storage)?;
532                            }
533                        }
534                    }
535
536                    Ok(())
537                })?;
538            }
539
540            // Update the Orchard note commitment tree with all newly read note commitments
541            #[cfg(feature = "orchard")]
542            {
543                let mut orchard_subtrees = orchard_subtrees.into_iter();
544                self.with_orchard_tree_mut::<_, _, Self::Error>(|orchard_tree| {
545                    orchard_tree.insert_frontier(
546                        from_state.final_orchard_tree().clone(),
547                        Retention::Checkpoint {
548                            id: from_state.block_height(),
549                            marking: Marking::Reference,
550                        },
551                    )?;
552
553                    for (tree, checkpoints) in &mut orchard_subtrees {
554                        orchard_tree.insert_tree(tree, checkpoints)?;
555                    }
556
557                    // Ensure we have an Orchard checkpoint for each checkpointed Sapling block height.
558                    // We skip all checkpoints below the minimum retained checkpoint in the
559                    // Orchard tree, because branches below this height may be pruned.
560                    {
561                        let min_checkpoint_height = orchard_tree
562                            .store()
563                            .min_checkpoint_id()
564                            .map_err(ShardTreeError::Storage)?
565                            .expect("At least one checkpoint was inserted (by insert_frontier)");
566
567                        for (height, checkpoint) in &missing_orchard_checkpoints {
568                            if *height > min_checkpoint_height {
569                                orchard_tree
570                                    .store_mut()
571                                    .add_checkpoint(*height, checkpoint.clone())
572                                    .map_err(ShardTreeError::Storage)?;
573                            }
574                        }
575                    }
576                    Ok(())
577                })?;
578            }
579
580            self.scan_complete(
581                Range {
582                    start: start_positions.height,
583                    end: last_scanned_height + 1,
584                },
585                &note_positions,
586            )?;
587        }
588
589        Ok(())
590    }

592    /// Adds a transparent UTXO received by the wallet to the data store.
593    fn put_received_transparent_utxo(
594        &mut self,
595        _output: &WalletTransparentOutput,
596    ) -> Result<Self::UtxoRef, Self::Error> {
597        tracing::debug!("put_received_transparent_utxo");
598        #[cfg(feature = "transparent-inputs")]
599        {
600            let address = _output.recipient_address();
601            if let Some(account_id) = self.find_account_for_transparent_address(address)? {
602                self.put_transparent_output(_output, &account_id, false)
603            } else {
604                Err(Error::AddressNotRecognized(*address))
605            }
606        }
607        #[cfg(not(feature = "transparent-inputs"))]
608        panic!(
609            "The wallet must be compiled with the transparent-inputs feature to use this method."
610        );
611    }

614        &mut self,
615        d_tx: DecryptedTransaction<Self::AccountId>,
616    ) -> Result<(), Self::Error> {
617        tracing::debug!("store_decrypted_tx");
618        self.tx_table.put_tx_data(d_tx.tx(), None, None);
619        if let Some(height) = d_tx.mined_height() {
620            self.set_transaction_status(d_tx.tx().txid(), TransactionStatus::Mined(height))?
621        }
622
623        let funding_accounts = self.get_funding_accounts(d_tx.tx())?;
624        // TODO(#1305): Correctly track accounts that fund each transaction output.
625        let funding_account = funding_accounts.iter().next().copied();
626        if funding_accounts.len() > 1 {
627            tracing::warn!(
628                "More than one wallet account detected as funding transaction {:?}, selecting {:?}",
629                d_tx.tx().txid(),
630                funding_account.unwrap()
631            )
632        }
633
634        // A flag used to determine whether it is necessary to query for transactions that
635        // provided transparent inputs to this transaction, in order to be able to correctly
636        // recover transparent transaction history.
637        #[cfg(feature = "transparent-inputs")]
638        let mut tx_has_wallet_outputs = false;
639
640        for output in d_tx.sapling_outputs() {
641            #[cfg(feature = "transparent-inputs")]
642            {
643                tx_has_wallet_outputs = true;
644            }
645
646            match output.transfer_type() {
647                TransferType::Outgoing => {
648                    let recipient = {
649                        let receiver = Receiver::Sapling(output.note().recipient());
650                        let wallet_address = self
651                            .accounts
652                            .get(*output.account())
653                            .map(|acc| {
654                                acc.select_receiving_address(self.params.network_type(), &receiver)
655                            })
656                            .transpose()?
657                            .flatten()
658                            .unwrap_or_else(|| {
659                                receiver.to_zcash_address(self.params.network_type())
660                            });
661
662                        Recipient::External {
663                            recipient_address: wallet_address,
664                            output_pool: PoolType::SAPLING,
665                        }
666                    };
667
668                    let sent_tx_output = SentTransactionOutput::from_parts(
669                        output.index(),
670                        recipient,
671                        output.note_value(),
672                        Some(output.memo().clone()),
673                    );
674                    self.sent_notes.put_sent_output(
675                        d_tx.tx().txid(),
676                        *output.account(),
677                        &sent_tx_output,
678                    );
679                }
680                TransferType::WalletInternal => {
681                    let recipient = Recipient::InternalAccount {
682                        receiving_account: *output.account(),
683                        external_address: None,
684                        note: Box::new(Note::Sapling(output.note().clone())),
685                    };
686                    let sent_tx_output = SentTransactionOutput::from_parts(
687                        output.index(),
688                        recipient,
689                        output.note_value(),
690                        Some(output.memo().clone()),
691                    );
692
693                    self.received_notes
694                        .insert_received_note(ReceivedNote::from_sent_tx_output(
695                            d_tx.tx().txid(),
696                            &sent_tx_output,
697                        )?);
698
699                    self.sent_notes.put_sent_output(
700                        d_tx.tx().txid(),
701                        *output.account(),
702                        &sent_tx_output,
703                    );
704                }
705                TransferType::Incoming => {
706                    todo!("store decrypted tx sapling incoming")
707                }
708            }
709        }
710
711        #[cfg(feature = "orchard")]
712        for output in d_tx.orchard_outputs() {
713            #[cfg(feature = "transparent-inputs")]
714            {
715                tx_has_wallet_outputs = true;
716            }
717            match output.transfer_type() {
718                TransferType::Outgoing => {
719                    let recipient = {
720                        let receiver = Receiver::Orchard(output.note().recipient());
721                        let wallet_address = self
722                            .accounts
723                            .get(*output.account())
724                            .map(|acc| {
725                                acc.select_receiving_address(self.params.network_type(), &receiver)
726                            })
727                            .transpose()?
728                            .flatten()
729                            .unwrap_or_else(|| {
730                                receiver.to_zcash_address(self.params.network_type())
731                            });
732
733                        Recipient::External {
734                            recipient_address: wallet_address,
735                            output_pool: PoolType::ORCHARD,
736                        }
737                    };
738
739                    let sent_tx_output = SentTransactionOutput::from_parts(
740                        output.index(),
741                        recipient,
742                        output.note_value(),
743                        Some(output.memo().clone()),
744                    );
745                    self.sent_notes.put_sent_output(
746                        d_tx.tx().txid(),
747                        *output.account(),
748                        &sent_tx_output,
749                    );
750                }
751                TransferType::WalletInternal => {
752                    let recipient = Recipient::InternalAccount {
753                        receiving_account: *output.account(),
754                        external_address: None,
755                        note: Box::new(Note::Orchard(*output.note())),
756                    };
757                    let sent_tx_output = SentTransactionOutput::from_parts(
758                        output.index(),
759                        recipient,
760                        output.note_value(),
761                        Some(output.memo().clone()),
762                    );
763
764                    self.received_notes
765                        .insert_received_note(ReceivedNote::from_sent_tx_output(
766                            d_tx.tx().txid(),
767                            &sent_tx_output,
768                        )?);
769
770                    self.sent_notes.put_sent_output(
771                        d_tx.tx().txid(),
772                        *output.account(),
773                        &sent_tx_output,
774                    );
775                }
776                TransferType::Incoming => {
777                    todo!("store decrypted tx orchard incoming")
778                }
779            }
780        }
781
782        // If any of the utxos spent in the transaction are ours, mark them as spent.
783        #[cfg(feature = "transparent-inputs")]
784        for txin in d_tx
785            .tx()
786            .transparent_bundle()
787            .iter()
788            .flat_map(|b| b.vin.iter())
789        {
790            self.mark_transparent_output_spent(&d_tx.tx().txid(), txin.prevout())?;
791        }
792
793        // This `if` is just an optimization for cases where we would do nothing in the loop.
794        if funding_account.is_some() || cfg!(feature = "transparent-inputs") {
795            for (output_index, txout) in d_tx
796                .tx()
797                .transparent_bundle()
798                .iter()
799                .flat_map(|b| b.vout.iter())
800                .enumerate()
801            {
802                if let Some(address) = txout.recipient_address() {
803                    tracing::debug!(
804                        "{:?} output {} has recipient {}",
805                        d_tx.tx().txid(),
806                        output_index,
807                        address.encode(self.params())
808                    );
809
810                    // The transaction is not necessarily mined yet, but we want to record
811                    // that an output to the address was seen in this tx anyway. This will
812                    // advance the gap regardless of whether it is mined, but an output in
813                    // an unmined transaction won't advance the range of safe indices.
814                    #[cfg(feature = "transparent-inputs")]
815                    self.accounts
816                        .mark_ephemeral_address_as_seen(&address, d_tx.tx().txid())?;
817
818                    // If the output belongs to the wallet, add it to `transparent_received_outputs`.
819                    #[cfg(feature = "transparent-inputs")]
820                    if let Some(account_id) = self
821                        .accounts
822                        .find_account_for_transparent_address(&address)?
823                    {
824                        tracing::debug!(
825                            "{:?} output {} belongs to account {:?}",
826                            d_tx.tx().txid(),
827                            output_index,
828                            account_id
829                        );
830                        let wallet_transparent_output = WalletTransparentOutput::from_parts(
831                            OutPoint::new(
832                                d_tx.tx().txid().into(),
833                                u32::try_from(output_index).unwrap(),
834                            ),
835                            txout.clone(),
836                            d_tx.mined_height(),
837                        )
838                        .unwrap();
839                        self.put_transparent_output(
840                            &wallet_transparent_output,
841                            &account_id,
842                            false,
843                        )?;
844
845                        // Since the wallet created the transparent output, we need to ensure
846                        // that any transparent inputs belonging to the wallet will be
847                        // discovered.
848                        tx_has_wallet_outputs = true;
849                    } else {
850                        tracing::debug!(
851                            "Address {} is not recognized as belonging to any of our accounts.",
852                            address.encode(self.params())
853                        );
854                    }
855
856                    // If a transaction we observe contains spends from our wallet, we will
857                    // store its transparent outputs in the same way they would be stored by
858                    // create_spend_to_address.
859                    if let Some(account_id) = funding_account {
860                        let receiver = Receiver::Transparent(address);
861
862                        #[cfg(feature = "transparent-inputs")]
863                        let recipient_addr = self
864                            .accounts
865                            .get(account_id)
866                            .map(|acc| {
867                                acc.select_receiving_address(self.params.network_type(), &receiver)
868                            })
869                            .transpose()?
870                            .flatten()
871                            .unwrap_or_else(|| {
872                                receiver.to_zcash_address(self.params.network_type())
873                            });
874
875                        #[cfg(not(feature = "transparent-inputs"))]
876                        let recipient_addr = receiver.to_zcash_address(self.params.network_type());
877
878                        let recipient = Recipient::External {
879                            recipient_address: recipient_addr,
880                            output_pool: PoolType::TRANSPARENT,
881                        };
882
883                        let sent_tx_output = SentTransactionOutput::from_parts(
884                            output_index,
885                            recipient,
886                            txout.value(),
887                            None,
888                        );
889                        self.sent_notes.put_sent_output(
890                            d_tx.tx().txid(),
891                            account_id,
892                            &sent_tx_output,
893                        );
894                        // Even though we know the funding account, we don't know that we have
895                        // information for all of the transparent inputs to the transaction.
896                        #[cfg(feature = "transparent-inputs")]
897                        {
898                            tx_has_wallet_outputs = true;
899                        }
900                    }
901                } else {
902                    tracing::warn!(
903                        "Unable to determine recipient address for tx {:?} output {}",
904                        d_tx.tx().txid(),
905                        output_index
906                    );
907                }
908            }
909        }
910
911        // If the transaction has outputs that belong to the wallet as well as transparent
912        // inputs, we may need to download the transactions corresponding to the transparent
913        // prevout references to determine whether the transaction was created (at least in
914        // part) by this wallet.
915        #[cfg(feature = "transparent-inputs")]
916        if tx_has_wallet_outputs && d_tx.tx().transparent_bundle().is_some() {
917            // queue the transparent inputs for enhancement
918            self.transaction_data_request_queue
919                .queue_status_retrieval(&d_tx.tx().txid());
920        }
921
922        #[cfg(feature = "transparent-inputs")]
923        {
924            let detectable_via_scanning = d_tx.tx().sapling_bundle().is_some();
925            #[cfg(feature = "orchard")]
926            let detectable_via_scanning =
927                detectable_via_scanning | d_tx.tx().orchard_bundle().is_some();
928
929            if d_tx.mined_height().is_none() && !detectable_via_scanning {
930                self.transaction_data_request_queue
931                    .queue_status_retrieval(&d_tx.tx().txid());
932            }
933        }
934        Ok(())
935    }

937    /// Truncates the database to the given height.
938    ///
939    /// If the requested height is greater than or equal to the height of the last scanned
940    /// block, this function does nothing.
941    ///
942    /// This should only be executed inside a transactional context.
943    fn truncate_to_height(&mut self, max_height: BlockHeight) -> Result<BlockHeight, Self::Error> {
944        let truncation_height = {
945            // This is the intersection of all the checkpoint heights from the sapling and orchard tree.
946            let mut checkpoint_heights = BTreeSet::new();
947            self.sapling_tree.store().for_each_checkpoint(
948                self.sapling_tree.store().checkpoint_count()?,
949                |height, _| {
950                    checkpoint_heights.insert(u32::from(*height));
951                    Ok(())
952                },
953            )?;
954            #[cfg(feature = "orchard")]
955            {
956                let mut orchard_checkpoint_heights = BTreeSet::new();
957                self.orchard_tree.store().for_each_checkpoint(
958                    self.orchard_tree.store().checkpoint_count()?,
959                    |height, _| {
960                        orchard_checkpoint_heights.insert(u32::from(*height));
961                        Ok(())
962                    },
963                )?;
964
965                checkpoint_heights = checkpoint_heights
966                    .intersection(&orchard_checkpoint_heights)
967                    .copied()
968                    .collect();
969            }
970            // All the checkpoints that are greater than the truncation height
971            let over = checkpoint_heights.split_off(&(u32::from(max_height + 1)));
972            if let Some(height) = checkpoint_heights.last().copied() {
973                Ok(BlockHeight::from(height))
974            } else {
975                // If there are no checkpoints that are less than or equal to the truncation height
976                // then we can't truncate the tree.
977                Err(Error::RequestedRewindInvalid(
978                    over.first().copied().map(Into::into),
979                    max_height,
980                ))
981            }
982        }?;
983
984        // Recall where we synced up to previously.
985        let last_scanned_height = self.blocks.keys().max().copied().unwrap_or_else(|| {
986            self.params
987                .activation_height(NetworkUpgrade::Sapling)
988                .expect("Sapling activation height must be available.")
989                - 1
990        });
991
992        // Delete from the scanning queue any range with a start height greater than the
993        // truncation height, and then truncate any remaining range by setting the end
994        // equal to the truncation height + 1. This sets our view of the chain tip back
995        // to the retained height.
996        self.scan_queue
997            .delete_starts_greater_than_equal_to(truncation_height + 1);
998        self.scan_queue.truncate_ends_to(truncation_height + 1);
999
1000        // Mark transparent utxos as un-mined. Since the TXO is now not mined, it would ideally be
1001        // considered to have been returned to the mempool; it _might_ be spendable in this state, but
1002        // we must also set its max_observed_unspent_height field to NULL because the transaction may
1003        // be rendered entirely invalid by a reorg that alters anchor(s) used in constructing shielded
1004        // spends in the transaction.
1005        self.transparent_received_outputs
1006            .iter_mut()
1007            .for_each(|(_, txo)| {
1008                if let Some(mined_height) = self
1009                    .tx_table
1010                    .get(&txo.transaction_id)
1011                    .and_then(|tx| tx.mined_height())
1012                {
1013                    if mined_height <= truncation_height {
1014                        txo.max_observed_unspent_height = Some(truncation_height)
1015                    } else {
1016                        txo.max_observed_unspent_height = None
1017                    }
1018                }
1019            });
1020
1021        // Un-mine transactions. This must be done outside of the last_scanned_height check because
1022        // transaction entries may be created as a consequence of receiving transparent TXOs.
1023        self.tx_table
1024            .unmine_transactions_greater_than(truncation_height);
1025
1026        // If we're removing scanned blocks, we need to truncate the note commitment tree and remove
1027        // affected block records from the database.
1028        if truncation_height < last_scanned_height {
1029            self.with_sapling_tree_mut(|tree| {
1030                tree.truncate_to_checkpoint(&truncation_height).map(|_| ())
1031            })?;
1032            #[cfg(feature = "orchard")]
1033            self.with_orchard_tree_mut(|tree| {
1034                tree.truncate_to_checkpoint(&truncation_height).map(|_| ())
1035            })?;
1036
1037            // Do not delete sent notes; this can contain data that is not recoverable
1038            // from the chain. Wallets must continue to operate correctly in the
1039            // presence of stale sent notes that link to unmined transactions.
1040            // Also, do not delete received notes; they may contain memo data that is
1041            // not recoverable; balance APIs must ensure that un-mined received notes
1042            // do not count towards spendability or transaction balalnce.
1043
1044            // Now that they aren't depended on, delete un-mined blocks.
1045            self.blocks.retain(|height, _| *height <= truncation_height);
1046
1047            // Delete from the nullifier map any entries with a locator referencing a block
1048            // height greater than the truncation height.
1049            // Willem: We don't need to do this I think..
1050        }
1051        Ok(truncation_height)
1052    }

1089    fn store_transactions_to_be_sent(
1090        &mut self,
1091        transactions: &[SentTransaction<Self::AccountId>],
1092    ) -> Result<(), Self::Error> {
1093        tracing::debug!("store_transactions_to_be_sent");
1094        for sent_tx in transactions {
1095            self.tx_table.put_tx_data(
1096                sent_tx.tx(),
1097                Some(sent_tx.fee_amount()),
1098                Some(sent_tx.target_height()),
1099            );
1100            let mut detectable_via_scanning = false;
1101            // Mark sapling notes as spent
1102            if let Some(bundle) = sent_tx.tx().sapling_bundle() {
1103                detectable_via_scanning = true;
1104                for spend in bundle.shielded_spends() {
1105                    self.mark_sapling_note_spent(*spend.nullifier(), sent_tx.tx().txid())?;
1106                }
1107            }
1108            // Mark orchard notes as spent
1109            if let Some(_bundle) = sent_tx.tx().orchard_bundle() {
1110                #[cfg(feature = "orchard")]
1111                {
1112                    detectable_via_scanning = true;
1113                    for action in _bundle.actions() {
1114                        match self.mark_orchard_note_spent(*action.nullifier(), sent_tx.tx().txid())
1115                        {
1116                            Ok(()) => {}
1117                            Err(Error::NoteNotFound) => {
1118                                // This is expected as some of the actions will be new outputs we don't have notes for
1119                                // The ones we do recognize will be marked as spent
1120                            }
1121                            Err(e) => return Err(e),
1122                        }
1123                    }
1124                }
1125
1126                #[cfg(not(feature = "orchard"))]
1127                panic!("Sent a transaction with Orchard Actions without `orchard` enabled?");
1128            }
1129            // Mark transparent UTXOs as spent
1130            #[cfg(feature = "transparent-inputs")]
1131            for utxo_outpoint in sent_tx.utxos_spent() {
1132                self.mark_transparent_output_spent(&sent_tx.tx().txid(), utxo_outpoint)?;
1133            }
1134
1135            for output in sent_tx.outputs() {
1136                self.sent_notes.insert_sent_output(sent_tx, output);
1137
1138                match output.recipient() {
1139                    Recipient::InternalAccount { .. } => {
1140                        self.received_notes.insert_received_note(
1141                            ReceivedNote::from_sent_tx_output(sent_tx.tx().txid(), output)?,
1142                        );
1143                    }
1144                    #[cfg(feature = "transparent-inputs")]
1145                    Recipient::EphemeralTransparent {
1146                        receiving_account,
1147                        ephemeral_address,
1148                        outpoint,
1149                    } => {
1150                        let txo = WalletTransparentOutput::from_parts(
1151                            outpoint.clone(),
1152                            TxOut::new(output.value(), ephemeral_address.script().into()),
1153                            None,
1154                        )
1155                        .unwrap();
1156                        self.put_transparent_output(&txo, receiving_account, true)?;
1157                        if let Some(account) = self.accounts.get_mut(*receiving_account) {
1158                            account.mark_ephemeral_address_as_used(
1159                                ephemeral_address,
1160                                sent_tx.tx().txid(),
1161                            )?
1162                        }
1163                    }
1164                    _ => {}
1165                }
1166            }
1167
1168            // Add the transaction to the set to be queried for transaction status. This is only necessary
1169            // at present for fully transparent transactions, because any transaction with a shielded
1170            // component will be detected via ordinary chain scanning and/or nullifier checking.
1171            if !detectable_via_scanning {
1172                self.transaction_data_request_queue
1173                    .queue_status_retrieval(&sent_tx.tx().txid());
1174            }
1175        }
1176
1177        Ok(())
1178    }

1190    fn reserve_next_n_ephemeral_addresses(
1191        &mut self,
1192        account_id: Self::AccountId,
1193        n: usize,
1194    ) -> Result<Vec<(TransparentAddress, TransparentAddressMetadata)>, Self::Error> {
1195        use zcash_keys::keys::AddressGenerationError;
1196        // TODO: We need to implement first_unsafe_index to make sure we dont violate gap invarient
1197
1198        use transparent::keys::NonHardenedChildIndex;
1199        let first_unsafe = self.first_unsafe_index(account_id)?;
1200        if let Some(account) = self.accounts.get_mut(account_id) {
1201            let first_unreserved = account.first_unreserved_index()?;
1202
1203            let allocation = range_from(first_unreserved, u32::try_from(n).unwrap());
1204
1205            if allocation.len() < n {
1206                return Err(AddressGenerationError::DiversifierSpaceExhausted.into());
1207            }
1208            if allocation.end > first_unsafe {
1209                return Err(Error::ReachedGapLimit(
1210                    account_id,
1211                    core::cmp::max(first_unreserved, first_unsafe),
1212                ));
1213            }
1214            let _reserved = account.reserve_until(allocation.end)?;
1215            self.get_known_ephemeral_addresses(
1216                account_id,
1217                Some(
1218                    NonHardenedChildIndex::from_index(allocation.start).expect("Bad Index")
1219                        ..NonHardenedChildIndex::from_index(allocation.end).expect("Bad Index"),
1220                ),
1221            )
1222        } else {
1223            Err(Self::Error::AccountUnknown(account_id))
1224        }
1225    }

1256fn range_from(i: u32, n: u32) -> Range<u32> {

1269fn ensure_checkpoints<'a, H, I: Iterator<Item = &'a BlockHeight>, const DEPTH: u8>(
1270    // An iterator of checkpoints heights for which we wish to ensure that
1271    // checkpoints exists.
1272    ensure_heights: I,
1273    // The map of checkpoint positions from which we will draw note commitment tree
1274    // position information for the newly created checkpoints.
1275    existing_checkpoint_positions: &BTreeMap<BlockHeight, Position>,
1276    // The frontier whose position will be used for an inserted checkpoint when
1277    // there is no preceding checkpoint in existing_checkpoint_positions.
1278    state_final_tree: &Frontier<H, DEPTH>,
1279) -> Vec<(BlockHeight, Checkpoint)> {
1280    ensure_heights
1281        .flat_map(|ensure_height| {
1282            existing_checkpoint_positions
1283                .range::<BlockHeight, _>(..=*ensure_height)
1284                .last()
1285                .map_or_else(
1286                    || {
1287                        Some((
1288                            *ensure_height,
1289                            state_final_tree
1290                                .value()
1291                                .map_or_else(Checkpoint::tree_empty, |t| {
1292                                    Checkpoint::at_position(t.position())
1293                                }),
1294                        ))
1295                    },
1296                    |(existing_checkpoint_height, position)| {
1297                        if *existing_checkpoint_height < *ensure_height {
1298                            Some((*ensure_height, Checkpoint::at_position(*position)))
1299                        } else {
1300                            // The checkpoint already exists, so we don't need to
1301                            // do anything.
1302                            None
1303                        }
1304                    },
1305                )
1306                .into_iter()
1307        })
1308        .collect::<Vec<_>>()
