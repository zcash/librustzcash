59    fn create_account(
60        &mut self,
61        account_name: &str,
62        seed: &SecretVec<u8>,
63        birthday: &AccountBirthday,
64        key_source: Option<&str>,
65    ) -> Result<(Self::AccountId, UnifiedSpendingKey), Self::Error> {
66        if cfg!(not(test)) {
67            unimplemented!(
68                "Memwallet does not support adding accounts from seed phrases. 
69    Instead derive the ufvk in the calling code and import it using `import_account_ufvk`"
70            )
71        } else {
72            let seed_fingerprint = SeedFingerprint::from_seed(seed.expose_secret())
73                .ok_or(Self::Error::InvalidSeedLength)?;
74            let account_index = self
75                .max_zip32_account_index(&seed_fingerprint)?
76                .map(|a| a.next().ok_or(Self::Error::AccountOutOfRange))
77                .transpose()?
78                .unwrap_or(zip32::AccountId::ZERO);
79
80            let usk =
81                UnifiedSpendingKey::from_seed(&self.params, seed.expose_secret(), account_index)?;
82            let ufvk = usk.to_unified_full_viewing_key();
83
84            let (id, _account) = self.add_account(
85                account_name,
86                AccountSource::Derived {
87                    derivation: Zip32Derivation::new(
88                        seed_fingerprint,
89                        account_index,
90                        #[cfg(feature = "zcashd-compat")]
91                        None,
92                    ),
93                    key_source: key_source.map(|s| s.to_string()),
94                },
95                ufvk,
96                birthday.clone(),
97            )?;
98
99            Ok((id, usk))
100        }
101    }

121    fn update_chain_tip(&mut self, tip_height: BlockHeight) -> Result<(), Self::Error> {
122        tracing::debug!("update_chain_tip");
123        // If the caller provided a chain tip that is before Sapling activation, do nothing.
124        let sapling_activation = match self.params.activation_height(NetworkUpgrade::Sapling) {
125            Some(h) if h <= tip_height => h,
126            _ => return Ok(()),
127        };
128
129        let max_scanned = self.block_height_extrema().map(|range| *range.end());
130        let wallet_birthday = self.get_wallet_birthday()?;
131
132        // If the chain tip is below the prior max scanned height, then the caller has caught
133        // the chain in the middle of a reorg. Do nothing; the caller will continue using the
134        // old scan ranges and either:
135        // - encounter an error trying to fetch the blocks (and thus trigger the same handling
136        //   logic as if this happened with the old linear scanning code); or
137        // - encounter a discontinuity error in `scan_cached_blocks`, at which point they will
138        //   call `WalletDb::truncate_to_height` as part of their reorg handling which will
139        //   resolve the problem.
140        //
141        // We don't check the shard height, as normal usage would have the caller update the
142        // shard state prior to this call, so it is possible and expected to be in a situation
143        // where we should update the tip-related scan ranges but not the shard-related ones.
144        match max_scanned {
145            Some(h) if tip_height < h => return Ok(()),
146            _ => (),
147        };
148
149        // `ScanRange` uses an exclusive upper bound.
150        let chain_end = tip_height + 1;
151
152        let sapling_shard_tip = self.sapling_tip_shard_end_height();
153        // TODO: Handle orchard case as well. See zcash_client_sqlite scanning.rs update_chain_tip
154        let min_shard_tip = sapling_shard_tip;
155
156        // Create a scanning range for the fragment of the last shard leading up to new tip.
157        // We set a lower bound at the wallet birthday (if known), because account creation
158        // requires specifying a tree frontier that ensures we don't need tree information
159        // prior to the birthday.
160        let tip_shard_entry = min_shard_tip.filter(|h| h < &chain_end).map(|h| {
161            let min_to_scan = wallet_birthday.filter(|b| b > &h).unwrap_or(h);
162            ScanRange::from_parts(min_to_scan..chain_end, ScanPriority::ChainTip)
163        });
164
165        // Create scan ranges to either validate potentially invalid blocks at the wallet's
166        // view of the chain tip, or connect the prior tip to the new tip.
167        let tip_entry = max_scanned.map_or_else(
168            || {
169                // No blocks have been scanned, so we need to anchor the start of the new scan
170                // range to something else.
171                wallet_birthday.map_or_else(
172                    // We don't have a wallet birthday, which means we have no accounts yet.
173                    // We can therefore ignore all blocks up to the chain tip.
174                    || ScanRange::from_parts(sapling_activation..chain_end, ScanPriority::Ignored),
175                    // We have a wallet birthday, so mark all blocks between that and the
176                    // chain tip as `Historic` (performing wallet recovery).
177                    |wallet_birthday| {
178                        ScanRange::from_parts(wallet_birthday..chain_end, ScanPriority::Historic)
179                    },
180                )
181            },
182            |max_scanned| {
183                // The scan range starts at the block after the max scanned height. Since
184                // `scan_cached_blocks` retrieves the metadata for the block being connected to
185                // (if it exists), the connectivity of the scan range to the max scanned block
186                // will always be checked if relevant.
187                let min_unscanned = max_scanned + 1;
188
189                // If we don't have shard metadata, this means we're doing linear scanning, so
190                // create a scan range from the prior tip to the current tip with `Historic`
191                // priority.
192                if tip_shard_entry.is_none() {
193                    ScanRange::from_parts(min_unscanned..chain_end, ScanPriority::Historic)
194                } else {
195                    // Determine the height to which we expect new blocks retrieved from the
196                    // block source to be stable and not subject to being reorg'ed.
197                    let stable_height = tip_height.saturating_sub(PRUNING_DEPTH);
198
199                    // If the wallet's max scanned height is above the stable height,
200                    // prioritize the range between it and the new tip as `ChainTip`.
201                    if max_scanned > stable_height {
202                        // We are in the steady-state case, where a wallet is close to the
203                        // chain tip and just needs to catch up.
204                        //
205                        // This overlaps the `tip_shard_entry` range and so will be coalesced
206                        // with it.
207                        ScanRange::from_parts(min_unscanned..chain_end, ScanPriority::ChainTip)
208                    } else {
209                        // In this case, the max scanned height is considered stable relative
210                        // to the chain tip. However, it may be stable or unstable relative to
211                        // the prior chain tip, which we could determine by looking up the
212                        // prior chain tip height from the scan queue. For simplicity we merge
213                        // these two cases together, and proceed as though the max scanned
214                        // block is unstable relative to the prior chain tip.
215                        //
216                        // To confirm its stability, prioritize the `VERIFY_LOOKAHEAD` blocks
217                        // above the max scanned height as `Verify`:
218                        //
219                        // - We use `Verify` to ensure that a connectivity check is performed,
220                        //   along with any required rewinds, before any `ChainTip` ranges
221                        //   (from this or any prior `update_chain_tip` call) are scanned.
222                        //
223                        // - We prioritize `VERIFY_LOOKAHEAD` blocks because this is expected
224                        //   to be 12.5 minutes, within which it is reasonable for a user to
225                        //   have potentially received a transaction (if they opened their
226                        //   wallet to provide an address to someone else, or spent their own
227                        //   funds creating a change output), without necessarily having left
228                        //   their wallet open long enough for the transaction to be mined and
229                        //   the corresponding block to be scanned.
230                        //
231                        // - We limit the range to at most the stable region, to prevent any
232                        //   `Verify` ranges from being susceptible to reorgs, and potentially
233                        //   interfering with subsequent `Verify` ranges defined by future
234                        //   calls to `update_chain_tip`. Any gap between `stable_height` and
235                        //   `shard_start_height` will be filled by the scan range merging
236                        //   logic with a `Historic` range.
237                        //
238                        // If `max_scanned == stable_height` then this is a zero-length range.
239                        // In this case, any non-empty `(stable_height+1)..shard_start_height`
240                        // will be marked `Historic`, minimising the prioritised blocks at the
241                        // chain tip and allowing for other ranges (for example, `FoundNote`)
242                        // to take priority.
243                        ScanRange::from_parts(
244                            min_unscanned
245                                ..std::cmp::min(
246                                    stable_height + 1,
247                                    min_unscanned + VERIFY_LOOKAHEAD,
248                                ),
249                            ScanPriority::Verify,
250                        )
251                    }
252                }
253            },
254        );
255        if let Some(entry) = &tip_shard_entry {
256            tracing::debug!("{} will update latest shard", entry);
257        }
258        tracing::debug!("{} will connect prior scanned state to new tip", tip_entry);
259
260        let query_range = match tip_shard_entry.as_ref() {
261            Some(se) => Range {
262                start: std::cmp::min(se.block_range().start, tip_entry.block_range().start),
263                end: std::cmp::max(se.block_range().end, tip_entry.block_range().end),
264            },
265            None => tip_entry.block_range().clone(),
266        };
267
268        self.scan_queue.replace_queue_entries(
269            &query_range,
270            tip_shard_entry.into_iter().chain(Some(tip_entry)),
271            false,
272        )?;
273        Ok(())
274    }

276    /// Adds a sequence of blocks to the data store.
277    ///
278    /// Assumes blocks will be here in order.
279    fn put_blocks(
280        &mut self,
281        from_state: &ChainState,
282        blocks: Vec<ScannedBlock<Self::AccountId>>,
283    ) -> Result<(), Self::Error> {
284        tracing::debug!("put_blocks");
285        let mut last_scanned_height = None;
286        struct BlockPositions {
287            height: BlockHeight,
288            sapling_start_position: Position,
289            #[cfg(feature = "orchard")]
290            orchard_start_position: Position,
291        }
292        let start_positions = blocks.first().map(|block| BlockPositions {
293            height: block.height(),
294            sapling_start_position: Position::from(
295                u64::from(block.sapling().final_tree_size())
296                    - u64::try_from(block.sapling().commitments().len()).unwrap(),
297            ),
298            #[cfg(feature = "orchard")]
299            orchard_start_position: Position::from(
300                u64::from(block.orchard().final_tree_size())
301                    - u64::try_from(block.orchard().commitments().len()).unwrap(),
302            ),
303        });
304
305        let mut sapling_commitments = vec![];
306        #[cfg(feature = "orchard")]
307        let mut orchard_commitments = vec![];
308        let mut note_positions = vec![];
309        for block in blocks.into_iter() {
310            let mut transactions = HashMap::new();
311            let mut memos = HashMap::new();
312            if last_scanned_height
313                .iter()
314                .any(|prev| block.height() != *prev + 1)
315            {
316                return Err(Error::NonSequentialBlocks);
317            }
318
319            for transaction in block.transactions().iter() {
320                let txid = transaction.txid();
321
322                // Mark the Sapling nullifiers of the spent notes as spent in the `sapling_spends` map.
323                for spend in transaction.sapling_spends() {
324                    println!(
325                        "marking note {:?} as spent in transaction {:?}",
326                        spend.nf(),
327                        txid
328                    );
329                    self.mark_sapling_note_spent(*spend.nf(), txid)?;
330                }
331
332                // Mark the Orchard nullifiers of the spent notes as spent in the `orchard_spends` map.
333                #[cfg(feature = "orchard")]
334                for spend in transaction.orchard_spends() {
335                    self.mark_orchard_note_spent(*spend.nf(), txid)?;
336                }
337
338                for output in transaction.sapling_outputs() {
339                    // Insert the memo into the `memos` map.
340                    let note_id = NoteId::new(
341                        txid,
342                        Sapling,
343                        u16::try_from(output.index())
344                            .expect("output indices are representable as u16"),
345                    );
346                    if let Ok(Some(memo)) = self.get_memo(note_id) {
347                        memos.insert(note_id, memo.encode());
348                    }
349                    // Check whether this note was spent in a later block range that
350                    // we previously scanned.
351                    let spent_in = output
352                        .nf()
353                        .and_then(|nf| self.nullifiers.get(&Nullifier::Sapling(*nf)))
354                        .and_then(|(height, tx_idx)| self.tx_locator.get(*height, *tx_idx))
355                        .copied();
356
357                    self.insert_received_sapling_note(note_id, output, spent_in);
358                }
359
360                #[cfg(feature = "orchard")]
361                for output in transaction.orchard_outputs().iter() {
362                    // Insert the memo into the `memos` map.
363                    let note_id = NoteId::new(
364                        txid,
365                        Orchard,
366                        u16::try_from(output.index())
367                            .expect("output indices are representable as u16"),
368                    );
369                    if let Ok(Some(memo)) = self.get_memo(note_id) {
370                        memos.insert(note_id, memo.encode());
371                    }
372                    // Check whether this note was spent in a later block range that
373                    // we previously scanned.
374                    let spent_in = output
375                        .nf()
376                        .and_then(|nf| self.nullifiers.get(&Nullifier::Orchard(*nf)))
377                        .and_then(|(height, tx_idx)| self.tx_locator.get(*height, *tx_idx))
378                        .copied();
379
380                    self.insert_received_orchard_note(note_id, output, spent_in)
381                }
382
383                transactions.insert(txid, transaction.clone());
384            }
385
386            // Insert the new nullifiers from this block into the nullifier map
387            self.insert_sapling_nullifier_map(block.height(), block.sapling().nullifier_map())?;
388            #[cfg(feature = "orchard")]
389            self.insert_orchard_nullifier_map(block.height(), block.orchard().nullifier_map())?;
390            note_positions.extend(block.transactions().iter().flat_map(|wtx| {
391                let iter = wtx.sapling_outputs().iter().map(|out| {
392                    (
393                        ShieldedProtocol::Sapling,
394                        out.note_commitment_tree_position(),
395                    )
396                });
397                #[cfg(feature = "orchard")]
398                let iter = iter.chain(wtx.orchard_outputs().iter().map(|out| {
399                    (
400                        ShieldedProtocol::Orchard,
401                        out.note_commitment_tree_position(),
402                    )
403                }));
404
405                iter
406            }));
407
408            let memory_block = MemoryWalletBlock {
409                height: block.height(),
410                hash: block.block_hash(),
411                block_time: block.block_time(),
412                _transactions: transactions.keys().cloned().collect(),
413                _memos: memos,
414                sapling_commitment_tree_size: Some(block.sapling().final_tree_size()),
415                sapling_output_count: Some(block.sapling().commitments().len().try_into().unwrap()),
416                #[cfg(feature = "orchard")]
417                orchard_commitment_tree_size: Some(block.orchard().final_tree_size()),
418                #[cfg(feature = "orchard")]
419                orchard_action_count: Some(block.orchard().commitments().len().try_into().unwrap()),
420            };
421
422            // Insert transaction metadata into the transaction table
423            transactions
424                .into_iter()
425                .for_each(|(_id, tx)| self.tx_table.put_tx_meta(tx, block.height()));
426
427            // Insert the block into the block map
428            self.blocks.insert(block.height(), memory_block);
429            last_scanned_height = Some(block.height());
430
431            let block_commitments = block.into_commitments();
432            sapling_commitments.extend(block_commitments.sapling.into_iter().map(Some));
433            #[cfg(feature = "orchard")]
434            orchard_commitments.extend(block_commitments.orchard.into_iter().map(Some));
435        }
436
437        if let Some((start_positions, last_scanned_height)) =
438            start_positions.zip(last_scanned_height)
439        {
440            // Create subtrees from the note commitments in parallel.
441            const CHUNK_SIZE: usize = 1024;
442            let sapling_subtrees = sapling_commitments
443                .par_chunks_mut(CHUNK_SIZE)
444                .enumerate()
445                .filter_map(|(i, chunk)| {
446                    let start = start_positions.sapling_start_position + (i * CHUNK_SIZE) as u64;
447                    let end = start + chunk.len() as u64;
448
449                    shardtree::LocatedTree::from_iter(
450                        start..end,
451                        SAPLING_SHARD_HEIGHT.into(),
452                        chunk.iter_mut().map(|n| n.take().expect("always Some")),
453                    )
454                })
455                .map(|res| (res.subtree, res.checkpoints))
456                .collect::<Vec<_>>();
457
458            #[cfg(feature = "orchard")]
459            let orchard_subtrees = orchard_commitments
460                .par_chunks_mut(CHUNK_SIZE)
461                .enumerate()
462                .filter_map(|(i, chunk)| {
463                    let start = start_positions.orchard_start_position + (i * CHUNK_SIZE) as u64;
464                    let end = start + chunk.len() as u64;
465
466                    shardtree::LocatedTree::from_iter(
467                        start..end,
468                        ORCHARD_SHARD_HEIGHT.into(),
469                        chunk.iter_mut().map(|n| n.take().expect("always Some")),
470                    )
471                })
472                .map(|res| (res.subtree, res.checkpoints))
473                .collect::<Vec<_>>();
474
475            // Collect the complete set of Sapling checkpoints
476            #[cfg(feature = "orchard")]
477            let sapling_checkpoint_positions: BTreeMap<BlockHeight, Position> = sapling_subtrees
478                .iter()
479                .flat_map(|(_, checkpoints)| checkpoints.iter())
480                .map(|(k, v)| (*k, *v))
481                .collect();
482
483            #[cfg(feature = "orchard")]
484            let orchard_checkpoint_positions: BTreeMap<BlockHeight, Position> = orchard_subtrees
485                .iter()
486                .flat_map(|(_, checkpoints)| checkpoints.iter())
487                .map(|(k, v)| (*k, *v))
488                .collect();
489
490            #[cfg(feature = "orchard")]
491            let (missing_sapling_checkpoints, missing_orchard_checkpoints) = (
492                ensure_checkpoints(
493                    orchard_checkpoint_positions.keys(),
494                    &sapling_checkpoint_positions,
495                    from_state.final_sapling_tree(),
496                ),
497                ensure_checkpoints(
498                    sapling_checkpoint_positions.keys(),
499                    &orchard_checkpoint_positions,
500                    from_state.final_orchard_tree(),
501                ),
502            );
503
504            // Update the Sapling note commitment tree with all newly read note commitments
505            {
506                let mut sapling_subtrees_iter = sapling_subtrees.into_iter();
507                self.with_sapling_tree_mut::<_, _, Self::Error>(|sapling_tree| {
508                    sapling_tree.insert_frontier(
509                        from_state.final_sapling_tree().clone(),
510                        Retention::Checkpoint {
511                            id: from_state.block_height(),
512                            marking: Marking::Reference,
513                        },
514                    )?;
515
516                    for (tree, checkpoints) in &mut sapling_subtrees_iter {
517                        sapling_tree.insert_tree(tree, checkpoints)?;
518                    }
519
520                    // Ensure we have a Sapling checkpoint for each checkpointed Orchard block height.
521                    // We skip all checkpoints below the minimum retained checkpoint in the
522                    // Sapling tree, because branches below this height may be pruned.
523                    #[cfg(feature = "orchard")]
524                    {
525                        let min_checkpoint_height = sapling_tree
526                            .store()
527                            .min_checkpoint_id()
528                            .map_err(ShardTreeError::Storage)?
529                            .expect("At least one checkpoint was inserted (by insert_frontier)");
530
531                        for (height, checkpoint) in &missing_sapling_checkpoints {
532                            if *height > min_checkpoint_height {
533                                sapling_tree
534                                    .store_mut()
535                                    .add_checkpoint(*height, checkpoint.clone())
536                                    .map_err(ShardTreeError::Storage)?;
537                            }
538                        }
539                    }
540
541                    Ok(())
542                })?;
543            }
544
545            // Update the Orchard note commitment tree with all newly read note commitments
546            #[cfg(feature = "orchard")]
547            {
548                let mut orchard_subtrees = orchard_subtrees.into_iter();
549                self.with_orchard_tree_mut::<_, _, Self::Error>(|orchard_tree| {
550                    orchard_tree.insert_frontier(
551                        from_state.final_orchard_tree().clone(),
552                        Retention::Checkpoint {
553                            id: from_state.block_height(),
554                            marking: Marking::Reference,
555                        },
556                    )?;
557
558                    for (tree, checkpoints) in &mut orchard_subtrees {
559                        orchard_tree.insert_tree(tree, checkpoints)?;
560                    }
561
562                    // Ensure we have an Orchard checkpoint for each checkpointed Sapling block height.
563                    // We skip all checkpoints below the minimum retained checkpoint in the
564                    // Orchard tree, because branches below this height may be pruned.
565                    {
566                        let min_checkpoint_height = orchard_tree
567                            .store()
568                            .min_checkpoint_id()
569                            .map_err(ShardTreeError::Storage)?
570                            .expect("At least one checkpoint was inserted (by insert_frontier)");
571
572                        for (height, checkpoint) in &missing_orchard_checkpoints {
573                            if *height > min_checkpoint_height {
574                                orchard_tree
575                                    .store_mut()
576                                    .add_checkpoint(*height, checkpoint.clone())
577                                    .map_err(ShardTreeError::Storage)?;
578                            }
579                        }
580                    }
581                    Ok(())
582                })?;
583            }
584
585            self.scan_complete(
586                Range {
587                    start: start_positions.height,
588                    end: last_scanned_height + 1,
589                },
590                &note_positions,
591            )?;
592        }
593
594        Ok(())
595    }

597    /// Adds a transparent UTXO received by the wallet to the data store.
598    fn put_received_transparent_utxo(
599        &mut self,
600        _output: &WalletTransparentOutput,
601    ) -> Result<Self::UtxoRef, Self::Error> {
602        tracing::debug!("put_received_transparent_utxo");
603        #[cfg(feature = "transparent-inputs")]
604        {
605            let address = _output.recipient_address();
606            if let Some(account_id) = self.find_account_for_transparent_address(address)? {
607                self.put_transparent_output(_output, &account_id, false)
608            } else {
609                Err(Error::AddressNotRecognized(*address))
610            }
611        }
612        #[cfg(not(feature = "transparent-inputs"))]
613        panic!(
614            "The wallet must be compiled with the transparent-inputs feature to use this method."
615        );
616    }

618    fn store_decrypted_tx(
619        &mut self,
620        d_tx: DecryptedTransaction<Self::AccountId>,
621    ) -> Result<(), Self::Error> {
622        tracing::debug!("store_decrypted_tx");
623        self.tx_table.put_tx_data(d_tx.tx(), None, None);
624        if let Some(height) = d_tx.mined_height() {
625            self.set_transaction_status(d_tx.tx().txid(), TransactionStatus::Mined(height))?
626        }
627
628        let funding_accounts = self.get_funding_accounts(d_tx.tx())?;
629        // TODO(#1305): Correctly track accounts that fund each transaction output.
630        let funding_account = funding_accounts.iter().next().copied();
631        if funding_accounts.len() > 1 {
632            tracing::warn!(
633                "More than one wallet account detected as funding transaction {:?}, selecting {:?}",
634                d_tx.tx().txid(),
635                funding_account.unwrap()
636            )
637        }
638
639        // A flag used to determine whether it is necessary to query for transactions that
640        // provided transparent inputs to this transaction, in order to be able to correctly
641        // recover transparent transaction history.
642        #[cfg(feature = "transparent-inputs")]
643        let mut tx_has_wallet_outputs = false;
644
645        for output in d_tx.sapling_outputs() {
646            #[cfg(feature = "transparent-inputs")]
647            {
648                tx_has_wallet_outputs = true;
649            }
650
651            match output.transfer_type() {
652                TransferType::Outgoing => {
653                    let recipient = {
654                        let receiver = Receiver::Sapling(output.note().recipient());
655                        let wallet_address = self
656                            .accounts
657                            .get(*output.account())
658                            .map(|acc| {
659                                acc.select_receiving_address(self.params.network_type(), &receiver)
660                            })
661                            .transpose()?
662                            .flatten()
663                            .unwrap_or_else(|| {
664                                receiver.to_zcash_address(self.params.network_type())
665                            });
666
667                        Recipient::External {
668                            recipient_address: wallet_address,
669                            output_pool: PoolType::SAPLING,
670                        }
671                    };
672
673                    let sent_tx_output = SentTransactionOutput::from_parts(
674                        output.index(),
675                        recipient,
676                        output.note_value(),
677                        Some(output.memo().clone()),
678                    );
679                    self.sent_notes.put_sent_output(
680                        d_tx.tx().txid(),
681                        *output.account(),
682                        &sent_tx_output,
683                    );
684                }
685                TransferType::WalletInternal => {
686                    let recipient = Recipient::InternalAccount {
687                        receiving_account: *output.account(),
688                        external_address: None,
689                        note: Box::new(Note::Sapling(output.note().clone())),
690                    };
691                    let sent_tx_output = SentTransactionOutput::from_parts(
692                        output.index(),
693                        recipient,
694                        output.note_value(),
695                        Some(output.memo().clone()),
696                    );
697
698                    self.received_notes
699                        .insert_received_note(ReceivedNote::from_sent_tx_output(
700                            d_tx.tx().txid(),
701                            &sent_tx_output,
702                        )?);
703
704                    self.sent_notes.put_sent_output(
705                        d_tx.tx().txid(),
706                        *output.account(),
707                        &sent_tx_output,
708                    );
709                }
710                TransferType::Incoming => {
711                    todo!("store decrypted tx sapling incoming")
712                }
713            }
714        }
715
716        #[cfg(feature = "orchard")]
717        for output in d_tx.orchard_outputs() {
718            #[cfg(feature = "transparent-inputs")]
719            {
720                tx_has_wallet_outputs = true;
721            }
722            match output.transfer_type() {
723                TransferType::Outgoing => {
724                    let recipient = {
725                        let receiver = Receiver::Orchard(output.note().recipient());
726                        let wallet_address = self
727                            .accounts
728                            .get(*output.account())
729                            .map(|acc| {
730                                acc.select_receiving_address(self.params.network_type(), &receiver)
731                            })
732                            .transpose()?
733                            .flatten()
734                            .unwrap_or_else(|| {
735                                receiver.to_zcash_address(self.params.network_type())
736                            });
737
738                        Recipient::External {
739                            recipient_address: wallet_address,
740                            output_pool: PoolType::ORCHARD,
741                        }
742                    };
743
744                    let sent_tx_output = SentTransactionOutput::from_parts(
745                        output.index(),
746                        recipient,
747                        output.note_value(),
748                        Some(output.memo().clone()),
749                    );
750                    self.sent_notes.put_sent_output(
751                        d_tx.tx().txid(),
752                        *output.account(),
753                        &sent_tx_output,
754                    );
755                }
756                TransferType::WalletInternal => {
757                    let recipient = Recipient::InternalAccount {
758                        receiving_account: *output.account(),
759                        external_address: None,
760                        note: Box::new(Note::Orchard(*output.note())),
761                    };
762                    let sent_tx_output = SentTransactionOutput::from_parts(
763                        output.index(),
764                        recipient,
765                        output.note_value(),
766                        Some(output.memo().clone()),
767                    );
768
769                    self.received_notes
770                        .insert_received_note(ReceivedNote::from_sent_tx_output(
771                            d_tx.tx().txid(),
772                            &sent_tx_output,
773                        )?);
774
775                    self.sent_notes.put_sent_output(
776                        d_tx.tx().txid(),
777                        *output.account(),
778                        &sent_tx_output,
779                    );
780                }
781                TransferType::Incoming => {
782                    todo!("store decrypted tx orchard incoming")
783                }
784            }
785        }
786
787        // If any of the utxos spent in the transaction are ours, mark them as spent.
788        #[cfg(feature = "transparent-inputs")]
789        for txin in d_tx
790            .tx()
791            .transparent_bundle()
792            .iter()
793            .flat_map(|b| b.vin.iter())
794        {
795            self.mark_transparent_output_spent(&d_tx.tx().txid(), txin.prevout())?;
796        }
797
798        // This `if` is just an optimization for cases where we would do nothing in the loop.
799        if funding_account.is_some() || cfg!(feature = "transparent-inputs") {
800            for (output_index, txout) in d_tx
801                .tx()
802                .transparent_bundle()
803                .iter()
804                .flat_map(|b| b.vout.iter())
805                .enumerate()
806            {
807                if let Some(address) = txout.recipient_address() {
808                    tracing::debug!(
809                        "{:?} output {} has recipient {}",
810                        d_tx.tx().txid(),
811                        output_index,
812                        address.encode(self.params())
813                    );
814
815                    // The transaction is not necessarily mined yet, but we want to record
816                    // that an output to the address was seen in this tx anyway. This will
817                    // advance the gap regardless of whether it is mined, but an output in
818                    // an unmined transaction won't advance the range of safe indices.
819                    #[cfg(feature = "transparent-inputs")]
820                    self.accounts
821                        .mark_ephemeral_address_as_seen(&address, d_tx.tx().txid())?;
822
823                    // If the output belongs to the wallet, add it to `transparent_received_outputs`.
824                    #[cfg(feature = "transparent-inputs")]
825                    if let Some(account_id) = self
826                        .accounts
827                        .find_account_for_transparent_address(&address)?
828                    {
829                        tracing::debug!(
830                            "{:?} output {} belongs to account {:?}",
831                            d_tx.tx().txid(),
832                            output_index,
833                            account_id
834                        );
835                        let wallet_transparent_output = WalletTransparentOutput::from_parts(
836                            OutPoint::new(
837                                d_tx.tx().txid().into(),
838                                u32::try_from(output_index).unwrap(),
839                            ),
840                            txout.clone(),
841                            d_tx.mined_height(),
842                        )
843                        .unwrap();
844                        self.put_transparent_output(
845                            &wallet_transparent_output,
846                            &account_id,
847                            false,
848                        )?;
849
850                        // Since the wallet created the transparent output, we need to ensure
851                        // that any transparent inputs belonging to the wallet will be
852                        // discovered.
853                        tx_has_wallet_outputs = true;
854                    } else {
855                        tracing::debug!(
856                            "Address {} is not recognized as belonging to any of our accounts.",
857                            address.encode(self.params())
858                        );
859                    }
860
861                    // If a transaction we observe contains spends from our wallet, we will
862                    // store its transparent outputs in the same way they would be stored by
863                    // create_spend_to_address.
864                    if let Some(account_id) = funding_account {
865                        let receiver = Receiver::Transparent(address);
866
867                        #[cfg(feature = "transparent-inputs")]
868                        let recipient_addr = self
869                            .accounts
870                            .get(account_id)
871                            .map(|acc| {
872                                acc.select_receiving_address(self.params.network_type(), &receiver)
873                            })
874                            .transpose()?
875                            .flatten()
876                            .unwrap_or_else(|| {
877                                receiver.to_zcash_address(self.params.network_type())
878                            });
879
880                        #[cfg(not(feature = "transparent-inputs"))]
881                        let recipient_addr = receiver.to_zcash_address(self.params.network_type());
882
883                        let recipient = Recipient::External {
884                            recipient_address: recipient_addr,
885                            output_pool: PoolType::TRANSPARENT,
886                        };
887
888                        let sent_tx_output = SentTransactionOutput::from_parts(
889                            output_index,
890                            recipient,
891                            txout.value(),
892                            None,
893                        );
894                        self.sent_notes.put_sent_output(
895                            d_tx.tx().txid(),
896                            account_id,
897                            &sent_tx_output,
898                        );
899                        // Even though we know the funding account, we don't know that we have
900                        // information for all of the transparent inputs to the transaction.
901                        #[cfg(feature = "transparent-inputs")]
902                        {
903                            tx_has_wallet_outputs = true;
904                        }
905                    }
906                } else {
907                    tracing::warn!(
908                        "Unable to determine recipient address for tx {:?} output {}",
909                        d_tx.tx().txid(),
910                        output_index
911                    );
912                }
913            }
914        }
915
916        // If the transaction has outputs that belong to the wallet as well as transparent
917        // inputs, we may need to download the transactions corresponding to the transparent
918        // prevout references to determine whether the transaction was created (at least in
919        // part) by this wallet.
920        #[cfg(feature = "transparent-inputs")]
921        if tx_has_wallet_outputs && d_tx.tx().transparent_bundle().is_some() {
922            // queue the transparent inputs for enhancement
923            self.transaction_data_request_queue
924                .queue_status_retrieval(&d_tx.tx().txid());
925        }
926
927        #[cfg(feature = "transparent-inputs")]
928        {
929            let detectable_via_scanning = d_tx.tx().sapling_bundle().is_some();
930            #[cfg(feature = "orchard")]
931            let detectable_via_scanning =
932                detectable_via_scanning | d_tx.tx().orchard_bundle().is_some();
933
934            if d_tx.mined_height().is_none() && !detectable_via_scanning {
935                self.transaction_data_request_queue
936                    .queue_status_retrieval(&d_tx.tx().txid());
937            }
938        }
939        Ok(())
940    }

946    /// Truncates the database to the given height.
947    ///
948    /// If the requested height is greater than or equal to the height of the last scanned
949    /// block, this function does nothing.
950    ///
951    /// This should only be executed inside a transactional context.
952    fn truncate_to_height(&mut self, max_height: BlockHeight) -> Result<BlockHeight, Self::Error> {
953        let truncation_height = {
954            // This is the intersection of all the checkpoint heights from the sapling and orchard tree.
955            let mut checkpoint_heights = BTreeSet::new();
956            self.sapling_tree.store().for_each_checkpoint(
957                self.sapling_tree.store().checkpoint_count()?,
958                |height, _| {
959                    checkpoint_heights.insert(u32::from(*height));
960                    Ok(())
961                },
962            )?;
963            #[cfg(feature = "orchard")]
964            {
965                let mut orchard_checkpoint_heights = BTreeSet::new();
966                self.orchard_tree.store().for_each_checkpoint(
967                    self.orchard_tree.store().checkpoint_count()?,
968                    |height, _| {
969                        orchard_checkpoint_heights.insert(u32::from(*height));
970                        Ok(())
971                    },
972                )?;
973
974                checkpoint_heights = checkpoint_heights
975                    .intersection(&orchard_checkpoint_heights)
976                    .copied()
977                    .collect();
978            }
979            // All the checkpoints that are greater than the truncation height
980            let over = checkpoint_heights.split_off(&(u32::from(max_height + 1)));
981            if let Some(height) = checkpoint_heights.last().copied() {
982                Ok(BlockHeight::from(height))
983            } else {
984                // If there are no checkpoints that are less than or equal to the truncation height
985                // then we can't truncate the tree.
986                Err(Error::RequestedRewindInvalid(
987                    over.first().copied().map(Into::into),
988                    max_height,
989                ))
990            }
991        }?;
992
993        // Recall where we synced up to previously.
994        let last_scanned_height = self.blocks.keys().max().copied().unwrap_or_else(|| {
995            self.params
996                .activation_height(NetworkUpgrade::Sapling)
997                .expect("Sapling activation height must be available.")
998                - 1
999        });
1000
1001        // Delete from the scanning queue any range with a start height greater than the
1002        // truncation height, and then truncate any remaining range by setting the end
1003        // equal to the truncation height + 1. This sets our view of the chain tip back
1004        // to the retained height.
1005        self.scan_queue
1006            .delete_starts_greater_than_equal_to(truncation_height + 1);
1007        self.scan_queue.truncate_ends_to(truncation_height + 1);
1008
1009        // Mark transparent utxos as un-mined. Since the TXO is now not mined, it would ideally be
1010        // considered to have been returned to the mempool; it _might_ be spendable in this state, but
1011        // we must also set its max_observed_unspent_height field to NULL because the transaction may
1012        // be rendered entirely invalid by a reorg that alters anchor(s) used in constructing shielded
1013        // spends in the transaction.
1014        self.transparent_received_outputs
1015            .iter_mut()
1016            .for_each(|(_, txo)| {
1017                if let Some(mined_height) = self
1018                    .tx_table
1019                    .get(&txo.transaction_id)
1020                    .and_then(|tx| tx.mined_height())
1021                {
1022                    if mined_height <= truncation_height {
1023                        txo.max_observed_unspent_height = Some(truncation_height)
1024                    } else {
1025                        txo.max_observed_unspent_height = None
1026                    }
1027                }
1028            });
1029
1030        // Un-mine transactions. This must be done outside of the last_scanned_height check because
1031        // transaction entries may be created as a consequence of receiving transparent TXOs.
1032        self.tx_table
1033            .unmine_transactions_greater_than(truncation_height);
1034
1035        // If we're removing scanned blocks, we need to truncate the note commitment tree and remove
1036        // affected block records from the database.
1037        if truncation_height < last_scanned_height {
1038            self.with_sapling_tree_mut(|tree| {
1039                tree.truncate_to_checkpoint(&truncation_height).map(|_| ())
1040            })?;
1041            #[cfg(feature = "orchard")]
1042            self.with_orchard_tree_mut(|tree| {
1043                tree.truncate_to_checkpoint(&truncation_height).map(|_| ())
1044            })?;
1045
1046            // Do not delete sent notes; this can contain data that is not recoverable
1047            // from the chain. Wallets must continue to operate correctly in the
1048            // presence of stale sent notes that link to unmined transactions.
1049            // Also, do not delete received notes; they may contain memo data that is
1050            // not recoverable; balance APIs must ensure that un-mined received notes
1051            // do not count towards spendability or transaction balalnce.
1052
1053            // Now that they aren't depended on, delete un-mined blocks.
1054            self.blocks.retain(|height, _| *height <= truncation_height);
1055
1056            // Delete from the nullifier map any entries with a locator referencing a block
1057            // height greater than the truncation height.
1058            // Willem: We don't need to do this I think..
1059        }
1060        Ok(truncation_height)
1061    }

1098    fn store_transactions_to_be_sent(
1099        &mut self,
1100        transactions: &[SentTransaction<Self::AccountId>],
1101    ) -> Result<(), Self::Error> {
1102        tracing::debug!("store_transactions_to_be_sent");
1103        for sent_tx in transactions {
1104            self.tx_table.put_tx_data(
1105                sent_tx.tx(),
1106                Some(sent_tx.fee_amount()),
1107                Some(sent_tx.target_height()),
1108            );
1109            let mut detectable_via_scanning = false;
1110            // Mark sapling notes as spent
1111            if let Some(bundle) = sent_tx.tx().sapling_bundle() {
1112                detectable_via_scanning = true;
1113                for spend in bundle.shielded_spends() {
1114                    self.mark_sapling_note_spent(*spend.nullifier(), sent_tx.tx().txid())?;
1115                }
1116            }
1117            // Mark orchard notes as spent
1118            if let Some(_bundle) = sent_tx.tx().orchard_bundle() {
1119                #[cfg(feature = "orchard")]
1120                {
1121                    detectable_via_scanning = true;
1122                    for action in _bundle.actions() {
1123                        match self.mark_orchard_note_spent(*action.nullifier(), sent_tx.tx().txid())
1124                        {
1125                            Ok(()) => {}
1126                            Err(Error::NoteNotFound) => {
1127                                // This is expected as some of the actions will be new outputs we don't have notes for
1128                                // The ones we do recognize will be marked as spent
1129                            }
1130                            Err(e) => return Err(e),
1131                        }
1132                    }
1133                }
1134
1135                #[cfg(not(feature = "orchard"))]
1136                panic!("Sent a transaction with Orchard Actions without `orchard` enabled?");
1137            }
1138            // Mark transparent UTXOs as spent
1139            #[cfg(feature = "transparent-inputs")]
1140            for utxo_outpoint in sent_tx.utxos_spent() {
1141                self.mark_transparent_output_spent(&sent_tx.tx().txid(), utxo_outpoint)?;
1142            }
1143
1144            for output in sent_tx.outputs() {
1145                self.sent_notes.insert_sent_output(sent_tx, output);
1146
1147                match output.recipient() {
1148                    Recipient::InternalAccount { .. } => {
1149                        self.received_notes.insert_received_note(
1150                            ReceivedNote::from_sent_tx_output(sent_tx.tx().txid(), output)?,
1151                        );
1152                    }
1153                    #[cfg(feature = "transparent-inputs")]
1154                    Recipient::EphemeralTransparent {
1155                        receiving_account,
1156                        ephemeral_address,
1157                        outpoint,
1158                    } => {
1159                        let txo = WalletTransparentOutput::from_parts(
1160                            outpoint.clone(),
1161                            TxOut::new(output.value(), ephemeral_address.script().into()),
1162                            None,
1163                        )
1164                        .unwrap();
1165                        self.put_transparent_output(&txo, receiving_account, true)?;
1166                        if let Some(account) = self.accounts.get_mut(*receiving_account) {
1167                            account.mark_ephemeral_address_as_used(
1168                                ephemeral_address,
1169                                sent_tx.tx().txid(),
1170                            )?
1171                        }
1172                    }
1173                    _ => {}
1174                }
1175            }
1176
1177            // Add the transaction to the set to be queried for transaction status. This is only necessary
1178            // at present for fully transparent transactions, because any transaction with a shielded
1179            // component will be detected via ordinary chain scanning and/or nullifier checking.
1180            if !detectable_via_scanning {
1181                self.transaction_data_request_queue
1182                    .queue_status_retrieval(&sent_tx.tx().txid());
1183            }
1184        }
1185
1186        Ok(())
1187    }

1198    #[cfg(feature = "transparent-inputs")]
1199    fn reserve_next_n_ephemeral_addresses(
1200        &mut self,
1201        account_id: Self::AccountId,
1202        n: usize,
1203    ) -> Result<Vec<(TransparentAddress, TransparentAddressMetadata)>, Self::Error> {
1204        use zcash_keys::keys::AddressGenerationError;
1205        // TODO: We need to implement first_unsafe_index to make sure we dont violate gap invarient
1206
1207        use transparent::keys::NonHardenedChildIndex;
1208        let first_unsafe = self.first_unsafe_index(account_id)?;
1209        if let Some(account) = self.accounts.get_mut(account_id) {
1210            let first_unreserved = account.first_unreserved_index()?;
1211
1212            let allocation = range_from(first_unreserved, u32::try_from(n).unwrap());
1213
1214            if allocation.len() < n {
1215                return Err(AddressGenerationError::DiversifierSpaceExhausted.into());
1216            }
1217            if allocation.end > first_unsafe {
1218                return Err(Error::ReachedGapLimit(
1219                    account_id,
1220                    core::cmp::max(first_unreserved, first_unsafe),
1221                ));
1222            }
1223            let _reserved = account.reserve_until(allocation.end)?;
1224            self.get_known_ephemeral_addresses(
1225                account_id,
1226                Some(
1227                    NonHardenedChildIndex::from_index(allocation.start).expect("Bad Index")
1228                        ..NonHardenedChildIndex::from_index(allocation.end).expect("Bad Index"),
1229                ),
1230            )
1231        } else {
1232            Err(Self::Error::AccountUnknown(account_id))
1233        }
1234    }

1278fn ensure_checkpoints<'a, H, I: Iterator<Item = &'a BlockHeight>, const DEPTH: u8>(
1279    // An iterator of checkpoints heights for which we wish to ensure that
1280    // checkpoints exists.
1281    ensure_heights: I,
1282    // The map of checkpoint positions from which we will draw note commitment tree
1283    // position information for the newly created checkpoints.
1284    existing_checkpoint_positions: &BTreeMap<BlockHeight, Position>,
1285    // The frontier whose position will be used for an inserted checkpoint when
1286    // there is no preceding checkpoint in existing_checkpoint_positions.
1287    state_final_tree: &Frontier<H, DEPTH>,
1288) -> Vec<(BlockHeight, Checkpoint)> {
1289    ensure_heights
1290        .flat_map(|ensure_height| {
1291            existing_checkpoint_positions
1292                .range::<BlockHeight, _>(..=*ensure_height)
1293                .last()
1294                .map_or_else(
1295                    || {
1296                        Some((
1297                            *ensure_height,
1298                            state_final_tree
1299                                .value()
1300                                .map_or_else(Checkpoint::tree_empty, |t| {
1301                                    Checkpoint::at_position(t.position())
1302                                }),
1303                        ))
1304                    },
1305                    |(existing_checkpoint_height, position)| {
1306                        if *existing_checkpoint_height < *ensure_height {
1307                            Some((*ensure_height, Checkpoint::at_position(*position)))
1308                        } else {
1309                            // The checkpoint already exists, so we don't need to
1310                            // do anything.
1311                            None
1312                        }
1313                    },
1314                )
1315                .into_iter()
1316        })
1317        .collect::<Vec<_>>()
