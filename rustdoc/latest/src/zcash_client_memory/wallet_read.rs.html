86    fn validate_seed(
87        &self,
88        account_id: Self::AccountId,
89        seed: &SecretVec<u8>,
90    ) -> Result<bool, Self::Error> {
91        tracing::debug!("validate_seed: {:?}", account_id);
92        if let Some(account) = self.get_account(account_id)? {
93            if let AccountSource::Derived { derivation, .. } = account.source() {
94                seed_matches_derived_account(
95                    &self.params,
96                    seed,
97                    derivation.seed_fingerprint(),
98                    derivation.account_index(),
99                    &account.uivk(),
100                )
101            } else {
102                Err(Error::UnknownZip32Derivation)
103            }
104        } else {
105            // Missing account is documented to return false.
106            Ok(false)
107        }

110    fn seed_relevance_to_derived_accounts(
111        &self,
112        seed: &SecretVec<u8>,
113    ) -> Result<SeedRelevance<Self::AccountId>, Self::Error> {
114        tracing::debug!("seed_relevance_to_derived_accounts");
115        let mut has_accounts = false;
116        let mut has_derived = false;
117        let mut relevant_account_ids = vec![];
118
119        for account_id in self.get_account_ids()? {
120            has_accounts = true;
121            let account = self.get_account(account_id)?.expect("account ID exists");
122
123            // If the account is imported, the seed _might_ be relevant, but the only
124            // way we could determine that is by brute-forcing the ZIP 32 account
125            // index space, which we're not going to do. The method name indicates to
126            // the caller that we only check derived accounts.
127            if let AccountSource::Derived { derivation, .. } = account.source() {
128                has_derived = true;
129
130                if seed_matches_derived_account(
131                    &self.params,
132                    seed,
133                    derivation.seed_fingerprint(),
134                    derivation.account_index(),
135                    &account.uivk(),
136                )? {
137                    // The seed is relevant to this account.
138                    relevant_account_ids.push(account_id);
139                }
140            }
141        }
142
143        Ok(
144            if let Some(account_ids) = NonEmpty::from_vec(relevant_account_ids) {
145                SeedRelevance::Relevant { account_ids }
146            } else if has_derived {
147                SeedRelevance::NotRelevant
148            } else if has_accounts {
149                SeedRelevance::NoDerivedAccounts
150            } else {
151                SeedRelevance::NoAccounts
152            },
153        )
154    }

194    fn get_wallet_summary(
195        &self,
196        confirmations_policy: ConfirmationsPolicy,
197    ) -> Result<Option<WalletSummary<Self::AccountId>>, Self::Error> {
198        tracing::debug!("get_wallet_summary");
199        let chain_tip_height = match self.chain_height()? {
200            Some(height) => height,
201            None => return Ok(None),
202        };
203        let target_height = TargetHeight::from(chain_tip_height + 1);
204        let birthday_height = self
205            .get_wallet_birthday()?
206            .expect("If a scan range exists, we know the wallet birthday.");
207
208        let fully_scanned_height = self
209            .block_fully_scanned()?
210            .map_or(birthday_height - 1, |m| m.block_height());
211
212        let mut account_balances = self
213            .accounts
214            .values()
215            .map(|account| (account.account_id(), AccountBalance::ZERO))
216            .collect::<HashMap<AccountId, AccountBalance>>();
217
218        for note in self.get_received_notes().iter() {
219            // don't count spent notes
220            if self.note_is_spent(note, target_height)? {
221                continue;
222            }
223            // don't count notes in unscanned ranges
224            let unscanned_ranges = self.unscanned_ranges();
225            for (_, _, start_position, end_position_exclusive) in unscanned_ranges {
226                if note.commitment_tree_position >= start_position
227                    && note.commitment_tree_position < end_position_exclusive
228                {
229                    continue; // note is in an unscanned range. Skip it
230                }
231            }
232            // don't count notes in unmined transactions or that have expired
233            if let Ok(Some(note_tx)) = self.get_transaction(note.txid) {
234                if note_tx.expiry_height() < BlockHeight::from(target_height) {
235                    continue;
236                }
237            }
238
239            let account_id = note.account_id();
240            // if this is the first note for this account add a new balance record
241            if let Entry::Vacant(entry) = account_balances.entry(account_id) {
242                entry.insert(AccountBalance::ZERO);
243            }
244            let account_balance = account_balances
245                .get_mut(&account_id)
246                .expect("Account balance should exist");
247
248            // Given a note update the balance for the account
249            // This includes determining if it is change, spendable, etc
250            let update_balance_with_note = |b: &mut Balance| -> Result<(), Error> {
251                match (
252                    self.note_is_pending(note, target_height, confirmations_policy)?,
253                    note.is_change,
254                ) {
255                    (true, true) => b.add_pending_change_value(note.note.value()),
256                    (true, false) => b.add_pending_spendable_value(note.note.value()),
257                    (false, _) => b.add_spendable_value(note.note.value()),
258                }?;
259                Ok(())
260            };
261
262            match note.pool() {
263                PoolType::SAPLING => {
264                    account_balance.with_sapling_balance_mut(update_balance_with_note)?;
265                }
266                PoolType::ORCHARD => {
267                    account_balance.with_orchard_balance_mut(update_balance_with_note)?;
268                }
269                _ => unimplemented!("Unknown pool type"),
270            }
271        }
272
273        #[cfg(feature = "transparent-inputs")]
274        for (account, balance) in account_balances.iter_mut() {
275            let transparent_balances =
276                self.get_transparent_balances(*account, target_height, confirmations_policy)?;
277            for (_, (_, bal)) in transparent_balances.into_iter() {
278                balance.with_unshielded_balance_mut(|b: &mut Balance| -> Result<(), Error> {
279                    *b = (*b + bal)?;
280                    Ok(())
281                })?;
282            }
283        }
284
285        let next_sapling_subtree_index = self
286            .sapling_tree
287            .store()
288            .last_shard()?
289            .map(|s| s.root_addr().index())
290            .unwrap_or(0);
291
292        #[cfg(feature = "orchard")]
293        let next_orchard_subtree_index = self
294            .orchard_tree
295            .store()
296            .last_shard()?
297            .map(|s| s.root_addr().index())
298            .unwrap_or(0);
299
300        // Treat Sapling and Orchard outputs as having the same cost to scan.
301        let sapling_scan_progress =
302            self.sapling_scan_progress(&birthday_height, &fully_scanned_height, &chain_tip_height)?;
303        #[cfg(feature = "orchard")]
304        let orchard_scan_progress =
305            self.orchard_scan_progress(&birthday_height, &fully_scanned_height, &chain_tip_height)?;
306        #[cfg(not(feature = "orchard"))]
307        let orchard_scan_progress: Option<Ratio<u64>> = None;
308
309        let scan_progress = sapling_scan_progress
310            .zip(orchard_scan_progress)
311            .map(|(s, o)| {
312                Ratio::new(
313                    s.numerator() + o.numerator(),
314                    s.denominator() + o.denominator(),
315                )
316            })
317            .or(sapling_scan_progress)
318            .or(orchard_scan_progress);
319
320        // TODO: This won't work
321        let scan_progress = Progress::new(scan_progress.unwrap(), Some(Ratio::new(0, 0)));
322
323        let summary = WalletSummary::new(
324            account_balances,
325            chain_tip_height,
326            fully_scanned_height,
327            scan_progress,
328            next_sapling_subtree_index,
329            #[cfg(feature = "orchard")]
330            next_orchard_subtree_index,
331        );
332        Ok(Some(summary))
333    }

377    fn block_fully_scanned(&self) -> Result<Option<BlockMetadata>, Self::Error> {
378        tracing::debug!("block_fully_scanned");
379        if let Some(birthday_height) = self.get_wallet_birthday()? {
380            // We assume that the only way we get a contiguous range of block heights in the `blocks` table
381            // starting with the birthday block, is if all scanning operations have been performed on those
382            // blocks. This holds because the `blocks` table is only altered by `WalletDb::put_blocks` via
383            // `put_block`, and the effective combination of intra-range linear scanning and the nullifier
384            // map ensures that we discover all wallet-related information within the contiguous range.
385            //
386            // We also assume that every contiguous range of block heights in the `blocks` table has a
387            // single matching entry in the `scan_queue` table with priority "Scanned". This requires no
388            // bugs in the scan queue update logic, which we have had before. However, a bug here would
389            // mean that we return a more conservative fully-scanned height, which likely just causes a
390            // performance regression.
391            //
392            // The fully-scanned height is therefore the last height that falls within the first range in
393            // the scan queue with priority "Scanned".
394            // SQL query problems.
395
396            let mut scanned_ranges: Vec<_> = self
397                .scan_queue
398                .iter()
399                .filter(|(_, _, p)| p == &ScanPriority::Scanned)
400                .collect();
401            scanned_ranges.sort_by(|(start_a, _, _), (start_b, _, _)| start_a.cmp(start_b));
402            if let Some(fully_scanned_height) = scanned_ranges.first().and_then(
403                |(block_range_start, block_range_end, _priority)| {
404                    // If the start of the earliest scanned range is greater than
405                    // the birthday height, then there is an unscanned range between
406                    // the wallet birthday and that range, so there is no fully
407                    // scanned height.
408                    if *block_range_start <= birthday_height {
409                        // Scan ranges are end-exclusive.
410                        Some(*block_range_end - 1)
411                    } else {
412                        None
413                    }
414                },
415            ) {
416                self.block_metadata(fully_scanned_height)
417            } else {
418                Ok(None)
419            }
420        } else {
421            Ok(None)
422        }
423    }

461
462            let anchor_height = sapling_anchor_height
463                .zip(orchard_anchor_height)
464                .map(|(s, o)| std::cmp::min(s, o))
465                .or(sapling_anchor_height)
466                .or(orchard_anchor_height);
467
468            Ok(anchor_height.map(|h| (TargetHeight::from(chain_tip_height + 1), h)))
469        } else {
470            Ok(None)
471        }

507    fn get_transaction(&self, txid: TxId) -> Result<Option<Transaction>, Self::Error> {
508        tracing::debug!("get_transaction: {:?}", txid);
509        self.tx_table
510            .get(&txid)
511            .map(|tx| (tx.status(), tx.expiry_height(), tx.raw()))
512            .map(|(status, expiry_height, raw)| {
513                let raw = raw.ok_or_else(|| {
514                    Self::Error::CorruptedData("Transaction raw data not found".to_string())
515                })?;
516
517                // We need to provide a consensus branch ID so that pre-v5 `Transaction` structs
518                // (which don't commit directly to one) can store it internally.
519                // - If the transaction is mined, we use the block height to get the correct one.
520                // - If the transaction is unmined and has a cached non-zero expiry height, we use
521                //   that (relying on the invariant that a transaction can't be mined across a network
522                //   upgrade boundary, so the expiry height must be in the same epoch).
523                // - Otherwise, we use a placeholder for the initial transaction parse (as the
524                //   consensus branch ID is not used there), and then either use its non-zero expiry
525                //   height or return an error.
526                if let TransactionStatus::Mined(height) = status {
527                    return Ok(Transaction::read(
528                        raw,
529                        BranchId::for_height(&self.params, height),
530                    )?);
531                }
532                if let Some(height) = expiry_height.filter(|h| h > &BlockHeight::from(0)) {
533                    return Ok(Transaction::read(
534                        raw,
535                        BranchId::for_height(&self.params, height),
536                    )?);
537                }
538
539                let tx_data = Transaction::read(raw, BranchId::Sprout)
540                    .map_err(Self::Error::from)?
541                    .into_data();
542
543                let expiry_height = tx_data.expiry_height();
544                if expiry_height > BlockHeight::from(0) {
545                    Ok(TransactionData::from_parts(
546                        tx_data.version(),
547                        BranchId::for_height(&self.params, expiry_height),
548                        tx_data.lock_time(),
549                        expiry_height,
550                        tx_data.transparent_bundle().cloned(),
551                        tx_data.sprout_bundle().cloned(),
552                        tx_data.sapling_bundle().cloned(),
553                        tx_data.orchard_bundle().cloned(),
554                    )
555                        .freeze()?)
556                } else {
557                    Err(Self::Error::CorruptedData(
558                        "Consensus branch ID not known, cannot parse this transaction until it is mined"
559                            .to_string(),
560                    ))
561                }
562            })
563            .transpose()
564    }

566    fn get_sapling_nullifiers(
567        &self,
568        query: NullifierQuery,
569    ) -> Result<Vec<(Self::AccountId, sapling::Nullifier)>, Self::Error> {
570        tracing::debug!("get_sapling_nullifiers");
571        let nullifiers = self.received_notes.get_sapling_nullifiers();
572        Ok(match query {
573            NullifierQuery::All => nullifiers
574                .map(|(account_id, _, nf)| (account_id, nf))
575                .collect(),
576            NullifierQuery::Unspent => nullifiers
577                .filter_map(|(account_id, _, nf)| {
578                    // find any tx we know of that spends this nullifier and if so require that it is unmined or expired
579                    if let Some((height, tx_index)) = self.nullifiers.get(&Nullifier::Sapling(nf)) {
580                        if let Some(spending_tx) =
581                            self.tx_table.get_by_height_and_index(*height, *tx_index)
582                        {
583                            if matches!(spending_tx.status(), TransactionStatus::Mined(_))
584                                || spending_tx.expiry_height().is_none()
585                            {
586                                None
587                            } else {
588                                Some((account_id, nf))
589                            }
590                        } else {
591                            None
592                        }
593                    } else {
594                        Some((account_id, nf))
595                    }
596                })
597                .collect(),
598        })
599    }

644        let account = self

646            .ok_or(Error::AccountUnknown(account_id))?;
647
648        if _include_change {
649            unimplemented!("include_change is not yet supported");
650        }
651        if _include_standalone {
652            unimplemented!("include_standalone is not yet supported");
653        }
654
655        let t_addresses = account
656            .addresses()
657            .iter()
658            .filter_map(|(diversifier_index, ua)| {
659                ua.transparent().map(|ta| {
660                    let metadata = TransparentAddressMetadata::derived(
661                        Scope::External.into(),
662                        NonHardenedChildIndex::from_index((*diversifier_index).try_into().unwrap())
663                            .expect(
664                                "diversifier index corresponds to a valid NonHardenedChildIndex",
665                            ),
666                        Exposure::Unknown,
667                        None,
668                    );
669
670                    (*ta, metadata)
671                })
672            })
673            .collect();
674        Ok(t_addresses)
675    }

677    /// Returns a mapping from each transparent receiver associated with the specified account
678    /// to its not-yet-shielded UTXO balance, including only the effects of transactions mined
679    /// at a block height less than or equal to `summary_height`.
680    ///
681    /// Only non-ephemeral transparent receivers with a non-zero balance at the summary height
682    /// will be included.
683    #[cfg(feature = "transparent-inputs")]
684    fn get_transparent_balances(
685        &self,
686        account_id: Self::AccountId,
687        target_height: TargetHeight,
688        confirmations_policy: ConfirmationsPolicy,
689    ) -> Result<HashMap<TransparentAddress, (TransparentKeyScope, Balance)>, Self::Error> {
690        tracing::debug!("get_transparent_balances");
691
692        let mut balances = HashMap::new();
693
694        for (outpoint, txo) in self.transparent_received_outputs.iter().filter(|(_, txo)| {
695            if let Ok(Some(txo_account_id)) =
696                self.find_account_for_transparent_address(&txo.address)
697            {
698                txo_account_id == account_id
699            } else {
700                false
701            }
702        }) {
703            if self.utxo_is_spendable(outpoint, target_height, confirmations_policy)? {
704                let address = txo.address;
705                let entry = balances
706                    .entry(address)
707                    .or_insert((txo.key_scope, Balance::ZERO));
708
709                entry.1.add_spendable_value(txo.txout.value())?;
710            }
711        }
712
713        Ok(balances)
714    }

743/// Copied from zcash_client_sqlite::wallet::seed_matches_derived_account
744fn seed_matches_derived_account<P: consensus::Parameters>(
745    params: &P,
746    seed: &SecretVec<u8>,
747    seed_fingerprint: &SeedFingerprint,
748    account_index: zip32::AccountId,
749    uivk: &UnifiedIncomingViewingKey,
750) -> Result<bool, Error> {
751    let seed_fingerprint_match =
752        &SeedFingerprint::from_seed(seed.expose_secret()).ok_or_else(|| {
753            Error::BadAccountData("Seed must be between 32 and 252 bytes in length.".to_owned())
754        })? == seed_fingerprint;
755
756    // Keys are not comparable with `Eq`, but addresses are, so we derive what should
757    // be equivalent addresses for each key and use those to check for key equality.
758    let uivk_match =
759        match UnifiedSpendingKey::from_seed(params, &seed.expose_secret()[..], account_index) {
760            // If we can't derive a USK from the given seed with the account's ZIP 32
761            // account index, then we immediately know the UIVK won't match because wallet
762            // accounts are required to have a known UIVK.
763            Err(_) => false,
764            Ok(usk) => {
765                usk.to_unified_full_viewing_key()
766                    .default_address(UnifiedAddressRequest::AllAvailableKeys)?
767                    == uivk.default_address(UnifiedAddressRequest::AllAvailableKeys)?
768            }
769        };
770
771    if seed_fingerprint_match != uivk_match {
772        // If these mismatch, it suggests database corruption.
773        Err(Error::CorruptedData(format!(
774            "Seed fingerprint match: {seed_fingerprint_match}, uivk match: {uivk_match}"
775        )))
776    } else {
777        Ok(seed_fingerprint_match && uivk_match)
778    }
779}
