7impl Params {
8    /// Returns `None` if the parameters are invalid.
9    pub(crate) fn new(n: u32, k: u32) -> Option<Self> {
10        // We place the following requirements on the parameters:
11        // - n is a multiple of 8, so the hash output has an exact byte length.
12        // - k >= 3 so the encoded solutions have an exact byte length.
13        // - k < n, so the collision bit length is at least 1.
14        // - n is a multiple of k + 1, so we have an integer collision bit length.
15        if (n % 8 == 0) && (k >= 3) && (k < n) && (n % (k + 1) == 0) {
16            Some(Params { n, k })
17        } else {
18            None
19        }
20    }
21    pub(crate) fn indices_per_hash_output(&self) -> u32 {
22        512 / self.n
23    }
24    pub(crate) fn hash_output(&self) -> u8 {
25        (self.indices_per_hash_output() * self.n / 8) as u8
26    }
27    pub(crate) fn collision_bit_length(&self) -> usize {
28        (self.n / (self.k + 1)) as usize
29    }
30    pub(crate) fn collision_byte_length(&self) -> usize {
31        (self.collision_bit_length() + 7) / 8
32    }
33    #[cfg(test)]
34    pub(crate) fn hash_length(&self) -> usize {
35        ((self.k as usize) + 1) * self.collision_byte_length()
36    }
